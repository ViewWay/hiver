1//! Hiver Cache - Caching abstraction module
2//! Hiver缓存 - 缓存抽象模块
3//!
4//! # Equivalent to Spring Boot / 等价于 Spring Boot
5//!
6//! - `@Cacheable` - Cacheable
7//! - `@CacheEvict` - `CacheEvict`
8//! - `@CachePut` - `CachePut`
9//! - `@Caching` - Caching (multi-operation)
10//! - `@CacheConfig` - `CacheConfig`
11//! - `CacheManager` - `CacheManager`
12//! - `@EnableCaching` - `EnableCaching`
13//!
14//! # Example / 示例
15//!
16//! ```rust,no_run,ignore
17//! use hiver_cache::{Cache, Cacheable, CachePut, CacheEvict};
18//! use std::sync::Arc;
19//!
20//! struct UserService {
21//!     cache: Arc<Cache<String, User>>,
22//! }
23//!
24//! impl UserService {
25//!     // Equivalent to @Cacheable("users")
26//!     #[cacheable("users")]
27//!     async fn get_user(&self, id: &str) -> Option<User> {
28//!         // Implementation
29//!         None
30//!     }
31//!
32//!     // Equivalent to @CachePut("users")
33//!     #[cache_put("users")]
34//!     async fn update_user(&self, user: User) -> User {
35//!         // Implementation
36//!         user
37//!     }
38//!
39//!     // Equivalent to @CacheEvict("users")
40//!     #[cache_evict("users")]
41//!     async fn delete_user(&self, id: &str) {
42//!         // Implementation
43//!     }
44//! }
45//! ```
