13/// Type-safe SQL parameter value
14/// 类型安全的 SQL 参数值
15///
16/// Used for parameterized queries (`$1, $2, ...`) to prevent SQL injection.
17/// 用于参数化查询（`$1, $2, ...`）以防止 SQL 注入。
18#[derive(Debug, Clone, PartialEq)]
19pub enum QueryParam
20{
21    /// NULL value / NULL 值
22    Null,
23    /// Boolean value / 布尔值
24    Bool(bool),
25    /// 32-bit integer / 32 位整数
26    I32(i32),
27    /// 64-bit integer / 64 位整数
28    I64(i64),
29    /// 64-bit float / 64 位浮点数
30    F64(f64),
31    /// Text string / 文本字符串
32    Text(String),
33    /// Binary data / 二进制数据
34    Bytes(Vec<u8>),
35}
36
37impl QueryParam
38{
39    /// Convert to an inline SQL literal (fallback for non-parameterized clients)
40    /// 转换为内联 SQL 字面量（非参数化客户端的回退）
41    pub fn to_sql_literal(&self) -> String
42    {
43        match self
44        {
45            Self::Null => "NULL".to_string(),
46            Self::Bool(b) => if *b { "TRUE" } else { "FALSE" }.to_string(),
47            Self::I32(n) => n.to_string(),
48            Self::I64(n) => n.to_string(),
49            Self::F64(n) => n.to_string(),
50            Self::Text(s) => format!("'{}'", s.replace('\'', "''").replace('\0', "")),
51            Self::Bytes(b) =>
52            {
53                let hex: String = b.iter().map(|byte| format!("{byte:02x}")).collect();
54                format!("'\\x{hex}'")
55            },
56        }
57    }
58}

128impl From<serde_json::Value> for QueryParam
129{
130    fn from(v: serde_json::Value) -> Self
131    {
132        match v
133        {
134            serde_json::Value::Null => Self::Null,
135            serde_json::Value::Bool(b) => Self::Bool(b),
136            serde_json::Value::Number(n) =>
137            {
138                if let Some(i) = n.as_i64()
139                {
140                    Self::I64(i)
141                }
142                else if let Some(f) = n.as_f64()
143                {
144                    Self::F64(f)
145                }
146                else
147                {
148                    Self::Text(n.to_string())
149                }
150            },
151            serde_json::Value::String(s) => Self::Text(s),
152            _ => Self::Text(v.to_string()),
153        }
154    }
155}

157/// Database client trait
158/// 数据库客户端 trait
159///
160/// Abstracts over different database backends (Postgres, MySQL, SQLite).
161/// 抽象不同数据库后端（Postgres, MySQL, SQLite）。
162#[async_trait::async_trait]
163pub trait DatabaseClient: Send + Sync
164{
165    /// Execute a query and return all rows
166    /// 执行查询并返回所有行
167    async fn fetch_all(&self, sql: &str) -> Result<Vec<Row>>;
168
169    /// Execute a query and return the first row
170    /// 执行查询并返回第一行
171    async fn fetch_one(&self, sql: &str) -> Result<Option<Row>>;
172
173    /// Execute a command (INSERT, UPDATE, DELETE) and return affected rows
174    /// 执行命令（INSERT, UPDATE, DELETE）并返回受影响行数
175    async fn execute_cmd(&self, sql: &str) -> Result<u64>;
176
177    /// Execute a parameterized query and return all rows
178    /// 执行参数化查询并返回所有行
179    #[allow(deprecated)]
180    async fn fetch_all_params(&self, sql: &str, params: &[QueryParam]) -> Result<Vec<Row>>
181    {
182        let interpolated = interpolate_params(sql, params);
183        self.fetch_all(&interpolated).await
184    }
185
186    /// Execute a parameterized query and return the first row
187    /// 执行参数化查询并返回第一行
188    #[allow(deprecated)]
189    async fn fetch_one_params(&self, sql: &str, params: &[QueryParam]) -> Result<Option<Row>>
190    {
191        let interpolated = interpolate_params(sql, params);
192        self.fetch_one(&interpolated).await
193    }
194
195    /// Execute a parameterized command and return affected rows
196    /// 执行参数化命令并返回受影响行数
197    #[allow(deprecated)]
198    async fn execute_params(&self, sql: &str, params: &[QueryParam]) -> Result<u64>
199    {
200        let interpolated = interpolate_params(sql, params);
201        self.execute_cmd(&interpolated).await
202    }
203
204    /// Begin a transaction
205    /// 开始事务
206    async fn begin_transaction(&self) -> Result<crate::Transaction>;
207
208    /// Get the current transaction from the active transaction context, if any.
209    /// 从活跃事务上下文中获取当前事务（如果有）。
210    ///
211    /// When the `tx-bridge` feature is enabled and a transaction has been started
212    /// via `RdbcTransactionManager`, this method returns it.
213    ///
214    /// 当启用 `tx-bridge` feature 且通过 `RdbcTransactionManager` 启动了事务时，
215    /// 此方法返回该事务。
216    fn current_transaction(&self) -> Option<crate::Transaction>
217    {
218        #[cfg(feature = "tx-bridge")]
219        {
220            crate::tx_bridge::try_current_transaction()
221        }
222        #[cfg(not(feature = "tx-bridge"))]
223        {
224            None
225        }
226    }
227
228    /// Ping the database
229    /// Ping 数据库
230    async fn ping(&self) -> Result<()>;
231
232    /// Close the client
233    /// 关闭客户端
234    async fn close(&self) -> Result<()>;
235}

237/// Replace `$1, $2, ...` placeholders with SQL literals.
238/// 将 `$1, $2, ...` 占位符替换为 SQL 字面量。
239///
240/// # Security Warning / 安全警告
241///
242/// This is a **fallback** that performs client-side string interpolation.
243/// It does NOT provide the same security guarantees as server-side parameter binding.
244/// Concrete `DatabaseClient` implementations SHOULD override `fetch_all_params`,
245/// `fetch_one_params`, and `execute_params` to use real parameterized queries.
246///
247/// 这是执行客户端字符串插值的**降级方案**，不提供与服务器端参数绑定相同的安全保证。
248/// 具体的 `DatabaseClient` 实现应覆盖 `fetch_all_params`、`fetch_one_params` 和 `execute_params`
249/// 以使用真正的参数化查询。
250#[deprecated = "Override fetch_all_params/fetch_one_params/execute_params in DatabaseClient \
251                implementations instead"]
252fn interpolate_params(sql: &str, params: &[QueryParam]) -> String
253{
254    let mut result = sql.to_string();
255    // Replace from highest index to lowest to prevent re-replacement
256    // when a parameter value contains `$N` patterns.
257    // 从最高索引到最低索引替换，防止参数值中的 `$N` 模式被二次替换。
258    for (i, param) in params.iter().enumerate().rev()
259    {
260        let placeholder = format!("${}", i + 1);
261        result = result.replace(&placeholder, &param.to_sql_literal());
262    }
263    result
264}

333    {
334        assert_eq!(QueryParam::Null.to_sql_literal(), "NULL");
335        assert_eq!(QueryParam::Bool(true).to_sql_literal(), "TRUE");
336        assert_eq!(QueryParam::I32(42).to_sql_literal(), "42");
337        assert_eq!(QueryParam::I64(100).to_sql_literal(), "100");
338        assert_eq!(QueryParam::F64(3.15).to_sql_literal(), "3.15");

340        assert_eq!(QueryParam::Text("it's".into()).to_sql_literal(), "'it''s'");
341        assert_eq!(QueryParam::Bytes(vec![0xDE, 0xAD]).to_sql_literal(), "'\\xdead'");

357    {
358        let sql = "SELECT * FROM users WHERE id = $1 AND name = $2";
359        let params = vec![QueryParam::I64(1), QueryParam::Text("Alice".into())];

363
364    #[test]
365    #[allow(deprecated)]
366    fn test_interpolate_params_sql_injection_safe()
367    {
368        let malicious = "'; DROP TABLE users; --";
369        let sql = "SELECT * FROM users WHERE name = $1";
370        let params = vec![QueryParam::Text(malicious.into())];
371        let result = interpolate_params(sql, &params);
372        // Single quotes are escaped (' → ''), wrapping in SQL quotes produces:
373        // WHERE name = '''; DROP TABLE users; --'
374        // Note: basic escaping alone is not a substitute for parameterized queries.
375        // The escaping neutralizes the injection by doubling the quote character.
376        assert!(result.contains("''';"), "leading quote should be escaped: {result}");
377    }
