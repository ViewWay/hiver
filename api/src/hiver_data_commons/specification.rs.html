1//! Specification API for type-safe dynamic queries
2//! 用于类型安全动态查询的 Specification API
3//!
4//! # Overview / 概述
5//!
6//! This module provides a JPA-inspired `Specification` pattern for building
7//! dynamic queries in a composable, type-safe manner.
8//! 本模块提供 JPA 风格的 `Specification` 模式，以可组合、类型安全的方式构建动态查询。
9//!
10//! # Equivalent to Spring / 等价于 Spring
11//!
12//! | Hiver | Spring Data JPA |
13//! |-------|-----------------|
14//! | `Specification` trait | `Specification<T>` |
15//! | `Specifications` builder | `Specifications` |
16//! | `Predicate` enum | `Predicate` |
17//! | `JpaSpecificationExecutor<T, ID>` | `JpaSpecificationExecutor<T>` |
18//!
19//! # Example / 示例
20//!
21//! ```rust,ignore
22//! use hiver_data_commons::specification::{Specifications, SpecValue};
23//!
24//! let spec = Specifications::where_clause()
25//!     .eq("status", SpecValue::String("active".into()))
26//!     .gt("age", SpecValue::I64(18))
27//!     .build();
28//! ```

34/// Literal value used in predicates.
35/// 谓词中使用的字面值。
36///
37/// A simplified value type for use in specification predicates.
38/// 用于 specification 谓词的简化值类型。
39///
40/// # Example / 示例
41///
42/// ```rust
43/// use hiver_data_commons::specification::SpecValue;
44///
45/// let v = SpecValue::String("Alice".into());
46/// let n = SpecValue::I64(42);
47/// let b = SpecValue::Bool(true);
48/// ```
49#[derive(Debug, Clone, PartialEq)]
50pub enum SpecValue
51{
52    /// String value / 字符串值
53    String(String),
54    /// 64-bit integer / 64 位整数
55    I64(i64),
56    /// 64-bit float / 64 位浮点数
57    F64(f64),
58    /// Boolean / 布尔值
59    Bool(bool),
60    /// Null value / 空值
61    Null,
62}
63
64impl fmt::Display for SpecValue
65{
66    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
67    {
68        match self
69        {
70            Self::String(s) => write!(f, "'{}'", s.replace('\'', "''")),
71            Self::I64(n) => write!(f, "{n}"),
72            Self::F64(n) => write!(f, "{n}"),
73            Self::Bool(b) => write!(f, "{b}"),
74            Self::Null => write!(f, "NULL"),
75        }
76    }

79/// Predicate representing a single filter condition.
80/// 表示单个过滤条件的谓词。
81///
82/// Predicates are the building blocks of specifications.
83/// They describe how entities should be filtered.
84/// 谓词是 specification 的构建块。它们描述实体应如何被过滤。
85///
86/// # Example / 示例
87///
88/// ```rust
89/// use hiver_data_commons::specification::{Predicate, SpecValue};
90///
91/// let pred = Predicate::eq("name", SpecValue::String("Alice".into()));
92/// let combined =
93///     Predicate::and(Box::new(pred), Box::new(Predicate::gt("age", SpecValue::I64(18))));
94/// ```
95#[derive(Debug, Clone, PartialEq)]
96pub enum Predicate
97{
98    /// Logical AND of two predicates / 两个谓词的逻辑与
99    And(Box<Predicate>, Box<Predicate>),
100    /// Logical OR of two predicates / 两个谓词的逻辑或
101    Or(Box<Predicate>, Box<Predicate>),
102    /// Logical NOT of a predicate / 谓词的逻辑非
103    Not(Box<Predicate>),
104    /// Equality: field = value / 等于：field = value
105    Eq(String, SpecValue),
106    /// Inequality: field != value / 不等于：field != value
107    Ne(String, SpecValue),
108    /// Greater than: field > value / 大于：field > value
109    Gt(String, SpecValue),
110    /// Greater than or equal: field >= value / 大于等于：field >= value
111    Gte(String, SpecValue),
112    /// Less than: field < value / 小于：field < value
113    Lt(String, SpecValue),
114    /// Less than or equal: field <= value / 小于等于：field <= value
115    Lte(String, SpecValue),
116    /// LIKE: field LIKE pattern / 模糊匹配：field LIKE pattern
117    Like(String, String),
118    /// IN: field IN (values...) / 包含于：field IN (values...)
119    In(String, Vec<SpecValue>),
120    /// NOT IN: field NOT IN (values...) / 不包含于：field NOT IN (values...)
121    NotIn(String, Vec<SpecValue>),
122    /// IS NULL / 为空
123    IsNull(String),
124    /// IS NOT NULL / 非空
125    IsNotNull(String),
126    /// BETWEEN: field BETWEEN low AND high / 范围：field BETWEEN low AND high
127    Between(String, SpecValue, SpecValue),
128}

238    /// Convert the predicate to an approximate SQL WHERE clause fragment.
239    /// 将谓词转换为近似的 SQL WHERE 子句片段。
240    ///
241    /// This is useful for debugging and logging.
242    /// 用于调试和日志记录。
243    pub fn to_sql(&self) -> String
244    {
245        match self
246        {
247            Self::And(l, r) => format!("({} AND {})", l.to_sql(), r.to_sql()),
248            Self::Or(l, r) => format!("({} OR {})", l.to_sql(), r.to_sql()),
249            Self::Not(inner) => format!("NOT ({})", inner.to_sql()),
250            Self::Eq(field, value) => format!("{} = {}", field, value),
251            Self::Ne(field, value) => format!("{} != {}", field, value),
252            Self::Gt(field, value) => format!("{} > {}", field, value),
253            Self::Gte(field, value) => format!("{} >= {}", field, value),
254            Self::Lt(field, value) => format!("{} < {}", field, value),
255            Self::Lte(field, value) => format!("{} <= {}", field, value),
256            Self::Like(field, pattern) => format!("{} LIKE '{}'", field, pattern),
257            Self::In(field, values) =>
258            {
259                let vals: Vec<String> = values.iter().map(|v| v.to_string()).collect();
260                format!("{} IN ({})", field, vals.join(", "))
261            },
262            Self::NotIn(field, values) =>
263            {
264                let vals: Vec<String> = values.iter().map(|v| v.to_string()).collect();
265                format!("{} NOT IN ({})", field, vals.join(", "))
266            },
267            Self::IsNull(field) => format!("{} IS NULL", field),
268            Self::IsNotNull(field) => format!("{} IS NOT NULL", field),
269            Self::Between(field, low, high) =>
270            {
271                format!("{} BETWEEN {} AND {}", field, low, high)
272            },
273        }
274    }
275}

280    {
281        write!(f, "{}", self.to_sql())
282    }

285/// Specification trait for composable query criteria.
286/// 可组合查询条件的 Specification trait。
287///
288/// Implementors produce a [`Predicate`] that can be translated
289/// into SQL, query parameters, or other query representations.
290/// 实现者生成一个 [`Predicate`]，可转换为 SQL、查询参数或其他查询表示。
291///
292/// This mirrors Spring Data JPA's `Specification<T>` interface.
293/// 这镜像了 Spring Data JPA 的 `Specification<T>` 接口。
294///
295/// # Example / 示例
296///
297/// ```rust,ignore
298/// use hiver_data_commons::specification::{Specification, Predicate, SpecValue};
299///
300/// struct ActiveUserSpec;
301///
302/// impl Specification for ActiveUserSpec {
303///     fn to_predicate(&self) -> Option<Predicate> {
304///         Some(Predicate::eq("status", SpecValue::String("active".into())))
305///     }
306/// }
307/// ```
308pub trait Specification: std::fmt::Debug + Send + Sync
309{
310    /// Convert this specification into a predicate.
311    /// 将此 specification 转换为谓词。
312    ///
313    /// Returns `None` if the specification has no constraints
314    /// (matches all entities).
315    /// 如果 specification 没有约束（匹配所有实体），返回 `None`。
316    fn to_predicate(&self) -> Option<Predicate>;
317}
318
319/// Blanket implementation: any `Fn() -> Option<Predicate>` is a Specification.
320/// 任何 `Fn() -> Option<Predicate>` 都是一个 Specification。
321impl<F> Specification for F
322where
323    F: Fn() -> Option<Predicate> + std::fmt::Debug + Send + Sync,
324{
325    fn to_predicate(&self) -> Option<Predicate>
326    {
327        self()
328    }
329}

522    /// Combine with another specification using AND.
523    /// 使用 AND 与另一个 specification 组合。
524    pub fn and(self, other: impl Specification) -> Self
525    {
526        match (self.predicate, other.to_predicate())
527        {
528            (Some(left), Some(right)) => Self {
529                predicate: Some(Predicate::and(Box::new(left), Box::new(right))),
530            },
531            (Some(p), None) => Self { predicate: Some(p) },
532            (None, Some(p)) => Self { predicate: Some(p) },
533            (None, None) => Self { predicate: None },
534        }
535    }

537    /// Combine with another specification using OR.

539    pub fn or(self, other: impl Specification) -> Self
540    {
541        match (self.predicate, other.to_predicate())
542        {
543            (Some(left), Some(right)) => Self {
544                predicate: Some(Predicate::or(Box::new(left), Box::new(right))),
545            },
546            (Some(p), None) => Self { predicate: Some(p) },
547            (None, Some(p)) => Self { predicate: Some(p) },
548            (None, None) => Self { predicate: None },
549        }
550    }

562            None => Self { predicate: None },
563        }
564    }

746/// A simple specification wrapping a single predicate.

748#[derive(Debug, Clone)]
749pub struct SimpleSpec(pub Predicate);

759/// AND composition of two specifications.
760/// 两个 specification 的 AND 组合。
761#[derive(Debug)]
762pub struct AndSpec
763{
764    left: Box<dyn Specification>,
765    right: Box<dyn Specification>,
766}
767
768impl Specification for AndSpec
769{
770    fn to_predicate(&self) -> Option<Predicate>
771    {
772        match (self.left.to_predicate(), self.right.to_predicate())
773        {
774            (Some(l), Some(r)) => Some(Predicate::and(Box::new(l), Box::new(r))),
775            (Some(p), None) => Some(p),
776            (None, Some(p)) => Some(p),
777            (None, None) => None,
778        }
779    }
780}
781
782/// OR composition of two specifications.
783/// 两个 specification 的 OR 组合。
784#[derive(Debug)]
785pub struct OrSpec
786{
787    left: Box<dyn Specification>,
788    right: Box<dyn Specification>,
789}
790
791impl Specification for OrSpec
792{
793    fn to_predicate(&self) -> Option<Predicate>
794    {
795        match (self.left.to_predicate(), self.right.to_predicate())
796        {
797            (Some(l), Some(r)) => Some(Predicate::or(Box::new(l), Box::new(r))),
798            (Some(p), None) => Some(p),
799            (None, Some(p)) => Some(p),
800            (None, None) => None,
801        }
802    }
803}
804
805/// NOT composition of a specification.
806/// specification 的 NOT 组合。
807#[derive(Debug)]
808pub struct NotSpec
809{
810    inner: Box<dyn Specification>,
811}
812
813impl Specification for NotSpec
814{
815    fn to_predicate(&self) -> Option<Predicate>
816    {
817        self.inner
818            .to_predicate()
819            .map(|p| Predicate::not(Box::new(p)))
820    }
821}

823/// Executor for specification-based queries.
824/// 基于 specification 的查询执行器。
825///
826/// This mirrors Spring Data JPA's `JpaSpecificationExecutor<T>` interface,
827/// allowing repositories to execute queries built from specifications.
828/// 这镜像了 Spring Data JPA 的 `JpaSpecificationExecutor<T>` 接口，
829/// 允许 repository 执行由 specification 构建的查询。
830///
831/// # Example / 示例
832///
833/// ```rust,ignore
834/// use hiver_data_commons::specification::{JpaSpecificationExecutor, Specification, SpecValue};
835/// use hiver_data_commons::PageRequest;
836///
837/// // Find active users / 查找活跃用户
838/// let spec = Specification::eq("status", SpecValue::String("active".into()));
839/// let users = executor.find_all_spec(spec).await?;
840///
841/// // Find with pagination / 分页查找
842/// let page = executor.find_all_spec_pageable(spec, PageRequest::of(0, 20)).await?;
843/// ```
844#[async_trait::async_trait]
845pub trait JpaSpecificationExecutor<T: Send + 'static, ID: Send + Sync + 'static>:
846    Send + Sync
847{
848    /// Error type for specification queries.
849    /// specification 查询的错误类型。
850    type Error: std::fmt::Debug + Send + Sync;
851
852    /// Find a single entity matching the specification.
853    /// 查找匹配 specification 的单个实体。
854    ///
855    /// Returns `Ok(None)` if no match is found.
856    /// 如果没有找到匹配项，返回 `Ok(None)`。
857    async fn find_one(&self, spec: impl Specification) -> Result<Option<T>, Self::Error>;
858
859    /// Find all entities matching the specification.
860    /// 查找所有匹配 specification 的实体。
861    async fn find_all_spec(&self, spec: impl Specification) -> Result<Vec<T>, Self::Error>;
862
863    /// Count entities matching the specification.
864    /// 统计匹配 specification 的实体数量。
865    async fn count_spec(&self, spec: impl Specification) -> Result<u64, Self::Error>;
866
867    /// Find entities matching the specification with pagination.
868    /// 分页查找匹配 specification 的实体。
869    async fn find_all_spec_pageable(
870        &self,
871        spec: impl Specification,
872        pageable: PageRequest,
873    ) -> Result<Page<T>, Self::Error>;
874}
