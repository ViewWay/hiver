7use std::sync::Arc;
8
9use hiver_http::Request;
10use tokio::sync::RwLock;
11
12use crate::{Transaction, TransactionStatus};

14/// `TransactionContext` extension for Request
15/// `Request的TransactionContext扩展`
16///
17/// This allows `TransactionContext` to be passed through Request extensions,
18/// making it available across async boundaries without `ThreadLocal`.
19///
20/// `这允许TransactionContext通过Request扩展传递`，
21/// 使其在异步边界间可用，无需ThreadLocal。
22///
23/// # Example / 示例
24///
25/// ```rust,no_run,ignore
26/// use hiver_tx::request_ext::TransactionContextExt;
27/// use hiver_http::Request;
28///
29/// async fn handler(req: Request) -> Result<Response> {
30///     // Get TransactionContext from Request
31///     let ctx = TransactionContextExt::from_request(&req)?;
32///     let tx = ctx.current_transaction().await;
33///     Ok(Response::json(tx))
34/// }
35/// ```
36#[derive(Clone)]
37pub struct TransactionContextExt
38{
39    /// Current transaction
40    /// 当前事务
41    current: Arc<RwLock<Option<Transaction>>>,
42
43    /// Transaction stack (for nested transactions)
44    /// 事务栈（用于嵌套事务）
45    stack: Arc<RwLock<Vec<Transaction>>>,
46}
47
48impl TransactionContextExt
49{
50    /// Create a new `TransactionContext` extension
51    /// `创建新的TransactionContext扩展`
52    pub fn new() -> Self
53    {
54        Self {
55            current: Arc::new(RwLock::new(None)),
56            stack: Arc::new(RwLock::new(Vec::new())),
57        }
58    }
59
60    /// Get `TransactionContext` from Request extensions
61    /// `从Request扩展中获取TransactionContext`
62    ///
63    /// Returns None if `TransactionContext` is not found in the request.
64    /// 如果请求中未找到TransactionContext，则返回None。
65    pub fn from_request(req: &Request) -> Option<Arc<Self>>
66    {
67        req.extensions().get::<Arc<Self>>().cloned()
68    }
69
70    /// Set `TransactionContext` to Request extensions
71    /// `将TransactionContext设置到Request扩展`
72    pub fn set_to_request(req: &mut Request) -> Arc<Self>
73    {
74        let ctx = Arc::new(Self::new());
75        req.extensions_mut().insert(ctx.clone());
76        ctx
77    }
78
79    /// Get current transaction
80    /// 获取当前事务
81    pub async fn current_transaction(&self) -> Option<Transaction>
82    {
83        self.current.read().await.clone()
84    }
85
86    /// Set current transaction
87    /// 设置当前事务
88    pub async fn set_current_transaction(&self, tx: Transaction)
89    {
90        let mut current = self.current.write().await;
91        *current = Some(tx);
92    }
93
94    /// Clear current transaction
95    /// 清除当前事务
96    pub async fn clear(&self)
97    {
98        let mut current = self.current.write().await;
99        *current = None;
100    }
101
102    /// Push transaction onto stack (for nested transactions)
103    /// 将事务压入栈（用于嵌套事务）
104    pub async fn push_transaction(&self, tx: Transaction)
105    {
106        let mut stack = self.stack.write().await;
107        stack.push(tx);
108    }
109
110    /// Pop transaction from stack
111    /// 从栈弹出事务
112    pub async fn pop_transaction(&self) -> Option<Transaction>
113    {
114        let mut stack = self.stack.write().await;
115        stack.pop()
116    }
117
118    /// Get stack depth
119    /// 获取栈深度
120    pub async fn stack_depth(&self) -> usize
121    {
122        self.stack.read().await.len()
123    }
124
125    /// Check if there is an active transaction
126    /// 检查是否有活动事务
127    pub async fn has_active_transaction(&self) -> bool
128    {
129        self.current
130            .read()
131            .await
132            .as_ref()
133            .is_some_and(Transaction::is_active)
134    }
135
136    /// Get transaction status
137    /// 获取事务状态
138    pub async fn transaction_status(&self) -> Option<TransactionStatus>
139    {
140        self.current
141            .read()
142            .await
143            .as_ref()
144            .map(|tx| tx.status().clone())
145    }
146}
147
148impl Default for TransactionContextExt
149{
150    fn default() -> Self
151    {
152        Self::new()
153    }
154}
155
156/// Convenience function: Get current transaction from Request
157/// 便捷函数：从Request获取当前事务
158///
159/// # Example / 示例
160///
161/// ```rust,no_run,ignore
162/// use hiver_tx::request_ext::get_transaction_from_request;
163/// use hiver_http::Request;
164///
165/// async fn handler(req: Request) -> Result<Response> {
166///     let tx = get_transaction_from_request(&req).await?;
167///     Ok(Response::json(tx))
168/// }
169/// ```
170pub async fn get_transaction_from_request(req: &Request) -> Option<Transaction>
171{
172    TransactionContextExt::from_request(req)?
173        .current_transaction()
174        .await
175}
176
177/// Convenience function: Check if request has active transaction
178/// 便捷函数：检查请求是否有活动事务
179pub async fn has_active_transaction_in_request(req: &Request) -> bool
180{
181    if let Some(ctx) = TransactionContextExt::from_request(req)
182    {
183        ctx.has_active_transaction().await
184    }
185    else
186    {
187        false
188    }
189}

199mod tests

201    use hiver_http::{Method, Request};
202
203    use super::*;
204    use crate::Transaction;
205
206    #[hiver_macros::test]
207    async fn test_transaction_context_ext()
208    {
209        let mut req = Request::from_method_uri(Method::GET, "/test");
210
211        // Set TransactionContext
212        let ctx = TransactionContextExt::set_to_request(&mut req);
213
214        // Get from Request
215        let ctx2 = TransactionContextExt::from_request(&req).unwrap();
216        assert_eq!(Arc::as_ptr(&ctx), Arc::as_ptr(&ctx2));
217
218        // Test transaction
219        let tx = Transaction::new("test");
220        ctx.set_current_transaction(tx.clone()).await;
221
222        assert!(ctx.has_active_transaction().await);
223
224        // Get from Request again
225        let tx_from_req = get_transaction_from_request(&req).await;
226        assert_eq!(tx_from_req.map(|t| t.status().name().to_string()), Some("test".to_string()));
227    }
