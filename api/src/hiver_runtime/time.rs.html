1//! Timer module
2//! 定时器模块
3//!
4//! # Overview / 概述
5//!
6//! This module provides efficient timer management using a hierarchical timing wheel.
7//! The timing wheel provides O(1) insertion, deletion, and advancement.
8//!
9//! 本模块使用分层时间轮提供高效的定时器管理。
10//! 时间轮提供O(1)插入、删除和推进操作。
11//!
12//! # Timer Wheel Algorithm / 时间轮算法
13//!
14//! The timer wheel is organized as a hierarchy of wheels with different granularities:
15//! - Wheel 0: 1ms resolution, 256 slots (256ms range)
16//! - Wheel 1: 256ms resolution, 64 slots (16.384s range)
17//! - Wheel 2: 16.384s resolution, 64 slots (1048.576s range)
18//! - Wheel 3: 1048.576s resolution, 64 slots (67108.864s range)
19//!
20//! 时间轮组织为具有不同粒度的层级：
21//! - 轮0：1ms分辨率，256个槽（256ms范围）
22//! - 轮1：256ms分辨率，64个槽（16.384s范围）
23//! - 轮2：16.384s分辨率，64个槽（1048.576s范围）
24//! - 轮3：1048.576s分辨率，64个槽（67108.864s范围）
25//!
26//! # Example / 示例
27//!
28//! ```rust,no_run,ignore
29//! use hiver_runtime::time::{sleep, Duration};
30//!
31//! async fn example() {
32//!     sleep(Duration::from_millis(100)).await;
33//!     println!("Woke up after 100ms");
34//! }
35//! ```

55/// Tick size in milliseconds (1ms)
56/// 滴答大小（毫秒）
57const TICK_MS: u64 = 1;
58
59/// Wheel 0: 1ms tick, 256 slots (256ms total)
60/// 轮0：1ms滴答，256个槽（总共256ms）
61const WHEEL0_SIZE: usize = 256;
62const WHEEL0_SHIFT: usize = 8; // 2^8 = 256
63const WHEEL0_MASK: usize = WHEEL0_SIZE - 1;
64
65/// Wheel 1: 256ms tick, 64 slots (16.384s total)
66/// 轮1：256ms滴答，64个槽（总共16.384s）
67const WHEEL1_SIZE: usize = 64;
68const WHEEL1_SHIFT: usize = 6; // 2^6 = 64
69const WHEEL1_MASK: usize = WHEEL1_SIZE - 1;
70
71/// Wheel 2: 16.384s tick, 64 slots (1048.576s total)
72/// 轮2：16.384s滴答，64个槽（总共1048.576s）
73const WHEEL2_SIZE: usize = 64;
74const WHEEL2_SHIFT: usize = 6;
75const WHEEL2_MASK: usize = WHEEL2_SIZE - 1;
76
77/// Wheel 3: 1048.576s tick, 64 slots (67108.864s total)
78/// 轮3：1048.576s滴答，64个槽（总共67108.864s）
79const WHEEL3_SIZE: usize = 64;
80#[allow(dead_code)]
81const WHEEL3_SHIFT: usize = 6;
82const WHEEL3_MASK: usize = WHEEL3_SIZE - 1;
83
84/// Maximum timeout in milliseconds (about 18.6 hours)
85/// 最大超时时间（毫秒，约18.6小时）
86#[allow(dead_code)]
87const MAX_TIMEOUT_MS: u64 =
88    (WHEEL0_SIZE * WHEEL1_SIZE * WHEEL2_SIZE * WHEEL3_SIZE) as u64 * TICK_MS;

90/// Timer entry in the timing wheel
91/// 时间轮中的定时器条目
92struct TimerEntry
93{
94    /// Unique identifier for this timer
95    /// 此定时器的唯一标识符
96    id: u64,
97    /// Expiration time in milliseconds (absolute)
98    /// 到期时间（毫秒，绝对值）
99    expiration_ms: u64,
100    /// Waker for this timer
101    /// 此定时器的waker
102    waker: Option<Waker>,
103    /// Whether this timer has been canceled
104    /// 此定时器是否已取消
105    #[allow(dead_code)]
106    canceled: Mutex<bool>,
107}

112/// A timing wheel slot containing timer entries
113/// 包含定时器条目的时间轮槽
114#[derive(Debug)]
115struct TimerSlot
116{
117    /// List of timer entries in this slot
118    /// 此槽中的定时器条目列表
119    timers: UnsafeCell<LinkedList<TimerEntry>>,
120}
121
122impl TimerSlot
123{
124    fn new() -> Self
125    {
126        Self {
127            timers: UnsafeCell::new(LinkedList::new()),
128        }
129    }
130
131    /// Add a timer to this slot
132    /// 向此槽添加定时器
133    unsafe fn push(&self, timer: TimerEntry)
134    {
135        let list = &mut *self.timers.get();
136        list.push_back(timer);
137    }
138
139    /// Get all timers from this slot, clearing it
140    /// 从此槽获取所有定时器并清空
141    unsafe fn take_all(&self) -> LinkedList<TimerEntry>
142    {
143        let list = &mut *self.timers.get();
144        std::mem::take(list)
145    }
146}

153/// Hierarchical timing wheel for efficient timer management
154/// 用于高效定时器管理的分层时间轮
155///
156/// Uses 4 wheels with different granularities to cover a wide range of timeouts.
157/// 使用4个具有不同粒度的轮来覆盖大范围的超时。
158pub struct TimerWheel
159{
160    /// Current time in ticks
161    /// 当前时间（滴答）
162    current_ticks: AtomicU64,
163    /// Wheel 0 (finest granularity)
164    /// 轮0（最细粒度）
165    wheel0: Box<[TimerSlot; WHEEL0_SIZE]>,
166    /// Wheel 1
167    /// 轮1
168    wheel1: Box<[TimerSlot; WHEEL1_SIZE]>,
169    /// Wheel 2
170    /// 轮2
171    wheel2: Box<[TimerSlot; WHEEL2_SIZE]>,
172    /// Wheel 3 (coarsest granularity)
173    /// 轮3（最粗粒度）
174    wheel3: Box<[TimerSlot; WHEEL3_SIZE]>,
175    /// Next timer ID
176    /// 下一个定时器ID
177    next_id: AtomicU64,
178    /// Active timer registry for cancellation (ID -> slot index)
179    /// 活跃定时器注册表用于取消（ID -> 槽索引）
180    timer_registry: Mutex<HashMap<u64, TimerLocation>>,
181}
182
183/// Location of a timer in the wheel
184/// 定时器在轮中的位置
185#[derive(Clone, Copy, Debug)]
186struct TimerLocation
187{
188    /// Wheel level (0-3)
189    #[allow(dead_code)]
190    wheel_level: u8,
191    /// Slot index within the wheel
192    #[allow(dead_code)]
193    slot_index: usize,
194}

196// SAFETY: TimerWheel uses atomic operations and interior mutability
197// TimerWheel使用原子操作和内部可变性
198unsafe impl Send for TimerWheel {}
199unsafe impl Sync for TimerWheel {}

203    /// Create a new timer wheel
204    /// 创建新的时间轮
205    pub fn new() -> Self
206    {
207        Self {
208            current_ticks: AtomicU64::new(0),
209            wheel0: (0..WHEEL0_SIZE)

211                .collect::<Vec<_>>()
212                .into_boxed_slice()
213                .try_into()
214                .unwrap(),
215            wheel1: (0..WHEEL1_SIZE)
216                .map(|_| TimerSlot::new())
217                .collect::<Vec<_>>()
218                .into_boxed_slice()
219                .try_into()
220                .unwrap(),
221            wheel2: (0..WHEEL2_SIZE)
222                .map(|_| TimerSlot::new())
223                .collect::<Vec<_>>()
224                .into_boxed_slice()
225                .try_into()
226                .unwrap(),
227            wheel3: (0..WHEEL3_SIZE)
228                .map(|_| TimerSlot::new())
229                .collect::<Vec<_>>()
230                .into_boxed_slice()
231                .try_into()
232                .unwrap(),
233            next_id: AtomicU64::new(1),
234            timer_registry: Mutex::new(HashMap::new()),
235        }
236    }

238    /// Cancel a timer by ID
239    /// 通过ID取消定时器
240    pub fn cancel_timer(&self, id: u64) -> bool
241    {
242        let mut registry = self.timer_registry.lock().unwrap();
243        if let Some(_location) = registry.remove(&id)
244        {
245            // Timer was found and removed from registry
246            // The actual cancellation will be checked when the timer expires
247            // 定时器已找到并从注册表中移除
248            // 实际取消将在定时器到期时检查
249            true
250        }
251        else
252        {
253            false
254        }
255    }

257    /// Get the current tick count
258    /// 获取当前滴答计数
259    #[inline]
260    pub fn current_ticks(&self) -> u64
261    {
262        self.current_ticks.load(Ordering::Acquire)
263    }
264
265    /// Advance the timer wheel by the specified number of ticks
266    /// 将时间轮推进指定数量的滴答
267    ///
268    /// Returns the number of timers that expired during this advancement.
269    /// 返回在此推进期间到期的定时器数量。
270    pub fn advance(&self, ticks: u64) -> usize
271    {
272        let mut expired = 0;
273        let _start = self.current_ticks.load(Ordering::Acquire);
274
275        for _ in 0..ticks
276        {
277            let tick = self.current_ticks.fetch_add(1, Ordering::AcqRel);
278            let pos0 = (tick & WHEEL0_MASK as u64) as usize;
279
280            // Process wheel 0
281            // 处理轮0
282            unsafe {
283                let timers = self.wheel0[pos0].take_all();
284                for timer in timers
285                {
286                    // Check if timer is still in registry (not canceled)
287                    // 检查定时器是否仍在注册表中（未取消）
288                    let is_active = {
289                        let mut registry = self.timer_registry.lock().unwrap();
290                        registry.remove(&timer.id).is_some()
291                    };
292
293                    if is_active
294                    {
295                        // Try to wake the timer
296                        // 尝试唤醒定时器
297                        if let Some(waker) = timer.waker
298                        {
299                            waker.wake();
300                        }
301                        expired += 1;
302                    }
303                }
304            }
305
306            // Cascade to wheel 1 every WHEEL0_SIZE ticks
307            // 每WHEEL0_SIZE个滴答级联到轮1
308            if tick & (WHEEL0_SIZE as u64 - 1) == 0
309            {
310                let pos1 = ((tick >> WHEEL0_SHIFT) & WHEEL1_MASK as u64) as usize;
311                unsafe {
312                    let timers = self.wheel1[pos1].take_all();
313                    for timer in timers
314                    {
315                        // Check if timer is still in registry
316                        // 检查定时器是否仍在注册表中
317                        let is_active = {
318                            let registry = self.timer_registry.lock().unwrap();
319                            registry.contains_key(&timer.id)
320                        };
321
322                        if is_active
323                        {
324                            // Re-insert into wheel 0
325                            // 重新插入轮0
326                            self.insert_timer_inner(timer);
327                        }
328                    }
329                }
330            }
331
332            // Cascade to wheel 2 every (WHEEL0_SIZE * WHEEL1_SIZE) ticks
333            // 每(WHEEL0_SIZE * WHEEL1_SIZE)个滴答级联到轮2
334            if tick & ((WHEEL0_SIZE * WHEEL1_SIZE) as u64 - 1) == 0
335            {
336                let pos2 = ((tick >> (WHEEL0_SHIFT + WHEEL1_SHIFT)) & WHEEL2_MASK as u64) as usize;
337                unsafe {
338                    let timers = self.wheel2[pos2].take_all();
339                    for timer in timers
340                    {
341                        // Check if timer is still in registry
342                        let is_active = {
343                            let registry = self.timer_registry.lock().unwrap();
344                            registry.contains_key(&timer.id)
345                        };
346
347                        if is_active
348                        {
349                            self.insert_timer_inner(timer);
350                        }
351                    }
352                }
353            }
354
355            // Cascade to wheel 3 every (WHEEL0_SIZE * WHEEL1_SIZE * WHEEL2_SIZE) ticks
356            // 每(WHEEL0_SIZE * WHEEL1_SIZE * WHEEL2_SIZE)个滴答级联到轮3
357            if tick & ((WHEEL0_SIZE * WHEEL1_SIZE * WHEEL2_SIZE) as u64 - 1) == 0
358            {
359                let pos3 = ((tick >> (WHEEL0_SHIFT + WHEEL1_SHIFT + WHEEL2_SHIFT))
360                    & WHEEL3_MASK as u64) as usize;
361                unsafe {
362                    let timers = self.wheel3[pos3].take_all();
363                    for timer in timers
364                    {
365                        // Check if timer is still in registry
366                        let is_active = {
367                            let registry = self.timer_registry.lock().unwrap();
368                            registry.contains_key(&timer.id)
369                        };
370
371                        if is_active
372                        {
373                            self.insert_timer_inner(timer);
374                        }
375                    }
376                }
377            }
378        }
379
380        expired
381    }

383    /// Insert a timer into the wheel
384    /// 向时间轮插入定时器
385    fn insert_timer_inner(&self, timer: TimerEntry)
386    {
387        let current = self.current_ticks.load(Ordering::Acquire);
388        let expiration = timer.expiration_ms / TICK_MS;
389        let id = timer.id;
390
391        if expiration <= current
392        {
393            // Already expired, wake immediately
394            // 已到期，立即唤醒
395            if let Some(waker) = timer.waker
396            {
397                waker.wake();
398            }
399            return;
400        }
401
402        // Determine wheel level and position before inserting
403        // 在插入前确定轮层级和位置
404        let ticks = expiration - current;
405        let (wheel_level, pos) = if ticks < WHEEL0_SIZE as u64
406        {
407            (0u8, ((current + ticks) & WHEEL0_MASK as u64) as usize)
408        }
409        else if ticks < (WHEEL0_SIZE * WHEEL1_SIZE) as u64
410        {
411            (1u8, (((current + ticks) >> WHEEL0_SHIFT) & WHEEL1_MASK as u64) as usize)
412        }
413        else if ticks < (WHEEL0_SIZE * WHEEL1_SIZE * WHEEL2_SIZE) as u64
414        {
415            (
416                2u8,
417                (((current + ticks) >> (WHEEL0_SHIFT + WHEEL1_SHIFT)) & WHEEL2_MASK as u64)
418                    as usize,
419            )
420        }
421        else
422        {
423            (
424                3u8,
425                (((current + ticks) >> (WHEEL0_SHIFT + WHEEL1_SHIFT + WHEEL2_SHIFT))
426                    & WHEEL3_MASK as u64) as usize,
427            )
428        };
429
430        // Add to registry before inserting into wheel
431        // 在插入到轮之前添加到注册表
432        {
433            let mut registry = self.timer_registry.lock().unwrap();
434            registry.insert(id, TimerLocation {
435                wheel_level,
436                slot_index: pos,
437            });
438        }
439
440        // Insert into appropriate wheel
441        // 插入到适当的轮中
442        match wheel_level
443        {
444            0 =>
445            unsafe { self.wheel0[pos].push(timer) },
446            1 =>
447            unsafe { self.wheel1[pos].push(timer) },
448            2 =>
449            unsafe { self.wheel2[pos].push(timer) },
450            _ =>
451            unsafe { self.wheel3[pos].push(timer) },
452        }
453    }

455    /// Insert a timer with the specified duration
456    /// 插入具有指定持续时间的定时器
457    pub fn insert_timer(&self, duration: Duration) -> TimerHandle
458    {
459        let duration_ms = duration.as_millis() as u64;
460        let current = self.current_ticks.load(Ordering::Acquire);
461        let expiration_ms = (current * TICK_MS) + duration_ms;
462
463        let id = self.next_id.fetch_add(1, Ordering::AcqRel);
464
465        let timer = TimerEntry {
466            id,
467            expiration_ms,
468            waker: None,
469            canceled: Mutex::new(false),
470        };
471
472        // Insert into wheel and registry
473        // 插入到轮和注册表中
474        self.insert_timer_inner(timer);
475
476        TimerHandle::new(id)
477    }

479    /// Insert a timer with the specified duration and associated waker
480    /// 插入具有指定持续时间和关联waker的定时器
481    pub fn insert_timer_with_waker(&self, duration: Duration, waker: Waker) -> TimerHandle
482    {
483        let duration_ms = duration.as_millis() as u64;
484        let current = self.current_ticks.load(Ordering::Acquire);
485        let expiration_ms = (current * TICK_MS) + duration_ms;
486
487        let id = self.next_id.fetch_add(1, Ordering::Relaxed);
488
489        let timer = TimerEntry {
490            id,
491            expiration_ms,
492            waker: Some(waker),
493            canceled: Mutex::new(false),
494        };
495
496        self.insert_timer_inner(timer);
497
498        TimerHandle::new(id)
499    }

572/// Sleep future that completes after the specified duration
573/// 在指定持续时间后完成的sleep future
574pub struct Sleep
575{
576    /// Duration to sleep / 睡眠持续时间
577    duration: Duration,
578    /// Whether the timer has been registered
579    /// 定时器是否已注册
580    registered: bool,
581    /// Start time / 开始时间
582    start: Option<Instant>,
583}
584
585impl Sleep
586{
587    /// Create a new sleep future
588    /// 创建新的sleep future
589    pub fn new(duration: Duration) -> Self
590    {
591        Self {
592            duration,
593            registered: false,
594            start: None,
595        }
596    }
597}
598
599impl Future for Sleep
600{
601    type Output = ();
602
603    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<()>
604    {
605        if self.registered
606        {
607            // Check if the duration has elapsed
608            // 检查持续时间是否已过
609            if let Some(start) = self.start
610                && start.elapsed() >= self.duration
611            {
612                return Poll::Ready(());
613            }
614            Poll::Pending
615        }
616        else
617        {
618            // First poll: register the timer
619            // 第一次轮询：注册定时器
620            self.registered = true;
621            self.start = Some(Instant::now());
622
623            // Insert timer into the global timer wheel
624            // 将定时器插入全局时间轮
625            global_timer().insert_timer_with_waker(self.duration, cx.waker().clone());
626
627            // Check if already expired
628            // 检查是否已到期
629            Poll::Pending
630        }
631    }
632}

652/// Sleep until the specified instant
653/// 睡眠直到指定时刻
654///
655/// # Example / 示例
656///
657/// ```rust,no_run,ignore
658/// use hiver_runtime::time::{sleep_until, Instant, Duration};
659///
660/// async fn example() {
661///     let deadline = Instant::now() + Duration::from_secs(5);
662///     sleep_until(deadline).await;
663///     println!("5 seconds have passed");
664/// }
665/// ```
666pub fn sleep_until(instant: Instant) -> SleepUntil
667{
668    let now = Instant::now();
669    let duration = if instant > now
670    {
671        instant.duration_since(now)
672    }
673    else
674    {
675        Duration::ZERO
676    };
677
678    SleepUntil {
679        sleep: sleep(duration),
680    }
681}

715/// ```
716pub fn interval(duration: Duration) -> Interval
717{
718    Interval {
719        duration,
720        next: Instant::now(),
721    }
722}
723
724/// Interval stream
725/// 间隔流
726pub struct Interval
727{
728    duration: Duration,
729    next: Instant,
730}
731
732impl Interval
733{
734    /// Wait for the next tick
735    /// 等待下一个滴答
736    pub async fn tick(&mut self) -> Instant
737    {
738        let now = Instant::now();
739        if now >= self.next
740        {
741            self.next = now + self.duration;
742        }
743
744        sleep_until(self.next).await;
745        self.next
746    }
747}
