use hiver_extractors::Query;
use serde::Deserialize;

#[derive(Deserialize)]
struct SearchParams {
    q: String,
    page: Option<u32>,
    size: Option<u32>,
}

// GET /search?q=rust&page=1&size=10
async fn search(Query(params): Query<SearchParams>) -> String {
    format!("Search: {}, Page: {:?}", params.q, params.page)
}
