10/// 基础 CRUD Repository trait
11/// Basic CRUD Repository trait
12///
13/// # Example / 示例
14///
15/// ```rust,no_run,ignore
16/// use hiver_data_annotations::CrudRepository;
17/// use hiver_data_annotations::Entity;
18///
19/// #[Entity]
20/// #[Table(name = "users")]
21/// pub struct User {
22///     #[Id]
23///     pub id: i64,
24///     pub username: String,
25/// }
26///
27/// #[async_trait]
28/// impl CrudRepository<User, i64> for UserRepository {
29///     async fn save(&self, entity: &User) -> Result<User, Error> {
30///         self.repository.insert(entity).await
31///     }
32///
33///     async fn find_by_id(&self, id: i64) -> Result<Option<User>, Error> {
34///         self.repository.find_by_id(id).await
35///     }
36///
37///     async fn find_all(&self) -> Result<Vec<User>, Error> {
38///         self.repository.find_all().await
39///     }
40///
41///     async fn delete_by_id(&self, id: i64) -> Result<bool, Error> {
42///         self.repository.delete_by_id(id).await
43///     }
44///
45///     async fn count(&self) -> Result<i64, Error> {
46///         self.repository.count().await
47///     }
48/// }
49/// ```
50#[async_trait]
51pub(crate) trait CrudRepository<T, ID>: Send + Sync
52where
53    T: Send + Sync,
54    ID: Send + Sync + 'static,
55{
56    /// 保存实体（新增或更新）
57    /// Save entity (insert or update)
58    async fn save(&self, entity: &T) -> Result<T, Error>;
59
60    /// 根据 ID 查找实体
61    /// Find entity by ID
62    async fn find_by_id(&self, id: ID) -> Result<Option<T>, Error>;
63
64    /// 查找所有实体
65    /// Find all entities
66    async fn find_all(&self) -> Result<Vec<T>, Error>;
67
68    /// 根据 ID 删除实体
69    /// Delete entity by ID
70    async fn delete_by_id(&self, id: ID) -> Result<bool, Error>;
71
72    /// 统计实体数量
73    /// Count all entities
74    async fn count(&self) -> Result<i64, Error>;
75
76    /// 检查实体是否存在
77    /// Check if entity exists by ID
78    async fn exists_by_id(&self, id: ID) -> Result<bool, Error>
79    {
80        match self.find_by_id(id).await
81        {
82            Ok(_) => Ok(true),
83            Err(_) => Ok(false),
84        }
85    }
86}

88/// 可分页的 Repository
89/// Pageable Repository
90///
91/// # Example / 示例
92///
93/// ```rust,no_run,ignore
94/// use hiver_data_annotations::PagingRepository;
95///
96/// #[async_trait]
97/// impl PagingRepository<User> for UserRepository {
98///     async fn find_all_pageable(
99///         &self,
100///         pageable: &PageRequest
101///     ) -> Result<Page<User>, Error> {
102///         // 实现分页查询
103///     }
104/// }
105/// ```
106#[async_trait]
107pub(crate) trait PagingRepository<T>: Send + Sync
108where
109    T: Send + Sync,
110{
111    /// 分页查询所有实体
112    /// Find all entities with pagination
113    async fn find_all_pageable(&self, pageable: &PageRequest) -> Result<Page<T>, Error>;
114
115    /// 根据条件分页查询
116    /// Find entities by criteria with pagination
117    async fn find_by_criteria_pageable(
118        &self,
119        criteria: &QueryCriteria,
120        pageable: &PageRequest,
121    ) -> Result<Page<T>, Error>;
122}

124/// 分页请求
125/// Page request
126#[derive(Clone, Debug)]
127pub(crate) struct PageRequest
128{
129    /// 页码（从 0 开始）/ Page number (0-indexed)
130    pub page: usize,
131
132    /// 每页大小 / Page size
133    pub size: usize,
134
135    /// 排序字段 / Sort field
136    pub sort: Option<String>,
137
138    /// 排序方向 / Sort direction
139    pub direction: SortDirection,
140}
141
142/// 排序方向
143/// Sort direction
144#[derive(Clone, Copy, Debug, PartialEq, Eq)]
145pub(crate) enum SortDirection
146{
147    /// 升序 / Ascending
148    Asc,
149
150    /// 降序 / Descending
151    Desc,
152}
153
154impl PageRequest
155{
156    /// 创建分页请求 / Create page request
157    ///
158    /// # Example / 示例
159    ///
160    /// ```rust,no_run,ignore
161    /// use hiver_data_annotations::PageRequest;
162    ///
163    /// // 第 0 页，每页 20 条，按 ID 降序
164    /// let page = PageRequest::new(0, 20)
165    ///     .with_sort("id", SortDirection::Desc);
166    /// ```
167    pub(crate) fn new(page: usize, size: usize) -> Self
168    {
169        Self {
170            page,
171            size,
172            sort: None,
173            direction: SortDirection::Asc,
174        }
175    }
176
177    /// 设置排序字段
178    /// Set sort field
179    pub(crate) fn with_sort(mut self, field: impl Into<String>, direction: SortDirection) -> Self
180    {
181        self.sort = Some(field.into());
182        self.direction = direction;
183        self
184    }
185
186    /// 获取偏移量 / Get offset
187    pub(crate) fn offset(&self) -> usize
188    {
189        self.page * self.size
190    }
191
192    /// 创建下一页请求 / Create next page request
193    pub(crate) fn next(&self) -> Option<Self>
194    {
195        Some(PageRequest::new(self.page + 1, self.size))
196    }
197
198    /// 创建上一页请求 / Create previous page request
199    pub(crate) fn previous(&self) -> Option<Self>
200    {
201        if self.page > 0
202        {
203            Some(PageRequest::new(self.page - 1, self.size))
204        }
205        else
206        {
207            None
208        }
209    }
210
211    /// 第一页 / First page
212    pub(crate) fn first(&self) -> Self
213    {
214        Self::new(0, self.size)
215    }
216}
217
218impl Default for PageRequest
219{
220    fn default() -> Self
221    {
222        Self::new(0, 20)
223    }
224}

226/// 分页结果
227/// Page result
228#[derive(Clone, Debug)]
229pub(crate) struct Page<T>
230{
231    /// 内容 / Content
232    pub content: Vec<T>,
233
234    /// 当前页码 / Current page number
235    pub number: usize,
236
237    /// 每页大小 / Page size
238    pub size: usize,
239
240    /// 总元素数 / Total elements
241    pub total_elements: i64,
242
243    /// 总页数 / Total pages
244    pub total_pages: usize,
245
246    /// 是否第一页 / Is first page
247    pub first: bool,
248
249    /// 是否最后一页 / Is last page
250    pub last: bool,
251
252    /// 是否有下一页 / Has next page
253    pub has_next: bool,
254
255    /// 是否有上一页 / Has previous page
256    pub has_previous: bool,
257}
258
259impl<T> Page<T>
260{
261    /// 创建分页结果 / Create page result
262    pub(crate) fn new(content: Vec<T>, number: usize, size: usize, total_elements: i64) -> Self
263    {
264        let total_pages = if total_elements == 0
265        {
266            0
267        }
268        else
269        {
270            ((total_elements as usize - 1) / size) + 1
271        };
272
273        let first = number == 0;
274        let last = number >= total_pages.saturating_sub(1);
275        let has_next = !last;
276        let has_previous = !first;
277
278        Self {
279            content,
280            number,
281            size,
282            total_elements,
283            total_pages,
284            first,
285            last,
286            has_next,
287            has_previous,
288        }
289    }
290
291    /// 获取空分页 / Get empty page
292    pub(crate) fn empty() -> Self
293    {
294        Self::new(Vec::new(), 0, 20, 0)
295    }
296
297    /// 获取总页数 / Get total pages
298    pub(crate) fn total_pages(&self) -> usize
299    {
300        self.total_pages
301    }
302
303    /// 获取元素总数 / Get total elements
304    pub(crate) fn total_elements(&self) -> i64
305    {
306        self.total_elements
307    }
308
309    /// 是否为空 / Is empty
310    pub(crate) fn is_empty(&self) -> bool
311    {
312        self.content.is_empty()
313    }
314
315    /// 获取元素数量 / Get number of elements
316    pub(crate) fn number_of_elements(&self) -> usize
317    {
318        self.content.len()
319    }
320
321    /// 转换内容类型 / Transform content
322    pub(crate) fn map<U, F>(self, f: F) -> Page<U>
323    where
324        T: Send + Sync,
325        U: Send + Sync,
326        F: Fn(T) -> U,
327    {
328        Page {
329            content: self.content.into_iter().map(f).collect(),
330            number: self.number,
331            size: self.size,
332            total_elements: self.total_elements,
333            total_pages: self.total_pages,
334            first: self.first,
335            last: self.last,
336            has_next: self.has_next,
337            has_previous: self.has_previous,
338        }
339    }
340}

342/// 查询条件
343/// Query criteria
344#[derive(Clone, Debug)]
345pub(crate) struct QueryCriteria
346{
347    /// 条件表达式 / Criteria expression
348    pub expression: String,
349
350    /// 参数绑定 / Parameter bindings
351    pub bindings: Vec<(String, serde_json::Value)>,
352}
353
354impl QueryCriteria
355{
356    /// 创建查询条件 / Create query criteria
357    pub(crate) fn new(expression: impl Into<String>) -> Self
358    {
359        Self {
360            expression: expression.into(),
361            bindings: Vec::new(),
362        }
363    }
364
365    /// 添加参数绑定 / Add parameter binding
366    pub(crate) fn bind(
367        mut self,
368        key: impl Into<String>,
369        value: impl Into<serde_json::Value>,
370    ) -> Self
371    {
372        self.bindings.push((key.into(), value.into()));
373        self
374    }
375}
