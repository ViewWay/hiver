1//! # HTTP Server Example (real socket) / HTTP 服务端示例（真实 socket）
2//!
3//! Proves the full end-to-end request path on Hiver's own stack:
4//! 1. `hiver-runtime` (custom reactor) drives async I/O with proper wakers.
5//! 2. `hiver-http::Server` binds a TCP socket, parses HTTP/1.1, dispatches.
6//! 3. `hiver-router::Router` matches the path and invokes the handler.
7//! 4. `#[get]` macro compiles (parameter extraction via `FromRequest`).
8//!
9//! 验证 Hiver 自有技术栈上的完整端到端请求路径:
10//! 1. `hiver-runtime`(自定义 reactor)以正确的 waker 驱动异步 I/O。
11//! 2. `hiver-http::Server` 绑定 TCP 套接字、解析 HTTP/1.1、分发。
12//! 3. `hiver-router::Router` 匹配路径并调用处理程序。
13//! 4. `#[get]` 宏可编译(经 `FromRequest` 提取参数)。
14//!
15//! ## Run / 运行
16//!
17//! ```bash
18//! cargo run -p hiver-examples --bin http_server_example
19//! # then in another terminal:
20//! curl http://127.0.0.1:8080/hello
21//! curl http://127.0.0.1:8080/echo
22//! ```
23
24use hiver_http::{IntoResponse, Request, Response, Result as HttpResult, StatusCode};
25use hiver_macros::get;
26use hiver_router::Router;
27use hiver_runtime::Runtime;
28
29/// Macro-annotated handler — proves `#[get]` + `FromRequest` compile.
30/// 宏标注的处理程序 —— 证明 `#[get]` + `FromRequest` 可编译。
31#[get("/hello")]
32async fn hello() -> &'static str
33{
34    "Hello from Hiver!"
35}
36
37/// Handler that takes the raw Request (identity extractor) and returns a
38/// `Result<Response>` — the signature `Router::get` expects.
39/// 接收原始 Request 的处理程序(身份提取器),返回 `Result<Response>` ——
40/// `Router::get` 所期望的签名。
41async fn echo(_req: Request) -> HttpResult<Response>
42{
43    Ok("echo".to_string().into_response())
44}
45
46fn main() -> std::io::Result<()>
47{
48    let mut runtime = Runtime::new()?;
49
50    println!("Hiver HTTP server starting on 127.0.0.1:8080 ...");
51    println!("Try: curl http://127.0.0.1:8080/hello");
52
53    runtime.block_on(async move {
54        // Build the router standalone via the builder API. (The `#[get]` macro
55        // above also compiles and registers via inventory when run inside an
56        // ApplicationContext; here we wire routes by hand for a standalone
57        // server that doesn't need the full starter.)
58        // 经构建器 API 独立构建 router。(上面的 `#[get]` 宏也可编译,并在
59        // ApplicationContext 内运行时经 inventory 注册;此处为无需完整
60        // starter 的独立服务端,手动接线路由。)
61        let router: Router<()> = Router::new()
62            .get("/hello", |req: Request| async {
63                // Wrap the macro-defined handler: it takes no args and returns
64                // &str, so we call it and convert.
65                // 包装宏定义的处理程序:它无参数且返回 &str,故调用并转换。
66                let _ = req;
67                HttpResult::Ok(hello().await.into_response())
68            })
69            .get("/echo", echo);
70
71        let _warmup: Response = "warmup".to_string().into_response();
72
73        let server = hiver_http::Server::bind("127.0.0.1:8080");
74        let _: HttpResult<()> = async { server.run(router).await }.await;
75    })?;
76
77    Ok(())
78}
