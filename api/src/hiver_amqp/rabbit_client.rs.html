1//! Real RabbitMQ client using lapin.
2//! 使用 lapin 的真实 RabbitMQ 客户端。
3//!
4//! # Equivalent to Spring / 等价于 Spring
5//!
6//! | Hiver | Spring AMQP |
7//! |-------|-------------|
8//! | `RabbitMqClient::new()` | `CachingConnectionFactory` + `RabbitTemplate` |
9//! | `client.publish()` | `rabbitTemplate.convertAndSend()` |
10//! | `client.consume()` | `@RabbitListener` / `SimpleMessageListenerContainer` |
11//!
12//! # Example / 示例
13//!
14//! ```rust,no_run,ignore
15//! use hiver_amqp::RabbitMqClient;
16//!
17//! let client = RabbitMqClient::connect("amqp://guest:guest@localhost:5672/%2f").await?;
18//!
19//! // Publish a message / 发布消息
20//! client.publish("my.exchange", "routing.key", b"Hello, RabbitMQ!").await?;
21//!
22//! // Consume messages / 消费消息
23//! client.consume("my_queue", |delivery| async move {
24//!     println!("Received: {:?}", String::from_utf8_lossy(&delivery.data));
25//!     Ok(())
26//! }).await?;
27//! ```

30mod inner
31{
32    use std::sync::Arc;

34    use lapin::{
35        BasicProperties, Channel, Connection, ConnectionProperties,
36        options::{
37            BasicAckOptions, BasicConsumeOptions, BasicNackOptions, BasicPublishOptions,
38            BasicQosOptions, ExchangeDeclareOptions, QueueBindOptions, QueueDeclareOptions,
39        },
40        types::FieldTable,
41    };
42    use tokio::sync::Mutex;
43    use tracing::{debug, error, info};

45    /// Error type for RabbitMQ operations.
46    /// RabbitMQ 操作错误类型。
47    #[derive(Debug, thiserror::Error)]
48    pub enum RabbitError
49    {
50        /// lapin library error. / lapin 库错误。
51        #[error("lapin error: {0}")]
52        Lapin(#[from] lapin::Error),
53        /// Channel has been closed. / 通道已关闭。
54        #[error("channel closed")]
55        ChannelClosed,
56        /// Connection attempt failed. / 连接尝试失败。
57        #[error("connection failed: {0}")]
58        ConnectionFailed(String),
59    }
60
61    pub(super) type Result<T> = std::result::Result<T, RabbitError>;
62
63    /// Shared lapin connection + channel pool.
64    /// 共享的 lapin 连接和通道池。
65    #[derive(Clone)]
66    pub struct RabbitMqClient
67    {
68        connection: Arc<Connection>,
69        channel: Arc<Mutex<Channel>>,
70        amqp_url: String,
71    }
72
73    impl RabbitMqClient
74    {
75        /// Connect to RabbitMQ and open a channel.
76        /// 连接到 RabbitMQ 并打开通道。
77        pub async fn connect(amqp_url: &str) -> Result<Self>
78        {
79            info!("Connecting to RabbitMQ at {}", amqp_url);
80            let conn = Connection::connect(amqp_url, ConnectionProperties::default())
81                .await
82                .map_err(|e| RabbitError::ConnectionFailed(e.to_string()))?;
83            let channel = conn.create_channel().await?;
84            info!("RabbitMQ channel opened");
85            Ok(Self {
86                connection: Arc::new(conn),
87                channel: Arc::new(Mutex::new(channel)),
88                amqp_url: amqp_url.to_string(),
89            })
90        }
91
92        /// Declare a durable queue. Idempotent.
93        /// 声明一个持久化队列（幂等）。
94        pub async fn declare_queue(&self, name: &str) -> Result<()>
95        {
96            let channel = self.channel.lock().await;
97            channel
98                .queue_declare(
99                    name,
100                    QueueDeclareOptions {
101                        durable: true,
102                        ..Default::default()
103                    },
104                    FieldTable::default(),
105                )
106                .await?;
107            debug!("Queue declared: {}", name);
108            Ok(())
109        }
110
111        /// Declare a durable topic exchange. Idempotent.
112        /// 声明一个持久化 topic 交换机（幂等）。
113        pub async fn declare_exchange(&self, name: &str, kind: &str) -> Result<()>
114        {
115            use lapin::ExchangeKind;
116            let kind = match kind
117            {
118                "direct" => ExchangeKind::Direct,
119                "fanout" => ExchangeKind::Fanout,
120                "headers" => ExchangeKind::Headers,
121                _ => ExchangeKind::Topic,
122            };
123            let channel = self.channel.lock().await;
124            channel
125                .exchange_declare(
126                    name,
127                    kind,
128                    ExchangeDeclareOptions {
129                        durable: true,
130                        ..Default::default()
131                    },
132                    FieldTable::default(),
133                )
134                .await?;
135            debug!("Exchange declared: {}", name);
136            Ok(())
137        }
138
139        /// Bind a queue to an exchange with a routing key.
140        /// 将队列绑定到交换机。
141        pub async fn bind_queue(&self, queue: &str, exchange: &str, routing_key: &str)
142        -> Result<()>
143        {
144            let channel = self.channel.lock().await;
145            channel
146                .queue_bind(
147                    queue,
148                    exchange,
149                    routing_key,
150                    QueueBindOptions::default(),
151                    FieldTable::default(),
152                )
153                .await?;
154            debug!("Bound queue={} exchange={} rk={}", queue, exchange, routing_key);
155            Ok(())
156        }
157
158        /// Publish a message to an exchange with a routing key.
159        /// 向交换机发布消息。
160        pub async fn publish(&self, exchange: &str, routing_key: &str, payload: &[u8])
161        -> Result<()>
162        {
163            let channel = self.channel.lock().await;
164            channel
165                .basic_publish(
166                    exchange,
167                    routing_key,
168                    BasicPublishOptions::default(),
169                    payload,
170                    BasicProperties::default().with_delivery_mode(2), // persistent
171                )
172                .await?
173                .await?; // wait for publisher confirm
174            debug!("Published {} bytes to {}#{}", payload.len(), exchange, routing_key);
175            Ok(())
176        }
177
178        /// Publish a JSON-serializable value.
179        /// 发布 JSON 可序列化的值。
180        pub async fn publish_json<T: serde::Serialize>(
181            &self,
182            exchange: &str,
183            routing_key: &str,
184            payload: &T,
185        ) -> Result<()>
186        {
187            let json = serde_json::to_vec(payload)
188                .map_err(|e| RabbitError::ConnectionFailed(format!("serialize: {e}")))?;
189            let channel = self.channel.lock().await;
190            channel
191                .basic_publish(
192                    exchange,
193                    routing_key,
194                    BasicPublishOptions::default(),
195                    &json,
196                    BasicProperties::default()
197                        .with_content_type("application/json".into())
198                        .with_delivery_mode(2),
199                )
200                .await?
201                .await?;
202            Ok(())
203        }
204
205        /// Consume messages from a queue.
206        /// 从队列消费消息。
207        ///
208        /// The handler receives raw bytes and must return `Ok(())` to acknowledge,
209        /// or `Err(...)` to nack.
210        /// 处理器接收原始字节，返回 `Ok(())` 表示确认，返回 `Err(...)` 表示拒绝。
211        pub async fn consume<F, Fut>(&self, queue: &str, prefetch: u16, handler: F) -> Result<()>
212        where
213            F: Fn(Vec<u8>, String) -> Fut + Send + Sync + 'static,
214            Fut: Future<Output = std::result::Result<(), String>> + Send + 'static,
215        {
216            // Open a dedicated channel for this consumer
217            let consumer_channel = self.connection.create_channel().await?;
218            consumer_channel
219                .basic_qos(prefetch, BasicQosOptions::default())
220                .await?;
221
222            let mut consumer = consumer_channel
223                .basic_consume(
224                    queue,
225                    &format!("hiver-consumer-{}", uuid::Uuid::new_v4()),
226                    BasicConsumeOptions::default(),
227                    FieldTable::default(),
228                )
229                .await?;
230
231            info!("Consuming from queue '{}'", queue);
232
233            let handler = Arc::new(handler);
234            tokio::spawn(async move {
235                use futures::StreamExt;
236                while let Some(delivery) = consumer.next().await
237                {
238                    match delivery
239                    {
240                        Ok(delivery) =>
241                        {
242                            let tag = delivery.delivery_tag;
243                            let routing_key = delivery.routing_key.to_string();
244                            let data = delivery.data.clone();
245                            let h = handler.clone();
246                            let ch = consumer_channel.clone();
247                            tokio::spawn(async move {
248                                match h(data, routing_key).await
249                                {
250                                    Ok(()) =>
251                                    {
252                                        if let Err(e) =
253                                            ch.basic_ack(tag, BasicAckOptions::default()).await
254                                        {
255                                            error!("ack failed: {e}");
256                                        }
257                                    },
258                                    Err(e) =>
259                                    {
260                                        error!("handler error: {e}");
261                                        let _ = ch
262                                            .basic_nack(tag, BasicNackOptions {
263                                                requeue: true,
264                                                ..Default::default()
265                                            })
266                                            .await;
267                                    },
268                                }
269                            });
270                        },
271                        Err(e) =>
272                        {
273                            error!("Consumer delivery error: {e}");
274                            break;
275                        },
276                    }
277                }
278                info!("Consumer loop ended for queue");
279            });
280
281            Ok(())
282        }
283
284        /// Check if the underlying connection is still open.
285        /// 检查底层连接是否仍然打开。
286        pub fn is_connected(&self) -> bool
287        {
288            self.connection.status().connected()
289        }
290
291        /// Return the AMQP URL used for this connection.
292        /// 返回此连接使用的 AMQP URL。
293        pub fn url(&self) -> &str
294        {
295            &self.amqp_url
296        }
297    }
