1//! Hiver Security - Security framework module
2//! Hiver安全 - 安全框架模块
3//!
4//! # Equivalent to Spring Boot / 等价于 Spring Boot
5//!
6//! - `@PreAuthorize` - `PreAuthorize`
7//! - `@Secured` - Secured
8//! - `@RolesAllowed` - `RolesAllowed`
9//! - `UserDetails` - User
10//! - `GrantedAuthority` - Permission/Role
11//! - `Authentication` - Auth
12//! - `SecurityContext` - `SecurityContext`
13//!
14//! # Example / 示例
15//!
16//! ```rust,no_run,ignore
17//! use hiver_security::{PreAuthorize, Secured, User, Role};
18//!
19//! struct UserService {
20//!     // ... fields
21//! }
22//!
23//! impl UserService {
24//!     #[pre_authorize("hasRole('ADMIN')")]
25//!     async fn delete_user(&self, id: u64) -> Result<(), Error> {
26//!         // Only accessible by users with ADMIN role
27//!         Ok(())
28//!     }
29//!
30//!     #[secured("ROLE_USER")]
31//!     async fn get_profile(&self) -> Result<Profile, Error> {
32//!         // Only accessible by authenticated users
33//!         Ok(Profile::default())
34//!     }
35//! }
36//! ```
