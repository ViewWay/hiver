39/// Config server client
40/// 配置服务器客户端
41///
42/// Fetches configuration from a remote Spring Cloud Config Server compatible
43/// endpoint.
44/// 从远程Spring Cloud Config Server兼容端点获取配置。
45///
46/// # Spring Equivalent / Spring等价物
47///
48/// ```java
49/// @Bean
50/// public ConfigServicePropertySourceLocator configLocator() {
51///     ConfigClientProperties props = new ConfigClientProperties();
52///     props.setUri("http://localhost:8888");
53///     return new ConfigServicePropertySourceLocator(props);
54/// }
55/// ```
56pub struct ConfigServerClient
57{
58    /// Config server base URL (e.g. `http://localhost:8888`)
59    /// 配置服务器基础URL
60    pub base_url: String,
61
62    /// HTTP client used for all requests
63    /// 用于所有请求的HTTP客户端
64    http_client: reqwest::Client,
65
66    /// Response cache: (app, profile, label) -> (properties, fetched_at)
67    /// 响应缓存：(app, profile, label) -> (属性, 获取时间)
68    cache: Arc<RwLock<HashMap<String, (HashMap<String, String>, DateTime<Utc>)>>>,
69
70    /// Cache TTL (time-to-live) for each entry
71    /// 每个缓存条目的TTL（存活时间）
72    cache_ttl: Duration,
73
74    /// Maximum number of retries on transient failures
75    /// 瞬态故障的最大重试次数
76    max_retries: u32,
77
78    /// Base backoff duration for exponential retry
79    /// 指数退避重试的基准时间
80    retry_backoff: Duration,
81}

83impl ConfigServerClient
84{
85    /// Create a new config server client targeting `base_url`.
86    /// 创建指向`base_url`的新配置服务器客户端。
87    ///
88    /// The client includes an in-memory cache with a default TTL of 60 seconds
89    /// and will retry up to 3 times on transient failures with exponential backoff.
90    /// 客户端包含一个默认TTL为60秒的内存缓存，并在瞬态故障时最多重试3次，采用指数退避。
91    pub fn new(base_url: impl Into<String>) -> Self
92    {
93        Self {
94            base_url: base_url.into(),
95            http_client: reqwest::Client::new(),
96            cache: Arc::new(RwLock::new(HashMap::new())),
97            cache_ttl: Duration::from_secs(DEFAULT_CACHE_TTL_SECS),
98            max_retries: DEFAULT_MAX_RETRIES,
99            retry_backoff: Duration::from_millis(DEFAULT_RETRY_BACKOFF_MS),
100        }
101    }
102
103    /// Set the cache TTL for config responses.
104    /// 设置配置响应的缓存TTL。
105    pub fn with_cache_ttl(mut self, ttl: Duration) -> Self
106    {
107        self.cache_ttl = ttl;
108        self
109    }
110
111    /// Set the max retries and base backoff for transient failures.
112    /// 设置瞬态故障的最大重试次数和基准退避时间。
113    pub fn with_retry(mut self, max_retries: u32, backoff: Duration) -> Self
114    {
115        self.max_retries = max_retries;
116        self.retry_backoff = backoff;
117        self
118    }

120    /// Build the full config endpoint URL.
121    /// 构建完整的配置端点URL。
122    ///
123    /// Pattern: `{base_url}/{application}/{profile}/{label}`
124    fn build_url(&self, application: &str, profile: &str, label: &str) -> String
125    {
126        format!("{}/{}/{}/{}", self.base_url.trim_end_matches('/'), application, profile, label)
127    }
128
129    /// Build the cache key from app/profile/label.
130    /// 从app/profile/label构建缓存键。
131    fn cache_key(app: &str, profile: &str, label: &str) -> String
132    {
133        format!("{}/{}/{}", app, profile, label)
134    }
135
136    /// Fetch configuration as a flat `HashMap<String, String>`.
137    /// 获取配置并返回扁平化的`HashMap<String, String>`。
138    ///
139    /// Results are cached for the configured TTL. On cache miss, a real HTTP
140    /// GET is performed with retries and exponential backoff on transient errors
141    /// (connection refused, timeout).
142    /// 结果会缓存至配置的TTL。缓存未命中时，执行真实的HTTP GET请求，
143    /// 在瞬态错误（连接拒绝、超时）时进行指数退避重试。
144    pub async fn get_config(
145        &self,
146        app: &str,
147        profile: &str,
148        label: &str,
149    ) -> Result<HashMap<String, String>, ConfigClientError>
150    {
151        let key = Self::cache_key(app, profile, label);
152
153        // Check cache first / 首先检查缓存
154        {
155            let cache = self.cache.read().await;
156            if let Some((props, fetched_at)) = cache.get(&key)
157            {
158                let age = Utc::now()
159                    .signed_duration_since(*fetched_at)
160                    .to_std()
161                    .unwrap_or(Duration::ZERO);
162                if age < self.cache_ttl
163                {
164                    tracing::debug!("Config cache hit for {} (age: {}ms)", key, age.as_millis());
165                    return Ok(props.clone());
166                }
167            }
168        }
169
170        // Cache miss — fetch from remote with retry / 缓存未命中 — 带重试从远程获取
171        let url = self.build_url(app, profile, label);
172        let props = self.fetch_with_retry(&url, app, profile, label).await?;
173
174        // Store in cache / 存入缓存
175        {
176            let mut cache = self.cache.write().await;
177            cache.insert(key, (props.clone(), Utc::now()));
178        }
179
180        Ok(props)
181    }
182
183    /// Perform the HTTP GET with exponential backoff retry on transient errors.
184    /// 执行HTTP GET，在瞬态错误时进行指数退避重试。
185    ///
186    /// Transient errors include connection refused, timeout, and server errors
187    /// (5xx). Non-transient errors (404, parse errors) are returned immediately.
188    /// 瞬态错误包括连接拒绝、超时和服务器错误（5xx）。
189    /// 非瞬态错误（404、解析错误）立即返回。
190    async fn fetch_with_retry(
191        &self,
192        url: &str,
193        app: &str,
194        profile: &str,
195        label: &str,
196    ) -> Result<HashMap<String, String>, ConfigClientError>
197    {
198        let mut last_err = None;
199
200        for attempt in 0..=self.max_retries
201        {
202            if attempt > 0
203            {
204                // Exponential backoff: base * 2^(attempt-1)
205                // 指数退避：base * 2^(attempt-1)
206                let delay = self.retry_backoff * 2u32.pow(attempt - 1);
207                tracing::warn!(
208                    "Config server retry #{}/{} for {}/{}/{}, waiting {}ms",
209                    attempt,
210                    self.max_retries,
211                    app,
212                    profile,
213                    label,
214                    delay.as_millis(),
215                );
216                tokio::time::sleep(delay).await;
217            }
218
219            match self.fetch_single(url, app, profile, label).await
220            {
221                Ok(props) => return Ok(props),
222                Err(e @ ConfigClientError::Connection(_)) =>
223                {
224                    // Transient — retry / 瞬态 — 重试
225                    tracing::warn!(
226                        "Config server transient error (attempt {}/{}): {}",
227                        attempt + 1,
228                        self.max_retries + 1,
229                        e
230                    );
231                    last_err = Some(e);
232                },
233                Err(e) =>
234                {
235                    // Non-transient — return immediately / 非瞬态 — 立即返回
236                    return Err(e);
237                },
238            }
239        }
240
241        // All retries exhausted / 所有重试耗尽
242        Err(last_err
243            .unwrap_or_else(|| ConfigClientError::Connection("All retries exhausted".to_string())))
244    }
245
246    /// Perform a single HTTP GET to the config server.
247    /// 执行单次HTTP GET到配置服务器。
248    async fn fetch_single(
249        &self,
250        url: &str,
251        app: &str,
252        profile: &str,
253        label: &str,
254    ) -> Result<HashMap<String, String>, ConfigClientError>
255    {
256        let response = self
257            .http_client
258            .get(url)
259            .timeout(Duration::from_secs(5))
260            .send()
261            .await
262            .map_err(|e| ConfigClientError::Connection(e.to_string()))?;
263
264        if !response.status().is_success()
265        {
266            let status = response.status();
267            if status.as_u16() == 404
268            {
269                return Err(ConfigClientError::NotFound(format!(
270                    "Config for {}/{}/{} not found",
271                    app, profile, label
272                )));
273            }
274            // 5xx errors are transient — map to Connection so retry kicks in
275            // 5xx错误是瞬态的 — 映射到Connection以触发重试
276            if status.as_u16() >= 500
277            {
278                return Err(ConfigClientError::Connection(format!(
279                    "Config server returned {}",
280                    status
281                )));
282            }
283            return Err(ConfigClientError::Connection(format!(
284                "Config server returned {}",
285                status
286            )));
287        }
288
289        // The server is expected to return JSON with a flat key-value structure
290        // or the Spring Cloud Config envelope `{ propertySources: [{ source: {...} }] }`.
291        // 服务器应返回扁平键值JSON或Spring Cloud Config信封格式。
292        let body: serde_json::Value = response
293            .json()
294            .await
295            .map_err(|e| ConfigClientError::Parse(e.to_string()))?;
296
297        flatten_json_value(&body)
298    }

300    /// Invalidate the entire config cache, forcing the next request to hit the server.

302    pub async fn invalidate_cache(&self)
303    {
304        let mut cache = self.cache.write().await;
305        cache.clear();
306    }

308    /// Invalidate a specific cache entry.
309    /// 使特定缓存条目失效。

314        cache.remove(&key);
315    }

326/// Recursively flatten a JSON value into `HashMap<String, String>`.
327/// 递归地将JSON值扁平化为`HashMap<String, String>`。
328fn flatten_json_value(
329    value: &serde_json::Value,
330) -> Result<HashMap<String, String>, ConfigClientError>
331{
332    let mut map = HashMap::new();
333
334    // If the value is the Spring Cloud Config envelope, drill into propertySources.
335    // Process in reverse order so the first source wins on duplicate keys.
336    // 如果值是Spring Cloud Config信封格式，深入propertySources。
337    // 反向处理以使第一个源的重复键优先。
338    if let Some(sources) = value.get("propertySources").and_then(|v| v.as_array())
339    {
340        for source in sources.iter().rev()
341        {
342            if let Some(obj) = source.get("source").and_then(|v| v.as_object())
343            {
344                flatten_object(obj, "", &mut map);
345            }
346        }
347    }
348    else if let Some(obj) = value.as_object()
349    {
350        flatten_object(obj, "", &mut map);
351    }
352
353    Ok(map)
354}
355
356#[allow(clippy::match_wildcard_for_single_variants)]
357fn flatten_object(
358    obj: &serde_json::Map<String, serde_json::Value>,
359    prefix: &str,
360    out: &mut HashMap<String, String>,
361)
362{
363    for (key, value) in obj
364    {
365        let full_key = if prefix.is_empty()
366        {
367            key.clone()
368        }
369        else
370        {
371            format!("{}.{}", prefix, key)
372        };
373        match value
374        {
375            serde_json::Value::Object(inner) =>
376            {
377                flatten_object(inner, &full_key, out);
378            },
379            other =>
380            {
381                // Convert non-string values to their string representation
382                let string_val = match other
383                {
384                    serde_json::Value::String(s) => s.clone(),
385                    serde_json::Value::Number(n) => n.to_string(),
386                    serde_json::Value::Bool(b) => b.to_string(),
387                    serde_json::Value::Null => String::new(),
388                    serde_json::Value::Array(arr) =>
389                    {
390                        // Serialize array as JSON string
391                        serde_json::to_string(arr).unwrap_or_default()
392                    },
393                    _ => other.to_string(),
394                };
395                out.insert(full_key, string_val);
396            },
397        }
398    }
399}

405/// Configuration source kind
406/// 配置源类型
407///
408/// Determines where configuration properties are loaded from.
409/// 决定从何处加载配置属性。
410#[derive(Debug, Clone)]
411pub enum ConfigSource
412{
413    /// Remote config server (e.g. Spring Cloud Config Server)
414    /// 远程配置服务器
415    Remote
416    {
417        /// Server URL / 服务器URL
418        server_url: String,
419        /// Application name (defaults to "application") / 应用名称（默认为"application"）
420        app: Option<String>,
421        /// Profile name (defaults to "default") / 配置文件名（默认为"default"）
422        profile: Option<String>,
423        /// Label / branch (defaults to "main") / 标签/分支（默认为"main"）
424        label: Option<String>,
425    },
426
427    /// Local file path (e.g. `./config/application.yml`)
428    /// 本地文件路径
429    Local
430    {
431        /// Path to the config file / 配置文件路径
432        path: PathBuf,
433    },
434
435    /// Environment variables (prefix filter optional)
436    /// 环境变量
437    Environment
438    {
439        /// Optional prefix filter (e.g. `"APP_"` to only capture `APP_*` vars)
440        /// 可选前缀过滤器
441        prefix: Option<String>,
442    },
443}
444
445#[async_trait]
446impl ConfigProvider for ConfigSource
447{
448    async fn load(&self) -> Result<HashMap<String, String>, ConfigClientError>
449    {
450        match self
451        {
452            ConfigSource::Remote {
453                server_url,
454                app,
455                profile,
456                label,
457            } =>
458            {
459                let client = ConfigServerClient::new(server_url.as_str());
460                let app_name = app.as_deref().unwrap_or("application");
461                let profile_name = profile.as_deref().unwrap_or("default");
462                let label_name = label.as_deref().unwrap_or("main");
463                client.get_config(app_name, profile_name, label_name).await
464            },
465            ConfigSource::Local { path } => load_local_config(path).await,
466            ConfigSource::Environment { prefix } => Ok(load_env_config(prefix.as_deref())),
467        }
468    }
469
470    fn name(&self) -> &str
471    {
472        match self
473        {
474            ConfigSource::Remote { .. } => "remote",
475            ConfigSource::Local { .. } => "local",
476            ConfigSource::Environment { .. } => "environment",
477        }
478    }
479}

481/// Load configuration from a local file.
482/// 从本地文件加载配置。
483///
484/// Supports `.json`, `.properties` (key=value per line), and `.yaml`/`.yml`.
485/// 支持`.json`、`.properties`和`.yaml`/`.yml`格式。
486async fn load_local_config(path: &PathBuf) -> Result<HashMap<String, String>, ConfigClientError>
487{
488    let content = tokio::fs::read_to_string(path)
489        .await
490        .map_err(|e| ConfigClientError::Io(e.to_string()))?;
491
492    let ext = path
493        .extension()
494        .and_then(|e| e.to_str())
495        .unwrap_or("")
496        .to_lowercase();
497
498    match ext.as_str()
499    {
500        "json" =>
501        {
502            let value: serde_json::Value = serde_json::from_str(&content)
503                .map_err(|e| ConfigClientError::Parse(e.to_string()))?;
504            flatten_json_value(&value)
505        },
506        "properties" =>
507        {
508            let mut map = HashMap::new();
509            for line in content.lines()
510            {
511                let line = line.trim();
512                if line.is_empty() || line.starts_with('#')
513                {
514                    continue;
515                }
516                if let Some((k, v)) = line.split_once('=')
517                {
518                    map.insert(k.trim().to_string(), v.trim().to_string());
519                }
520            }
521            Ok(map)
522        },
523        "yml" | "yaml" =>
524        {
525            // Basic YAML parsing: try to parse as JSON first (subset of YAML)
526            // In production, use the `serde_yaml` crate for full YAML support.
527            let value: serde_json::Value = serde_json::from_str(&content).map_err(|e| {
528                ConfigClientError::Parse(format!("YAML parsing requires serde_yaml: {}", e))
529            })?;
530            flatten_json_value(&value)
531        },
532        _ => Err(ConfigClientError::Parse(format!("Unsupported config file format: {}", ext))),
533    }
534}

536/// Load configuration from environment variables.
537/// 从环境变量加载配置。
538fn load_env_config(prefix: Option<&str>) -> HashMap<String, String>
539{
540    let mut map = HashMap::new();
541    for (key, value) in std::env::vars()
542    {
543        if let Some(p) = prefix
544        {
545            if !key.starts_with(p)
546            {
547                continue;
548            }
549            // Strip the prefix and lowercase for consistency
550            let stripped = key.strip_prefix(p).unwrap_or(&key);
551            map.insert(stripped.to_lowercase(), value);
552        }
553        else
554        {
555            map.insert(key.to_lowercase(), value);
556        }
557    }
558    map
559}

565/// Composite configuration source that merges multiple sources with priority.
566/// 合并多个配置源的复合配置源，按优先级排序。
567///
568/// Sources listed first have the **highest** priority: when two sources define
569/// the same key, the value from the higher-priority source wins.
570/// 列在前面的源具有**最高**优先级：当两个源定义相同的键时，高优先级源的值胜出。
571///
572/// # Spring Equivalent / Spring等价物
573///
574/// ```java
575/// @Bean
576/// public CompositePropertySource compositeSource() {
577///     CompositePropertySource source = new CompositePropertySource("composite");
578///     source.addPropertySource(remoteSource);
579///     source.addPropertySource(localSource);
580///     return source;
581/// }
582/// ```
583pub struct CompositeConfigSource
584{
585    /// Ordered list of sources (first = highest priority)
586    /// 有序的源列表（第一个=最高优先级）
587    sources: Vec<Box<dyn ConfigProvider>>,
588}
589
590impl CompositeConfigSource
591{
592    /// Create an empty composite source
593    /// 创建空的复合源
594    pub fn new() -> Self
595    {
596        Self {
597            sources: Vec::new(),
598        }
599    }
600
601    /// Add a source. Sources added first have higher priority.
602    /// 添加源。先添加的源具有更高优先级。
603    pub fn with_source(mut self, source: Box<dyn ConfigProvider>) -> Self
604    {
605        self.sources.push(source);
606        self
607    }
608
609    /// Add a `ConfigSource` variant.
610    /// 添加`ConfigSource`变体。
611    pub fn with_config_source(self, source: ConfigSource) -> Self
612    {
613        self.with_source(Box::new(source))
614    }
615
616    /// Load and merge all sources.
617    /// 加载并合并所有源。
618    ///
619    /// Higher-priority sources overwrite lower-priority ones.
620    /// 高优先级源覆盖低优先级源。
621    pub async fn load_merged(&self) -> Result<HashMap<String, String>, ConfigClientError>
622    {
623        let mut merged = HashMap::new();
624
625        for source in &self.sources
626        {
627            let props = source.load().await?;
628            // Higher priority overwrites / 高优先级覆盖
629            merged.extend(props);
630        }
631
632        Ok(merged)
633    }
634
635    /// Get the number of registered sources.
636    /// 获取已注册源的数量。
637    pub fn source_count(&self) -> usize
638    {
639        self.sources.len()
640    }
641}

673/// Polling-based config refresher.
674/// 基于轮询的配置刷新器。
675///
676/// Periodically reloads configuration from a `CompositeConfigSource` and
677/// notifies listeners when properties change.
678/// 定期从`CompositeConfigSource`重新加载配置，并在属性更改时通知监听器。
679pub struct PollingConfigRefresher
680{
681    /// Source to poll
682    /// 轮询的源
683    source: Arc<CompositeConfigSource>,
684
685    /// Current snapshot of properties
686    /// 当前属性快照
687    current: Arc<RwLock<HashMap<String, String>>>,
688
689    /// Last refresh timestamp
690    /// 上次刷新时间戳
691    last_refresh: Arc<RwLock<Option<DateTime<Utc>>>>,
692
693    /// Polling interval
694    /// 轮询间隔
695    interval: Duration,
696
697    /// Shutdown signal
698    /// 关闭信号
699    shutdown: tokio::sync::watch::Sender<bool>,
700}
701
702impl PollingConfigRefresher
703{
704    /// Create a new polling refresher
705    /// 创建新的轮询刷新器
706    pub fn new(source: Arc<CompositeConfigSource>, interval: Duration) -> Self
707    {
708        let (shutdown, _) = tokio::sync::watch::channel(false);
709        Self {
710            source,
711            current: Arc::new(RwLock::new(HashMap::new())),
712            last_refresh: Arc::new(RwLock::new(None)),
713            interval,
714            shutdown,
715        }
716    }
717
718    /// Start the polling loop in the background
719    /// 在后台启动轮询循环
720    pub async fn start(&self)
721    {
722        // Perform initial load / 执行初始加载
723        self.refresh().await;
724
725        let source = self.source.clone();
726        let current = self.current.clone();
727        let last_refresh = self.last_refresh.clone();
728        let mut shutdown_rx = self.shutdown.subscribe();
729        let interval = self.interval;
730
731        tokio::spawn(async move {
732            let mut ticker = tokio::time::interval(interval);
733            loop
734            {
735                tokio::select! {
736                    _ = ticker.tick() => {
737                        match source.load_merged().await {
738                            Ok(new_props) => {
739                                *current.write().await = new_props;
740                                *last_refresh.write().await = Some(Utc::now());
741                            }
742                            Err(e) => {
743                                tracing::warn!("Config refresh failed: {}", e);
744                            }
745                        }
746                    }
747                    _ = shutdown_rx.changed() => {
748                        tracing::info!("Config polling stopped");
749                        break;
750                    }
751                }
752            }
753        });
754    }
755
756    /// Manually trigger a refresh
757    /// 手动触发刷新
758    pub async fn refresh(&self)
759    {
760        match self.source.load_merged().await
761        {
762            Ok(props) =>
763            {
764                *self.current.write().await = props;
765                *self.last_refresh.write().await = Some(Utc::now());
766            },
767            Err(e) =>
768            {
769                tracing::warn!("Config refresh failed: {}", e);
770            },
771        }
772    }
773
774    /// Get the current property snapshot
775    /// 获取当前属性快照
776    pub async fn current(&self) -> HashMap<String, String>
777    {
778        self.current.read().await.clone()
779    }
780
781    /// Get a single property value
782    /// 获取单个属性值
783    pub async fn get(&self, key: &str) -> Option<String>
784    {
785        self.current.read().await.get(key).cloned()
786    }
787
788    /// Get the last refresh timestamp
789    /// 获取上次刷新时间戳
790    pub async fn last_refresh(&self) -> Option<DateTime<Utc>>
791    {
792        *self.last_refresh.read().await
793    }
794
795    /// Stop the polling loop
796    /// 停止轮询循环
797    pub fn stop(&self)
798    {
799        let _ = self.shutdown.send(true);
800    }
801}
