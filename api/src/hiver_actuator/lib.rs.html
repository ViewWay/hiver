1//! Hiver Actuator - Spring Boot Actuator equivalent features
2//! Hiver Actuator - Spring Boot Actuator 等价功能
3//!
4//! # Equivalent to Spring Boot / 等价于 Spring Boot
5//!
6//! - `/actuator/health` - Health check endpoint
7//! - `/actuator/info` - Application information
8//! - `/actuator/metrics` - Metrics endpoint
9//! - `/actuator/env` - Environment information
10//! - `/actuator` - Actuator index
11//!
12//! # Example / 示例
13//!
14//! ```rust,no_run,ignore
15//! use hiver_actuator::Actuator;
16//! use hiver_http::Server;
17//! use hiver_router::Router;
18//!
19//! let actuator = Actuator::new()
20//!     .info("my-app", "1.0.0")
21//!     .enable_health(true)
22//!     .enable_metrics(true);
23//!
24//! let app = Router::new()
25//!     .nest("/actuator", actuator.routes());
26//!
27//! Server::bind("127.0.0.1:8080")
28//!     .run(app)
29//!     .await?;
30//! ```
