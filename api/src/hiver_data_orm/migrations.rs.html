1//! Database migrations
2//! 数据库迁移
3//!
4//! # Overview / 概述
5//!
6//! This module provides database migration support backed by a `DatabaseClient`.
7//! 本模块提供由 `DatabaseClient` 支持的数据库迁移支持。
8//!
9//! # Equivalent to Spring / 等价于 Spring
10//!
11//! | Hiver | Spring / Flyway |
12//! |-------|-----------------|
13//! | `Migration` | `FlywayMigration` / `Liquibase` |
14//! | `Migrator` | `Flyway` / `Liquibase` |
15//! | `Schema` | `SchemaCreator` / `JPA DDL` |
16//!
17//! # Example / 示例
18//!
19//! ```rust,no_run,ignore
20//! use hiver_data_orm::migrations::{Migration, Migrator};
21//!
22//! let migration = Migration::new("001_create_users")
23//!     .up("CREATE TABLE users (id SERIAL PRIMARY KEY, name TEXT);")
24//!     .down("DROP TABLE users;");
25//!
26//! let mut migrator = Migrator::new();
27//! migrator.register(migration);
28//! migrator.up(&client).await?;
29//! ```

37/// Migration — a single database migration with up and down SQL.
38/// 迁移 — 具有向上和向下 SQL 的单个数据库迁移。
39#[derive(Debug, Clone)]
40pub struct Migration
41{
42    /// Migration name/identifier / 迁移名称/标识符
43    pub name: String,
44    /// Migration version / 迁移版本
45    pub version: String,
46    /// Description of what the migration does / 迁移所做事情的描述
47    pub description: String,
48    /// SQL to apply the migration / 应用迁移的 SQL
49    pub up_sql: String,
50    /// SQL to rollback the migration / 回滚迁移的 SQL
51    pub down_sql: String,
52    /// Whether this migration has been applied / 此迁移是否已应用
53    pub applied: bool,
54    /// Migration attributes / 迁移属性
55    pub attributes: HashMap<String, String>,
56}
57
58impl Migration
59{
60    /// Create a new migration / 创建新迁移
61    pub fn new(name: impl Into<String>) -> Self
62    {
63        let name = name.into();
64        Self {
65            version: Self::extract_version(&name),
66            name: name.clone(),
67            description: String::new(),
68            up_sql: String::new(),
69            down_sql: String::new(),
70            applied: false,
71            attributes: HashMap::new(),
72        }
73    }
74
75    /// Create a new migration with explicit version / 创建具有显式版本的新迁移
76    pub fn with_version(version: impl Into<String>, name: impl Into<String>) -> Self
77    {
78        Self {
79            name: name.into(),
80            version: version.into(),
81            description: String::new(),
82            up_sql: String::new(),
83            down_sql: String::new(),
84            applied: false,
85            attributes: HashMap::new(),
86        }
87    }
88
89    /// Set the up SQL / 设置向上 SQL
90    pub fn up(mut self, sql: impl Into<String>) -> Self
91    {
92        self.up_sql = sql.into();
93        self
94    }
95
96    /// Set the down SQL / 设置向下 SQL
97    pub fn down(mut self, sql: impl Into<String>) -> Self
98    {
99        self.down_sql = sql.into();
100        self
101    }
102
103    /// Set the description / 设置描述
104    pub fn description(mut self, desc: impl Into<String>) -> Self
105    {
106        self.description = desc.into();
107        self
108    }
109
110    /// Add an attribute / 添加属性
111    pub fn attribute(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
112    {
113        self.attributes.insert(key.into(), value.into());
114        self
115    }
116
117    /// Extract version from migration name / 从迁移名称提取版本
118    fn extract_version(name: &str) -> String
119    {
120        if let Some(stripped) = name.strip_prefix('V')
121            && let Some(end) = stripped.find("__")
122        {
123            return format!("V{}", &stripped[..end]);
124        }
125        if let Some(end) = name.find('_')
126        {
127            let prefix = &name[..end];
128            if prefix.chars().all(|c| c.is_ascii_digit())
129            {
130                return prefix.to_string();
131            }
132        }
133        name.to_string()
134    }
135
136    /// Validate the migration / 验证迁移
137    pub fn validate(&self) -> Result<()>
138    {
139        if self.up_sql.is_empty()
140        {
141            return Err(Error::validation(format!("Migration {} has no up_sql", self.name)));
142        }
143        if self.down_sql.is_empty()
144        {
145            return Err(Error::validation(format!("Migration {} has no down_sql", self.name)));
146        }
147        Ok(())
148    }
149}

161/// Migrator — manages and executes database migrations.
162/// 迁移器 — 管理和执行数据库迁移。
163pub struct Migrator
164{
165    /// Registered migrations / 已注册的迁移
166    migrations: Vec<Migration>,
167    /// Migration table name / 迁移表名
168    migration_table: String,
169}
170
171impl Migrator
172{
173    /// Create a new migrator / 创建新迁移器
174    pub fn new() -> Self
175    {
176        Self {
177            migrations: Vec::new(),
178            migration_table: "_migrations".to_string(),
179        }
180    }
181
182    /// Register a migration / 注册迁移
183    pub fn register(&mut self, migration: Migration)
184    {
185        self.migrations.push(migration);
186    }
187
188    /// Register multiple migrations / 注册多个迁移
189    pub fn register_all(&mut self, migrations: Vec<Migration>)
190    {
191        self.migrations.extend(migrations);
192    }
193
194    /// Build the migrator with a custom migration table name / 设置迁移表名
195    pub fn migration_table(mut self, table: impl Into<String>) -> Self
196    {
197        self.migration_table = table.into();
198        self
199    }
200
201    /// Get all migrations / 获取所有迁移
202    pub fn migrations(&self) -> &[Migration]
203    {
204        &self.migrations
205    }
206
207    /// Get pending migrations / 获取待执行的迁移
208    pub fn pending(&self) -> Vec<&Migration>
209    {
210        self.migrations.iter().filter(|m| !m.applied).collect()
211    }
212
213    /// Get applied migrations / 获取已应用的迁移
214    pub fn applied(&self) -> Vec<&Migration>
215    {
216        self.migrations.iter().filter(|m| m.applied).collect()
217    }
218
219    /// Ensure the migration tracking table exists.
220    /// 确保迁移跟踪表存在。
221    async fn ensure_migration_table<C: DatabaseClient>(&self, client: &C) -> Result<()>
222    {
223        let sql = format!(
224            "CREATE TABLE IF NOT EXISTS {} (version TEXT PRIMARY KEY, applied_at TIMESTAMP \
225             DEFAULT CURRENT_TIMESTAMP)",
226            self.migration_table
227        );
228        client
229            .execute_cmd(&sql)
230            .await
231            .map_err(|e| Error::migration(format!("Failed to create migration table: {e}")))?;
232        Ok(())
233    }
234
235    /// Run all pending migrations.
236    /// 运行所有待执行的迁移。
237    pub async fn up<C: DatabaseClient>(&mut self, client: &C) -> Result<usize>
238    {
239        self.ensure_migration_table(client).await?;
240
241        let mut applied_count = 0;
242        for migration in &mut self.migrations
243        {
244            if migration.applied
245            {
246                continue;
247            }
248            migration.validate()?;
249
250            client.execute_cmd(&migration.up_sql).await.map_err(|e| {
251                Error::migration(format!("Migration {} failed: {}", migration.name, e))
252            })?;
253
254            // Record the migration as applied
255            let record_sql = format!(
256                "INSERT INTO {} (version) VALUES ('{}')",
257                self.migration_table, migration.version,
258            );
259            client.execute_cmd(&record_sql).await.map_err(|e| {
260                Error::migration(format!("Failed to record migration {}: {}", migration.name, e))
261            })?;
262
263            migration.applied = true;
264            applied_count += 1;
265        }
266        Ok(applied_count)
267    }
268
269    /// Rollback the last applied migration.
270    /// 回滚最后的迁移。
271    pub async fn down<C: DatabaseClient>(&mut self, client: &C) -> Result<bool>
272    {
273        if let Some(last) = self.migrations.iter_mut().rev().find(|m| m.applied)
274        {
275            last.validate()?;
276
277            client.execute_cmd(&last.down_sql).await.map_err(|e| {
278                Error::migration(format!("Rollback of {} failed: {}", last.name, e))
279            })?;
280
281            // Remove migration record
282            let delete_sql =
283                format!("DELETE FROM {} WHERE version = '{}'", self.migration_table, last.version,);
284            client.execute_cmd(&delete_sql).await.map_err(|e| {
285                Error::migration(format!("Failed to unrecord migration {}: {}", last.name, e))
286            })?;
287
288            last.applied = false;
289            Ok(true)
290        }
291        else
292        {
293            Ok(false)
294        }
295    }
296
297    /// Rollback a specific number of migrations.
298    /// 回滚指定数量的迁移。
299    pub async fn rollback<C: DatabaseClient>(&mut self, client: &C, steps: usize) -> Result<usize>
300    {
301        let mut rolled_back = 0;
302        for _ in 0..steps
303        {
304            if self.down(client).await?
305            {
306                rolled_back += 1;
307            }
308            else
309            {
310                break;
311            }
312        }
313        Ok(rolled_back)
314    }
315
316    /// Refresh — rollback all then reapply.
317    /// 刷新 — 回滚所有迁移并重新应用。
318    pub async fn refresh<C: DatabaseClient>(&mut self, client: &C) -> Result<usize>
319    {
320        let to_rollback = self.applied().len();
321        for _ in 0..to_rollback
322        {
323            self.down(client).await?;
324        }
325        self.up(client).await
326    }
327
328    /// Reset — rollback all migrations.
329    /// 重置 — 回滚所有迁移。
330    pub async fn reset<C: DatabaseClient>(&mut self, client: &C) -> Result<usize>
331    {
332        let mut count = 0;
333        while self.down(client).await?
334        {
335            count += 1;
336        }
337        Ok(count)
338    }
339}

349/// Schema builder — provides a fluent interface for building database schemas.
350/// Schema 构建器 — 提供用于构建数据库模式的流畅接口。
351pub struct Schema
352{
353    operations: Vec<SchemaOperation>,

356/// Schema operation / 模式操作
357#[derive(Debug, Clone)]
358#[allow(missing_docs)]
359pub enum SchemaOperation
360{
361    /// Create table / 创建表
362    CreateTable
363    {
364        name: String,
365        columns: Vec<ColumnDefinition>,
366    },
367    /// Drop table / 删除表
368    DropTable
369    {
370        name: String
371    },
372    /// Add column / 添加列
373    AddColumn
374    {
375        table: String,
376        column: ColumnDefinition,
377    },
378    /// Drop column / 删除列
379    DropColumn
380    {
381        table: String, name: String
382    },
383    /// Rename table / 重命名表
384    RenameTable
385    {
386        from: String, to: String
387    },
388    /// Add index / 添加索引
389    AddIndex
390    {
391        table: String,
392        name: String,
393        columns: Vec<String>,
394        unique: bool,
395    },
396    /// Drop index / 删除索引
397    DropIndex
398    {
399        table: String, name: String
400    },
401}

406pub struct ColumnDefinition

446impl Schema
447{
448    /// Create a table operation / 创建表操作
449    pub fn create_table(name: impl Into<String>) -> Schema
450    {
451        let mut schema = Self::new();
452        schema.operations.push(SchemaOperation::CreateTable {
453            name: name.into(),
454            columns: Vec::new(),
455        });
456        schema
457    }
458
459    /// Add a column to the current create table / 添加列到当前创建表
460    pub fn column(mut self, name: &str, type_: crate::ColumnType) -> Self
461    {
462        if let Some(SchemaOperation::CreateTable { columns, .. }) = self.operations.last_mut()
463        {
464            columns.push(ColumnDefinition {
465                name: name.to_string(),
466                type_,
467                is_primary_key: false,
468                is_nullable: false,
469                is_unique: false,
470                default: None,
471                max_length: None,
472                references: None,
473            });
474        }
475        self
476    }
477
478    /// Execute the schema operations against a DatabaseClient.
479    /// 对 DatabaseClient 执行模式操作。
480    pub async fn execute<C: DatabaseClient>(&self, client: &C) -> Result<()>
481    {
482        for op in &self.operations
483        {
484            let sql = schema_op_to_sql(op);
485            client
486                .execute_cmd(&sql)
487                .await
488                .map_err(|e| Error::migration(format!("Schema execution failed: {e}")))?;
489        }
490        Ok(())
491    }
492}

494/// Convert a SchemaOperation to SQL / 将 SchemaOperation 转换为 SQL
495fn schema_op_to_sql(op: &SchemaOperation) -> String
496{
497    match op
498    {
499        SchemaOperation::CreateTable { name, columns } =>
500        {
501            let col_defs: Vec<String> = columns
502                .iter()
503                .map(|c| {
504                    let mut def = format!(
505                        "{} {}",
506                        c.name,
507                        c.type_.as_sql(crate::model::SqlDialect::PostgreSQL)
508                    );
509                    if c.is_primary_key
510                    {
511                        def.push_str(" PRIMARY KEY");
512                    }
513                    if !c.is_nullable
514                    {
515                        def.push_str(" NOT NULL");
516                    }
517                    if c.is_unique
518                    {
519                        def.push_str(" UNIQUE");
520                    }
521                    if let Some(d) = &c.default
522                    {
523                        def.push_str(&format!(" DEFAULT {}", d));
524                    }
525                    def
526                })
527                .collect();
528            format!("CREATE TABLE {} ({})", name, col_defs.join(", "))
529        },
530        SchemaOperation::DropTable { name } => format!("DROP TABLE IF EXISTS {}", name),
531        SchemaOperation::AddColumn { table, column } =>
532        {
533            format!(
534                "ALTER TABLE {} ADD COLUMN {} {}",
535                table,
536                column.name,
537                column.type_.as_sql(crate::model::SqlDialect::PostgreSQL),
538            )
539        },
540        SchemaOperation::DropColumn { table, name } =>
541        {
542            format!("ALTER TABLE {} DROP COLUMN {}", table, name)
543        },
544        SchemaOperation::RenameTable { from, to } =>
545        {
546            format!("ALTER TABLE {} RENAME TO {}", from, to)
547        },
548        SchemaOperation::AddIndex {
549            table,
550            name,
551            columns,
552            unique,
553        } =>
554        {
555            let unique_str = if *unique { "UNIQUE " } else { "" };
556            format!("CREATE {}INDEX {} ON {} ({})", unique_str, name, table, columns.join(", "),)
557        },
558        SchemaOperation::DropIndex { table, name } =>
559        {
560            format!("DROP INDEX IF EXISTS {}.{}", table, name)
561        },
562    }
563}
