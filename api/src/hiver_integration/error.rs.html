4/// Integration framework error
5/// 集成框架错误
6pub type Result<T> = std::result::Result<T, IntegrationError>;

8/// Integration error types
9/// 集成错误类型
10#[derive(Debug, thiserror::Error)]
11pub enum IntegrationError
12{
13    /// Message error
14    /// 消息错误
15    #[error("Message error: {0}")]
16    Message(String),
17
18    /// Channel error
19    /// 通道错误
20    #[error("Channel error: {0}")]
21    Channel(String),
22
23    /// Transformation error
24    /// 转换错误
25    #[error("Transformation error: {0}")]
26    Transformation(String),
27
28    /// Routing error
29    /// 路由错误
30    #[error("Routing error: {0}")]
31    Routing(String),
32
33    /// Timeout error
34    /// 超时错误
35    #[error("Operation timed out after {0:?}")]
36    Timeout(std::time::Duration),
37
38    /// Serialization error
39    /// 序列化错误
40    #[error("Serialization error: {0}")]
41    Serialization(String),
42
43    /// Deserialization error
44    /// 反序列化错误
45    #[error("Deserialization error: {0}")]
46    Deserialization(String),
47
48    /// Service error
49    /// 服务错误
50    #[error("Service error: {0}")]
51    Service(String),
52
53    /// Configuration error
54    /// 配置错误
55    #[error("Configuration error: {0}")]
56    Configuration(String),
57
58    /// Channel closed
59    /// 通道已关闭
60    #[error("Channel '{0}' is closed")]
61    ChannelClosed(String),
62
63    /// Channel full
64    /// 通道已满
65    #[error("Channel '{0}' is full")]
66    ChannelFull(String),
67
68    /// Handler error
69    /// 处理器错误
70    #[error("Handler error: {0}")]
71    Handler(String),
72
73    /// Payload error
74    /// 载荷错误
75    #[error("Payload error: {0}")]
76    Payload(String),
77
78    /// Aggregation error
79    /// 聚合错误
80    #[error("Aggregation error: {0}")]
81    Aggregation(String),
82}
83
84impl IntegrationError
85{
86    /// Create a message error
87    /// 创建消息错误
88    pub fn message(msg: impl Into<String>) -> Self
89    {
90        Self::Message(msg.into())
91    }
92
93    /// Create a channel error
94    /// 创建通道错误
95    pub fn channel(msg: impl Into<String>) -> Self
96    {
97        Self::Channel(msg.into())
98    }
99
100    /// Create a transformation error
101    /// 创建转换错误
102    pub fn transformation(msg: impl Into<String>) -> Self
103    {
104        Self::Transformation(msg.into())
105    }
106
107    /// Create a routing error
108    /// 创建路由错误
109    pub fn routing(msg: impl Into<String>) -> Self
110    {
111        Self::Routing(msg.into())
112    }
113}

115impl From<serde_json::Error> for IntegrationError
116{
117    fn from(err: serde_json::Error) -> Self
118    {
119        Self::Serialization(err.to_string())
120    }
121}
