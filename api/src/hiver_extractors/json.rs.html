33/// JSON body extractor
34/// JSON body提取器
35///
36/// Equivalent to Spring's `@RequestBody` with JSON content.
37/// 等价于Spring的`@RequestBody`与JSON内容。
38///
39/// # Type Parameters / 类型参数
40///
41/// - `T` - The type to deserialize from JSON. Must implement `Deserialize`. 要从 JSON
42///   反序列化的类型。必须实现 `Deserialize`。
43///
44/// # Example / 示例
45///
46/// ```rust,no_run,ignore
47/// use serde::Deserialize;
48///
49/// #[derive(Deserialize)]
50/// struct User {
51///     name: String,
52///     email: String,
53/// }
54///
55/// async fn create_user(Json(user): Json<User>) -> String {
56///     format!("User: {}", user.name)
57/// }
58/// ```
59pub struct Json<T>(pub T);
60
61impl<T> Json<T>
62{
63    /// Consume the JSON extractor and get the inner value
64    /// 消耗JSON提取器并获取内部值
65    pub fn into_inner(self) -> T
66    {
67        self.0
68    }
69
70    /// Get reference to the inner value
71    /// 获取内部值的引用
72    pub fn get(&self) -> &T
73    {
74        &self.0
75    }
76
77    /// Get mutable reference to the inner value
78    /// 获取内部值的可变引用
79    pub fn get_mut(&mut self) -> &mut T
80    {
81        &mut self.0
82    }
83}

85impl<T> std::fmt::Debug for Json<T>
86where
87    T: std::fmt::Debug,
88{
89    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result

105// Implement FromRequest for deserializable types
106impl<T> FromRequest for Json<T>
107where
108    T: for<'de> Deserialize<'de> + Send + 'static,
109{
110    fn from_request(req: &Request) -> ExtractorFuture<Self>
111    {
112        let body_bytes = req.body().as_bytes().map(<[u8]>::to_vec);
113        let content_type = req.header("content-type").unwrap_or("").to_string();
114
115        Box::pin(async move {
116            // Validate content type
117            if !content_type.starts_with("application/json")
118                && !content_type.starts_with("application/")
119                && !content_type.is_empty()
120            {
121                return Err(ExtractorError::Invalid(format!(
122                    "Expected JSON content type, got: {}",
123                    content_type
124                )));
125            }
126
127            let body = body_bytes.ok_or_else(|| {
128                ExtractorError::Invalid("Request body is not available".to_string())
129            })?;
130
131            // Parse JSON
132            serde_json::from_slice::<T>(&body)
133                .map(Json)
134                .map_err(ExtractorError::from)
135        })
136    }
137}
