21/// Flyway database migration executor
22/// Flyway 数据库迁移执行器
23///
24/// # Spring Equivalent / Spring等价物
25///
26/// ```java
27/// @Bean
28/// public Flyway flyway(DataSource dataSource) {
29///     return Flyway.configure()
30///         .dataSource(dataSource)
31///         .locations("db/migration")
32///         .load();
33/// }
34///
35/// @Bean
36/// public FlywayMigrationStrategy flywayMigrationStrategy(Flyway flyway) {
37///     return new FlywayMigrationStrategy(flyway);
38/// }
39/// ```
40///
41/// # Example / 示例
42///
43/// ```rust,ignore
44/// use hiver_flyway::{Flyway, Config};
45///
46/// let config = Config::builder()
47///     .datasource_url("postgresql://localhost:5432/mydb")
48///     .build()?;
49///
50/// let flyway = Flyway::new(config).await?;
51///
52/// // Migrate to latest version
53/// let result = flyway.migrate().await?;
54/// println!("Executed {} migrations", result.executed_count());
55///
56/// // Get migration info
57/// let info = flyway.info().await?;
58/// println!("Current version: {:?}", info.current_version());
59/// ```

101    /// Migrate to the latest version
102    /// 迁移到最新版本
103    ///
104    /// # Spring Equivalent / Spring等价物
105    ///
106    /// ```java
107    /// flyway.migrate();
108    /// ```
109    pub async fn migrate(&self) -> Result<MigrationResult>
110    {
111        info!("Starting database migration on {}", self.db_type);
112
113        let start = Instant::now();
114
115        // Ensure schema history table exists
116        self.ensure_schema_history_table().await?;
117
118        // Load migrations from file system
119        let migrations = self.load_migrations()?;
120
121        // Get current migration state
122        let current_info = self.get_info().await?;
123
124        // Find pending migrations
125        let pending = self.find_pending_migrations(&current_info, &migrations)?;
126
127        if pending.is_empty()
128        {
129            info!("No pending migrations to apply");
130            return Ok(MigrationResult::success(
131                Vec::new(),
132                current_info.current_version().cloned(),
133                start.elapsed().as_millis() as i64,
134            ));
135        }
136
137        debug!("Found {} pending migrations", pending.len());
138
139        // Execute pending migrations
140        let mut executed = Vec::new();
141        let warnings = Vec::new();
142
143        for &idx in &pending
144        {
145            let migration = &migrations[idx];
146            match self.execute_migration(migration).await
147            {
148                Ok(version) =>
149                {
150                    info!(
151                        "Applied migration: {} - {}",
152                        migration.version(),
153                        migration.description()
154                    );
155                    executed.push(version);
156                },
157                Err(e) =>
158                {
159                    warn!("Migration failed: {} - {}", migration.version(), e);
160                    return Ok(MigrationResult::failed(format!(
161                        "Migration {} failed: {}",
162                        migration.version(),
163                        e
164                    )));
165                },
166            }
167        }
168
169        let target_version = executed.last().map(|v| v.version.clone());
170
171        Ok(MigrationResult {
172            migrations_executed: executed,
173            target_version,
174            total_execution_time_ms: start.elapsed().as_millis() as i64,
175            success: true,
176            warnings,
177        })
178    }

180    /// Get current migration information
181    /// 获取当前迁移信息
182    ///
183    /// # Spring Equivalent / Spring等价物
184    ///
185    /// ```java
186    /// MigrationInfoService infoService = flyway.info();
187    /// ```
188    pub async fn info(&self) -> Result<Info>
189    {
190        let schema_exists = self.schema_history_table_exists().await?;
191
192        if !schema_exists
193        {
194            return Ok(Info::new());
195        }
196
197        let applied = self.get_applied_migrations().await?;
198        let migrations = self.load_migrations()?;
199
200        let current_version = applied.keys().max().cloned();
201
202        // Find pending migrations
203        let pending: Vec<MigrationEntry> = migrations
204            .iter()
205            .filter(|m| !applied.contains_key(m.version()))
206            .map(|m| MigrationEntry {
207                installed_rank: 0,
208                version: m.version().clone(),
209                description: m.description().clone(),
210                migration_type: m.migration_type(),
211                checksum: m.checksum(),
212                installed_by: None,
213                installed_on: None,
214                execution_time: 0,
215                success: true,
216            })
217            .collect();
218
219        let all: Vec<MigrationEntry> = applied.values().cloned().chain(pending.clone()).collect();
220
221        Ok(Info {
222            schema_exists,
223            current_version,
224            all,
225            pending,
226            applied,
227        })
228    }

230    /// Validate applied migrations
231    /// 校验已应用的迁移
232    ///
233    /// # Spring Equivalent / Spring等价物
234    ///
235    /// ```java
236    /// flyway.validate();
237    /// ```
238    pub async fn validate(&self) -> Result<()>
239    {
240        let info = self.info().await?;
241
242        if !info.schema_exists
243        {
244            return Err(crate::FlywayError::ValidationError(
245                "Schema history table does not exist".to_string(),
246            ));
247        }
248
249        // Check checksums
250        for entry in &info.all
251        {
252            if let Some(stored_checksum) = entry.checksum
253            {
254                // Load migration from file and verify checksum
255                let migrations = self.load_migrations()?;
256                if let Some(migration) = migrations.iter().find(|m| m.version() == &entry.version)
257                    && let Some(file_checksum) = migration.checksum()
258                    && stored_checksum != file_checksum
259                {
260                    return Err(crate::FlywayError::ChecksumMismatch {
261                        version: entry.version.clone(),
262                        expected: stored_checksum,
263                        actual: file_checksum,
264                    });
265                }
266            }
267        }
268
269        info!("Validation passed");
270        Ok(())
271    }

273    /// Baseline an existing database
274    /// 对现有数据库设置基线
275    ///
276    /// # Spring Equivalent / Spring等价物
277    ///
278    /// ```java
279    /// flyway.baseline();
280    /// ```
281    pub async fn baseline(&self) -> Result<BaselineInfo>
282    {
283        let info = self.info().await?;
284
285        if info.applied_count() > 0
286        {
287            return Err(crate::FlywayError::ValidationError(
288                "Cannot baseline: migrations already applied".to_string(),
289            ));
290        }
291
292        let baseline_info = BaselineInfo::new(self.config.baseline_version.clone());
293
294        self.insert_baseline(&baseline_info).await?;
295
296        info!("Baseline set to version {}", baseline_info.version);
297
298        Ok(baseline_info)
299    }

302    /// 清理数据库（删除所有对象）
303    ///
304    /// # Spring Equivalent / Spring等价物
305    ///
306    /// ```java
307    /// flyway.clean();
308    /// ```

310    {
311        // This is a destructive operation - only allow in development
312        #[cfg(debug_assertions)]
313        {
314            warn!("Cleaning database - dropping all objects");
315            let ddl = self.db_type.clean_ddl();
316            sqlx::query(ddl)
317                .execute(&self.pool)
318                .await
319                .map_err(|e| crate::FlywayError::MigrationError(e.to_string()))?;
320        }
321
322        #[cfg(not(debug_assertions))]
323        {
324            return Err(crate::FlywayError::ValidationError(
325                "Clean operation not allowed in production".to_string(),
326            ));
327        }
328
329        Ok(())
330    }

373    /// Load migrations from file system
374    /// 从文件系统加载迁移
375    ///
376    /// Supports database-specific migration files with naming convention:
377    /// 支持数据库特定的迁移文件命名约定：
378    ///
379    /// - `V1__init.sql` - runs on all databases (所有数据库)
380    /// - `V1__init.postgresql.sql` - runs only on PostgreSQL (仅 PostgreSQL)
381    /// - `V1__init.mysql.sql` - runs only on MySQL (仅 MySQL)
382    /// - `V1__init.sqlite.sql` - runs only on SQLite (仅 SQLite)
383    fn load_migrations(&self) -> Result<MigrationVec>
384    {
385        let migrations_dir = self.config.migrations_dir();
386
387        if !migrations_dir.exists()
388        {
389            debug!("Migrations directory does not exist: {:?}", migrations_dir);
390            return Ok(Vec::new());
391        }
392
393        let mut migrations = Vec::new();
394        let target_suffix = self.db_type.file_suffix();
395
396        // Track which base versions have been overridden by a dialect-specific file
397        // 跟踪被方言特定文件覆盖的基础版本
398        let mut dialect_overrides = std::collections::HashSet::new();
399
400        let entries: Vec<_> = std::fs::read_dir(&migrations_dir)
401            .map_err(|_e| crate::FlywayError::FileNotFound(migrations_dir.clone()))?
402            .filter_map(|e| e.ok())
403            .collect();
404
405        // First pass: identify dialect-specific overrides
406        // 第一遍：识别方言特定的覆盖
407        for entry in &entries
408        {
409            let path = entry.path();
410            if path.is_dir()
411            {
412                continue;
413            }
414            let file_name = path.file_name().unwrap().to_string_lossy().to_string();
415            if let Some((base_name, _desc)) = parse_migration_filename_with_dialect(&file_name)
416                && let Some(dialect) = extract_dialect_suffix(&base_name)
417                && dialect == target_suffix
418            {
419                // This dialect-specific file matches our DB
420                // 此方言特定文件匹配我们的数据库
421                let base_version = strip_dialect_suffix(&base_name);
422                dialect_overrides.insert(base_version);
423            }
424        }
425
426        // Second pass: collect applicable migrations
427        // 第二遍：收集适用的迁移
428        for entry in &entries
429        {
430            let path = entry.path();
431            if path.is_dir() || path.file_name().is_none()
432            {
433                continue;
434            }
435
436            let file_name = path.file_name().unwrap().to_string_lossy().to_string();
437
438            if let Some((base_name, description)) =
439                parse_migration_filename_with_dialect(&file_name)
440            {
441                let (effective_name, applicable) = if let Some(dialect) =
442                    extract_dialect_suffix(&base_name)
443                {
444                    // Dialect-specific file: only applicable if dialect matches
445                    // 方言特定文件：仅当方言匹配时适用
446                    let applicable = dialect == target_suffix;
447                    let effective_name = strip_dialect_suffix(&base_name);
448                    (effective_name, applicable)
449                }
450                else
451                {
452                    // Generic file: applicable unless overridden by a dialect-specific file
453                    // 通用文件：除非被方言特定文件覆盖，否则适用
454                    let overridden = dialect_overrides.contains(&base_name);
455                    (base_name, !overridden)
456                };
457
458                if !applicable
459                {
460                    debug!(
461                        "Skipping migration {} (not applicable for {})",
462                        file_name, self.db_type
463                    );
464                    continue;
465                }
466
467                let sql = std::fs::read_to_string(&path).map_err(crate::FlywayError::Io)?;
468
469                let migration: Migration =
470                    SqlMigration::new(effective_name, description, sql).into();
471                migrations.push(migration);
472            }
473        }
474
475        // Sort by version
476        migrations.sort_by(|a, b| a.version().cmp(b.version()));
477
478        Ok(migrations)
479    }

481    /// Get applied migrations from database
482    /// 从数据库获取已应用的迁移
483    async fn get_applied_migrations(
484        &self,
485    ) -> Result<std::collections::HashMap<String, MigrationEntry>>
486    {
487        let query = format!(
488            "SELECT installed_rank, version, description, type, checksum,
489                    installed_by, installed_on, execution_time, success
490             FROM {}
491             ORDER BY installed_rank",
492            self.config.table
493        );
494
495        let rows = sqlx::query(&query)
496            .fetch_all(&self.pool)
497            .await
498            .map_err(|e| crate::FlywayError::MigrationError(e.to_string()))?;
499
500        let mut applied = std::collections::HashMap::new();
501
502        for row in rows
503        {
504            let success_val: bool = if self.db_type == DatabaseType::Sqlite
505            {
506                // SQLite stores booleans as integers
507                let int_val: i32 = row.get("success");
508                int_val != 0
509            }
510            else
511            {
512                row.get("success")
513            };
514
515            let entry = MigrationEntry {
516                installed_rank: row.get("installed_rank"),
517                version: row.get("version"),
518                description: row.get("description"),
519                migration_type: match &row.get::<String, _>("type")[..]
520                {
521                    "SQL" => MigrationType::SQL,
522                    "Rust" => MigrationType::Rust,
523                    _ => MigrationType::Repeatable,
524                },
525                checksum: row.get("checksum"),
526                installed_by: row.get("installed_by"),
527                installed_on: row.get::<Option<String>, _>("installed_on").and_then(|s| {
528                    chrono::DateTime::parse_from_rfc3339(&s)
529                        .ok()
530                        .map(|d| d.with_timezone(&chrono::Utc))
531                }),
532                execution_time: row.get("execution_time"),
533                success: success_val,
534            };
535            applied.insert(entry.version.clone(), entry);
536        }
537
538        Ok(applied)
539    }

548    /// Find pending migrations
549    /// 查找待执行的迁移
550    fn find_pending_migrations(
551        &self,
552        current_info: &Info,
553        migrations: &[Migration],
554    ) -> Result<Vec<usize>>
555    {
556        let mut pending_indices = Vec::new();
557
558        for (idx, migration) in migrations.iter().enumerate()
559        {
560            if !current_info.is_applied(migration.version())
561            {
562                pending_indices.push(idx);
563            }
564        }
565
566        // Sort by version (migrations are already sorted)
567        Ok(pending_indices)
568    }

570    /// Execute a single migration
571    /// 执行单个迁移
572    async fn execute_migration(&self, migration: &Migration) -> Result<MigratedVersion>
573    {
574        let start = Instant::now();
575
576        // Begin transaction
577        let mut tx = self.pool.begin().await?;
578
579        // Execute migration SQL
580        migration.execute_on(&mut tx).await?;
581
582        // Record migration in history
583        self.record_migration(&mut tx, migration, start.elapsed().as_millis() as i64)
584            .await?;
585
586        // Commit transaction
587        tx.commit().await?;
588
589        Ok(MigratedVersion {
590            version: migration.version().clone(),
591            description: migration.description().clone(),
592            execution_time_ms: start.elapsed().as_millis() as i64,
593            success: true,
594        })
595    }

597    /// Record migration in history table
598    /// 在历史表中记录迁移
599    async fn record_migration(
600        &self,
601        tx: &mut sqlx::Transaction<'_, Any>,
602        migration: &Migration,
603        execution_time: i64,
604    ) -> Result<()>
605    {
606        // Get next installed rank
607        let rank_query = self.db_type.next_rank_sql(&self.config.table);
608        let next_rank: i32 = sqlx::query_scalar(&rank_query)
609            .fetch_one(&mut **tx)
610            .await
611            .unwrap_or(0);
612
613        let (insert_query, _) = self.db_type.record_migration_sql(&self.config.table);
614
615        // SQLite does not support chrono::DateTime directly; convert to string
616        // SQLite 不直接支持 chrono::DateTime；转换为字符串
617        let now = Utc::now();
618        match self.db_type
619        {
620            DatabaseType::Sqlite =>
621            {
622                sqlx::query(&insert_query)
623                    .bind(next_rank)
624                    .bind(migration.version())
625                    .bind(migration.description())
626                    .bind(migration.migration_type().to_string())
627                    .bind(migration.checksum())
628                    .bind("hiver-flyway")
629                    .bind(now.to_rfc3339())
630                    .bind(execution_time)
631                    .bind(1i32) // SQLite: 1 = true
632                    .execute(&mut **tx)
633                    .await
634                    .map_err(|e| crate::FlywayError::MigrationError(e.to_string()))?;
635            },
636            DatabaseType::Postgres | DatabaseType::Mysql =>
637            {
638                sqlx::query(&insert_query)
639                    .bind(next_rank)
640                    .bind(migration.version())
641                    .bind(migration.description())
642                    .bind(migration.migration_type().to_string())
643                    .bind(migration.checksum())
644                    .bind("hiver-flyway")
645                    .bind(now.to_rfc3339())
646                    .bind(execution_time)
647                    .bind(true)
648                    .execute(&mut **tx)
649                    .await
650                    .map_err(|e| crate::FlywayError::MigrationError(e.to_string()))?;
651            },
652        }
653
654        Ok(())
655    }

696/// Parse Flyway migration filename with optional database dialect suffix
697/// 解析带可选数据库方言后缀的 Flyway 迁移文件名
698///
699/// # Format / 格式
700///
701/// - `V1__Description.sql` - Generic migration (通用迁移)
702/// - `V1__Description.postgresql.sql` - PostgreSQL-specific (PostgreSQL 专用)
703/// - `V1__Description.mysql.sql` - MySQL-specific (MySQL 专用)
704/// - `V1__Description.sqlite.sql` - SQLite-specific (SQLite 专用)
705/// - `V1.1__Description.sql` - Versioned migration with sub-version
706/// - `R__Description.sql` - Repeatable migration
707fn parse_migration_filename_with_dialect(filename: &str) -> Option<(String, String)>
708{
709    // Must end with .sql
710    if !filename.ends_with(".sql")
711    {
712        return None;
713    }
714
715    let without_ext = &filename[..filename.len() - 4];
716
717    if without_ext.starts_with("V")
718    {
719        // Versioned migration: V1__Description or V1__Description.postgresql
720        let parts: Vec<&str> = without_ext.splitn(2, "__").collect();
721        if parts.len() == 2
722        {
723            Some((parts[0].to_string(), parts[1].to_string()))
724        }
725        else
726        {
727            None
728        }
729    }
730    else if without_ext.starts_with("R__")
731    {
732        let description = without_ext.strip_prefix("R__")?;
733        Some(("R__".to_string(), description.to_string()))
734    }
735    else
736    {
737        None
738    }
739}

741/// Extract the dialect suffix from a base name like "V1.postgresql"
742/// 从基础名称（如 "V1.postgresql"）中提取方言后缀
743fn extract_dialect_suffix(base_name: &str) -> Option<&str>
744{
745    // Known dialect suffixes
746    // 已知的方言后缀
747    let suffixes = ["postgresql", "mysql", "sqlite"];
748
749    suffixes
750        .into_iter()
751        .find(|&suffix| {
752            base_name.ends_with(suffix)
753                && base_name.as_bytes().get(base_name.len() - suffix.len() - 1) == Some(&b'.')
754        })
755        .map(|v| v as _)
756}
757
758/// Strip the dialect suffix from a base name: "V1.postgresql" -> "V1"
759/// 去除基础名称中的方言后缀
760fn strip_dialect_suffix(base_name: &str) -> String
761{
762    if let Some(dialect) = extract_dialect_suffix(base_name)
763    {
764        let end = base_name.len() - dialect.len() - 1; // -1 for the dot
765        base_name[..end].to_string()
766    }
767    else
768    {
769        base_name.to_string()
770    }
771}
