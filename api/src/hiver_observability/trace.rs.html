1//! Distributed tracing module
2//! 分布式追踪模块
3//!
4//! # Overview / 概述
5//!
6//! This module provides distributed tracing functionality compatible with `OpenTelemetry`.
7//! It includes span management, trace context propagation, and W3C Trace Context format.
8//!
9//! 本模块提供与 `OpenTelemetry` 兼容的分布式追踪功能。
10//! 包括 span 管理、追踪上下文传播和 W3C Trace Context 格式。
11//!
12//! # Equivalent to Spring Boot / 等价于 Spring Boot
13//!
14//! - Spring Cloud Sleuth + `OpenTelemetry` Tracing
15//! - Micrometer Tracing
16//! - @`NewSpan` annotation
17//!
18//! # Example / 示例
19//!
20//! ```rust,no_run,ignore
21//! use hiver_observability::trace::{Tracer, SpanBuilder};
22//!
23//! let tracer = Tracer::new("my-service");
24//!
25//! // Create a root span
26//! let span = tracer.span("handle_request")
27//!     .with_attribute("http.method", "GET")
28//!     .with_attribute("http.path", "/api/users")
29//!     .start();
30//!
31//! // Do some work...
32//! span.end();
33//! ```

47/// Global trace ID generator
48/// 全局追踪ID生成器
49static TRACE_ID_GENERATOR: std::sync::LazyLock<Arc<IdGenerator>> =
50    std::sync::LazyLock::new(|| Arc::new(IdGenerator::new()));

52/// ID generator for trace and span IDs
53/// ID生成器，用于追踪和span ID
54struct IdGenerator
55{
56    counter: AtomicU64,
57}
58
59impl IdGenerator
60{
61    fn new() -> Self
62    {
63        Self {
64            counter: AtomicU64::new(1),
65        }
66    }
67
68    /// Generate next trace ID (128-bit)
69    /// 生成下一个追踪ID（128位）
70    #[allow(clippy::expect_used)]
71    fn next_trace_id(&self) -> TraceId
72    {
73        let timestamp = SystemTime::now()
74            .duration_since(UNIX_EPOCH)
75            .expect("system clock before UNIX_EPOCH")
76            .as_nanos() as u64;
77        let random = self.counter.fetch_add(1, Ordering::Relaxed);
78        TraceId::from_parts(timestamp, random)
79    }
80
81    /// Generate next span ID (64-bit)
82    /// 生成下一个span ID（64位）
83    fn next_span_id(&self) -> SpanId
84    {
85        let id = self.counter.fetch_add(1, Ordering::Relaxed);
86        SpanId::from_u64(id)
87    }
88}

90/// Trace ID (128-bit)
91/// 追踪ID（128位）
92///
93/// Uniquely identifies a distributed trace.
94/// 唯一标识分布式追踪。
95#[derive(Clone, Copy, PartialEq, Eq, Hash)]
96pub struct TraceId([u8; 16]);
97
98impl TraceId
99{
100    /// Create a new trace ID
101    /// 创建新的追踪ID
102    pub fn new() -> Self
103    {
104        TRACE_ID_GENERATOR.next_trace_id()
105    }
106
107    /// Create from high and low u64 parts
108    /// 从高位和低位u64部分创建
109    pub fn from_parts(high: u64, low: u64) -> Self
110    {
111        let mut bytes = [0u8; 16];
112        bytes[0..8].copy_from_slice(&high.to_be_bytes());
113        bytes[8..16].copy_from_slice(&low.to_be_bytes());
114        Self(bytes)
115    }
116
117    /// Create from u64 (zero-extended)
118    /// 从u64创建（零扩展）
119    pub fn from_u64(value: u64) -> Self
120    {
121        Self::from_parts(0, value)
122    }
123
124    /// Parse from hex string
125    /// 从十六进制字符串解析
126    pub fn from_hex(hex: &str) -> Option<Self>
127    {
128        let bytes = if hex.len() == 32
129        {
130            hex.as_bytes()
131                .chunks(2)
132                .map(|chunk| u8::from_str_radix(std::str::from_utf8(chunk).ok()?, 16).ok())
133                .collect::<Option<Vec<_>>>()?
134        }
135        else
136        {
137            return None;
138        };
139        Some(Self(bytes.try_into().ok()?))
140    }
141
142    /// Get high 64 bits
143    /// 获取高64位
144    #[allow(clippy::expect_used)]
145    pub fn high(&self) -> u64
146    {
147        u64::from_be_bytes(self.0[0..8].try_into().expect("slice has correct length"))
148    }
149
150    /// Get low 64 bits
151    /// 获取低64位
152    #[allow(clippy::expect_used)]
153    pub fn low(&self) -> u64
154    {
155        u64::from_be_bytes(self.0[8..16].try_into().expect("slice has correct length"))
156    }
157
158    /// Convert to hex string
159    /// 转换为十六进制字符串
160    #[allow(clippy::format_collect)]
161    pub fn to_hex(&self) -> String
162    {
163        self.0.iter().map(|b| format!("{:02x}", b)).collect()
164    }
165}

316/// Trace context for distributed tracing
317/// 分布式追踪的追踪上下文
318///
319/// Contains the trace ID, span ID, and flags needed to propagate
320/// trace information across service boundaries.
321///
322/// 包含跨服务边界传播追踪信息所需的追踪ID、span ID和标志。
323#[derive(Clone, PartialEq, Eq)]
324pub struct TraceContext
325{
326    /// Trace ID
327    /// 追踪ID
328    pub trace_id: TraceId,
329
330    /// Current span ID
331    /// 当前span ID
332    pub span_id: SpanId,
333
334    /// Parent span ID (if any)
335    /// 父span ID（如果有）
336    pub parent_span_id: Option<SpanId>,
337
338    /// Trace flags
339    /// 追踪标志
340    pub flags: SpanFlags,
341
342    /// Trace state (for W3C tracestate)
343    /// 追踪状态（用于W3C tracestate）
344    pub trace_state: Vec<(String, String)>,
345}
346
347impl TraceContext
348{
349    /// Create a new trace context (root span)
350    /// 创建新的追踪上下文（根span）
351    pub fn new() -> Self
352    {
353        Self {
354            trace_id: TraceId::new(),
355            span_id: SpanId::new(),
356            parent_span_id: None,
357            flags: SpanFlags::SAMPLED,
358            trace_state: Vec::new(),
359        }
360    }
361
362    /// Create a child context from this one
363    /// 从此上下文创建子上下文
364    pub fn child(&self) -> Self
365    {
366        Self {
367            trace_id: self.trace_id,
368            span_id: SpanId::new(),
369            parent_span_id: Some(self.span_id),
370            flags: self.flags,
371            trace_state: self.trace_state.clone(),
372        }
373    }
374
375    /// Create from W3C traceparent header value
376    /// 从W3C traceparent头值创建
377    ///
378    /// Format: `version-trace_id-span_id-flags`
379    /// Example: `00-4bf92f3577b34da6a3ce929d0e0e4736-00f067aa0ba902b7-01`
380    pub fn from_traceparent(value: &str) -> Result<Self, TraceContextError>
381    {
382        let parts: Vec<&str> = value.split('-').collect();
383        if parts.len() != 4
384        {
385            return Err(TraceContextError::InvalidFormat);
386        }
387
388        // SAFETY: length verified to be 4 above
389        #[allow(clippy::indexing_slicing)]
390        let version = parts[0];
391        if version != "00"
392        {
393            return Err(TraceContextError::UnsupportedVersion(version.to_string()));
394        }
395
396        // SAFETY: length verified to be 4 above
397        #[allow(clippy::indexing_slicing)]
398        let trace_id = TraceId::from_hex(parts[1])
399            .ok_or(TraceContextError::InvalidTraceId(parts[1].to_string()))?;
400
401        // SAFETY: length verified to be 4 above
402        #[allow(clippy::indexing_slicing)]
403        let span_id = SpanId::from_hex(parts[2])
404            .ok_or(TraceContextError::InvalidSpanId(parts[2].to_string()))?;
405
406        // SAFETY: length verified to be 4 above
407        #[allow(clippy::indexing_slicing)]
408        let flags = u8::from_str_radix(parts[3], 16)
409            .map_err(|_| TraceContextError::InvalidFlags(parts[3].to_string()))?;
410
411        Ok(Self {
412            trace_id,
413            span_id,
414            parent_span_id: None,
415            flags: SpanFlags(flags),
416            trace_state: Vec::new(),
417        })
418    }
419
420    /// Convert to W3C traceparent header value
421    /// 转换为W3C traceparent头值
422    pub fn to_traceparent(&self) -> String
423    {
424        format!("00-{}-{}-{:02x}", self.trace_id.to_hex(), self.span_id.to_hex(), self.flags.0)
425    }
426
427    /// Check if this trace is sampled
428    /// 检查此追踪是否被采样
429    pub fn is_sampled(&self) -> bool
430    {
431        self.flags.is_sampled()
432    }
433
434    /// Add a trace state entry
435    /// 添加追踪状态条目
436    pub fn with_trace_state(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
437    {
438        self.trace_state.push((key.into(), value.into()));
439        self
440    }
441}

445    fn default() -> Self
446    {
447        Self::new()
448    }

465/// Trace context error
466/// 追踪上下文错误
467#[derive(Debug, Clone)]
468pub enum TraceContextError
469{
470    /// Invalid traceparent format
471    /// 无效的traceparent格式
472    InvalidFormat,
473
474    /// Unsupported traceparent version
475    /// 不支持的traceparent版本
476    UnsupportedVersion(String),
477
478    /// Invalid trace ID
479    /// 无效的追踪ID
480    InvalidTraceId(String),
481
482    /// Invalid span ID
483    /// 无效的span ID
484    InvalidSpanId(String),
485
486    /// Invalid flags
487    /// 无效的标志
488    InvalidFlags(String),
489}
490
491impl fmt::Display for TraceContextError
492{
493    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
494    {
495        match self
496        {
497            Self::InvalidFormat => write!(f, "Invalid traceparent format"),
498            Self::UnsupportedVersion(v) => write!(f, "Unsupported traceparent version: {}", v),
499            Self::InvalidTraceId(id) => write!(f, "Invalid trace ID: {}", id),
500            Self::InvalidSpanId(id) => write!(f, "Invalid span ID: {}", id),
501            Self::InvalidFlags(flg) => write!(f, "Invalid flags: {}", flg),
502        }
503    }
504}

786    /// Get span duration in nanoseconds (None if not ended)
787    /// 获取span持续时间（以纳秒为单位，如果未结束则为None）
788    pub fn duration(&self) -> Option<u64>
789    {
790        self.end_time.map(|end| end.saturating_sub(self.start_time))
791    }
792
793    /// Check if the span is active (not ended)
794    /// 检查span是否处于活动状态（未结束）
795    pub fn is_active(&self) -> bool
796    {
797        self.end_time.is_none()
798    }
799
800    /// Check if the span is recording
801    /// 检查span是否正在记录
802    pub fn is_recording(&self) -> bool
803    {
804        self.is_recording
805    }
806
807    /// End the span
808    /// 结束span
809    #[allow(clippy::unwrap_used)]
810    pub fn end(&mut self)
811    {
812        if self.is_active()
813        {
814            self.end_time = Some(
815                SystemTime::now()
816                    .duration_since(UNIX_EPOCH)
817                    .unwrap()
818                    .as_nanos() as u64,
819            );
820        }
821    }
822
823    /// End the span with a status
824    /// 使用状态结束span
825    pub fn end_with_status(&mut self, status: SpanStatus)
826    {
827        self.status = Some(status);
828        self.end();
829    }
830
831    /// Create a child span
832    /// 创建子span
833    pub fn child(&self, name: impl Into<String>) -> Self
834    {
835        let child_ctx = self.context.child();
836        Self::with_context(name, child_ctx)
837    }
838}

858/// Builder for creating spans
859/// 用于创建span的构建器
860pub struct SpanBuilder
861{
862    name: String,
863    context: Option<TraceContext>,
864    kind: SpanKind,
865    attributes: Vec<(String, String)>,
866}
867
868impl SpanBuilder
869{
870    /// Create a new span builder
871    /// 创建新的span构建器
872    pub fn new(name: impl Into<String>) -> Self
873    {
874        Self {
875            name: name.into(),
876            context: None,
877            kind: SpanKind::Internal,
878            attributes: Vec::new(),
879        }
880    }
881
882    /// Set the trace context
883    /// 设置追踪上下文
884    pub fn with_context(mut self, context: TraceContext) -> Self
885    {
886        self.context = Some(context);
887        self
888    }
889
890    /// Set the span kind
891    /// 设置span类型
892    pub fn with_kind(mut self, kind: SpanKind) -> Self
893    {
894        self.kind = kind;
895        self
896    }
897
898    /// Add an attribute
899    /// 添加属性
900    pub fn with_attribute(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
901    {
902        self.attributes.push((key.into(), value.into()));
903        self
904    }
905
906    /// Build the span
907    /// 构建span
908    pub fn build(self) -> Span
909    {
910        let context = self.context.unwrap_or_default();
911        let mut span = Span::with_context(self.name, context).with_kind(self.kind);
912
913        for (key, value) in self.attributes
914        {
915            span.add_attribute(key, value);
916        }
917
918        span
919    }
920
921    /// Build and start the span
922    /// 构建并启动span
923    pub fn start(self) -> Span
924    {
925        self.build()
926    }
927}

929/// Tracer for creating and managing spans
930/// 用于创建和管理span的追踪器
931///
932/// The tracer is the entry point for creating spans and managing
933/// distributed tracing.
934///
935/// 追踪器是创建span和管理分布式追踪的入口点。
936#[derive(Clone)]
937pub struct Tracer
938{
939    /// Service name
940    /// 服务名称
941    service_name: String,
942
943    /// Whether to sample traces
944    /// 是否对追踪进行采样
945    sample_rate: f64,
946}
947
948impl Tracer
949{
950    /// Create a new tracer
951    /// 创建新的追踪器
952    pub fn new(service_name: impl Into<String>) -> Self
953    {
954        Self {
955            service_name: service_name.into(),
956            sample_rate: 1.0,
957        }
958    }
959
960    /// Set the sample rate (0.0 to 1.0)
961    /// 设置采样率（0.0到1.0）
962    pub fn with_sample_rate(mut self, rate: f64) -> Self
963    {
964        self.sample_rate = rate.clamp(0.0, 1.0);
965        self
966    }
967
968    /// Get the service name
969    /// 获取服务名称
970    pub fn service_name(&self) -> &str
971    {
972        &self.service_name
973    }
974
975    /// Create a span builder
976    /// 创建span构建器
977    pub fn span(&self, name: impl Into<String>) -> SpanBuilder
978    {
979        SpanBuilder::new(name)
980    }
981
982    /// Create a root span
983    /// 创建根span
984    pub fn root_span(&self, name: impl Into<String>) -> Span
985    {
986        self.span(name)
987            .with_kind(SpanKind::Server)
988            .with_attribute("service.name", self.service_name.clone())
989            .start()
990    }
991
992    /// Create a child span from a context
993    /// 从上下文创建子span
994    pub fn child_span(&self, name: impl Into<String>, parent: &TraceContext) -> Span
995    {
996        let child_ctx = parent.child();
997        Span::with_context(name, child_ctx)
998    }
999
1000    /// Extract trace context from headers
1001    /// 从头中提取追踪上下文
1002    pub fn extract(&self, headers: &[(String, String)]) -> Result<TraceContext, TraceContextError>
1003    {
1004        // Look for traceparent header
1005        for (key, value) in headers
1006        {
1007            if key.eq_ignore_ascii_case("traceparent")
1008            {
1009                return TraceContext::from_traceparent(value);
1010            }
1011        }
1012        Err(TraceContextError::InvalidFormat)
1013    }
1014
1015    /// Inject trace context into headers
1016    /// 将追踪上下文注入头中
1017    pub fn inject(&self, context: &TraceContext, headers: &mut Vec<(String, String)>)
1018    {
1019        headers.push(("traceparent".to_string(), context.to_traceparent()));
1020    }
1021}
1022
1023impl fmt::Debug for Tracer
1024{
1025    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
1026    {
1027        f.debug_struct("Tracer")
1028            .field("service_name", &self.service_name)
1029            .field("sample_rate", &self.sample_rate)
1030            .finish()
1031    }
1032}

1034/// Global tracer instance
1035/// 全局追踪器实例
1036static GLOBAL_TRACER: std::sync::LazyLock<std::sync::RwLock<Option<Tracer>>> =
1037    std::sync::LazyLock::new(|| std::sync::RwLock::new(None));
1038
1039/// Initialize the global tracer
1040/// 初始化全局追踪器
1041#[allow(clippy::expect_used)]
1042pub fn init_tracer(service_name: impl Into<String>) -> Tracer
1043{
1044    let tracer = Tracer::new(service_name);
1045    *GLOBAL_TRACER.write().expect("lock poisoned") = Some(tracer.clone());
1046    tracer
1047}
1048
1049/// Get the global tracer
1050/// 获取全局追踪器
1051#[allow(clippy::expect_used)]
1052pub fn global_tracer() -> Option<Tracer>
1053{
1054    GLOBAL_TRACER
1055        .read()
1056        .expect("lock poisoned")
1057        .as_ref()
1058        .cloned()
1059}
