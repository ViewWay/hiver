24/// Task ID type — a newtype around `u64` so it cannot be confused with
25/// other `u64` values (timestamps, counters) at call sites.
26/// 任务ID类型 —— `u64` 的 newtype，避免在调用点与其他 `u64` 值
27/// （时间戳、计数器）混淆。
28#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
29pub struct TaskId(u64);
30
31impl TaskId
32{
33    /// Sentinel for "no task" — returned when a task has no assigned core.
34    /// "无任务"哨兵 —— 当任务无分配 core 时返回。
35    pub const UNKNOWN: TaskId = TaskId(0);
36
37    /// Create a TaskId from a raw u64 (internal use only).
38    /// 从原始 u64 创建 TaskId（仅供内部使用）。
39    #[must_use]
40    pub const fn from_u64(id: u64) -> Self
41    {
42        Self(id)
43    }
44
45    /// Get the raw u64 value.
46    /// 获取原始 u64 值。
47    #[must_use]
48    pub const fn as_u64(self) -> u64
49    {
50        self.0
51    }
52}
53
54impl std::fmt::Display for TaskId
55{
56    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result
57    {
58        write!(f, "TaskId({})", self.0)
59    }
60}
61
62/// Generate a new unique task ID
63/// 生成新的唯一任务ID
64#[must_use]
65pub fn gen_task_id() -> TaskId
66{
67    use std::sync::atomic::{AtomicU64, Ordering};
68    static COUNTER: AtomicU64 = AtomicU64::new(1);
69    TaskId(COUNTER.fetch_add(1, Ordering::Relaxed))
70}
