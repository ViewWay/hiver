39/// Error handling example / 错误处理示例
40///
41/// Demonstrates different error types and how to create them.
42/// 演示不同的错误类型以及如何创建它们。
43fn error_handling_example()
44{
45    // Not Found error / 404错误
46    let not_found = Error::not_found("User not found");
47    println!("  404 Not Found: {}", not_found);
48    println!("    Kind: {:?}", not_found.kind());
49
50    // Bad Request error / 400错误
51    let bad_request = Error::with_message(ErrorKind::BadRequest, "Invalid input");
52    println!("  400 Bad Request: {}", bad_request);
53    println!("    Kind: {:?}", bad_request.kind());
54
55    // Internal Server Error / 500错误
56    let internal = Error::internal("Database connection failed");
57    println!("  500 Internal: {}", internal);
58    println!("    Kind: {:?}", internal.kind());
59
60    // Unauthorized error / 401错误
61    let unauthorized = Error::with_message(ErrorKind::Unauthorized, "Authentication required");
62    println!("  401 Unauthorized: {}", unauthorized);
63    println!("    Kind: {:?}", unauthorized.kind());
64
65    // Forbidden error / 403错误
66    let forbidden = Error::with_message(ErrorKind::Forbidden, "Access denied");
67    println!("  403 Forbidden: {}", forbidden);
68    println!("    Kind: {:?}", forbidden.kind());
69}

71/// Extensions example / 扩展示例
72///
73/// Demonstrates type-safe extensions storage and retrieval.
74/// 演示类型安全的扩展存储和检索。
75///
76/// Equivalent to Spring's `@RequestScope` beans or `Model` attributes.
77/// 等价于 Spring 的 `@RequestScope` bean 或 `Model` 属性。
78fn extensions_example()
79{
80    let mut ext = Extensions::new();
81
82    // Store different types / 存储不同类型
83    ext.insert("Hello, World!".to_string());
84    ext.insert(42i32);
85    ext.insert(vec![1, 2, 3]);
86    ext.insert(User {
87        id: 1,
88        name: "Alice".to_string(),
89    });
90
91    // Retrieve values / 检索值
92    if let Some(s) = ext.get::<String>()
93    {
94        println!("  String: {}", s);
95    }
96
97    if let Some(n) = ext.get::<i32>()
98    {
99        println!("  Integer: {}", n);
100    }
101
102    if let Some(v) = ext.get::<Vec<i32>>()
103    {
104        println!("  Vec: {:?}", v);
105    }
106
107    if let Some(user) = ext.get::<User>()
108    {
109        println!("  User: {} (id={})", user.name, user.id);
110    }
111
112    // Check if type exists / 检查类型是否存在
113    println!("  Has String?: {}", ext.contains::<String>());
114    println!("  Has bool?: {}", ext.contains::<bool>());
115}

117/// Error conversion example / 错误转换示例
118///
119/// Demonstrates how to convert application errors to HTTP errors.
120/// 演示如何将应用错误转换为 HTTP 错误。
121fn error_conversion_example()
122{
123    // Simulating database error / 模拟数据库错误
124    let db_error = "Connection refused";
125    let http_error = Error::internal(format!("Database error: {}", db_error));
126    println!("  DB → HTTP: {}", http_error);
127
128    // Simulating validation error / 模拟验证错误
129    let validation_error = "Email is required";
130    let http_error = Error::with_message(
131        ErrorKind::BadRequest,
132        format!("Validation failed: {}", validation_error),
133    );
134    println!("  Validation → HTTP: {}", http_error);
135
136    // Creating error from ErrorKind / 从 ErrorKind 创建错误
137    let custom_error = Error::new(ErrorKind::NotFound("Resource".to_string()));
138    println!("  Custom: {}", custom_error);
139
140    // Get status code from ErrorKind / 从ErrorKind获取状态码
141    let status = ErrorKind::NotFound("test".to_string()).status_code();
142    println!("  Status code for NotFound: {}", status);
143}
