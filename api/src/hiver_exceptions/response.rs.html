9/// Error response builder
10/// 错误响应构建器
11///
12/// # Spring Equivalent / Spring 等价物
13///
14/// Equivalent to Spring's `ResponseEntity` with error status codes.
15///
16/// # Example / 示例
17///
18/// ```rust,no_run,ignore
19/// use hiver_exceptions::ErrorResponse;
20///
21/// // Simple error response
22/// let response = ErrorResponse::bad_request("Invalid input");
23///
24/// // With error code
25/// let response = ErrorResponse::bad_request_code("VALIDATION_ERROR", "Username is required");
26///
27/// // Full error response
28/// let response = ErrorResponse::new(400)
29///     .error("VALIDATION_ERROR")
30///     .message("Username is required")
31///     .path("/api/users")
32///     .build();
33/// ```
34pub struct ErrorResponse
35{
36    status_code: u16,
37    error: Option<String>,
38    message: Option<String>,
39    timestamp: Option<String>,
40    path: Option<String>,
41    details: Option<Value>,
42}
43
44impl ErrorResponse
45{
46    /// Create a new error response builder
47    /// 创建新的错误响应构建器
48    pub fn new(status: u16) -> Self
49    {
50        Self {
51            status_code: status,
52            error: None,
53            message: None,
54            timestamp: None,
55            path: None,
56            details: None,
57        }
58    }
59
60    /// Create a bad request error response (400)
61    /// 创建 bad request 错误响应 (400)
62    pub fn bad_request(message: impl Into<String>) -> Response
63    {
64        let body = ErrorBody::bad_request("BAD_REQUEST", message);
65        Self::build_response(body)
66    }
67
68    /// Create a bad request error with custom error code (400)
69    /// 创建带自定义错误代码的 bad request 错误 (400)
70    pub fn bad_request_code(error: impl Into<String>, message: impl Into<String>) -> Response
71    {
72        let body = ErrorBody::bad_request(error, message);
73        Self::build_response(body)
74    }
75
76    /// Create an unauthorized error response (401)
77    /// 创建 unauthorized 错误响应 (401)
78    pub fn unauthorized(message: impl Into<String>) -> Response
79    {
80        let body = ErrorBody::unauthorized(message);
81        Self::build_response(body)
82    }
83
84    /// Create a forbidden error response (403)
85    /// 创建 forbidden 错误响应 (403)
86    pub fn forbidden(message: impl Into<String>) -> Response
87    {
88        let body = ErrorBody::forbidden(message);
89        Self::build_response(body)
90    }
91
92    /// Create a not found error response (404)
93    /// 创建 not found 错误响应 (404)
94    pub fn not_found(resource: impl Into<String>) -> Response
95    {
96        let body = ErrorBody::not_found(resource);
97        Self::build_response(body)
98    }
99
100    /// Create an internal server error response (500)
101    /// 创建 internal server error 错误响应 (500)
102    pub fn internal(message: impl Into<String>) -> Response
103    {
104        let body = ErrorBody::internal(message);
105        Self::build_response(body)
106    }
107
108    /// Set the error code
109    /// 设置错误代码
110    pub fn error(mut self, error: impl Into<String>) -> Self
111    {
112        self.error = Some(error.into());
113        self
114    }
115
116    /// Set the error message
117    /// 设置错误消息
118    pub fn message(mut self, message: impl Into<String>) -> Self
119    {
120        self.message = Some(message.into());
121        self
122    }
123
124    /// Set the timestamp
125    /// 设置时间戳
126    pub fn timestamp(mut self, timestamp: impl Into<String>) -> Self
127    {
128        self.timestamp = Some(timestamp.into());
129        self
130    }
131
132    /// Set the request path
133    /// 设置请求路径
134    pub fn path(mut self, path: impl Into<String>) -> Self
135    {
136        self.path = Some(path.into());
137        self
138    }
139
140    /// Set additional details
141    /// 设置额外详情
142    pub fn details(mut self, details: Value) -> Self
143    {
144        self.details = Some(details);
145        self
146    }
147
148    /// Build the response
149    /// 构建响应
150    pub fn build(self) -> Response
151    {
152        let error = self.error.unwrap_or_else(|| {
153            StatusCode::from_u16(self.status_code)
154                .canonical_reason()
155                .unwrap_or("ERROR")
156                .to_string()
157        });
158
159        let message = self.message.unwrap_or_else(|| error.clone());
160
161        let mut body = ErrorBody::new(error, message, self.status_code);
162
163        if let Some(ts) = self.timestamp
164        {
165            body = body.with_timestamp(ts);
166        }
167        if let Some(p) = self.path
168        {
169            body = body.with_path(p);
170        }
171        if let Some(d) = self.details
172        {
173            body = body.with_details(d);
174        }
175
176        Self::build_response(body)
177    }
178
179    /// Build response from `ErrorBody`
180    /// 从 `ErrorBody` 构建响应
181    #[allow(clippy::needless_pass_by_value)]
182    fn build_response(body: ErrorBody) -> Response
183    {
184        // Convert to JSON bytes
185        match serde_json::to_vec(&body)
186        {
187            Ok(bytes) =>
188            {
189                let status = StatusCode::from_u16(body.status);
190                Response::new(status).with_body(hiver_http::Body::from(bytes))
191            },
192            Err(_) => Response::internal_server_error()
193                .with_body(hiver_http::Body::from("{\"error\":\"SERIALIZATION_ERROR\"}")),
194        }
195    }
196}

198/// Extension trait to easily convert errors to responses
199/// 轻松将错误转换为响应的扩展 trait
200pub trait ToErrorResponse
201{
202    /// Convert this error to an error response
203    /// 将此错误转换为错误响应
204    fn to_error_response(&self) -> Response;
205}
206
207impl ToErrorResponse for String
208{
209    fn to_error_response(&self) -> Response
210    {
211        ErrorResponse::bad_request_code("ERROR", self)
212    }
213}
214
215impl ToErrorResponse for &str
216{
217    fn to_error_response(&self) -> Response
218    {
219        ErrorResponse::bad_request_code("ERROR", *self)
220    }
221}
222
223impl ToErrorResponse for std::io::Error
224{
225    fn to_error_response(&self) -> Response
226    {
227        ErrorResponse::internal(self.to_string())
228    }
229}
