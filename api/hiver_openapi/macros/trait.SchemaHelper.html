§Example / 示例

ⓘ
use hiver_openapi::SchemaHelper;
use hiver_openapi::ToSchema;

#[derive(ToSchema)]
struct User {
    id: u64,
    name: String,
}

impl SchemaHelper for User {
    fn description() -> &'static str {
        "User entity"
    }
}
