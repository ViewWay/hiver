1//! Hiver Extractors - Request data extractors
2//! Hiver提取器 - 请求数据提取器
3//!
4//! # Overview / 概述
5//!
6//! This crate provides type-safe extractors for pulling data from HTTP requests.
7//! Each extractor corresponds to a Spring Boot annotation for easy migration.
8//!
9//! 本crate提供类型安全的提取器，用于从HTTP请求中提取数据。
10//! 每个提取器对应一个Spring Boot注解，便于迁移。
11//!
12//! # Features / 功能特性
13//!
14//! - Path variable extraction (`@PathVariable`)
15//! - Query parameter extraction (`@RequestParam`)
16//! - JSON body extraction (`@RequestBody`)
17//! - Form data extraction (`@ModelAttribute`)
18//! - Header extraction (`@RequestHeader`)
19//! - Cookie extraction (`@CookieValue`)
20//! - Matrix variable extraction (`@MatrixVariable`)
21//! - Application state injection (`@Autowired`)
22//! - Request attribute access (`@RequestAttribute`)
23//! - Multipart file upload (`MultipartFile`)
24//!
25//! # Equivalent to Spring Boot / 等价于 Spring Boot
26//!
27//! | Hiver Extractor | Spring Annotation | Description / 描述 |
28//! |---|---|---|
29//! | `Path<T>` | `@PathVariable` | 路径变量 / Path variables |
30//! | `Query<T>` | `@RequestParam` | 查询参数 / Query parameters |
31//! | `Json<T>` | `@RequestBody` | JSON请求体 / JSON request body |
32//! | `Form<T>` | `@ModelAttribute` (form) | 表单数据 / Form data |
33//! | `ModelAttribute<T>` | `@ModelAttribute` | 查询+表单 / Query + form |
34//! | `QueryParams<T>` | `@RequestParam` (query only) | 仅查询参数 / Query parameters only |
35//! | `State<T>` | `@Autowired` | 应用状态 / Application state |
36//! | `RequestAttribute<T>` | `@RequestAttribute` | 请求属性 / Request attributes |
37//! | `Header<T>` | `@RequestHeader` | 请求头 / Request headers |
38//! | `Cookie<T>` | `@CookieValue` | Cookie值 / Cookie values |
39//! | `MatrixVariable<T>` | `@MatrixVariable` | 矩阵变量 / Matrix variables |
40//!
41//! # Example / 示例
42//!
43//! ```rust,no_run,ignore
44//! use hiver_extractors::*;
45//! use serde::Deserialize;
46//!
47//! #[derive(Deserialize)]
48//! struct UserQuery {
49//!     name: String,
50//!     page: Option<u32>,
51//! }
52//!
53//! // GET /search?name=Alice&page=1
54//! async fn search(Query(params): Query<UserQuery>) -> String {
55//!     format!("Searching for: {}", params.name)
56//! }
57//! ```

109/// Extractor trait
110/// 提取器trait
111///
112/// # Description / 描述
113///
114/// Core trait implemented by all request data extractors.
115/// Each extractor pulls a specific piece of data from the HTTP request
116/// and produces a typed result, returning an [`ExtractorError`] on failure.
117///
118/// 所有请求数据提取器实现的核心trait。
119/// 每个提取器从HTTP请求中提取特定类型的数据并产生类型化结果，
120/// 失败时返回 [`ExtractorError`]。
121///
122/// # Equivalent to Spring Boot / 等价于 Spring Boot
123///
124/// This trait corresponds to Spring's method parameter resolution mechanism
125/// used by `@PathVariable`, `@RequestParam`, `@RequestBody`, `@RequestHeader`,
126/// `@CookieValue`, `@ModelAttribute`, `@RequestAttribute`, and `@MatrixVariable`.
127///
128/// 该trait对应Spring的方法参数解析机制，用于 `@PathVariable`、`@RequestParam`、
129/// `@RequestBody`、`@RequestHeader`、`@CookieValue`、`@ModelAttribute`、
130/// `@RequestAttribute` 和 `@MatrixVariable`。
131///
132/// # Example / 示例
133///
134/// ```rust,no_run,ignore
135/// use hiver_extractors::{FromRequest, Query};
136/// use serde::Deserialize;
137///
138/// #[derive(Deserialize)]
139/// struct Params { id: u64 }
140///
141/// async fn handler(Query(params): Query<Params>) -> String {
142///     format!("ID: {}", params.id)
143/// }
144/// ```
145pub trait FromRequest: Sized
146{
147    /// Extract from request
148    /// 从请求提取
149    ///
150    /// Attempts to extract a value of type `Self` from the given request.
151    /// Returns a future that resolves to `Result<Self, ExtractorError>`.
152    ///
153    /// 尝试从给定请求中提取 `Self` 类型的值。
154    /// 返回一个future，解析为 `Result<Self, ExtractorError>`。
155    fn from_request(req: &Request) -> ExtractorFuture<Self>;
156}
157
158/// Future type returned by `FromRequest`.
159/// `FromRequest` 返回的 Future 类型。
160///
161/// A pinned, boxed, Send-safe future that resolves to either the extracted
162/// value `T` or an [`ExtractorError`].
163///
164/// 一个固定化的、装箱的、Send安全的 future，解析为提取的值 `T`
165/// 或 [`ExtractorError`]。
166pub type ExtractorFuture<T> = Pin<Box<dyn Future<Output = Result<T, ExtractorError>> + Send>>;
167
168/// Extractor error
169/// 提取器错误
170///
171/// # Description / 描述
172///
173/// Errors that can occur during request data extraction.
174/// Each variant represents a different category of extraction failure.
175///
176/// 请求数据提取过程中可能发生的错误。
177/// 每个变体代表不同类别的提取失败。
178#[derive(Debug, thiserror::Error)]
179pub enum ExtractorError
180{
181    /// Missing parameter
182    /// 缺少参数
183    ///
184    /// A required parameter was not found in the request.
185    /// 请求中未找到所需参数。
186    #[error("Missing parameter: {0}")]
187    Missing(String),
188
189    /// Invalid parameter format
190    /// 无效参数格式
191    ///
192    /// A parameter was found but could not be parsed or converted
193    /// to the target type.
194    /// 找到了参数，但无法解析或转换为目标类型。
195    #[error("Invalid parameter format: {0}")]
196    Invalid(String),
197
198    /// IO error
199    /// IO错误
200    ///
201    /// An I/O error occurred while reading the request body.
202    /// 读取请求主体时发生I/O错误。
203    #[error("IO error: {0}")]
204    Io(#[from] std::io::Error),
205
206    /// JSON error
207    /// JSON错误
208    ///
209    /// A JSON deserialization error occurred while parsing the
210    /// request body.
211    /// 解析请求主体时发生JSON反序列化错误。
212    #[error("JSON error: {0}")]
213    Json(#[from] serde_json::Error),
214
215    /// Other error
216    /// 其他错误
217    ///
218    /// A generic error that does not fit into the other categories.
219    /// 不适合其他类别的通用错误。
220    #[error("Error: {0}")]
221    Other(String),
222}
