1//! Route registry — compile-time collected route descriptors via `inventory`.
2//! 路由注册表 — 通过 `inventory` 在编译期收集路由描述符。
3//!
4//! Equivalent to Spring Boot's automatic route discovery.
5//! 等价于 Spring Boot 的自动路由发现。
6
7/// HTTP method for route registration.
8/// 路由注册的 HTTP 方法。
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10pub enum HttpMethod
11{
12    /// GET
13    Get,
14    /// POST
15    Post,
16    /// PUT
17    Put,
18    /// DELETE
19    Delete,
20    /// PATCH
21    Patch,
22    /// HEAD
23    Head,
24    /// OPTIONS
25    Options,
26    /// TRACE
27    Trace,
28}
29
30/// Compile-time route descriptor submitted by `#[get]`, `#[post]`, etc. macros.
31/// 由 `#[get]`、`#[post]` 等宏在编译期提交的路由描述符。
32///
33/// At startup, `RouterAutoConfiguration` collects all descriptors via
34/// `inventory::iter::<RouteDescriptor>()` and builds the router automatically.
35///
36/// 启动时，`RouterAutoConfiguration` 通过 `inventory::iter::<RouteDescriptor>()`
37/// 自动收集所有描述符并构建路由器。
38pub struct RouteDescriptor
39{
40    /// HTTP method.
41    /// HTTP 方法。
42    pub method: HttpMethod,
43
44    /// URL path pattern (e.g., "/users/{id}").
45    /// URL 路径模式（如 "/users/{id}"）。
46    pub path: &'static str,
47
48    /// Function that registers this route onto a `Router<()>`.
49    /// 将此路由注册到 `Router<()>` 的函数。
50    pub register: fn(hiver_router::Router) -> hiver_router::Router,
51}
52
53inventory::collect!(RouteDescriptor);
54
55/// Collect all inventory-registered routes and return a built router.
56/// 收集所有 inventory 注册的路由并返回构建好的路由器。
57///
58/// Equivalent to Spring Boot's `RequestMappingHandlerMapping`.
59/// 等价于 Spring Boot 的 `RequestMappingHandlerMapping`。
60pub fn collect_routes() -> hiver_router::Router
61{
62    let descriptors: Vec<&RouteDescriptor> = inventory::iter::<RouteDescriptor>().collect();
63    let mut router = hiver_router::Router::new();
64
65    for desc in &descriptors
66    {
67        let method_name = match desc.method
68        {
69            HttpMethod::Get => "GET",
70            HttpMethod::Post => "POST",
71            HttpMethod::Put => "PUT",
72            HttpMethod::Delete => "DELETE",
73            HttpMethod::Patch => "PATCH",
74            HttpMethod::Head => "HEAD",
75            HttpMethod::Options => "OPTIONS",
76            HttpMethod::Trace => "TRACE",
77        };
78        tracing::debug!("Registering route: {} {}", method_name, desc.path);
79        router = (desc.register)(router);
80    }
81
82    let count = descriptors.len();
83    if count > 0
84    {
85        tracing::info!("Registered {} routes from inventory", count);
86    }
87
88    router
89}
