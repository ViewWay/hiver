1//! Builder derive macro implementation
2//! Builder 派生宏实现
3
4use proc_macro2::TokenStream;
5use quote::quote;
6use syn::{Data, DataStruct, DeriveInput, Fields};
7
8/// Implement #[Builder] derive macro
9/// 实现 #[Builder] 派生宏
10///
11/// Generates a builder pattern for the struct.
12/// 为结构体生成 builder 模式。
13pub fn impl_builder(input: DeriveInput) -> TokenStream
14{
15    let struct_name = &input.ident;
16    let builder_name = format!("{}Builder", struct_name);
17    let builder_ident = syn::Ident::new(&builder_name, struct_name.span());
18
19    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
20
21    // Extract fields from struct
22    // 从结构体中提取字段
23    let fields = match &input.data
24    {
25        Data::Struct(DataStruct {
26            fields: Fields::Named(fields),
27            ..
28        }) => &fields.named,
29        _ =>
30        {
31            return syn::Error::new_spanned(
32                struct_name,
33                "#[Builder] can only be used on structs with named fields",
34            )
35            .to_compile_error()
36            .into();
37        },
38    };
39
40    // Get field names and types
41    // 获取字段名和类型
42    let field_names: Vec<_> = fields.iter().filter_map(|f| f.ident.as_ref()).collect();
43
44    let field_types: Vec<_> = fields.iter().map(|f| &f.ty).collect();
45
46    // Detect Option<T> fields for optional handling
47    // 检测 Option<T> 字段以进行可选处理
48    let is_option: Vec<_> = field_types
49        .iter()
50        .map(|ty| {
51            if let syn::Type::Path(type_path) = ty
52            {
53                type_path
54                    .path
55                    .segments
56                    .last()
57                    .map_or(false, |seg| seg.ident == "Option")
58            }
59            else
60            {
61                false
62            }
63        })
64        .collect();
65
66    // Generate Builder struct: Option<T> fields stored directly, others wrapped in Option
67    let builder_fields: Vec<_> = field_names
68        .iter()
69        .zip(field_types.iter())
70        .zip(is_option.iter())
71        .map(|((name, ty), is_opt)| {
72            if *is_opt
73            {
74                quote! { #name: #ty }
75            }
76            else
77            {
78                quote! { #name: Option<#ty> }
79            }
80        })
81        .collect();
82
83    let builder_struct = quote! {
84        #[derive(Default, Debug)]
85        #[doc = concat!("Builder for `", stringify!(#struct_name), "`")]
86        pub struct #builder_ident #impl_generics #ty_generics #where_clause {
87            #(#builder_fields,)*
88        }
89    };
90
91    // Generate builder methods on original struct
92    let builder_method = quote! {
93        impl #impl_generics #struct_name #ty_generics #where_clause {
94            #[inline]
95            #[doc = "Creates a new builder for this struct.\n"]
96            #[doc = "为此结构体创建新的 builder。"]
97            pub fn builder() -> #builder_ident #ty_generics {
98                #builder_ident::default()
99            }
100        }
101    };
102
103    // Generate setter methods on Builder
104    let builder_setters = field_names
105        .iter()
106        .zip(field_types.iter())
107        .zip(is_option.iter())
108        .map(|((name, ty), is_opt)| {
109            if *is_opt
110            {
111                quote! {
112                    #[inline]
113                    pub fn #name(mut self, #name: #ty) -> Self {
114                        self.#name = #name;
115                        self
116                    }
117                }
118            }
119            else
120            {
121                quote! {
122                    #[inline]
123                    pub fn #name(mut self, #name: #ty) -> Self {
124                        self.#name = Some(#name);
125                        self
126                    }
127                }
128            }
129        });
130
131    // Generate build method: Option fields default to None, others are required
132    let build_assignments: Vec<_> = field_names
133        .iter()
134        .zip(is_option.iter())
135        .map(|(name, is_opt)| {
136            if *is_opt
137            {
138                quote! { #name: self.#name }
139            }
140            else
141            {
142                quote! {
143                    #name: self.#name
144                        .ok_or_else(|| concat!(stringify!(#name), " is required").to_string())?
145                }
146            }
147        })
148        .collect();
149
150    let build_method = quote! {
151        impl #impl_generics #builder_ident #ty_generics #where_clause {
152            #[inline]
153            #[doc = "Builds the struct.\n"]
154            #[doc = "构建结构体。"]
155            pub fn build(self) -> Result<#struct_name #ty_generics, String> {
156                Ok(#struct_name {
157                    #(#build_assignments,)*
158                })
159            }
160        }
161    };
162
163    // Combine all expansions
164    // 合并所有展开
165    let expanded = quote! {
166        #builder_struct
167
168        #builder_method
169
170        impl #impl_generics #builder_ident #ty_generics #where_clause {
171            #(#builder_setters)*
172        }
173
174        #build_method
175    };
176
177    TokenStream::from(expanded)
178}
