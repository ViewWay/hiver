§Example / 示例

ⓘ
use hiver_http::client::WebClient;

let client = WebClient::new("https://api.example.com")
    .timeout(std::time::Duration::from_secs(30))
    .header("Authorization", "Bearer token");

let user: User = client.get("/users/1")
    .send()
    .await?
    .json()
    .await?;
