17/// KV v1 secret backend / KV v1 密钥后端
18///
19/// KV v1 is the simplest secret backend — it stores secrets as key-value pairs
20/// without versioning.
21///
22/// KV v1 是最简单的密钥后端 — 它以键值对的形式存储密钥，不带版本控制。
23///
24/// # Example
25///
26/// ```rust,no_run,ignore
27/// use hiver_vault::VaultClient;
28///
29/// async fn example(client: &VaultClient) -> Result<(), Box<dyn std::error::Error>> {
30///     let kv = client.kv_v1("secret");
31///
32///     // Write a secret / 写入密钥
33///     kv.write("myapp/config", serde_json::json!({"username": "admin", "password": "secret"})).await?;
34///
35///     // Read a secret / 读取密钥
36///     let data = kv.read("myapp/config").await?;
37///     println!("{:?}", data);
38///
39///     Ok(())
40/// }
41/// ```
42#[derive(Debug)]
43pub struct KvV1<'a>
44{
45    client: &'a VaultClient,
46    mount: String,
47}
48
49impl<'a> KvV1<'a>
50{
51    /// Create a new KV v1 backend handle / 创建新的 KV v1 后端句柄
52    pub fn new(client: &'a VaultClient, mount: &str) -> Self
53    {
54        Self {
55            client,
56            mount: mount.to_string(),
57        }
58    }
59
60    /// Read a secret / 读取密钥
61    ///
62    /// Reads the secret at the given path within the KV v1 mount.
63    /// 读取 KV v1 挂载中指定路径的密钥。
64    pub async fn read(&self, path: &str) -> VaultResult<serde_json::Value>
65    {
66        let full_path = format!("{}/{}", self.mount, path);
67        let resp = self.client.get(&full_path).await?;
68        let body: serde_json::Value = resp.json().await?;
69
70        body.get("data")
71            .cloned()
72            .ok_or_else(|| VaultError::InvalidResponse("Missing data field".into()))
73    }
74
75    /// Write a secret / 写入密钥
76    ///
77    /// Writes the given data to the specified path within the KV v1 mount.
78    /// 将给定数据写入 KV v1 挂载中指定路径。
79    pub async fn write(&self, path: &str, data: serde_json::Value) -> VaultResult<()>
80    {
81        let full_path = format!("{}/{}", self.mount, path);
82        let body = serde_json::json!({ "data": data });
83        self.client.post(&full_path, &body).await?;
84        Ok(())
85    }
86
87    /// Delete a secret / 删除密钥
88    ///
89    /// Deletes the secret at the given path.
90    /// 删除指定路径的密钥。
91    pub async fn delete(&self, path: &str) -> VaultResult<()>
92    {
93        let full_path = format!("{}/{}", self.mount, path);
94        self.client.delete(&full_path).await?;
95        Ok(())
96    }
97
98    /// List secrets / 列出密钥
99    ///
100    /// Lists all secrets at the given path prefix.
101    /// 列出给定路径前缀下的所有密钥。
102    pub async fn list(&self, path: &str) -> VaultResult<Vec<String>>
103    {
104        let full_path = format!("{}/{}", self.mount, path);
105        crate::secret::list(self.client, &full_path).await
106    }
107}

113/// KV v2 metadata for a secret / KV v2 密钥的元数据
114#[derive(Debug, Clone, Serialize, Deserialize)]
115pub struct KvV2Metadata
116{
117    /// Time of creation / 创建时间
118    pub created_time: Option<String>,
119    /// Time of last update / 最后更新时间
120    #[serde(rename = "deletion_time")]
121    pub deletion_time: Option<String>,
122    /// Whether the secret is destroyed / 密钥是否已被销毁
123    pub destroyed: bool,
124    /// Current version number / 当前版本号
125    pub version: i64,
126}

128/// KV v2 secret read result / KV v2 密钥读取结果
129#[derive(Debug, Clone)]
130pub struct KvV2Secret
131{
132    /// The secret data / 密钥数据
133    pub data: serde_json::Value,
134    /// Metadata about the secret version / 密钥版本的元数据
135    pub metadata: KvV2Metadata,
136}

138/// KV v2 secret response from Vault API / Vault API 返回的 KV v2 密钥响应
139#[derive(Debug, Clone, Serialize, Deserialize)]
140struct KvV2Response

142    data: KvV2ResponseInner,
143}

145#[derive(Debug, Clone, Serialize, Deserialize)]
146struct KvV2ResponseInner
147{

159/// KV v2 secret backend / KV v2 密钥后端
160///
161/// KV v2 supports versioned secrets with metadata, soft delete, and undelete.
162/// Equivalent to Spring Vault's `KeyValueTemplate` for version 2.
163///
164/// KV v2 支持带元数据的版本化密钥、软删除和恢复删除。
165/// 等价于 Spring Vault 版本 2 的 `KeyValueTemplate`。
166///
167/// # Example
168///
169/// ```rust,no_run,ignore
170/// use hiver_vault::VaultClient;
171///
172/// async fn example(client: &VaultClient) -> Result<(), Box<dyn std::error::Error>> {
173///     let kv = client.kv_v2("secret");
174///
175///     // Write a secret / 写入密钥
176///     kv.write("myapp/config", serde_json::json!({"db_url": "postgres://..."})).await?;
177///
178///     // Read latest version / 读取最新版本
179///     let secret = kv.read("myapp/config").await?;
180///     println!("{:?}", secret.data);
181///
182///     // Read specific version / 读取指定版本
183///     let v1 = kv.read_version("myapp/config", 1).await?;
184///
185///     // List secrets / 列出密钥
186///     let keys = kv.list("myapp").await?;
187///
188///     Ok(())
189/// }
190/// ```
191#[derive(Debug)]
192pub struct KvV2<'a>
193{
194    client: &'a VaultClient,
195    mount: String,
196}
197
198impl<'a> KvV2<'a>
199{
200    /// Create a new KV v2 backend handle / 创建新的 KV v2 后端句柄
201    pub fn new(client: &'a VaultClient, mount: &str) -> Self
202    {
203        Self {
204            client,
205            mount: mount.to_string(),
206        }
207    }
208
209    /// Read the latest version of a secret / 读取密钥的最新版本
210    pub async fn read(&self, path: &str) -> VaultResult<KvV2Secret>
211    {
212        let full_path = format!("{}/data/{}", self.mount, path);
213        let resp = self.client.get(&full_path).await?;
214        let body: KvV2Response = resp.json().await?;
215        Ok(KvV2Secret {
216            data: body.data.data,
217            metadata: body.data.metadata,
218        })
219    }
220
221    /// Read a specific version of a secret / 读取密钥的指定版本
222    pub async fn read_version(&self, path: &str, version: i64) -> VaultResult<KvV2Secret>
223    {
224        let full_path = format!("{}/data/{}", self.mount, path);
225        let url = self.client.url(&full_path)?;
226        let url_with_version = format!("{}?version={}", url, version);
227        let url =
228            Url::parse(&url_with_version).map_err(|e| VaultError::InvalidAddress(e.to_string()))?;
229
230        let mut req = self.client.http_client().get(url);
231        req = add_auth_header(self.client, req)?;
232
233        let resp = req.send().await?;
234        if !resp.status().is_success()
235        {
236            let status = resp.status().as_u16();
237            let body = resp.text().await.unwrap_or_default();
238            return Err(VaultError::from_status(status, &body));
239        }
240
241        let body: KvV2Response = resp.json().await?;
242        Ok(KvV2Secret {
243            data: body.data.data,
244            metadata: body.data.metadata,
245        })
246    }
247
248    /// Write (or update) a secret / 写入（或更新）密钥
249    pub async fn write(&self, path: &str, data: serde_json::Value) -> VaultResult<KvV2Metadata>
250    {
251        let full_path = format!("{}/data/{}", self.mount, path);
252        let body = serde_json::json!({ "data": data });
253        let resp = self.client.post(&full_path, &body).await?;
254        let resp_body: serde_json::Value = resp.json().await?;
255
256        let metadata = resp_body
257            .get("data")
258            .and_then(|d| serde_json::from_value(d.clone()).ok())
259            .unwrap_or(KvV2Metadata {
260                created_time: None,
261                deletion_time: None,
262                destroyed: false,
263                version: 1,
264            });
265
266        Ok(metadata)
267    }
268
269    /// Patch (partial update) a secret / 补丁（部分更新）密钥
270    ///
271    /// Merges the provided data with the existing secret data.
272    /// 将提供的数据与现有密钥数据合并。
273    pub async fn patch(&self, path: &str, data: serde_json::Value) -> VaultResult<()>
274    {
275        let full_path = format!("{}/data/{}", self.mount, path);
276        let body = serde_json::json!({ "data": data });
277        self.client.post(&full_path, &body).await?;
278        Ok(())
279    }
280
281    /// Delete the latest version of a secret (soft delete) / 删除密钥的最新版本（软删除）
282    pub async fn delete(&self, path: &str) -> VaultResult<()>
283    {
284        let full_path = format!("{}/data/{}", self.mount, path);
285        self.client.delete(&full_path).await?;
286        Ok(())
287    }
288
289    /// Delete specific versions of a secret / 删除密钥的指定版本
290    pub async fn delete_versions(&self, path: &str, versions: &[i64]) -> VaultResult<()>
291    {
292        let full_path = format!("{}/delete/{}", self.mount, path);
293        let body = KvV2VersionRequest {
294            versions: versions.to_vec(),
295        };
296        self.client.post(&full_path, &body).await?;
297        Ok(())
298    }
299
300    /// Undelete specific versions of a secret / 恢复密钥的指定版本
301    pub async fn undelete_versions(&self, path: &str, versions: &[i64]) -> VaultResult<()>
302    {
303        let full_path = format!("{}/undelete/{}", self.mount, path);
304        let body = KvV2VersionRequest {
305            versions: versions.to_vec(),
306        };
307        self.client.post(&full_path, &body).await?;
308        Ok(())
309    }
310
311    /// Permanently destroy versions of a secret / 永久销毁密钥的指定版本
312    pub async fn destroy_versions(&self, path: &str, versions: &[i64]) -> VaultResult<()>
313    {
314        let full_path = format!("{}/destroy/{}", self.mount, path);
315        let body = KvV2VersionRequest {
316            versions: versions.to_vec(),
317        };
318        self.client.post(&full_path, &body).await?;
319        Ok(())
320    }
321
322    /// List secrets / 列出密钥
323    pub async fn list(&self, path: &str) -> VaultResult<Vec<String>>
324    {
325        let full_path = format!("{}/metadata/{}", self.mount, path);
326        crate::secret::list(self.client, &full_path).await
327    }
328
329    /// Read secret metadata / 读取密钥元数据
330    pub async fn read_metadata(&self, path: &str) -> VaultResult<KvV2Metadata>
331    {
332        let full_path = format!("{}/metadata/{}", self.mount, path);
333        let resp = self.client.get(&full_path).await?;
334        let body: serde_json::Value = resp.json().await?;
335
336        body.get("data")
337            .map(|d| {
338                // KV v2 metadata response has versions map, get max version
339                let max_version = d
340                    .get("max_version")
341                    .and_then(serde_json::Value::as_i64)
342                    .unwrap_or(0);
343                KvV2Metadata {
344                    created_time: d
345                        .get("created_time")
346                        .and_then(|v| v.as_str())
347                        .map(String::from),
348                    deletion_time: None,
349                    destroyed: false,
350                    version: max_version,
351                }
352            })
353            .ok_or_else(|| VaultError::InvalidResponse("Missing metadata".into()))
354    }
355
356    /// Delete all versions and metadata for a secret / 删除密钥的所有版本和元数据
357    pub async fn delete_metadata(&self, path: &str) -> VaultResult<()>
358    {
359        let full_path = format!("{}/metadata/{}", self.mount, path);
360        self.client.delete(&full_path).await?;
361        Ok(())
362    }
363}

365/// Helper to add auth header (to avoid needing client internals)
366/// 添加认证头的辅助函数
367fn add_auth_header(
368    client: &VaultClient,
369    mut req: reqwest::RequestBuilder,
370) -> VaultResult<reqwest::RequestBuilder>
371{
372    if let Some(token) = client.token()
373    {
374        req = req.header("Authorization", format!("Bearer {token}"));
375    }
376    Ok(req)
377}
