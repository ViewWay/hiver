1//! # Annotated Application Example (full `#[hiver_main]` path)
2//! # 注解驱动应用示例(完整 `#[hiver_main]` 路径)
3//!
4//! Demonstrates the Spring-Boot-style fully-annotated entry point, where the
5//! framework auto-wires everything:
6//! 1. `#[hiver_main]` boots Hiver's own runtime, loads config, runs auto-configurations, collects
7//!    routes via `inventory`, and binds + serves.
8//! 2. `#[controller]` registers the controller as a singleton bean.
9//! 3. `#[get]` contributes routes discovered at startup via `inventory`.
10//!
11//! 演示 Spring Boot 风格的全注解入口,框架自动接通一切:
12//! 1. `#[hiver_main]` 启动 Hiver 自有 runtime、加载配置、运行自动配置、经 `inventory`
13//!    收集路由,并绑定 + 提供服务。
14//! 2. `#[controller]` 将控制器注册为单例 bean。
15//! 3. `#[get]` 经 `inventory` 贡献启动时被发现的路由。
16//!
17//! ## Run / 运行
18//!
19//! ```bash
20//! cargo run -p hiver-examples --bin annotated_app_example
21//! curl http://127.0.0.1:8080/hello
22//! curl http://127.0.0.1:8080/greet/world
23//! ```
24//!
25//! Unlike `http_server_example` (which wires routes by hand via `.get()`),
26//! this one uses *only* annotations — no manual router construction.
27//! 与 `http_server_example`(手动经 `.get()` 接线路由)不同,本示例*仅*使用
28//! 注解 —— 无手动 router 构造。
29
30use hiver_http::Request;
31use hiver_macros::{controller, get, hiver_main};
32
33/// The application entry point. `#[hiver_main]` generates `MyApp::run()` which
34/// boots the runtime, runs auto-configurations, collects all `#[get]` routes,
35/// and serves on `server.port` (default 8080).
36/// 应用入口。`#[hiver_main]` 生成 `MyApp::run()`,启动 runtime、运行自动配置、
37/// 收集所有 `#[get]` 路由,并在 `server.port`(默认 8080)上服务。
38#[hiver_main]
39struct MyApp;
40
41/// A REST controller, registered as a singleton bean. It exists to show the
42/// `#[controller]` stereotype working end-to-end; the routes below are free
43/// functions discovered via `inventory`.
44/// REST 控制器,注册为单例 bean。其存在是为展示 `#[controller]` 构造型端到端
45/// 生效;下面的路由是经 `inventory` 发现的自由函数。
46#[derive(Default)]
47#[controller]
48struct HelloController;
49
50/// `GET /hello` — no-arg handler returning a static string. Registered via the
51/// `#[get]` macro + `inventory`; `#[hiver_main]`'s `RouterAutoConfiguration`
52/// collects it automatically.
53/// `GET /hello` —— 无参处理程序,返回静态字符串。经 `#[get]` 宏 + `inventory`
54/// 注册;`#[hiver_main]` 的 `RouterAutoConfiguration` 自动收集它。
55#[get("/hello")]
56async fn hello() -> &'static str
57{
58    "Hello from annotated Hiver!"
59}
60
61/// `GET /greet/{name}` — demonstrates a path-parameter extractor via the raw
62/// `Request` (identity extractor). Returns a static string to keep the handler
63/// signature aligned with the macro's `IntoResponse` path.
64/// `GET /greet/{name}` —— 经原始 `Request`(身份提取器)演示路径参数提取。
65/// 返回静态字符串,使处理程序签名与宏的 `IntoResponse` 路径一致。
66#[get("/greet/{name}")]
67async fn greet(_req: Request) -> &'static str
68{
69    "Hello, friend!"
70}
71
72/// Standard binary entry point — just delegate to the macro-generated runner.
73/// 标准二进制入口 —— 仅委托给宏生成的运行器。
74fn main() -> anyhow::Result<()>
75{
76    MyApp::run()
77}
