3//! Hiver Config - Configuration management module
4//! Hiver配置 - 配置管理模块
5//!
6//! # Equivalent to Spring Boot / 等价于 Spring Boot
7//!
8//! - `@ConfigurationProperties` - `PropertiesConfig`
9//! - `@Value` - Value extractor
10//! - `Environment` - Environment
11//! - `PropertySource` - `PropertySource`
12//! - `@Profile` - Profile
13//! - `ConfigFileApplicationListener` - `ConfigLoader`
14//!
15//! # Example / 示例
16//!
17//! ```rust,no_run,ignore
18//! use hiver_config::{Config, PropertiesConfig};
19//! use serde::Deserialize;
20//!
21//! #[derive(PropertiesConfig, Deserialize)]
22//! #[prefix = "app.datasource"]
23//! struct DataSourceConfig {
24//!     url: String,
25//!     username: String,
26//!     password: String,
27//! }
28//!
29//! #[derive(PropertiesConfig, Deserialize)]
30//! #[prefix = "server"]
31//! struct ServerConfig {
32//!     port: u16,
33//!     host: String,
34//! }
35//!
36//! let config = Config::load().unwrap();
37//! let server = config.get::<ServerConfig>().unwrap();
38//! ```
