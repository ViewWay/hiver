27// SubmitEntry/CompletionEntry are internal driver types — not re-exported
28// publicly to avoid leaking unsafe raw I/O structs to downstream crates.
29// SubmitEntry/CompletionEntry 是内部 driver 类型 —— 不公开重导出，
30// 避免向下游 crate 泄漏不安全的原始 I/O 结构。
31pub(crate) use queue::{CompletionEntry, SubmitEntry};

33/// Core driver trait for async I/O operations
34/// 异步I/O操作的核心driver trait
35///
36/// This trait abstracts different I/O polling mechanisms (io-uring, epoll, kqueue)
37/// providing a unified interface for the runtime.
38///
39/// 此trait抽象了不同的I/O轮询机制（io-uring、epoll、kqueue），
40/// 为运行时提供统一接口。
41// Driver is an internal trait — only implemented by iouring/epoll/kqueue
42// inside hiver-runtime. Not part of the public API.
43pub(crate) trait Driver: Send + Sync + AsRawFd
44{
45    /// Submit queued operations to the kernel
46    /// 将队列中的操作提交给内核
47    ///
48    /// Returns the number of operations submitted.
49    /// 返回已提交的操作数量。
50    fn submit(&self) -> std::io::Result<usize>;
51
52    /// Wait for completion events indefinitely
53    /// 无限等待完成事件
54    ///
55    /// Returns the number of completion events available.
56    /// 返回可用的完成事件数量。
57    fn wait(&self) -> std::io::Result<usize>;
58
59    /// Wait for completion events with a timeout
60    /// 带超时等待完成事件
61    ///
62    /// Returns `(events_count, timed_out)` where:
63    /// - `events_count`: number of completion events / 完成事件数量
64    /// - `timed_out`: true if timeout occurred, false if events arrived / 是否超时
65    fn wait_timeout(&self, duration: std::time::Duration) -> std::io::Result<(usize, bool)>;
66
67    /// Get a mutable reference to the next available submission entry
68    /// 获取下一个可用提交条目的可变引用
69    ///
70    /// Returns `None` if the submission queue is full.
71    /// 如果提交队列已满则返回 `None`。
72    fn get_submission(&self) -> Option<&mut SubmitEntry>;
73
74    /// Get a reference to the next available completion entry
75    /// 获取下一个可用完成条目的引用
76    ///
77    /// Returns `None` if the completion queue is empty.
78    /// 如果完成队列为空则返回 `None`。
79    fn get_completion(&self) -> Option<&CompletionEntry>;
80
81    /// Advance the completion queue cursor, consuming the current entry
82    /// 前进完成队列游标，消费当前条目
83    fn advance_completion(&self);
84
85    /// Register interest in a file descriptor
86    /// 注册对文件描述符的兴趣
87    ///
88    /// This tells the driver to monitor the FD for specific events.
89    /// 这告诉driver监控文件描述符的特定事件。
90    fn register(&self, fd: std::os::fd::RawFd, interest: Interest) -> std::io::Result<()>;
91
92    /// Deregister interest in a file descriptor
93    /// 取消对文件描述符的兴趣注册
94    fn deregister(&self, fd: std::os::fd::RawFd) -> std::io::Result<()>;
95
96    /// Modify the interest for a registered file descriptor
97    /// 修改已注册文件描述符的兴趣
98    fn modify(&self, fd: std::os::fd::RawFd, interest: Interest) -> std::io::Result<()>;
99
100    /// Get the capacity of the submission queue
101    /// 获取提交队列的容量
102    fn submission_capacity(&self) -> usize;
103
104    /// Get the capacity of the completion queue
105    /// 获取完成队列的容量
106    fn completion_capacity(&self) -> usize;
107
108    /// Check if the driver supports the specified operation
109    /// 检查driver是否支持指定操作
110    fn supports_operation(&self, opcode: u8) -> bool;
111}
