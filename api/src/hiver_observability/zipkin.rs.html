1//! Zipkin v2 span exporter
2//! Zipkin v2 span 导出器
3//!
4//! Exports trace spans to a Zipkin server via the HTTP v2 API.
5//! Equivalent to Spring Cloud Sleuth's Zipkin integration.
6//!
7//! 通过 HTTP v2 API 将追踪 span 导出到 Zipkin 服务器。等价于 Spring Cloud Sleuth 的 Zipkin 集成。
8//!
9//! # Example / 示例
10//!
11//! ```rust,no_run,ignore
12//! use hiver_observability::zipkin::{ZipkinExporter, ZipkinConfig};
13//!
14//! let config = ZipkinConfig::new("http://localhost:9411", "my-service");
15//! let exporter = ZipkinExporter::new(config);
16//!
17//! // Export spans
18//! exporter.export(&spans).await?;
19//! ```

31/// Configuration for Zipkin exporter
32/// Zipkin 导出器配置
33#[derive(Clone, Debug)]
34pub struct ZipkinConfig
35{
36    /// Zipkin server base URL (e.g. "http://localhost:9411")
37    /// Zipkin 服务器基础 URL
38    pub endpoint: String,
39    /// Service name reported to Zipkin
40    /// 报告给 Zipkin 的服务名称
41    pub service_name: String,
42    /// Connection timeout in milliseconds
43    /// 连接超时（毫秒）
44    pub timeout_ms: u64,
45}
46
47impl ZipkinConfig
48{
49    /// Create a new Zipkin config
50    /// 创建新的 Zipkin 配置
51    pub fn new(endpoint: impl Into<String>, service_name: impl Into<String>) -> Self
52    {
53        Self {
54            endpoint: endpoint.into(),
55            service_name: service_name.into(),
56            timeout_ms: 5000,
57        }
58    }
59
60    /// Set connection timeout
61    /// 设置连接超时
62    pub fn timeout(mut self, ms: u64) -> Self
63    {
64        self.timeout_ms = ms;
65        self
66    }
67
68    /// Full spans API URL
69    /// 完整的 spans API URL
70    pub fn spans_url(&self) -> String
71    {
72        format!("{}{}", self.endpoint.trim_end_matches('/'), ZIPKIN_API_PATH)
73    }
74}

76/// Error type for Zipkin export operations
77/// Zipkin 导出操作错误类型
78#[derive(Debug)]
79pub enum ZipkinError
80{
81    /// HTTP request failed
82    /// HTTP 请求失败
83    Http(String),
84    /// JSON serialization failed
85    /// JSON 序列化失败
86    Serialization(String),
87    /// Configuration error
88    /// 配置错误
89    Config(String),
90}
91
92impl fmt::Display for ZipkinError
93{
94    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
95    {
96        match self
97        {
98            Self::Http(msg) => write!(f, "Zipkin HTTP error: {}", msg),
99            Self::Serialization(msg) => write!(f, "Zipkin serialization error: {}", msg),
100            Self::Config(msg) => write!(f, "Zipkin config error: {}", msg),
101        }
102    }
103}

135/// Zipkin v2 span format
136/// Zipkin v2 span 格式
137#[derive(Debug, Clone, Serialize)]
138pub struct ZipkinSpan
139{
140    /// Trace ID (hex, 16 or 32 chars)
141    /// 追踪 ID（十六进制，16 或 32 字符）
142    #[serde(rename = "traceId")]
143    pub trace_id: String,
144    /// Span ID (hex, 16 chars)
145    /// Span ID（十六进制，16 字符）
146    pub id: String,
147    /// Parent span ID (optional)
148    /// 父 span ID（可选）
149    #[serde(rename = "parentId", skip_serializing_if = "Option::is_none")]
150    pub parent_id: Option<String>,
151    /// Operation name
152    /// 操作名称
153    pub name: String,
154    /// Span kind
155    /// Span 类型
156    #[serde(skip_serializing_if = "Option::is_none")]
157    pub kind: Option<String>,
158    /// Start timestamp in microseconds
159    /// 开始时间戳（微秒）
160    #[serde(skip_serializing_if = "Option::is_none")]
161    pub timestamp: Option<u64>,
162    /// Duration in microseconds
163    /// 持续时间（微秒）
164    #[serde(skip_serializing_if = "Option::is_none")]
165    pub duration: Option<u64>,
166    /// Local endpoint
167    /// 本地端点
168    #[serde(rename = "localEndpoint")]
169    pub local_endpoint: ZipkinEndpoint,
170    /// Tags
171    /// 标签
172    #[serde(skip_serializing_if = "Option::is_none")]
173    pub tags: Option<HashMap<String, String>>,
174    /// Annotations
175    /// 注解
176    #[serde(skip_serializing_if = "Option::is_none")]
177    pub annotations: Option<Vec<ZipkinAnnotation>>,
178}

180impl ZipkinSpan
181{
182    /// Convert a Hiver Span to a Zipkin Span
183    /// 将 Hiver Span 转换为 Zipkin Span
184    pub fn from_hiver(span: &Span, service_name: &str) -> Self
185    {
186        let ctx = span.context();
187        let kind = match span.kind()
188        {
189            SpanKind::Server => Some("SERVER".to_string()),
190            SpanKind::Client => Some("CLIENT".to_string()),
191            SpanKind::Producer => Some("PRODUCER".to_string()),
192            SpanKind::Consumer => Some("CONSUMER".to_string()),
193            SpanKind::Internal => None,
194        };
195
196        let timestamp = Some(span.start_time_ns() / 1000);
197        let duration = span.duration_ns().map(|d| d / 1000);
198
199        let mut tags: HashMap<String, String> = span
200            .attributes()
201            .iter()
202            .map(|(k, v)| (k.clone(), v.clone()))
203            .collect();
204
205        if let Some(status) = span.status()
206        {
207            match status
208            {
209                SpanStatus::Ok =>
210                {
211                    tags.insert("otel.status_code".to_string(), "OK".to_string());
212                },
213                SpanStatus::UnknownError =>
214                {
215                    tags.insert("otel.status_code".to_string(), "ERROR".to_string());
216                    tags.insert("error".to_string(), "true".to_string());
217                },
218                SpanStatus::Cancelled =>
219                {
220                    tags.insert("otel.status_code".to_string(), "ERROR".to_string());
221                    tags.insert("error".to_string(), "cancelled".to_string());
222                },
223            }
224        }
225
226        let annotations = if span.events().is_empty()
227        {
228            None
229        }
230        else
231        {
232            Some(
233                span.events()
234                    .iter()
235                    .map(|e| ZipkinAnnotation {
236                        timestamp: e.timestamp / 1000,
237                        value: e.name.clone(),
238                    })
239                    .collect(),
240            )
241        };
242
243        Self {
244            trace_id: ctx.trace_id.to_hex(),
245            id: ctx.span_id.to_hex(),
246            parent_id: ctx.parent_span_id.map(|p| p.to_hex()),
247            name: span.name().to_string(),
248            kind,
249            timestamp,
250            duration,
251            local_endpoint: ZipkinEndpoint {
252                service_name: service_name.to_string(),
253            },
254            tags: if tags.is_empty() { None } else { Some(tags) },
255            annotations,
256        }
257    }
258}

260/// Span reporter that collects spans for batch export
261/// Span 报告器，收集 span 用于批量导出
262pub trait SpanReporter: Send + Sync
263{
264    /// Report a span
265    /// 报告 span
266    fn report(&self, span: ZipkinSpan);
267    /// Flush pending spans
268    /// 刷新待处理 span
269    fn flush(&self);
270}

272/// In-memory span reporter for testing
273/// 内存 span 报告器，用于测试
274pub struct InMemoryReporter
275{
276    spans: std::sync::Mutex<Vec<ZipkinSpan>>,
277}
278
279impl InMemoryReporter
280{
281    /// Create a new in-memory reporter
282    /// 创建新的内存报告器
283    pub fn new() -> Self
284    {
285        Self {
286            spans: std::sync::Mutex::new(Vec::new()),
287        }
288    }
289
290    /// Get all reported spans
291    /// 获取所有已报告的 span
292    #[allow(clippy::expect_used)]
293    pub fn spans(&self) -> Vec<ZipkinSpan>
294    {
295        self.spans.lock().expect("not poisoned").clone()
296    }
297
298    /// Clear all reported spans
299    /// 清除所有已报告的 span
300    #[allow(clippy::expect_used)]
301    pub fn clear(&self)
302    {
303        self.spans.lock().expect("not poisoned").clear();
304    }
305}
306
307impl Default for InMemoryReporter
308{
309    fn default() -> Self
310    {
311        Self::new()
312    }
313}
314
315impl SpanReporter for InMemoryReporter
316{
317    #[allow(clippy::expect_used)]
318    fn report(&self, span: ZipkinSpan)
319    {
320        self.spans.lock().expect("not poisoned").push(span);
321    }
322
323    fn flush(&self) {}
324}

326/// Zipkin exporter that sends spans to a Zipkin server
327/// Zipkin 导出器，将 span 发送到 Zipkin 服务器
328pub struct ZipkinExporter
329{
330    config: ZipkinConfig,
331    reporter: Arc<dyn SpanReporter>,
332}
333
334impl ZipkinExporter
335{
336    /// Create a new Zipkin exporter
337    /// 创建新的 Zipkin 导出器
338    pub fn new(config: ZipkinConfig) -> Self
339    {
340        let reporter = Arc::new(InMemoryReporter::new());
341        Self { config, reporter }
342    }
343
344    /// Create exporter with a custom reporter
345    /// 使用自定义报告器创建导出器
346    pub fn with_reporter(config: ZipkinConfig, reporter: Arc<dyn SpanReporter>) -> Self
347    {
348        Self { config, reporter }
349    }
350
351    /// Get the exporter config
352    /// 获取导出器配置
353    pub fn config(&self) -> &ZipkinConfig
354    {
355        &self.config
356    }
357
358    /// Convert and export a batch of Hiver spans
359    /// 转换并导出一批 Hiver span
360    pub fn export(&self, spans: &[Span]) -> Result<()>
361    {
362        let zipkin_spans: Vec<ZipkinSpan> = spans
363            .iter()
364            .map(|s| ZipkinSpan::from_hiver(s, &self.config.service_name))
365            .collect();
366
367        for span in zipkin_spans
368        {
369            self.reporter.report(span);
370        }
371
372        Ok(())
373    }
374
375    /// Export a single span
376    /// 导出单个 span
377    pub fn export_one(&self, span: &Span) -> Result<()>
378    {
379        self.export(std::slice::from_ref(span))
380    }
381
382    /// Serialize spans to Zipkin v2 JSON
383    /// 将 span 序列化为 Zipkin v2 JSON
384    pub fn serialize_spans(spans: &[ZipkinSpan]) -> Result<String>
385    {
386        serde_json::to_string(spans).map_err(|e| ZipkinError::Serialization(e.to_string()))
387    }
388
389    /// Flush pending spans
390    /// 刷新待处理 span
391    pub fn flush(&self)
392    {
393        self.reporter.flush();
394    }
395}

397impl fmt::Debug for ZipkinExporter
398{
399    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
400    {
401        f.debug_struct("ZipkinExporter")
402            .field("config", &self.config)
403            .field("reporter", &"<SpanReporter>")
404            .finish()
405    }
