1//! Task module
2//! 任务模块
3//!
4//! Provides the task spawning and joining primitives. In the
5//! async-executor/async-io backend, `spawn` submits to the runtime's
6//! [`async_executor::Executor`] (reached via [`crate::runtime::Handle`]) and
7//! returns a [`JoinHandle`] that wraps the [`async_executor::Task`].
8//! Fire-and-forget semantics are supported via [`JoinHandle::Drop`], which
9//! detaches the underlying task so it keeps running when the handle is
10//! dropped without being awaited.
11//!
12//! 提供任务 spawn 与 join 原语。在 async-executor/async-io 后端,`spawn` 将任务
13//! 提交到 runtime 的 [`async_executor::Executor`]（经 [`crate::runtime::Handle`]
14//! 访问）,并返回包裹 [`async_executor::Task`] 的 [`JoinHandle`]。
15//! fire-and-forget 语义经由 [`JoinHandle::Drop`] 支持 —— 它 detach 底层任务,
16//! 使句柄在未被 await 即丢弃时任务仍继续运行。

95/// Error returned by [`JoinHandle::wait`] when a task fails to produce its
96/// output.
97/// [`JoinHandle::wait`] 在任务未能产出其输出时返回的错误。
98#[derive(Debug, Clone, PartialEq, Eq)]
99pub enum JoinError
100{
101    /// The task was cancelled (its handle was awaited after the task was
102    /// detached/already consumed).
103    /// 任务被取消（任务被 detach/已消费后才 await 其句柄）。
104    TaskCancelled,
105    /// The task panicked before completing.
106    /// 任务在完成前 panic。
107    TaskPanic,
108}

113    {
114        match self
115        {
116            Self::TaskCancelled => write!(f, "Task was cancelled"),
117            Self::TaskPanic => write!(f, "Task panicked"),
118        }
119    }
120}

124/// Join handle for a spawned task.
125/// 被 spawn 任务的 join 句柄。
126///
127/// Allows awaiting task completion and retrieving the result. Dropping the
128/// handle without awaiting **detaches** the task (it keeps running) rather than
129/// cancelling it — this is what enables fire-and-forget `spawn`.
130///
131/// 允许等待任务完成并获取结果。丢弃句柄而未 await 会 **detach** 任务
132/// （任务继续运行）而非取消它 —— 这正是支持 fire-and-forget `spawn` 的机制。
133pub struct JoinHandle<T>
134{
135    /// The async-executor task, when spawned inside a runtime OR via fallback.
136    /// 在 runtime 内 spawn 或经回退 spawn 时持有的 async-executor 任务。
137    task: Option<async_executor::Task<T>>,
138    /// For the fallback path (no runtime context): the ephemeral executor that
139    /// owns `task`. `wait()` drives it via `executor.run(...)`. `None` when the
140    /// task runs on a runtime's executor (which `block_on` already drives).
141    /// 回退路径（无 runtime 上下文）专用:拥有 `task` 的临时执行器。`wait()` 通过
142    /// `executor.run(...)` 驱动它。当任务运行在 runtime 的执行器上时为 `None`
143    /// （`block_on` 已在驱动它）。
144    fallback_executor: Option<&'static async_executor::Executor<'static>>,
145    /// Task id assigned at spawn time.
146    /// spawn 时分配的任务 id。
147    id: TaskId,
148}
149
150impl<T> JoinHandle<T>
151{
152    /// Get the task ID.
153    /// 获取任务 ID。
154    #[must_use]
155    pub fn id(&self) -> TaskId
156    {
157        self.id
158    }
159
160    /// Check if the task has finished (completed, cancelled, or panicked).
161    /// 检查任务是否已完成（成功完成、已取消或发生 panic）。
162    pub fn is_finished(&self) -> bool
163    {
164        self.task
165            .as_ref()
166            .is_some_and(async_executor::Task::is_finished)
167    }
168
169    /// Wait for the task to complete and retrieve its result.
170    /// 等待任务完成并获取其结果。
171    ///
172    /// If the task panicked, returns [`JoinError::TaskPanic`]. If the handle
173    /// was already consumed, returns [`JoinError::TaskCancelled`].
174    ///
175    /// 若任务 panic,返回 [`JoinError::TaskPanic`]。若句柄已被消费,
176    /// 返回 [`JoinError::TaskCancelled`]。
177    pub async fn wait(mut self) -> Result<T, JoinError>
178    {
179        let task = self.task.take().ok_or(JoinError::TaskCancelled)?;
180
181        // If this is a fallback task (no runtime driving an executor), we must
182        // drive its ephemeral executor ourselves here.
183        // 若为回退任务（无 runtime 驱动执行器）,必须在此自行驱动其临时执行器。
184        if let Some(executor) = self.fallback_executor
185        {
186            // `executor.run(task)` drives both the task and the executor on the
187            // current thread and returns the task's output. It borrows the
188            // executor, so it is not `UnwindSafe`; wrap in `AssertUnwindSafe`
189            // (safe: on unwind we just drop the executor and task — no partial
190            // state escapes) and `catch_unwind` turns a task panic into
191            // `JoinError::TaskPanic`.
192            // `executor.run(task)` 在当前线程同时驱动任务与执行器,返回任务输出。
193            // 它借用执行器,故非 `UnwindSafe`;用 `AssertUnwindSafe` 包裹
194            //（安全:unwind 时直接丢弃执行器与任务,无部分状态逃逸）,
195            // `catch_unwind` 将任务 panic 转为 `JoinError::TaskPanic`。
196            use futures::FutureExt;
197            return match std::panic::AssertUnwindSafe(executor.run(task))
198                .catch_unwind()
199                .await
200            {
201                Ok(value) => Ok(value),
202                Err(_) => Err(JoinError::TaskPanic),
203            };
204        }
205
206        // In-runtime path: the runtime's `block_on` is already driving the
207        // executor, so we can simply await the task. `Task` is `Unpin`, so
208        // `catch_unwind` works directly.
209        // runtime 内路径:runtime 的 `block_on` 已在驱动执行器,故可直接 await 任务。
210        // `Task` 是 `Unpin`,故 `catch_unwind` 可直接使用。
211        use futures::FutureExt;
212        match task.catch_unwind().await
213        {
214            Ok(value) => Ok(value),
215            Err(_) => Err(JoinError::TaskPanic),
216        }
217    }
218}

220impl<T> Drop for JoinHandle<T>
221{
222    fn drop(&mut self)
223    {
224        // CRITICAL for fire-and-forget semantics: `async_executor::Task`
225        // CANCELS its task when dropped (per async-task docs: "tasks get
226        // canceled when dropped, use `.detach()` to run them in the
227        // background"). If the caller did not `wait()` on this handle (which
228        // takes the task), we must detach it so the spawned task keeps running
229        // to completion in the background.
230        // 对 fire-and-forget 语义至关重要:`async_executor::Task` 在 drop 时会
231        // 取消其任务（依 async-task 文档:"tasks get canceled when dropped, use
232        // .detach() to run them in the background"）。若调用方未对本句柄调用
233        // `wait()`（它会取走 task）,必须 detach,使 spawn 的任务在后台继续运行至完成。
234        if let Some(task) = self.task.take()
235        {
236            task.detach();
237        }
238    }
239}

241// ─── spawn ──────────────────────────────────────────────────────────────────

243/// Spawn a new async task on the current runtime.
244/// 在当前 runtime 上 spawn 新的异步任务。
245///
246/// If called within a `block_on` context, the task is submitted to the runtime's
247/// [`async_executor::Executor`] and runs concurrently with the main future.
248/// If called outside any runtime, the task runs on a fresh ephemeral executor
249/// driven by [`JoinHandle::wait`].
250///
251/// 若在 `block_on` 上下文内调用,任务提交到 runtime 的 [`async_executor::Executor`]
252/// 并与主 future 并发运行。若在任何 runtime 之外调用,任务运行于由
253/// [`JoinHandle::wait`] 驱动的全新临时执行器上。
254///
255/// # Panics / 恐慌
256///
257/// Panics if called outside of a runtime context **and** the ephemeral executor
258/// cannot be constructed (effectively never).
259/// 若在 runtime 上下文之外调用 **且** 临时执行器无法构造（实际不会发生）则恐慌。
260///
261/// # Example / 示例
262///
263/// ```rust,no_run,ignore
264/// use hiver_runtime::task::spawn;
265///
266/// async fn my_task() -> i32 {
267///     42
268/// }
269///
270/// async fn main() {
271///     let handle = spawn(my_task());
272///     let result = handle.wait().await.unwrap();
273///     assert_eq!(result, 42);
274/// }
275/// ```
276pub fn spawn<F, T>(future: F) -> JoinHandle<T>
277where
278    F: Future<Output = T> + Send + 'static,
279    T: Send + 'static,
280{
281    // Try to use the runtime's executor if a runtime context is available.
282    // 若运行时上下文可用,尝试使用 runtime 的执行器。
283    if let Some(handle) = crate::runtime::Handle::try_current()
284    {
285        if let Some(executor) = handle.executor()
286        {
287            let task = executor.spawn(future);
288            return JoinHandle {
289                task: Some(task),
290                fallback_executor: None,
291                id: gen_task_id(),
292            };
293        }
294    }
295
296    // Fallback: drive the future on a fresh ephemeral executor (no runtime
297    // context). The previous thread-per-task fallback busy-poll-slept; using a
298    // real executor here is both correct and efficient. `JoinHandle::wait`
299    // drives this ephemeral executor via `executor.run(task)`.
300    // 回退:在新建的临时执行器上驱动 future（无运行时上下文）。旧的每任务一线程
301    // 回退会忙轮询-休眠;此处使用真正的执行器既正确又高效。`JoinHandle::wait`
302    // 通过 `executor.run(task)` 驱动此临时执行器。
303    let executor: &'static async_executor::Executor<'static> =
304        Box::leak(Box::new(async_executor::Executor::new()));
305    let task = executor.spawn(future);
306
307    JoinHandle {
308        task: Some(task),
309        fallback_executor: Some(executor),
310        id: gen_task_id(),
311    }
312}

314// ─── spawn_blocking ─────────────────────────────────────────────────────────

316/// Run a blocking closure on a dedicated thread pool, returning a future that
317/// resolves to its result.
318/// 在专用线程池上运行阻塞闭包,返回一个解析为其结果的 future。
319///
320/// This is the Hiver equivalent of `tokio::task::spawn_blocking`. It offloads
321/// synchronous, potentially-blocking work (DB calls, file I/O, CPU-bound crypto)
322/// off the async reactor thread so it doesn't stall the event loop. The closure
323/// runs on the `blocking` crate's internal thread pool (part of the smol
324/// ecosystem, compatible with async-io).
325///
326/// 这是 Hiver 版的 `tokio::task::spawn_blocking`。它将同步、可能阻塞的工作
327///（DB 调用、文件 I/O、CPU 密集加密）从异步 reactor 线程卸载,使其不会卡住事件
328/// 循环。闭包运行在 `blocking` crate 的内部线程池上(smol 生态的一部分,
329/// 与 async-io 兼容）。
330///
331/// # Example / 示例
332///
333/// ```rust,no_run,ignore
334/// use hiver_runtime::task::spawn_blocking;
335///
336/// async fn handler() -> std::io::Result<String> {
337///     // This sync call runs off the async thread.
338///     let result = spawn_blocking(|| {
339///         std::fs::read_to_string("/etc/hostname").unwrap_or_default()
340///     }).await;
341///     Ok(result)
342/// }
343/// ```
344pub fn spawn_blocking<F, R>(f: F) -> blocking::Task<R>
345where
346    F: FnOnce() -> R + Send + 'static,
347    R: Send + 'static,
348{
349    blocking::unblock(f)
350}

352// ─── standalone block_on (thread + mpsc) ────────────────────────────────────

354/// No-op raw waker vtable used by the standalone [`block_on`].
355/// 独立 [`block_on`] 使用的无操作 raw waker vtable。
356const NOOP_RAW_WAKER_VTABLE: RawWakerVTable = RawWakerVTable::new(
357    |_| RawWaker::new(ptr::null(), &NOOP_RAW_WAKER_VTABLE), // clone
358    |_| {},                                                 // drop
359    |_| {},                                                 // wake
360    |_| {},                                                 // wake_by_ref
361);
362
363/// Block on a future to completion.
364/// 阻塞等待 future 完成。
365///
366/// This is the standalone (non-runtime) entry point used by code that is not
367/// already inside a `Runtime::block_on` (e.g. `hiver_runtime::task::block_on`).
368/// It spawns a dedicated thread that polls the future in a tight loop and sends
369/// the result back over a channel. For runtime-driven execution use
370/// [`crate::Runtime::block_on`] instead.
371///
372/// 这是独立（非 runtime）入口点,供不在 `Runtime::block_on` 内的代码使用
373/// （如 `hiver_runtime::task::block_on`）。它 spawn 一个专用线程,在紧密循环中
374/// 轮询 future 并经通道回传结果。runtime 驱动的执行请改用
375/// [`crate::Runtime::block_on`]。
376///
377/// # Example / 示例
378///
379/// ```rust,no_run,ignore
380/// use hiver_runtime::task::block_on;
381///
382/// block_on(async {
383///     println!("Hello from async!");
384/// });
385/// ```
386pub fn block_on<F, T>(future: F) -> T
387where
388    F: Future<Output = T> + Send + 'static,
389    T: Send + 'static,
390{
391    // Channel to communicate the result.
392    // 通道用于通信结果。
393    let (sender, receiver) = mpsc::channel();
394
395    // Create a no-op waker (we poll in a tight loop).
396    // 创建一个无操作的 waker（我们在紧密循环中轮询）。
397    let waker = unsafe { Waker::from_raw(RawWaker::new(ptr::null(), &NOOP_RAW_WAKER_VTABLE)) };
398
399    // Spawn a thread to run the future.
400    // 生成一个线程来运行 future。
401    thread::spawn(move || {
402        let mut future = Box::pin(future);
403        let mut cx = Context::from_waker(&waker);
404
405        // Poll until complete.
406        // 轮询直到完成。
407        loop
408        {
409            match Pin::as_mut(&mut future).poll(&mut cx)
410            {
411                Poll::Ready(result) =>
412                {
413                    // Send result (ignore send errors - receiver may be dropped).
414                    // 发送结果（忽略发送错误 - 接收器可能已被删除）。
415                    let _ = sender.send(result);
416                    break;
417                },
418                Poll::Pending =>
419                {
420                    // Yield to avoid busy-wait burning CPU.
421                    // A 1ms sleep is a reasonable trade-off between
422                    // responsiveness and CPU usage for a blocking executor.
423                    // 让出 CPU 避免忙等烧 CPU。
424                    // 1ms 休眠在响应性和 CPU 使用之间是合理的折衷。
425                    thread::sleep(std::time::Duration::from_millis(1));
426                },
427            }
428        }
429    });
430
431    // Block until result is ready.
432    // 阻塞直到结果就绪。
433    receiver
434        .recv()
435        .unwrap_or_else(|_| panic!("block_on: Failed to receive result from executor"))
436}
