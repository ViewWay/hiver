1//! Chat client with fluent API for building and sending AI requests.
2//! 用于构建和发送 AI 请求的流畅 API 聊天客户端。
3//!
4//! The `ChatClient` provides a high-level, builder-style interface for
5//! interacting with chat models. It simplifies common patterns like
6//! setting system prompts, adding user messages, and calling the model.
7//!
8//! `ChatClient` 提供高级的 builder 风格接口用于与聊天模型交互。
9//! 它简化了设置系统提示、添加用户消息和调用模型等常见模式。
10//!
11//! # Example / 示例
12//!
13//! ```rust,ignore
14//! use hiver_ai::chat_client::ChatClient;
15//!
16//! let response = client
17//!     .prompt("Translate to French: Hello")
18//!     .system("You are a translator.")
19//!     .call()
20//!     .await?;
21//! ```
22

31/// A high-level client for interacting with chat models.
32/// 用于与聊天模型交互的高级客户端。
33///
34/// Wraps a `ChatModel` and provides a fluent builder API for
35/// constructing requests and processing responses.
36///
37/// 包装 `ChatModel` 并提供流畅的 builder API 用于构造请求和处理响应。
38pub struct ChatClient
39{
40    /// The underlying chat model implementation.
41    /// 底层聊天模型实现。
42    model: Arc<dyn ChatModel>,
43    /// Default system prompt applied to all requests.
44    /// 应用于所有请求的默认系统提示。
45    default_system_prompt: Option<String>,
46    /// Default temperature for requests.
47    /// 请求的默认温度。
48    default_temperature: Option<f64>,
49    /// Default model identifier.
50    /// 默认模型标识符。
51    default_model: Option<String>,
52}
53
54impl ChatClient
55{
56    /// Creates a new chat client wrapping the given model.
57    /// 创建包装给定模型的新聊天客户端。
58    #[must_use]
59    pub fn new(model: impl ChatModel + 'static) -> Self
60    {
61        Self {
62            model: Arc::new(model),
63            default_system_prompt: None,
64            default_temperature: None,
65            default_model: None,
66        }
67    }
68
69    /// Returns a builder for configuring the chat client.
70    /// 返回用于配置聊天客户端的 builder。
71    pub fn builder(model: impl ChatModel + 'static) -> ChatClientBuilder
72    {
73        ChatClientBuilder::new(model)
74    }
75
76    /// Creates a new request builder starting with the given user prompt.
77    /// 使用给定的用户提示创建新的请求 builder。
78    pub fn prompt(&self, content: impl Into<String>) -> ChatClientRequest<'_>
79    {
80        ChatClientRequest::new(self).user(content)
81    }
82
83    /// Creates a new request builder starting with a system prompt.
84    /// 使用系统提示创建新的请求 builder。
85    pub fn system(&self, content: impl Into<String>) -> ChatClientRequest<'_>
86    {
87        ChatClientRequest::new(self).system(content)
88    }
89
90    /// Creates a new request builder starting with a user message.
91    /// 使用用户消息创建新的请求 builder。
92    pub fn user(&self, content: impl Into<String>) -> ChatClientRequest<'_>
93    {
94        ChatClientRequest::new(self).user(content)
95    }
96
97    /// Sends a simple prompt and returns the response content as a string.
98    /// 发送简单提示并返回响应内容字符串。
99    ///
100    /// This is a convenience method for single-turn interactions.
101    /// 这是用于单轮交互的便捷方法。
102    pub async fn call(&self, prompt: impl Into<String>) -> Result<String, ModelError>
103    {
104        let response = self.prompt(prompt).call().await?;
105        Ok(response.content)
106    }
107
108    /// Returns a reference to the underlying model.
109    /// 返回底层模型的引用。
110    #[must_use]
111    pub fn model(&self) -> &dyn ChatModel
112    {
113        self.model.as_ref()
114    }
115}

190/// A fluent request builder for constructing chat requests.
191/// 用于构造聊天请求的流畅请求 builder。
192///
193/// Created from `ChatClient::prompt()`, `ChatClient::system()`, or
194/// `ChatClient::user()`, this builder collects messages and options
195/// before sending the request.
196///
197/// 从 `ChatClient::prompt()`、`ChatClient::system()` 或 `ChatClient::user()` 创建，
198/// 此 builder 在发送请求之前收集消息和选项。
199pub struct ChatClientRequest<'a>
200{
201    client: &'a ChatClient,
202    messages: Vec<ChatMessage>,
203    temperature: Option<f64>,
204    max_tokens: Option<u32>,
205    model: Option<String>,
206}
207
208impl<'a> ChatClientRequest<'a>
209{
210    /// Creates a new request builder for the given client.
211    /// 为给定客户端创建新的请求 builder。
212    fn new(client: &'a ChatClient) -> Self
213    {
214        Self {
215            client,
216            messages: Vec::new(),
217            temperature: None,
218            max_tokens: None,
219            model: None,
220        }
221    }
222
223    /// Adds a system message to the request.
224    /// 向请求添加系统消息。

230    }

233    /// 向请求添加用户消息。

238        self
239    }

242    /// 向请求添加助手消息。

295    /// Builds and sends the request, returning the full response.
296    /// 构建并发送请求，返回完整响应。
297    pub async fn call(self) -> Result<ChatResponse, ModelError>
298    {
299        let request = self.build_request();
300        self.client.model.complete(request).await
301    }
302
303    /// Builds and sends the request, returning only the response content.
304    /// 构建并发送请求，仅返回响应内容。
305    pub async fn content(self) -> Result<String, ModelError>
306    {
307        let response = self.call().await?;
308        Ok(response.content)
309    }
310
311    /// Builds and sends the request as a streaming response.
312    /// 构建并发送请求作为流式响应。
313    pub async fn stream(self) -> Result<ChatStream, ModelError>
314    {
315        let request = self.build_request();
316        self.client.model.stream(request).await
317    }
318
319    /// Collects all streaming chunks into a single string.
320    /// 将所有流式块收集为单个字符串。
321    pub async fn stream_content(self) -> Result<String, ModelError>
322    {
323        let stream = self.stream().await?;
324        let mut result = String::new();
325        let mut stream = std::pin::pin!(stream);
326
327        while let Some(chunk) = stream.next().await
328        {
329            let chunk = chunk?;
330            result.push_str(&chunk.content);
331        }
332
333        Ok(result)
334    }
335
336    /// Builds the internal `ChatRequest` from this builder's state.
337    /// 从此 builder 的状态构建内部 `ChatRequest`。
338    fn build_request(&self) -> ChatRequest
339    {
340        let mut messages = Vec::new();
341
342        // Add default system prompt if configured and no system message is present
343        // 如果已配置且不存在系统消息，则添加默认系统提示
344        if let Some(ref sys_prompt) = self.client.default_system_prompt
345        {
346            let has_system = self.messages.iter().any(|m| m.role == Role::System);
347            if !has_system
348            {
349                messages.push(ChatMessage::system(sys_prompt.clone()));
350            }
351        }
352
353        messages.extend(self.messages.clone());
354
355        let mut request = ChatRequest::new();
356        request.messages = messages;
357
358        // Apply overrides: request-level > client-level
359        // 应用覆盖：请求级 > 客户端级
360        request.temperature = self.temperature.or(self.client.default_temperature);
361        request.model = self
362            .model
363            .clone()
364            .or_else(|| self.client.default_model.clone());
365        request.max_tokens = self.max_tokens;
366
367        request
368    }
369}

371/// Collects a stream of chat chunks into a single string.
372/// 将聊天块流收集为单个字符串。
373pub async fn collect_stream(mut stream: ChatStream) -> Result<String, ModelError>
374{
375    let mut result = String::new();
376    while let Some(chunk) = stream.next().await
377    {
378        let chunk = chunk?;
379        result.push_str(&chunk.content);
380    }
381    Ok(result)
382}
