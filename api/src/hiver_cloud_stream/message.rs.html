1//! Stream message types — unified message format for all binders.
2//! 流消息类型 — 所有 Binder 的统一消息格式。
3
4use std::collections::HashMap;
5
6use serde::{Deserialize, Serialize};
7
8/// A unified stream message — equivalent to Spring Cloud Stream's `Message<T>`.
9/// 统一流消息 — 等价于 Spring Cloud Stream 的 `Message<T>`。
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct StreamMessage
12{
13    /// Message payload as raw bytes.
14    /// 消息载荷（原始字节）。
15    pub payload: Vec<u8>,
16
17    /// Message headers / metadata.
18    /// 消息头部/元数据。
19    pub headers: HashMap<String, String>,
20
21    /// Optional partition/routing key.
22    /// 可选的分区/路由键。
23    #[serde(skip_serializing_if = "Option::is_none")]
24    pub key: Option<String>,
25
26    /// Destination this message was received from or should be sent to.
27    /// 消息接收来源或发送目标。
28    #[serde(skip_serializing_if = "Option::is_none")]
29    pub destination: Option<String>,
30}
31
32impl StreamMessage
33{
34    /// Create a new message with raw bytes payload.
35    /// 创建带原始字节载荷的新消息。
36    pub fn new(payload: impl Into<Vec<u8>>) -> Self
37    {
38        Self {
39            payload: payload.into(),
40            headers: HashMap::new(),
41            key: None,
42            destination: None,
43        }
44    }
45
46    /// Create a message from a serializable payload.
47    /// 从可序列化载荷创建消息。
48    pub fn from_json<T: Serialize>(payload: &T) -> serde_json::Result<Self>
49    {
50        Ok(Self {
51            payload: serde_json::to_vec(payload)?,
52            headers: HashMap::new(),
53            key: None,
54            destination: None,
55        })
56    }
57
58    /// Deserialize the payload as a typed value.
59    /// 将载荷反序列化为类型化值。
60    pub fn as_json<T: for<'de> Deserialize<'de>>(&self) -> serde_json::Result<T>
61    {
62        serde_json::from_slice(&self.payload)
63    }
64
65    /// Get the payload as a UTF-8 string.
66    /// 获取载荷的 UTF-8 字符串。
67    pub fn as_str(&self) -> Option<&str>
68    {
69        std::str::from_utf8(&self.payload).ok()
70    }
71
72    /// Set a header.
73    /// 设置头部。
74    pub fn with_header(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
75    {
76        self.headers.insert(key.into(), value.into());
77        self
78    }
79
80    /// Set the routing key.
81    /// 设置路由键。
82    pub fn with_key(mut self, key: impl Into<String>) -> Self
83    {
84        self.key = Some(key.into());
85        self
86    }
87
88    /// Set the destination.
89    /// 设置目标。
90    pub fn with_destination(mut self, dest: impl Into<String>) -> Self
91    {
92        self.destination = Some(dest.into());
93        self
94    }
95
96    /// Get a header value.
97    /// 获取头部值。
98    pub fn header(&self, key: &str) -> Option<&str>
99    {
100        self.headers.get(key).map(String::as_str)
101    }
102
103    /// Payload length in bytes.
104    /// 载荷字节长度。
105    pub fn len(&self) -> usize
106    {
107        self.payload.len()
108    }
109
110    /// Whether the payload is empty.
111    /// 载荷是否为空。
112    pub fn is_empty(&self) -> bool
113    {
114        self.payload.is_empty()
115    }
116}

118#[cfg(test)]
119mod tests
120{
121    use super::*;
122
123    #[test]
124    fn test_new_message()
125    {
126        let msg = StreamMessage::new(b"hello".to_vec());
127        assert_eq!(msg.as_str(), Some("hello"));
128        assert!(msg.key.is_none());
129    }
130
131    #[test]
132    fn test_json_payload()
133    {
134        #[derive(Serialize, Deserialize, Debug, PartialEq)]
135        struct User
136        {
137            name: String,
138            age: u32,
139        }
140
141        let user = User {
142            name: "alice".to_string(),
143            age: 30,
144        };
145        let msg = StreamMessage::from_json(&user).unwrap();
146        let decoded: User = msg.as_json().unwrap();
147        assert_eq!(decoded, user);
148    }
149
150    #[test]
151    fn test_headers()
152    {
153        let msg = StreamMessage::new(b"data".to_vec())
154            .with_header("content-type", "application/json")
155            .with_key("user-123");
156        assert_eq!(msg.header("content-type"), Some("application/json"));
157        assert_eq!(msg.key.as_deref(), Some("user-123"));
158    }
159
160    #[test]
161    fn test_empty_payload()
162    {
163        let msg = StreamMessage::new(Vec::<u8>::new());
164        assert!(msg.is_empty());
165    }
166
167    #[test]
168    fn test_serialization_roundtrip()
169    {
170        let msg = StreamMessage::new(b"payload".to_vec())
171            .with_header("key", "value")
172            .with_key("route-1");
173        let json = serde_json::to_string(&msg).unwrap();
174        let decoded: StreamMessage = serde_json::from_str(&json).unwrap();
175        assert_eq!(decoded.payload, b"payload".to_vec());
176        assert_eq!(decoded.header("key"), Some("value"));
177    }
178}
