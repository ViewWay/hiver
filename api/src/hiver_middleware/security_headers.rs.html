1//! Security headers middleware
2//! 安全响应头中间件
3//!
4//! # Equivalent to Spring Boot / 等价于 Spring Boot
5//!
6//! - Spring Security's `HeadersConfigurer`
7//! - `Content-Security-Policy`, `X-Frame-Options`, `X-Content-Type-Options`
8//! - `Strict-Transport-Security`, `Referrer-Policy`, `Permissions-Policy`
9//!
10//! # Example / 示例
11//!
12//! ```rust,no_run,ignore
13//! use hiver_middleware::SecurityHeadersConfig;
14//! use std::sync::Arc;
15//!
16//! let config = SecurityHeadersConfig::new()
17//!     .content_security_policy("default-src 'self'")
18//!     .frame_options("DENY")
19//!     .hsts_max_age(31536000)
20//!     .referrer_policy("strict-origin-when-cross-origin");
21//! ```

31/// Security headers configuration
32/// 安全响应头配置
33///
34/// Equivalent to Spring Security's `HeadersConfigurer<HttpSecurity>`.
35/// 等价于 Spring Security 的 `HeadersConfigurer<HttpSecurity>`。
36#[derive(Debug, Clone)]
37pub struct SecurityHeadersConfig
38{
39    /// Content-Security-Policy header value
40    /// Content-Security-Policy 头值
41    pub content_security_policy: Option<String>,
42
43    /// X-Frame-Options header value: "DENY", "SAMEORIGIN", or "ALLOW-FROM uri"
44    /// X-Frame-Options 头值
45    pub frame_options: Option<String>,
46
47    /// X-Content-Type-Options header value (typically "nosniff")
48    /// X-Content-Type-Options 头值
49    pub content_type_options: Option<String>,
50
51    /// Strict-Transport-Security max-age in seconds
52    /// Strict-Transport-Security max-age（秒）
53    pub hsts_max_age: Option<u64>,
54
55    /// Whether HSTS includes subdomains
56    /// HSTS 是否包含子域名
57    pub hsts_include_subdomains: bool,
58
59    /// Whether HSTS includes preload
60    /// HSTS 是否包含 preload
61    pub hsts_preload: bool,
62
63    /// Referrer-Policy header value
64    /// Referrer-Policy 头值
65    pub referrer_policy: Option<String>,
66
67    /// Permissions-Policy header value
68    /// Permissions-Policy 头值
69    pub permissions_policy: Option<String>,
70
71    /// X-XSS-Protection header value (deprecated but still useful for legacy browsers)
72    /// X-XSS-Protection 头值（已弃用，但对旧浏览器仍有用）
73    pub xss_protection: Option<String>,
74
75    /// Cache-Control header for security (disables caching)
76    /// 安全相关的 Cache-Control 头（禁用缓存）
77    pub no_cache: bool,
78}
79
80impl SecurityHeadersConfig
81{
82    /// Create a new config with sensible defaults.
83    /// 创建使用合理默认值的新配置。
84    ///
85    /// Defaults:
86    /// - X-Content-Type-Options: nosniff
87    /// - X-Frame-Options: DENY
88    /// - Referrer-Policy: strict-origin-when-cross-origin
89    pub fn new() -> Self
90    {
91        Self {
92            content_security_policy: None,
93            frame_options: Some("DENY".to_string()),
94            content_type_options: Some("nosniff".to_string()),
95            hsts_max_age: None,
96            hsts_include_subdomains: false,
97            hsts_preload: false,
98            referrer_policy: Some("strict-origin-when-cross-origin".to_string()),
99            permissions_policy: None,
100            xss_protection: Some("0".to_string()),
101            no_cache: false,
102        }
103    }
104
105    /// Set Content-Security-Policy.
106    /// 设置 Content-Security-Policy。
107    pub fn content_security_policy(mut self, policy: impl Into<String>) -> Self
108    {
109        self.content_security_policy = Some(policy.into());
110        self
111    }
112
113    /// Set X-Frame-Options.
114    /// 设置 X-Frame-Options。
115    pub fn frame_options(mut self, value: impl Into<String>) -> Self
116    {
117        self.frame_options = Some(value.into());
118        self
119    }
120
121    /// Set X-Content-Type-Options (default: "nosniff").
122    /// 设置 X-Content-Type-Options。
123    pub fn content_type_options(mut self, value: impl Into<String>) -> Self
124    {
125        self.content_type_options = Some(value.into());
126        self
127    }
128
129    /// Enable Strict-Transport-Security with the given max-age.
130    /// 启用 Strict-Transport-Security 并设置 max-age。
131    pub fn hsts_max_age(mut self, max_age: u64) -> Self
132    {
133        self.hsts_max_age = Some(max_age);
134        self
135    }
136
137    /// Include subdomains in HSTS.
138    /// HSTS 包含子域名。
139    pub fn hsts_include_subdomains(mut self) -> Self
140    {
141        self.hsts_include_subdomains = true;
142        self
143    }
144
145    /// Include preload in HSTS.
146    /// HSTS 包含 preload。
147    pub fn hsts_preload(mut self) -> Self
148    {
149        self.hsts_preload = true;
150        self
151    }
152
153    /// Set Referrer-Policy.
154    /// 设置 Referrer-Policy。
155    pub fn referrer_policy(mut self, policy: impl Into<String>) -> Self
156    {
157        self.referrer_policy = Some(policy.into());
158        self
159    }
160
161    /// Set Permissions-Policy.
162    /// 设置 Permissions-Policy。
163    pub fn permissions_policy(mut self, policy: impl Into<String>) -> Self
164    {
165        self.permissions_policy = Some(policy.into());
166        self
167    }
168
169    /// Set X-XSS-Protection.
170    /// 设置 X-XSS-Protection。
171    pub fn xss_protection(mut self, value: impl Into<String>) -> Self
172    {
173        self.xss_protection = Some(value.into());
174        self
175    }
176
177    /// Disable caching for responses.
178    /// 禁用响应缓存。
179    pub fn no_cache(mut self) -> Self
180    {
181        self.no_cache = true;
182        self
183    }
184
185    /// Disable all headers (empty config).
186    /// 禁用所有头（空配置）。
187    pub fn disabled() -> Self
188    {
189        Self {
190            content_security_policy: None,
191            frame_options: None,
192            content_type_options: None,
193            hsts_max_age: None,
194            hsts_include_subdomains: false,
195            hsts_preload: false,
196            referrer_policy: None,
197            permissions_policy: None,
198            xss_protection: None,
199            no_cache: false,
200        }
201    }
202
203    /// Build HSTS header value from config.
204    /// 从配置构建 HSTS 头值。
205    fn hsts_value(&self) -> Option<String>
206    {
207        self.hsts_max_age.map(|max_age| {
208            let mut val = format!("max-age={}", max_age);
209            if self.hsts_include_subdomains
210            {
211                val.push_str("; includeSubDomains");
212            }
213            if self.hsts_preload
214            {
215                val.push_str("; preload");
216            }
217            val
218        })
219    }
220}

226        Self::new()
227    }

230/// Middleware that adds security headers to HTTP responses.
231/// 向 HTTP 响应添加安全头的中间件。
232///
233/// Equivalent to Spring Security's `HeaderWriterFilter`.
234/// 等价于 Spring Security 的 `HeaderWriterFilter`。
235#[derive(Debug, Clone)]
236pub struct SecurityHeadersMiddleware
237{
238    config: SecurityHeadersConfig,
239}
240
241impl SecurityHeadersMiddleware
242{
243    /// Create a new security headers middleware with default config.
244    /// 使用默认配置创建安全头中间件。
245    pub fn new() -> Self
246    {
247        Self {
248            config: SecurityHeadersConfig::new(),
249        }
250    }
251
252    /// Create with custom configuration.
253    /// 使用自定义配置创建。
254    pub fn with_config(config: SecurityHeadersConfig) -> Self
255    {
256        Self { config }
257    }
258
259    /// Returns a reference to the configuration.
260    /// 返回配置引用。
261    pub fn config(&self) -> &SecurityHeadersConfig
262    {
263        &self.config
264    }
265}
266
267impl Default for SecurityHeadersMiddleware
268{
269    fn default() -> Self
270    {
271        Self::new()
272    }
273}

275impl Middleware<()> for SecurityHeadersMiddleware
276{
277    fn call(
278        &self,
279        req: Request,
280        state: Arc<()>,
281        next: Next<()>,
282    ) -> Pin<Box<dyn Future<Output = Result<Response>> + Send>>
283    {
284        let config = self.config.clone();
285        Box::pin(async move {
286            let mut response = next.call(req, state).await?;
287
288            if let Some(csp) = &config.content_security_policy
289            {
290                response.insert_header("content-security-policy", csp.as_str());
291            }
292            if let Some(fo) = &config.frame_options
293            {
294                response.insert_header("x-frame-options", fo.as_str());
295            }
296            if let Some(cto) = &config.content_type_options
297            {
298                response.insert_header("x-content-type-options", cto.as_str());
299            }
300            if let Some(hsts) = &config.hsts_value()
301            {
302                response.insert_header("strict-transport-security", hsts.as_str());
303            }
304            if let Some(rp) = &config.referrer_policy
305            {
306                response.insert_header("referrer-policy", rp.as_str());
307            }
308            if let Some(pp) = &config.permissions_policy
309            {
310                response.insert_header("permissions-policy", pp.as_str());
311            }
312            if let Some(xss) = &config.xss_protection
313            {
314                response.insert_header("x-xss-protection", xss.as_str());
315            }
316            if config.no_cache
317            {
318                response.insert_header("cache-control", "no-store, no-cache, must-revalidate");
319                response.insert_header("pragma", "no-cache");
320            }
321
322            Ok(response)
323        })
324    }
325}

339    #[test]
340    fn test_default_config()
341    {
342        let config = SecurityHeadersConfig::new();
343        assert_eq!(config.frame_options.as_deref(), Some("DENY"));
344        assert_eq!(config.content_type_options.as_deref(), Some("nosniff"));
345        assert_eq!(config.referrer_policy.as_deref(), Some("strict-origin-when-cross-origin"));
346        assert_eq!(config.xss_protection.as_deref(), Some("0"));
347        assert!(config.content_security_policy.is_none());
348        assert!(config.hsts_max_age.is_none());
349    }

351    #[test]
352    fn test_builder_pattern()
353    {
354        let config = SecurityHeadersConfig::new()
355            .content_security_policy("default-src 'self'")
356            .frame_options("SAMEORIGIN")
357            .hsts_max_age(31536000)
358            .hsts_include_subdomains()
359            .hsts_preload()
360            .referrer_policy("no-referrer")
361            .permissions_policy("camera=(), microphone=()")
362            .no_cache();
363
364        assert_eq!(config.content_security_policy.as_deref(), Some("default-src 'self'"));
365        assert_eq!(config.frame_options.as_deref(), Some("SAMEORIGIN"));
366        assert_eq!(config.hsts_max_age, Some(31536000));
367        assert!(config.hsts_include_subdomains);
368        assert!(config.hsts_preload);
369        assert!(config.no_cache);
370    }
