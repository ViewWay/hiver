33/// JWT claims
34/// JWT 声明
35///
36/// Contains all standard JWT claims (RFC 7519) plus application-specific fields.
37/// 包含所有标准JWT声明（RFC 7519）加应用特定字段。
38///
39/// # Standard Claims / 标准声明
40///
41/// - `iss` (Issuer) / 签发者
42/// - `sub` (Subject) / 主题
43/// - `aud` (Audience) / 受众
44/// - `exp` (Expiration) / 过期时间
45/// - `nbf` (Not Before) / 生效时间
46/// - `iat` (Issued At) / 签发时间
47/// - `jti` (JWT ID) / JWT标识符
48///
49/// # Spring Equivalent / Spring等价物
50///
51/// ```java
52/// public class JwtClaims {
53///     private String iss;      // Issuer
54///     private String sub;      // Subject (user ID)
55///     private String aud;      // Audience
56///     private long exp;        // Expiration
57///     private long nbf;        // Not Before
58///     private long iat;        // Issued at
59///     private String jti;      // JWT ID
60///     private String username; // Username
61///     private List<String> authorities; // Roles/permissions
62/// }
63/// ```
64#[derive(Debug, Clone, Serialize, Deserialize)]
65pub struct JwtClaims
66{
67    /// Subject (user ID)
68    /// 主体（用户ID）
69    pub sub: String,
70
71    /// Username
72    /// 用户名
73    pub username: String,
74
75    /// Authorities/roles
76    /// 权限/角色
77    pub authorities: Vec<String>,
78
79    /// Issued at (seconds since epoch)
80    /// 签发时间（自纪元以来的秒数）
81    pub iat: i64,
82
83    /// Expiration (seconds since epoch)
84    /// 过期时间（自纪元以来的秒数）
85    pub exp: i64,
86
87    /// Issuer
88    /// 签发者
89    #[serde(skip_serializing_if = "Option::is_none")]
90    pub iss: Option<String>,
91
92    /// Audience (recipient(s) the JWT is intended for)
93    /// 受众（JWT的预期接收者）
94    #[serde(skip_serializing_if = "Option::is_none")]
95    pub aud: Option<serde_json::Value>,
96
97    /// Not Before (seconds since epoch; token is not valid before this time)
98    /// 生效时间（自纪元以来的秒数；此时间之前令牌无效）
99    #[serde(skip_serializing_if = "Option::is_none")]
100    pub nbf: Option<i64>,
101
102    /// JWT ID (unique identifier for the token)
103    /// JWT标识符（令牌的唯一标识符）
104    #[serde(skip_serializing_if = "Option::is_none")]
105    pub jti: Option<String>,
106
107    /// Custom claims (application-specific key-value pairs)
108    /// 自定义声明（应用特定的键值对）
109    #[serde(flatten)]
110    pub custom: HashMap<String, serde_json::Value>,
111}

347/// Supported JWT signing algorithms
348/// 支持的JWT签名算法
349#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
350pub enum JwtAlgorithm
351{
352    /// HMAC using SHA-256
353    #[default]
354    Hs256,
355    /// HMAC using SHA-384
356    Hs384,
357    /// HMAC using SHA-512
358    Hs512,
359    /// RSASSA-PKCS1-v1_5 using SHA-256
360    Rs256,
361}
362
363impl JwtAlgorithm
364{
365    /// Convert to jsonwebtoken Algorithm
366    /// 转换为jsonwebtoken库的Algorithm
367    pub fn to_algorithm(&self) -> jsonwebtoken::Algorithm
368    {
369        match self
370        {
371            JwtAlgorithm::Hs256 => jsonwebtoken::Algorithm::HS256,
372            JwtAlgorithm::Hs384 => jsonwebtoken::Algorithm::HS384,
373            JwtAlgorithm::Hs512 => jsonwebtoken::Algorithm::HS512,
374            JwtAlgorithm::Rs256 => jsonwebtoken::Algorithm::RS256,
375        }
376    }

463    /// Create JWT token with custom expiration
464    /// 创建带自定义过期时间的JWT token
465    ///
466    /// # Arguments / 参数
467    ///
468    /// * `user_id` - User ID / 用户ID
469    /// * `username` - Username / 用户名
470    /// * `authorities` - User authorities / 用户权限
471    /// * `expiration_hours` - Token expiration in hours / token过期时间（小时）
472    pub fn create_token_with_expiration(
473        user_id: impl Into<String>,
474        username: impl Into<String>,
475        authorities: &[Authority],
476        expiration_hours: i64,
477    ) -> SecurityResult<String>
478    {
479        let claims = JwtClaims::new(user_id, username, authorities, expiration_hours);
480
481        let secret = Self::get_secret();
482        let encoding_key = EncodingKey::from_secret(secret.as_ref());
483
484        encode(&Header::default(), &claims, &encoding_key)
485            .map_err(|e| SecurityError::TokenError(format!("Failed to encode token: {}", e)))
486    }

488    /// Verify and parse JWT token
489    /// 验证并解析JWT token
490    ///
491    /// # Arguments / 参数
492    ///
493    /// * `token` - JWT token string / JWT token字符串
494    ///
495    /// # Returns / 返回
496    ///
497    /// Parsed JWT claims / 解析后的JWT声明
498    ///
499    /// # Errors / 错误
500    ///
501    /// Returns error if token is invalid or expired / 如果token无效或过期则返回错误
502    pub fn verify_token(token: &str) -> SecurityResult<JwtClaims>
503    {
504        let secret = Self::get_secret();
505        let decoding_key = DecodingKey::from_secret(secret.as_ref());
506        let validation = Validation::new(jsonwebtoken::Algorithm::HS256);
507
508        decode::<JwtClaims>(token, &decoding_key, &validation)
509            .map(|data| {
510                let claims = data.claims;
511
512                // Check expiration manually for better error messages
513                if claims.is_expired()
514                {
515                    return Err(SecurityError::TokenExpired("Token has expired".to_string()));
516                }
517
518                Ok(claims)
519            })
520            .map_err(|e| match e.kind()
521            {
522                jsonwebtoken::errors::ErrorKind::ExpiredSignature =>
523                {
524                    SecurityError::TokenExpired("Token signature has expired".to_string())
525                },
526                _ => SecurityError::InvalidToken(format!("Invalid token: {}", e)),
527            })?
528    }

530    /// Refresh JWT token
531    /// 刷新JWT token

533    /// Creates a new token with the same user information but extended expiration.
534    /// 创建具有相同用户信息但延长过期时间的新token。

537    ///
538    /// * `token` - Old JWT token / 旧的JWT token
539    pub fn refresh_token(token: &str) -> SecurityResult<String>
540    {
541        let claims = Self::verify_token(token)?;
542
543        // Parse authorities back from strings
544        let authorities: Vec<Authority> = claims
545            .authorities
546            .iter()
547            .filter_map(|s| Authority::from_string(s))

549
550        Self::create_token(&claims.sub, &claims.username, &authorities)

566    /// Decode token without any signature or claim validation
567    /// 不进行任何签名或声明验证地解码令牌
568    ///
569    /// Reads the claims payload from the token without checking the signature,
570    /// expiration, or any other validation rules. Useful for inspecting token
571    /// contents in non-security contexts.
572    /// 从令牌中读取声明负载而不检查签名、过期时间或任何其他验证规则。
573    /// 适用于在非安全上下文中检查令牌内容。
574    ///
575    /// # Warning / 警告
576    ///
577    /// Do NOT use this for authentication decisions.
578    /// 不要将此用于身份验证决策。
579    pub fn decode_without_validation(token: &str) -> SecurityResult<JwtClaims>
580    {
581        use base64::Engine;
582        let parts: Vec<&str> = token.split('.').collect();
583        if parts.len() != 3
584        {
585            return Err(SecurityError::InvalidToken(
586                "Invalid token format: expected 3 parts".to_string(),
587            ));
588        }
589
590        let payload = parts
591            .get(1)
592            .ok_or_else(|| SecurityError::InvalidToken("Token payload part missing".to_string()))?;
593        let decoded = base64::engine::general_purpose::URL_SAFE_NO_PAD
594            .decode(payload)
595            .map_err(|_| {
596                SecurityError::InvalidToken("Failed to decode token payload".to_string())
597            })?;
598
599        let claims: JwtClaims = serde_json::from_slice(&decoded)
600            .map_err(|e| SecurityError::InvalidToken(format!("Failed to parse claims: {}", e)))?;
601
602        Ok(claims)
603    }

605    /// Fully validate and decode a JWT token
606    /// 完全验证并解码JWT令牌
607    ///
608    /// Performs signature verification and validates all standard claims:
609    /// 执行签名验证并验证所有标准声明：
610    /// - Signature is valid for the given secret
611    /// - `exp` (expiration) has not passed
612    /// - `nbf` (not-before) has passed if present
613    ///
614    /// Optionally validates `iss` (issuer) and `aud` (audience) if provided.
615    /// 如果提供了`iss`（签发者）和`aud`（受众），则可选地验证它们。
616    pub fn decode_and_validate(
617        token: &str,
618        secret: &str,
619        algorithm: &JwtAlgorithm,
620        issuer: Option<&str>,
621        audience: Option<&str>,
622    ) -> SecurityResult<JwtClaims>
623    {
624        let decoding_key = DecodingKey::from_secret(secret.as_ref());
625        let mut validation = Validation::new(algorithm.to_algorithm());
626
627        if let Some(iss) = issuer
628        {
629            validation.set_issuer(&[iss]);
630        }
631        if let Some(aud) = audience
632        {
633            validation.set_audience(&[aud]);
634        }
635
636        decode::<JwtClaims>(token, &decoding_key, &validation)
637            .map(|data| data.claims)
638            .map_err(|e| match e.kind()
639            {
640                jsonwebtoken::errors::ErrorKind::ExpiredSignature =>
641                {
642                    SecurityError::TokenExpired("Token has expired".to_string())
643                },
644                jsonwebtoken::errors::ErrorKind::InvalidToken =>
645                {
646                    SecurityError::InvalidToken("Token is invalid".to_string())
647                },
648                jsonwebtoken::errors::ErrorKind::InvalidSignature =>
649                {
650                    SecurityError::InvalidToken("Invalid token signature".to_string())
651                },
652                _ => SecurityError::InvalidToken(format!("Token validation failed: {}", e)),
653            })
654    }

656    /// Refresh token if it will expire within the given threshold
657    /// 如果令牌将在给定阈值内过期，则刷新令牌
658    ///
659    /// If the token's remaining lifetime is less than `threshold_secs` seconds,
660    /// a new token is created with the same claims but a fresh expiration.
661    /// Otherwise, the original token string is returned unchanged.
662    /// 如果令牌的剩余生存期少于`threshold_secs`秒，
663    /// 则创建具有相同声明但具有新过期时间的新令牌。
664    /// 否则，原令牌字符串不变地返回。
665    ///
666    /// # Arguments / 参数
667    ///
668    /// * `token` - The current JWT token / 当前的JWT令牌
669    /// * `threshold_secs` - Seconds before expiration to trigger refresh / 触发刷新的过期前秒数
670    ///
671    /// # Returns / 返回
672    ///
673    /// A tuple of (token_string, was_refreshed) / 一个元组（令牌字符串，是否已刷新）
674    pub fn refresh_if_needed(token: &str, threshold_secs: i64) -> SecurityResult<(String, bool)>
675    {
676        // Verify signature first — do NOT trust unverified claims.
677        // 先验证签名 — 不要信任未验证的 claims。
678        let claims = Self::verify_token(token)?;
679
680        let now = Utc::now().timestamp();
681        let remaining = claims.exp - now;
682
683        if remaining < threshold_secs
684        {
685            // Token is close to expiry or already expired; refresh it
686            // 令牌即将过期或已经过期；刷新它
687            let authorities: Vec<Authority> = claims
688                .authorities
689                .iter()
690                .filter_map(|s| Authority::from_string(s))
691                .collect();
692            let new_token = Self::create_token(&claims.sub, &claims.username, &authorities)?;
693            Ok((new_token, true))
694        }
695        else
696        {
697            Ok((token.to_string(), false))
698        }
699    }
700}

702/// JWT token provider
703/// JWT token 提供者
704///
705/// Equivalent to Spring's `JwtTokenProvider`.
706/// `等价于Spring的JwtTokenProvider`。
707///
708/// Supports HMAC (HS256/HS384/HS512) and RSA (RS256) algorithms.
709/// 支持HMAC（HS256/HS384/HS512）和RSA（RS256）算法。
710///
711/// # Spring Equivalent / Spring等价物
712///
713/// ```java
714/// public class JwtTokenProvider {
715///     public String generateToken(Authentication authentication) { ... }
716///     public boolean validateToken(String token) { ... }
717///     public Authentication getAuthentication(String token) { ... }
718/// }
719/// ```
720#[derive(Clone)]
721pub struct JwtTokenProvider
722{
723    /// Secret key for signing tokens (HMAC) or PEM-encoded RSA private key
724    /// 签名令牌的密钥（HMAC）或PEM编码的RSA私钥
725    secret: String,
726
727    /// PEM-encoded RSA public key for RS256 verification (optional)
728    /// PEM编码的RSA公钥，用于RS256验证（可选）
729    rsa_public_key_pem: Option<String>,
730
731    /// Token expiration in hours
732    /// Token过期时间（小时）
733    expiration_hours: i64,
734
735    /// Signing algorithm
736    /// 签名算法
737    algorithm: JwtAlgorithm,
738
739    /// Expected issuer for validation
740    /// 用于验证的预期签发者
741    issuer: Option<String>,
742
743    /// Expected audience for validation
744    /// 用于验证的预期受众
745    audience: Option<String>,
746}
747
748impl JwtTokenProvider
749{
750    /// Create new JWT token provider with default settings
751    /// 使用默认设置创建新的JWT令牌提供者
752    pub fn new() -> Self
753    {
754        Self {
755            secret: JwtUtil::get_secret(),
756            rsa_public_key_pem: None,
757            expiration_hours: JwtUtil::get_default_expiration(),
758            algorithm: JwtAlgorithm::default(),
759            issuer: Some("hiver-security".to_string()),
760            audience: None,
761        }
762    }
763
764    /// Create with custom secret and expiration
765    /// 使用自定义密钥和过期时间创建
766    pub fn with_settings(secret: impl Into<String>, expiration_hours: i64) -> Self
767    {
768        Self {
769            secret: secret.into(),
770            rsa_public_key_pem: None,
771            expiration_hours,
772            algorithm: JwtAlgorithm::default(),
773            issuer: Some("hiver-security".to_string()),
774            audience: None,
775        }
776    }
777
778    /// Set the signing algorithm
779    /// 设置签名算法
780    pub fn with_algorithm(mut self, algorithm: JwtAlgorithm) -> Self
781    {
782        self.algorithm = algorithm;
783        self
784    }
785
786    /// Set the expected issuer for validation
787    /// 设置用于验证的预期签发者
788    pub fn with_issuer(mut self, issuer: impl Into<String>) -> Self
789    {
790        self.issuer = Some(issuer.into());
791        self
792    }
793
794    /// Set the expected audience for validation
795    /// 设置用于验证的预期受众
796    pub fn with_audience(mut self, audience: impl Into<String>) -> Self
797    {
798        self.audience = Some(audience.into());
799        self
800    }
801
802    /// Set RSA public key PEM for RS256 verification
803    /// 设置用于RS256验证的RSA公钥PEM
804    ///
805    /// When using RS256, the private key is used for signing and
806    /// the public key is used for verification.
807    /// 使用RS256时，私钥用于签名，公钥用于验证。
808    pub fn with_rsa_public_key(mut self, pem: impl Into<String>) -> Self
809    {
810        self.rsa_public_key_pem = Some(pem.into());
811        self
812    }
813
814    /// Get the encoding key based on the algorithm
815    /// 根据算法获取编码密钥
816    fn encoding_key(&self) -> SecurityResult<EncodingKey>
817    {
818        match self.algorithm
819        {
820            JwtAlgorithm::Rs256 => EncodingKey::from_rsa_pem(self.secret.as_bytes())
821                .map_err(|e| SecurityError::Jwt(format!("Invalid RSA private key: {}", e))),
822            _ => Ok(EncodingKey::from_secret(self.secret.as_ref())),
823        }
824    }
825
826    /// Get the decoding key based on the algorithm
827    /// 根据算法获取解码密钥
828    fn decoding_key(&self) -> SecurityResult<DecodingKey>
829    {
830        match self.algorithm
831        {
832            JwtAlgorithm::Rs256 =>
833            {
834                let pem = self.rsa_public_key_pem.as_deref().unwrap_or(&self.secret);
835                DecodingKey::from_rsa_pem(pem.as_bytes())
836                    .map_err(|e| SecurityError::Jwt(format!("Invalid RSA public key: {}", e)))
837            },
838            _ => Ok(DecodingKey::from_secret(self.secret.as_ref())),
839        }
840    }
841
842    /// Build the validation rules
843    /// 构建验证规则
844    fn validation(&self) -> Validation
845    {
846        let mut validation = Validation::new(self.algorithm.to_algorithm());
847        if let Some(ref iss) = self.issuer
848        {
849            validation.set_issuer(&[iss.as_str()]);
850        }
851        if let Some(ref aud) = self.audience
852        {
853            validation.set_audience(&[aud.as_str()]);
854        }
855        else
856        {
857            // If no audience is configured, disable audience validation
858            // 如果未配置受众，则禁用受众验证
859            validation.set_audience::<&str>(&[]);
860        }
861        validation
862    }
863
864    /// Generate token from authentication
865    /// 从认证生成token
866    pub fn generate_token(
867        &self,
868        user_id: impl Into<String>,
869        username: impl Into<String>,
870        authorities: &[Authority],
871    ) -> SecurityResult<String>
872    {
873        let mut claims = JwtClaims::new(user_id, username, authorities, self.expiration_hours);
874
875        // Apply provider-level issuer and audience to claims
876        // 将提供者级别的签发者和受众应用于声明
877        if self.issuer.is_some()
878        {
879            claims.iss.clone_from(&self.issuer);
880        }
881        if let Some(ref aud) = self.audience
882        {
883            claims.aud = Some(serde_json::Value::String(aud.clone()));
884        }
885
886        let encoding_key = self.encoding_key()?;
887        let header = Header::new(self.algorithm.to_algorithm());
888
889        encode(&header, &claims, &encoding_key)
890            .map_err(|e| SecurityError::TokenError(format!("Failed to encode token: {}", e)))
891    }
892
893    /// Validate token, returning true if valid
894    /// 验证令牌，有效则返回true
895    pub fn validate_token(&self, token: &str) -> SecurityResult<bool>
896    {
897        match self.decode_and_validate(token)
898        {
899            Ok(_) => Ok(true),
900            Err(_) => Ok(false),
901        }
902    }
903
904    /// Get authentication from token
905    /// 从token获取认证
906    pub fn get_authentication(&self, token: &str) -> SecurityResult<JwtClaims>
907    {
908        self.decode_and_validate(token)
909    }
910
911    /// Refresh token
912    /// 刷新token
913    pub fn refresh_token(&self, token: &str) -> SecurityResult<String>
914    {
915        let claims = self.decode_and_validate(token)?;
916        let authorities: Vec<Authority> = claims
917            .authorities
918            .iter()
919            .filter_map(|s| Authority::from_string(s))
920            .collect();
921        self.generate_token(&claims.sub, &claims.username, &authorities)
922    }
923
924    /// Full validation: verify signature, check exp/nbf, optionally check issuer/audience
925    /// 完整验证：验证签名，检查exp/nbf，可选检查签发者/受众
926    ///
927    /// Returns the decoded claims if all validations pass.
928    /// 如果所有验证都通过，则返回解码后的声明。
929    pub fn decode_and_validate(&self, token: &str) -> SecurityResult<JwtClaims>
930    {
931        let decoding_key = self.decoding_key()?;
932        let validation = self.validation();
933
934        decode::<JwtClaims>(token, &decoding_key, &validation)
935            .map(|data| data.claims)
936            .map_err(|e| match e.kind()
937            {
938                jsonwebtoken::errors::ErrorKind::ExpiredSignature =>
939                {
940                    SecurityError::TokenExpired("Token has expired".to_string())
941                },
942                jsonwebtoken::errors::ErrorKind::InvalidSignature =>
943                {
944                    SecurityError::InvalidToken("Invalid token signature".to_string())
945                },
946                _ => SecurityError::InvalidToken(format!("Token validation failed: {}", e)),
947            })
948    }
949
950    /// Decode token without validation (reads claims without checking signature)
951    /// 不验证地解码令牌（读取声明而不检查签名）
952    pub fn decode_without_validation(&self, token: &str) -> SecurityResult<JwtClaims>
953    {
954        JwtUtil::decode_without_validation(token)
955    }
956
957    /// Refresh token if it will expire within the given threshold
958    /// 如果令牌将在给定阈值内过期，则刷新令牌
959    ///
960    /// Returns (token_string, was_refreshed).
961    /// 返回（令牌字符串，是否已刷新）。
962    pub fn refresh_if_needed(
963        &self,
964        token: &str,
965        threshold_secs: i64,
966    ) -> SecurityResult<(String, bool)>
967    {
968        let claims = JwtUtil::decode_without_validation(token)?;
969
970        let now = Utc::now().timestamp();
971        let remaining = claims.exp - now;
972
973        if remaining < threshold_secs
974        {
975            let authorities: Vec<Authority> = claims
976                .authorities
977                .iter()
978                .filter_map(|s| Authority::from_string(s))
979                .collect();
980            let new_token = self.generate_token(&claims.sub, &claims.username, &authorities)?;
981            Ok((new_token, true))
982        }
983        else
984        {
985            Ok((token.to_string(), false))
986        }
987    }

1005    /// Generate an OAuth2 access token embedding `scope` and `client_id` custom claims.
1006    /// 生成包含 `scope` 和 `client_id` 自定义声明的 OAuth2 访问令牌。
1007    pub fn generate_oauth2_token(
1008        &self,
1009        subject: &str,
1010        client_id: &str,
1011        scope: &str,
1012        ttl: std::time::Duration,
1013    ) -> SecurityResult<String>
1014    {
1015        let now = Utc::now().timestamp();
1016        let exp = now + ttl.as_secs() as i64;
1017        let mut custom: HashMap<String, serde_json::Value> = HashMap::new();
1018        custom.insert("scope".into(), serde_json::Value::String(scope.to_string()));
1019        custom.insert("client_id".into(), serde_json::Value::String(client_id.to_string()));
1020        let claims = JwtClaims {
1021            sub: subject.to_string(),
1022            username: subject.to_string(),
1023            authorities: Vec::new(),
1024            iat: now,
1025            exp,
1026            iss: self.issuer.clone(),
1027            aud: None,
1028            nbf: None,
1029            jti: None,
1030            custom,
1031        };
1032        let encoding_key = self.encoding_key()?;
1033        let header = Header::new(self.algorithm.to_algorithm());
1034        encode(&header, &claims, &encoding_key)
1035            .map_err(|e| SecurityError::TokenError(format!("Failed to encode OAuth2 token: {e}")))
1036    }

1039    /// 生成最小化的 OIDC ID 令牌（主体 + 受众）。
1040    pub fn generate_id_token(&self, subject: &str, client_id: &str) -> SecurityResult<String>
1041    {

1043        let exp = now + 3600;
1044        let claims = JwtClaims {
1045            sub: subject.to_string(),
1046            username: subject.to_string(),
1047            authorities: Vec::new(),
1048            iat: now,
1049            exp,
1050            iss: self.issuer.clone(),
1051            aud: Some(serde_json::Value::String(client_id.to_string())),
1052            nbf: None,
1053            jti: None,
1054            custom: HashMap::new(),
1055        };
1056        let encoding_key = self.encoding_key()?;
1057        let header = Header::new(self.algorithm.to_algorithm());
1058        encode(&header, &claims, &encoding_key)
1059            .map_err(|e| SecurityError::TokenError(format!("Failed to encode ID token: {e}")))
1060    }
