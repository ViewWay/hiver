1//! Multi-producer, single-consumer channels for async communication
2//! 用于异步通信的多生产者、单消费者通道
3//!
4//! # Overview / 概述
5//!
6//! This module provides multi-producer, single-consumer (mpsc) channels
7//! for asynchronous communication between tasks.
8//!
9//! 本模块提供用于任务间异步通信的多生产者、单消费者(mpsc)通道。
10//!
11//! # Example / 示例
12//!
13//! ```rust,no_run,ignore
14//! use hiver_runtime::channel::unbounded;
15//!
16//! async fn example() {
17//!     let (tx, mut rx) = unbounded();
18//!
19//!     // Spawn a sender task
20//!     // 生成发送器任务
21//!     hiver_runtime::spawn(async move {
22//!         tx.send("Hello").await.unwrap();
23//!         tx.send("World").await.unwrap();
24//!     });
25//!
26//!     // Receive messages
27//!     // 接收消息
28//!     while let Some(msg) = rx.recv().await {
29//!         println!("Received: {}", msg);
30//!     }
31//! }
32//! ```

84/// Error type for receiving
85/// 接收错误类型

87pub enum RecvError
88{
89    /// The channel is empty but not closed (senders still exist).

92    /// The channel is empty and closed (all senders dropped).
93    /// 通道为空且已关闭（所有发送者已 drop）。

105        }

111/// Unbounded mpsc channel
112/// 无边界mpsc通道
113///
114/// Creates a channel with an unbounded buffer.
115/// 创建具有无界缓冲区的通道。
116///
117/// # Example / 示例
118///
119/// ```rust,no_run,ignore
120/// use hiver_runtime::channel::unbounded;
121///
122/// let (tx, rx) = unbounded::<i32>();
123/// ```
124#[must_use]
125pub fn unbounded<T>() -> (Sender<T>, Receiver<T>)
126{
127    let shared = Arc::new(ChannelShared {
128        buffer: Mutex::new(VecDeque::new()),
129        sender_count: AtomicUsize::new(1),
130        is_receiver_alive: AtomicBool::new(true),
131        recv_waker: Mutex::new(None),
132    });
133
134    let sender = Sender {
135        shared: shared.clone(),
136    };
137    let receiver = Receiver { shared };
138
139    (sender, receiver)
140}

142/// Bounded mpsc channel with real backpressure.
143/// 具有真实背压的有界 mpsc 通道。
144///
145/// Creates a channel with a bounded buffer of `cap` slots. Unlike the old
146/// implementation (which silently ignored the cap), this delegates to
147/// [`async_channel::bounded`], which provides true backpressure: `send().await`
148/// will **park the sender** when the buffer is full, until a receiver consumes
149/// a slot. This prevents unbounded memory growth under fast-producer /
150/// slow-consumer patterns.
151///
152/// 创建一个具有 `cap` 容量槽的有界缓冲通道。与旧实现（静默忽略容量）不同,
153/// 此处委托给 [`async_channel::bounded`],提供真实背压:当缓冲区满时,
154/// `send().await` 会 **park 发送者**,直到接收者消费一个槽。这防止了快生产者/
155/// 慢消费者模式下的无界内存增长。
156///
157/// The returned `Sender`/`Receiver` are `async_channel` types with async
158/// `send().await` and `recv().await`. `try_recv` returns
159/// `Result<T, async_channel::TryRecvError>` where `TryRecvError::Empty`
160/// distinguishes "temporarily empty" from `Closed`.
161///
162/// 返回的 `Sender`/`Receiver` 是 `async_channel` 类型,具有异步
163/// `send().await` 与 `recv().await`。`try_recv` 返回
164/// `Result<T, async_channel::TryRecvError>`,其中 `TryRecvError::Empty`
165/// 区分 "暂时为空" 与 `Closed`。
166///
167/// # Example / 示例
168///
169/// ```rust,no_run,ignore
170/// use hiver_runtime::channel::bounded;
171///
172/// let (tx, rx) = bounded::<i32>(16);
173/// // tx.send(42).await.unwrap();  // parks if buffer is full
174/// ```
175#[must_use]
176pub fn bounded<T>(cap: usize) -> (async_channel::Sender<T>, async_channel::Receiver<T>)
177{
178    async_channel::bounded(cap)
179}

181/// Shared state for the channel
182/// 通道的共享状态
183struct ChannelShared<T>
184{
185    /// Message buffer
186    /// 消息缓冲区
187    buffer: Mutex<VecDeque<T>>,
188    /// Number of active senders
189    /// 活跃发送器数量
190    sender_count: AtomicUsize,
191    /// Whether the receiver is still alive
192    /// 接收器是否仍然存活
193    is_receiver_alive: AtomicBool,
194    /// Waker for pending receive operations
195    /// 挂起接收操作的waker
196    recv_waker: Mutex<Option<Waker>>,
197}
198
199/// Sender side of the channel
200/// 通道的发送端
201///
202/// Can be cloned to create multiple senders.
203/// 可以克隆以创建多个发送器。
204pub struct Sender<T>
205{
206    shared: Arc<ChannelShared<T>>,
207}
208
209impl<T> Clone for Sender<T>
210{
211    fn clone(&self) -> Self
212    {
213        // Increment sender count
214        // 增加发送器计数
215        self.shared.sender_count.fetch_add(1, Ordering::Relaxed);
216        Self {
217            shared: Arc::clone(&self.shared),
218        }
219    }
220}
221
222impl<T> Sender<T>
223{
224    /// Send a value synchronously to the channel
225    /// 向通道同步发送值
226    ///
227    /// # Errors
228    ///
229    /// Returns `SendError::Closed` if the receiver has been dropped.
230    /// 如果接收器已丢弃则返回 `SendError::Closed`。
231    ///
232    /// # Panics
233    ///
234    /// Panics if the internal mutex is poisoned (should never happen in normal operation).
235    /// 如果内部互斥锁被污染则恐慌（正常操作中不应发生）。
236    pub fn send(&self, value: T) -> Result<(), SendError<T>>
237    {
238        if !self.shared.is_receiver_alive.load(Ordering::Acquire)
239        {
240            return Err(SendError::Closed(value));
241        }
242
243        let mut buffer = self.shared.buffer.lock().unwrap();
244        buffer.push_back(value);
245
246        // Wake the receiver if it's waiting
247        // 如果接收器在等待，唤醒它
248        if let Some(waker) = self.shared.recv_waker.lock().unwrap().take()
249        {
250            drop(buffer);
251            waker.wake();
252        }
253
254        Ok(())
255    }
256
257    /// Check if the channel is closed (receiver dropped)
258    /// 检查通道是否已关闭（接收器已丢弃）
259    #[must_use]
260    pub fn is_closed(&self) -> bool
261    {
262        !self.shared.is_receiver_alive.load(Ordering::Acquire)
263    }
264
265    /// Get the number of active senders
266    /// 获取活跃发送器数量
267    #[must_use]
268    pub fn sender_count(&self) -> usize
269    {
270        self.shared.sender_count.load(Ordering::Acquire)
271    }
272}
273
274impl<T> Drop for Sender<T>
275{
276    fn drop(&mut self)
277    {
278        // Decrease sender count
279        // 减少发送器计数
280        let prev = self.shared.sender_count.fetch_sub(1, Ordering::AcqRel);
281
282        if prev == 1
283        {
284            // Last sender dropped, wake the receiver
285            // 最后一个发送器丢弃，唤醒接收器
286            if let Some(waker) = self.shared.recv_waker.lock().unwrap().take()
287            {
288                waker.wake();
289            }
290        }
291    }
292}

294/// Receiver side of the channel
295/// 通道的接收端
296pub struct Receiver<T>
297{
298    shared: Arc<ChannelShared<T>>,
299}
300
301impl<T> Receiver<T>
302{
303    /// Receive a value from the channel
304    /// 从通道接收值
305    pub fn recv(&mut self) -> RecvFuture<'_, T>
306    {
307        RecvFuture::new(self)
308    }
309
310    /// Try to receive a value without blocking
311    /// 尝试接收值而不阻塞
312    ///
313    /// # Errors
314    ///
315    /// Returns `Err(RecvError::Closed)` if the channel is empty and all senders are dropped.
316    /// 如果通道为空且所有发送器已丢弃则返回 `Err(RecvError::Closed)`。
317    ///
318    /// Returns `Err(RecvError::Empty)` if the channel is empty but senders still exist.
319    /// 如果通道为空但发送器仍然存在则返回 `Err(RecvError::Empty)`。
320    ///
321    /// # Panics
322    ///
323    /// Panics if the internal mutex is poisoned (should never happen in normal operation).
324    /// 如果内部互斥锁被污染则恐慌（正常操作中不应发生）。
325    pub fn try_recv(&mut self) -> Result<T, RecvError>
326    {
327        let mut buffer = self.shared.buffer.lock().unwrap();
328
329        if let Some(value) = buffer.pop_front()
330        {
331            Ok(value)
332        }
333        else if self.shared.sender_count.load(Ordering::Acquire) == 0
334        {
335            // No senders left — channel is closed.
336            // 没有发送器了 —— 通道已关闭。
337            Err(RecvError::Closed)
338        }
339        else
340        {
341            // Channel is empty but senders still exist — try again later.
342            // 通道为空但发送器仍然存在 —— 稍后重试。
343            Err(RecvError::Empty)
344        }
345    }
346
347    /// Get the number of messages in the channel buffer
348    /// 获取通道缓冲区中的消息数量
349    ///
350    /// # Panics
351    ///
352    /// Panics if the internal mutex is poisoned (should never happen in normal operation).
353    /// 如果内部互斥锁被污染则恐慌（正常操作中不应发生）。
354    #[must_use]
355    pub fn len(&self) -> usize
356    {
357        self.shared.buffer.lock().unwrap().len()
358    }
359
360    /// Check if the channel is empty
361    /// 检查通道是否为空
362    #[must_use]
363    pub fn is_empty(&self) -> bool
364    {
365        self.len() == 0
366    }
367}
368
369impl<T> Drop for Receiver<T>
370{
371    fn drop(&mut self)
372    {
373        // Mark receiver as dropped
374        // 标记接收器已丢弃
375        self.shared
376            .is_receiver_alive
377            .store(false, Ordering::Release);
378    }
379}

381/// Receive future
382/// 接收future
383pub struct RecvFuture<'a, T>
384{
385    /// Reference to the receiver's shared state
386    /// 接收器共享状态的引用
387    shared: Arc<ChannelShared<T>>,
388    /// Marker for the lifetime
389    /// 生命周期标记
390    _marker: std::marker::PhantomData<&'a mut Receiver<T>>,
391}
392
393impl<'a, T> RecvFuture<'a, T>
394{
395    /// Create a new receive future
396    fn new(receiver: &'a mut Receiver<T>) -> Self
397    {
398        // We extract the Arc since the receiver only holds it
399        // This is safe because the future borrows the receiver mutably
400        Self {
401            shared: Arc::clone(&receiver.shared),
402            _marker: std::marker::PhantomData,
403        }
404    }
405}
406
407unsafe impl<T: Send> Send for RecvFuture<'_, T> {}
408unsafe impl<T: Sync> Sync for RecvFuture<'_, T> {}
409
410impl<T> Future for RecvFuture<'_, T>
411{
412    type Output = Option<T>;
413
414    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output>
415    {
416        let mut buffer = self.shared.buffer.lock().unwrap();
417
418        if let Some(value) = buffer.pop_front()
419        {
420            Poll::Ready(Some(value))
421        }
422        else if self.shared.sender_count.load(Ordering::Acquire) == 0
423        {
424            // No senders left and buffer empty
425            // 没有发送器了且缓冲区为空
426            Poll::Ready(None)
427        }
428        else
429        {
430            // No value available, register waker
431            // 没有可用值，注册waker
432            *self.shared.recv_waker.lock().unwrap() = Some(cx.waker().clone());
433            Poll::Pending
434        }
435    }
436}
