11/// Column type
12/// 列类型
13#[derive(Debug, Clone, PartialEq, Eq)]
14pub enum ColumnType
15{
16    /// Boolean type
17    Bool,
18    /// 8-bit signed integer
19    I8,
20    /// 16-bit signed integer
21    I16,
22    /// 32-bit signed integer
23    I32,
24    /// 64-bit signed integer
25    I64,
26    /// 128-bit signed integer
27    I128,
28    /// 8-bit unsigned integer
29    U8,
30    /// 16-bit unsigned integer
31    U16,
32    /// 32-bit unsigned integer
33    U32,
34    /// 64-bit unsigned integer
35    U64,
36    /// 32-bit float
37    F32,
38    /// 64-bit float
39    F64,
40    /// String type
41    String,
42    /// Text type (long string)
43    Text,
44    /// Bytes type
45    Bytes,
46    /// UUID type
47    Uuid,
48    /// Date type
49    Date,
50    /// Time type
51    Time,
52    /// Timestamp type
53    Timestamp,
54    /// JSON type
55    Json,
56    /// Decimal type
57    Decimal,
58    /// Enum type
59    Enum,
60    /// Array type
61    Array,
62    /// Custom type
63    Custom(String),
64}
65
66impl ColumnType
67{
68    /// Get the SQL type name for a given database
69    pub fn as_sql(&self, dialect: SqlDialect) -> &str
70    {
71        match (self, dialect)
72        {
73            (ColumnType::Bool, _) => "BOOLEAN",
74            (ColumnType::I32, SqlDialect::PostgreSQL) => "INTEGER",
75            (ColumnType::I32, SqlDialect::MySQL) => "INT",
76            (ColumnType::I64, SqlDialect::PostgreSQL) => "BIGINT",
77            (ColumnType::I64, SqlDialect::MySQL) => "BIGINT",
78            (ColumnType::String, SqlDialect::PostgreSQL) => "VARCHAR",
79            (ColumnType::String, SqlDialect::MySQL) => "VARCHAR",
80            (ColumnType::Text, _) => "TEXT",
81            (ColumnType::Json, SqlDialect::PostgreSQL) => "JSONB",
82            (ColumnType::Json, _) => "JSON",
83            _ => "TEXT", // Default fallback
84        }
85    }
86}

100/// Column metadata
101#[derive(Debug, Clone)]
102pub struct Column
103{
104    /// Column name
105    pub name: String,
106    /// Column type
107    pub type_: ColumnType,
108    /// Whether this is a primary key
109    pub is_primary_key: bool,
110    /// Whether this column is nullable
111    pub is_nullable: bool,
112    /// Whether this column has a unique constraint
113    pub is_unique: bool,
114    /// Default value expression
115    pub default: Option<String>,
116    /// Maximum length for string types
117    pub max_length: Option<usize>,
118}
119
120impl Column
121{
122    /// Create a new column definition
123    pub fn new(name: impl Into<String>, type_: ColumnType) -> Self
124    {
125        Self {
126            name: name.into(),
127            type_,
128            is_primary_key: false,
129            is_nullable: false,
130            is_unique: false,
131            default: None,
132            max_length: None,
133        }
134    }
135
136    /// Mark this column as a primary key
137    pub fn primary_key(mut self) -> Self
138    {
139        self.is_primary_key = true;
140        self
141    }
142
143    /// Mark this column as nullable
144    pub fn nullable(mut self) -> Self
145    {
146        self.is_nullable = true;
147        self
148    }
149
150    /// Mark this column as unique
151    pub fn unique(mut self) -> Self
152    {
153        self.is_unique = true;
154        self
155    }
156
157    /// Set the default value for this column
158    pub fn with_default(mut self, default: impl Into<String>) -> Self
159    {
160        self.default = Some(default.into());
161        self
162    }
163}

165/// Model metadata
166#[derive(Debug, Clone)]
167pub struct ModelMeta
168{
169    /// Table name
170    pub table_name: String,
171    /// Column definitions
172    pub columns: Vec<Column>,
173}
174
175impl ModelMeta
176{
177    /// Create new model metadata with the given table name
178    pub fn new(table_name: impl Into<String>) -> Self
179    {
180        Self {
181            table_name: table_name.into(),
182            columns: Vec::new(),
183        }
184    }
185
186    /// Add a column to the model metadata
187    pub fn add_column(mut self, column: Column) -> Self
188    {
189        self.columns.push(column);
190        self
191    }
192
193    /// Get the table name
194    pub fn table_name(&self) -> &str
195    {
196        &self.table_name
197    }
198}

200/// Model trait
201/// Model trait
202pub trait Model: Send + Sync
203{
204    /// Get the model metadata
205    fn meta() -> ModelMeta;
206
207    /// Get the table name
208    fn table_name() -> String
209    where
210        Self: Sized,
211    {
212        Self::meta().table_name().to_string()
213    }
214
215    /// Get the primary key value (placeholder)
216    fn primary_key(&self) -> Result<String>
217    {
218        Err(Error::unknown("Primary key not implemented"))
219    }
220
221    /// Set the primary key value (placeholder)
222    fn set_primary_key(&mut self, _value: String) -> Result<()>
223    {
224        Err(Error::unknown("Set primary key not implemented"))
225    }
226
227    /// Validate the model
228    fn validate(&self) -> Result<()>
229    {
230        Ok(())
231    }
232
233    /// Get all column names in declaration order.
234    /// 获取声明顺序的所有列名。
235    ///
236    /// Returns a static slice of column name strings.
237    /// 返回列名字符串的静态切片。
238    fn column_names() -> &'static [&'static str]
239    where
240        Self: Sized,
241    {
242        &[]
243    }
244
245    /// Get the number of columns.
246    /// 获取列数量。
247    fn column_count() -> usize
248    where
249        Self: Sized,
250    {
251        Self::column_names().len()
252    }
253
254    /// Construct the model from a database row.
255    /// 从数据库行构造模型实例。
256    ///
257    /// Uses `Row::get_as` to extract typed values by column name.
258    /// 使用 `Row::get_as` 按列名提取类型化值。
259    fn from_row(row: &hiver_data_rdbc::Row) -> Result<Self>
260    where
261        Self: Sized,
262    {
263        let _ = row;
264        Err(Error::unknown("from_row not implemented"))
265    }
266
267    /// Convert the model to column-value pairs for INSERT/UPDATE.
268    /// 将模型转换为列-值对，用于 INSERT/UPDATE。
269    ///
270    /// Returns a vector of (column_name, boxed value) pairs.
271    /// 返回 (列名, 装箱值) 对的向量。
272    fn to_row(&self) -> Vec<(&'static str, Box<dyn std::any::Any + Send>)>
273    where
274        Self: Sized,
275    {
276        Vec::new()
277    }
278
279    /// Get the relationship definitions for this model.
280    /// 获取此模型的关系定义。
281    ///
282    /// Override this method to declare cascading relationships.
283    /// 重写此方法以声明级联关系。
284    fn relations() -> Vec<crate::Relation>
285    where
286        Self: Sized,
287    {
288        Vec::new()
289    }
290}
