29/// Represents a join point in the program execution
30/// 表示程序执行中的连接点
31///
32/// A join point is a well-defined point in the execution of a program,
33/// such as a method call or exception handler.
34///
35/// 连接点是程序执行中一个明确定义的点，例如方法调用或异常处理程序。
36///
37/// # Example / 示例
38///
39/// ```rust,no_run,ignore
40/// use hiver_aop::runtime::JoinPoint;
41///
42/// #[Before("execution(* com.example..*.*(..))")]
43/// fn log_before(join_point: &JoinPoint) {
44///     println!("Calling: {}", join_point.method_name());
45///     println!("Args: {:?}", join_point.args());
46/// }
47/// ```
48pub struct JoinPoint
49{
50    /// Target object (self)
51    /// 目标对象 (self)
52    target: Arc<dyn Any + Send + Sync>,
53
54    /// Method name
55    /// 方法名
56    method_name: String,
57
58    /// Method arguments
59    /// 方法参数
60    args: Vec<Arc<dyn Any + Send + Sync>>,
61
62    /// Method signature
63    /// 方法签名
64    signature: String,
65
66    /// Target class name
67    /// 目标类名
68    target_class: String,
69}

278/// Ordered advice to be applied to a join point.
279/// 要应用到连接点的有序通知。
280#[derive(Debug, Clone)]
281pub struct AdviceChain
282{
283    /// Before advices (in order).
284    /// 前置通知（按顺序）。
285    pub before: Vec<String>,
286    /// Around advices (in order).
287    /// 环绕通知（按顺序）。
288    pub around: Vec<String>,
289    /// After advices (in order).
290    /// 后置通知（按顺序）。
291    pub after: Vec<String>,
292    /// After-returning advices.
293    /// 返回后通知。
294    pub after_returning: Vec<String>,
295    /// After-throwing advices.
296    /// 异常后通知。
297    pub after_throwing: Vec<String>,
298}
299
300impl AdviceChain
301{
302    /// Create an empty chain.
303    /// 创建空链。
304    pub fn new() -> Self
305    {
306        Self {
307            before: Vec::new(),
308            around: Vec::new(),
309            after: Vec::new(),
310            after_returning: Vec::new(),
311            after_throwing: Vec::new(),
312        }
313    }
314
315    /// Build a chain from matched advice.
316    /// 从匹配的通知构建链。
317    pub fn from_matches(matches: &[(AdviceType, String, String)]) -> Self
318    {
319        let mut chain = Self::new();
320        for (advice_type, _aspect, method) in matches
321        {
322            match advice_type
323            {
324                AdviceType::Before => chain.before.push(method.clone()),
325                AdviceType::Around => chain.around.push(method.clone()),
326                AdviceType::After => chain.after.push(method.clone()),
327                AdviceType::AfterReturning => chain.after_returning.push(method.clone()),
328                AdviceType::AfterThrowing => chain.after_throwing.push(method.clone()),
329            }
330        }
331        chain
332    }
333
334    /// Total number of advices in the chain.
335    /// 链中通知总数。
336    pub fn total(&self) -> usize
337    {
338        self.before.len()
339            + self.around.len()
340            + self.after.len()
341            + self.after_returning.len()
342            + self.after_throwing.len()
343    }
344
345    /// Check if the chain is empty.
346    /// 检查链是否为空。
347    pub fn is_empty(&self) -> bool
348    {
349        self.total() == 0
350    }
351}

365/// Represents a pointcut expression
366/// 表示切点表达式
367///
368/// Pointcut expressions define join points where advice should be applied.
369/// Supports: `execution()`, `within()`, `@annotation()`, `&&`, `||`
370///
371/// 切点表达式定义应该应用通知的连接点。
372/// 支持：`execution()`、`within()`、`@annotation()`、`&&`、`||`
373///
374/// # Syntax / 语法
375///
376/// ```text
377/// execution(RET_TYPE PACKAGE..CLASS.METHOD(PARAMS))
378/// within(FULLY_QUALIFIED_CLASS)
379/// @annotation(FULLY_QUALIFIED_ANNOTATION)
380/// ```
381///
382/// Wildcards: `*` matches any single segment, `..` matches zero or more segments.
383/// 通配符：`*` 匹配单个段，`..` 匹配零或多个段。
384#[derive(Debug, Clone)]
385pub struct PointcutExpression
386{
387    /// The expression string
388    /// 表达式字符串
389    expression: String,
390
391    /// Parsed expression components
392    /// 解析后的表达式组件
393    components: Vec<ExpressionComponent>,
394
395    /// Whether this expression uses AND semantics (all must match)
396    /// 此表达式是否使用 AND 语义（所有组件必须匹配）
397    is_conjunction: bool,
398}
399
400/// Components of a pointcut expression
401/// 切点表达式的组件
402#[derive(Debug, Clone, PartialEq)]
403enum ExpressionComponent
404{
405    /// Execution pointcut: matches method execution
406    /// 执行切点：匹配方法执行
407    Execution
408    {
409        /// Package pattern (may contain `..` and `*`)
410        /// 包模式（可包含 `..` 和 `*`）
411        package: String,
412        /// Class pattern (may be `*`)
413        /// 类模式（可为 `*`）
414        class: String,
415        /// Method pattern (may be `*`)
416        /// 方法模式（可为 `*`）
417        method: String,
418        /// Parameter pattern
419        /// 参数模式
420        params: String,
421    },
422    /// Within pointcut: matches type boundary
423    /// Within 切点：匹配类型边界
424    Within(String),
425    /// Annotation pointcut: matches annotated elements
426    /// 注解切点：匹配带注解的元素
427    Annotation(String),
428}
429
430impl PointcutExpression
431{
432    /// Create a new pointcut expression
433    /// 创建新的切点表达式
434    pub fn new(expression: String) -> Self
435    {
436        let (components, is_conjunction) = Self::parse_expression(&expression);
437        Self {
438            expression,
439            components,
440            is_conjunction,
441        }
442    }
443
444    /// Get the expression string
445    /// 获取表达式字符串
446    pub fn expression(&self) -> &str
447    {
448        &self.expression
449    }
450
451    /// Parse a pointcut expression into components.
452    /// 解析切点表达式为组件。
453    fn parse_expression(expr: &str) -> (Vec<ExpressionComponent>, bool)
454    {
455        let expr = expr.trim();
456
457        // Handle OR: split and parse each branch independently
458        if expr.contains(" || ")
459        {
460            let components: Vec<ExpressionComponent> = expr
461                .split(" || ")
462                .flat_map(|part| Self::parse_single(part.trim()))
463                .collect();
464            return (components, false); // OR: any match is sufficient
465        }
466
467        // Handle AND: parse each branch, all must match
468        if expr.contains(" && ")
469        {
470            let mut components = Vec::new();
471            for part in expr.split(" && ")
472            {
473                components.extend(Self::parse_single(part.trim()));
474            }
475            return (components, true); // AND: all must match
476        }
477
478        // Single expression
479        (Self::parse_single(expr), false)
480    }
481
482    /// Parse a single pointcut designator (no logical operators).
483    /// 解析单个切点指示符（无逻辑运算符）。
484    fn parse_single(expr: &str) -> Vec<ExpressionComponent>
485    {
486        let mut components = Vec::new();
487
488        if let Some(inner) = Self::extract_parens(expr, "execution(")
489        {
490            components.push(Self::parse_execution(inner));
491        }
492
493        if let Some(inner) = Self::extract_parens(expr, "within(")
494        {
495            components.push(ExpressionComponent::Within(inner.to_string()));
496        }
497
498        if let Some(inner) = Self::extract_parens(expr, "@annotation(")
499        {
500            components.push(ExpressionComponent::Annotation(inner.to_string()));
501        }
502
503        components
504    }
505
506    /// Parse execution() inner content: `RET_TYPE QUALIFIED_NAME(PARAMS)`
507    /// 解析 execution() 内部内容：`返回类型 全限定名(参数)`
508    fn parse_execution(inner: &str) -> ExpressionComponent
509    {
510        // Split at '(' to separate pattern from params
511        let (pattern_part, params) = if let Some(pos) = inner.find('(')
512        {
513            (&inner[..pos], inner[pos + 1..].trim_end_matches(')').to_string())
514        }
515        else
516        {
517            (inner, String::new())
518        };
519
520        // Split into ret_type and qualified_name
521        let mut parts = pattern_part.splitn(2, ' ');
522        let _ret_type = parts.next().unwrap_or("*");
523        let qualified = parts.next().unwrap_or("").trim();
524
525        // Split qualified name: last segment is method, second-to-last is class, rest is package
526        let segments: Vec<&str> = qualified.split('.').collect();
527        let (package, class, method) = match segments.len()
528        {
529            0 => (String::new(), String::new(), String::new()),
530            1 => (String::new(), String::new(), segments[0].to_string()),
531            2 =>
532            {
533                let (first, second) = (segments[0], segments[1]);
534                // Check if first segment is a package wildcard (ends with ..)
535                if first == "*" || first.is_empty()
536                {
537                    (String::new(), first.to_string(), second.to_string())
538                }
539                else
540                {
541                    (first.to_string(), String::new(), second.to_string())
542                }
543            },
544            _ =>
545            {
546                let method = segments[segments.len() - 1].to_string();
547                let class = segments[segments.len() - 2].to_string();
548                let pkg = segments[..segments.len() - 2].join(".");
549                (pkg, class, method)
550            },
551        };
552
553        ExpressionComponent::Execution {
554            package,
555            class,
556            method,
557            params,
558        }
559    }
560
561    /// Extract content between `prefix(` and `)`.
562    /// 提取 `prefix(` 和 `)` 之间的内容。
563    fn extract_parens<'a>(expr: &'a str, prefix: &str) -> Option<&'a str>
564    {
565        let start = expr.find(prefix)?;
566        let content_start = start + prefix.len();
567        let end = expr[content_start..].find(')')?;
568        Some(&expr[content_start..content_start + end])
569    }
570
571    /// Check if this pointcut matches a join point.
572    /// 检查此切点是否匹配连接点。
573    pub fn matches(&self, join_point: &JoinPoint) -> bool
574    {
575        if self.components.is_empty()
576        {
577            return false;
578        }
579
580        if self.is_conjunction
581        {
582            // AND: every component must match
583            self.components
584                .iter()
585                .all(|c| Self::matches_component(c, join_point))
586        }
587        else
588        {
589            // OR / single: any component matching is sufficient
590            self.components
591                .iter()
592                .any(|c| Self::matches_component(c, join_point))
593        }
594    }
595
596    /// Check if a single component matches a join point.
597    /// 检查单个组件是否匹配连接点。
598    fn matches_component(component: &ExpressionComponent, join_point: &JoinPoint) -> bool
599    {
600        match component
601        {
602            ExpressionComponent::Execution {
603                package,
604                class,
605                method,
606                params: _,
607            } =>
608            {
609                // Method matching
610                if *method != "*" && method != join_point.method_name()
611                {
612                    return false;
613                }
614
615                let target = join_point.target_class();
616
617                // Class matching
618                if *class != "*" && !class.is_empty() && !Self::matches_segment(class, target)
619                {
620                    return false;
621                }
622
623                // Package matching: skip if target has no package prefix
624                // 包匹配：如果 target 没有包前缀则跳过
625                if package.is_empty() || *package == "*"
626                {
627                    return true;
628                }
629                if !target.contains('.')
630                {
631                    // Target is a simple class name (no package), skip package check
632                    // target 是简单类名（无包名），跳过包检查
633                    return true;
634                }
635
636                // Handle .. wildcard in package pattern
637                // 处理包模式中的 .. 通配符
638                let pkg_normalized = package.trim_end_matches('.');
639                if pkg_normalized.is_empty() || pkg_normalized == "*"
640                {
641                    true
642                }
643                else if pkg_normalized.contains("..")
644                {
645                    // "com.example.." means com.example and any sub-packages
646                    let prefix = pkg_normalized.replace("..", ".");
647                    let prefix = prefix.trim_end_matches('.');
648                    target.starts_with(prefix)
649                }
650                else
651                {
652                    target.starts_with(pkg_normalized)
653                        || target.starts_with(&format!("{}.", pkg_normalized))
654                }
655            },
656            ExpressionComponent::Within(pattern) =>
657            {
658                if *pattern == "*"
659                {
660                    return true;
661                }
662                let target = join_point.target_class();
663                target == *pattern
664                    || target.ends_with(&format!(".{}", pattern))
665                    || Self::matches_wildcard(pattern, target)
666            },
667            ExpressionComponent::Annotation(_ann) =>
668            {
669                // Cannot check annotations at runtime without metadata
670                // 无法在没有元数据的情况下在运行时检查注解
671                false
672            },
673        }
674    }
675
676    /// Check if a pattern segment matches a target string.
677    /// `*` matches any single name segment.
678    fn matches_segment(pattern: &str, target: &str) -> bool
679    {
680        if pattern == target
681        {
682            return true;
683        }
684        // Check if target ends with ".Class" where Class matches the pattern
685        if let Some(dot_pos) = target.rfind('.')
686        {
687            &target[dot_pos + 1..] == pattern
688        }
689        else
690        {
691            false
692        }
693    }
694
695    /// Check if a wildcard pattern (with `*`) matches a target.
696    fn matches_wildcard(pattern: &str, target: &str) -> bool
697    {
698        if !pattern.contains('*')
699        {
700            return false;
701        }
702        let parts: Vec<&str> = pattern.split('*').collect();
703        let mut idx = 0;
704        for (i, part) in parts.iter().enumerate()
705        {
706            if part.is_empty()
707            {
708                continue;
709            }
710            if let Some(pos) = target[idx..].find(part)
711            {
712                idx += pos + part.len();
713            }
714            else
715            {
716                return false;
717            }
718            if i == 0 && !pattern.starts_with('*') && target[idx..].find(part) != Some(0)
719            {
720                return false;
721            }
722        }
723        true
724    }
725}

778/// Type-erased callable advice closures.
779/// 类型擦除的可调用通知闭包。

983    /// Build an InterceptChain from all callable advice matching the join point.
984    /// 从所有匹配连接点的可调用通知构建拦截链。
985    pub fn build_chain(&self, join_point: &JoinPoint) -> InterceptChain
986    {
987        let advices = self.callable_advices.read().unwrap();
988        let mut chain = InterceptChain::new();
989
990        for advice in advices.iter()
991        {
992            if !advice.pointcut.matches(join_point)
993            {
994                continue;
995            }
996
997            match advice.advice_type
998            {
999                AdviceType::Before =>
1000                {
1001                    if let Ok(fn_ptr) = advice.callback.clone().downcast::<BeforeFn>()
1002                    {
1003                        let f = Arc::clone(&fn_ptr);
1004                        chain.before(move |jp| f(jp));
1005                    }
1006                },
1007                AdviceType::After =>
1008                {
1009                    if let Ok(fn_ptr) = advice.callback.clone().downcast::<AfterFn>()
1010                    {
1011                        let f = Arc::clone(&fn_ptr);
1012                        chain.after(move |jp| f(jp));
1013                    }
1014                },
1015                AdviceType::Around =>
1016                {
1017                    if let Ok(fn_ptr) = advice.callback.clone().downcast::<AroundFn>()
1018                    {
1019                        let f = Arc::clone(&fn_ptr);
1020                        chain.around(move |pjp| f(pjp));
1021                    }
1022                },
1023                AdviceType::AfterReturning =>
1024                {
1025                    if let Ok(fn_ptr) = advice.callback.clone().downcast::<AfterReturningFn>()
1026                    {
1027                        let f = Arc::clone(&fn_ptr);
1028                        chain.after_returning(move |jp, val| f(jp, val));
1029                    }
1030                },
1031                AdviceType::AfterThrowing =>
1032                {
1033                    // AfterThrowing not yet supported in InterceptChain
1034                    // AfterThrowing 尚未在 InterceptChain 中支持
1035                },
1036            }
1037        }
1038
1039        chain
1040    }

1132/// Runtime interception chain that applies ordered advice around a target method.
1133/// 在目标方法周围应用有序通知的运行时拦截链。
1134///
1135/// Execution order: Before → Around → Target → After → AfterReturning
1136/// 执行顺序：Before → Around → 目标 → After → AfterReturning
1137///
1138/// # Example / 示例
1139///
1140/// ```rust,no_run,ignore
1141/// use hiver_aop::runtime::{InterceptChain, JoinPoint};
1142///
1143/// let mut chain = InterceptChain::new();
1144/// chain.before(|jp| println!("Before: {}", jp.method_name()));
1145/// chain.after(|jp| println!("After: {}", jp.method_name()));
1146///
1147/// let jp = JoinPoint::new(/* ... */);
1148/// let result = chain.invoke(jp, || Some(Arc::new(42)));
1149/// ```
1150pub struct InterceptChain
1151{
1152    /// @Before advice callbacks
1153    /// @Before 通知回调
1154    before: Vec<Arc<dyn Fn(&JoinPoint) + Send + Sync>>,
1155
1156    /// @After advice callbacks (always executed)
1157    /// @After 通知回调（始终执行）
1158    after: Vec<Arc<dyn Fn(&JoinPoint) + Send + Sync>>,
1159
1160    /// @Around advice callbacks
1161    /// @Around 通知回调
1162    around: Vec<Arc<dyn Fn(&mut ProceedingJoinPoint) + Send + Sync>>,
1163
1164    /// @AfterReturning advice callbacks (executed on success)
1165    /// @AfterReturning 通知回调（成功时执行）
1166    after_returning: Vec<Arc<dyn Fn(&JoinPoint, &Arc<dyn Any + Send + Sync>) + Send + Sync>>,
1167}
1168
1169impl InterceptChain
1170{
1171    /// Create an empty intercept chain.
1172    /// 创建空拦截链。
1173    pub fn new() -> Self
1174    {
1175        Self {
1176            before: Vec::new(),
1177            after: Vec::new(),
1178            around: Vec::new(),
1179            after_returning: Vec::new(),
1180        }
1181    }
1182
1183    /// Add @Before advice.
1184    /// 添加 @Before 通知。
1185    pub fn before(&mut self, advice: impl Fn(&JoinPoint) + Send + Sync + 'static)
1186    {
1187        self.before.push(Arc::new(advice));
1188    }
1189
1190    /// Add @After advice (always executed, like finally).
1191    /// 添加 @After 通知（始终执行，类似 finally）。
1192    pub fn after(&mut self, advice: impl Fn(&JoinPoint) + Send + Sync + 'static)
1193    {
1194        self.after.push(Arc::new(advice));
1195    }
1196
1197    /// Add @Around advice.
1198    /// 添加 @Around 通知。
1199    pub fn around(&mut self, advice: impl Fn(&mut ProceedingJoinPoint) + Send + Sync + 'static)
1200    {
1201        self.around.push(Arc::new(advice));
1202    }
1203
1204    /// Add @AfterReturning advice (executed when target returns successfully).
1205    /// 添加 @AfterReturning 通知（目标成功返回时执行）。
1206    pub fn after_returning(
1207        &mut self,
1208        advice: impl Fn(&JoinPoint, &Arc<dyn Any + Send + Sync>) + Send + Sync + 'static,
1209    )
1210    {
1211        self.after_returning.push(Arc::new(advice));
1212    }
1213
1214    /// Total number of advice in the chain.
1215    /// 链中通知总数。
1216    pub fn total(&self) -> usize
1217    {
1218        self.before.len() + self.after.len() + self.around.len() + self.after_returning.len()
1219    }
1220
1221    /// Check if the chain has no advice.
1222    /// 检查链是否没有通知。
1223    pub fn is_empty(&self) -> bool
1224    {
1225        self.total() == 0
1226    }
1227
1228    /// Execute the chain around a target method.
1229    /// 在目标方法周围执行拦截链。
1230    ///
1231    /// Execution order:
1232    /// 1. All @Before advice (in registration order)
1233    /// 2. @Around advice (if any, controls whether target runs)
1234    /// 3. Target method (if @Around allows or no @Around)
1235    /// 4. All @After advice (always, in registration order)
1236    /// 5. @AfterReturning advice (if target succeeded and returned a value)
1237    ///
1238    /// 执行顺序：
1239    /// 1. 所有 @Before 通知（按注册顺序）
1240    /// 2. @Around 通知（如果有的话，控制目标是否执行）
1241    /// 3. 目标方法（如果 @Around 允许或没有 @Around）
1242    /// 4. 所有 @After 通知（始终执行，按注册顺序）
1243    /// 5. @AfterReturning 通知（如果目标成功并返回了值）
1244    pub fn invoke<F>(&self, join_point: JoinPoint, target: F) -> InterceptResult
1245    where
1246        F: FnOnce() -> Option<Arc<dyn Any + Send + Sync>>,
1247    {
1248        // 1. Execute @Before advice
1249        // 执行 @Before 通知
1250        for advice in &self.before
1251        {
1252            advice(&join_point);
1253        }
1254
1255        // 2. Execute @Around advice (if any)
1256        // 执行 @Around 通知（如果有的话）
1257        let should_proceed = if self.around.is_empty()
1258        {
1259            true
1260        }
1261        else
1262        {
1263            let mut pjp = ProceedingJoinPoint::new(join_point.clone());
1264            for advice in &self.around
1265            {
1266                advice(&mut pjp);
1267            }
1268            pjp.is_proceeded()
1269        };
1270
1271        // 3. Execute target if @Around allows
1272        // 如果 @Around 允许，执行目标
1273        let result = if should_proceed { target() } else { None };
1274
1275        // 4. Execute @After advice (always)
1276        // 执行 @After 通知（始终执行）
1277        for advice in &self.after
1278        {
1279            advice(&join_point);
1280        }
1281
1282        // 5. Execute @AfterReturning advice (on success with value)
1283        // 执行 @AfterReturning 通知（成功返回时）
1284        if let Some(ref value) = result
1285        {
1286            for advice in &self.after_returning
1287            {
1288                advice(&join_point, value);
1289            }
1290        }
1291
1292        InterceptResult::new(result)
1293    }
1294}
