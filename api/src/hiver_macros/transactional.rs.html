7use proc_macro::TokenStream;
8use quote::quote;
9use syn::{Expr, ItemFn, parse_macro_input};

11/// #[transactional] macro implementation
12/// #[transactional]宏实现
13///
14/// This is the actual implementation of the transactional macro.
15/// The public wrapper is in lib.rs with the #[`proc_macro_attribute`] tag.
16/// 这是transactional宏的实际实现。
17/// 公共包装器在lib.rs中，带有#[`proc_macro_attribute`]标签。
18///
19/// Equivalent to Spring's @Transactional annotation.
20/// 等价于Spring的@Transactional注解。
21///
22/// # Example / 示例
23///
24/// ```rust,no_run,ignore
25/// #[transactional]
26/// async fn create_user(user: User) -> Result<User> {
27///     // Transaction logic
28/// }
29///
30/// #[transactional(propagation = "REQUIRES_NEW", isolation = "SERIALIZABLE")]
31/// async fn transfer_money(from: u64, to: u64, amount: f64) -> Result<()> {
32///     // Transaction logic
33/// }
34/// ```
35pub(crate) fn transactional_impl(attr: &TokenStream, item: TokenStream) -> TokenStream
36{
37    // Parse attribute if not empty
38    // 解析属性（如果不为空）
39    let options = if attr.is_empty()
40    {
41        TransactionalOptionsParsed::default()
42    }
43    else
44    {
45        parse_transactional_options_attr(attr)
46    };
47
48    let function = parse_macro_input!(item as ItemFn);
49
50    let fn_name = &function.sig.ident;
51    let fn_vis = &function.vis;
52    let fn_async = &function.sig.asyncness;
53    let fn_inputs = &function.sig.inputs;
54    let fn_output = &function.sig.output;
55    let fn_block = &function.block;
56    let fn_attrs = &function.attrs;
57
58    // Build propagation configuration
59    // 构建传播配置
60    let propagation_config = if let Some(prop) = &options.propagation
61    {
62        quote! { definition = #prop; }
63    }
64    else
65    {
66        quote! {}
67    };
68
69    // Build isolation configuration
70    // 构建隔离级别配置
71    let isolation_config = if let Some(iso) = &options.isolation
72    {
73        quote! { definition = #iso; }
74    }
75    else
76    {
77        quote! {}
78    };
79
80    // Build timeout configuration
81    // 构建超时配置
82    let timeout_config = if let Some(timeout) = &options.timeout_secs
83    {
84        quote! { definition = #timeout; }
85    }
86    else
87    {
88        quote! {}
89    };
90
91    // Build read_only configuration
92    // 构建只读配置
93    let readonly_config = if options.read_only
94    {
95        quote! { definition = definition.read_only(true); }
96    }
97    else
98    {
99        quote! {}
100    };
101
102    // Generate transaction wrapper code
103    // 生成事务包装代码
104    let expanded = quote! {
105        #(#fn_attrs)*
106        #fn_vis #fn_async fn #fn_name(#fn_inputs) #fn_output {
107            use hiver_tx::{TransactionManager, TransactionDefinition, Propagation, IsolationLevel};
108
109            // Get transaction manager via OnceLock lazy initialization
110            // 通过 OnceLock 懒加载获取事务管理器
111            let tx_mgr = ::hiver_tx::global_tx_manager()
112                .unwrap_or_else(|| ::std::sync::Arc::new(::hiver_tx::NoopTransactionManager));
113
114            // Create transaction definition
115            // 创建事务定义
116            let mut definition = TransactionDefinition::new(stringify!(#fn_name));
117
118            #propagation_config
119            #isolation_config
120            #timeout_config
121            #readonly_config
122
123            // Wrap the entire begin→execute→commit/rollback cycle in a task-local scope so
124            // that bind_transaction and take_transaction share the same HashMap.
125            // 将整个 begin→execute→commit/rollback 周期包裹在任务本地 scope 中，
126            // 使 bind_transaction 和 take_transaction 共享同一个 HashMap。
127            ::hiver_tx::synchronization::with_transaction_scope(async move {
128                // Begin transaction
129                // 开始事务
130                let status = match tx_mgr.begin(&definition).await {
131                    Ok(s) => s,
132                    Err(e) => return Err(e.into()),
133                };
134
135                // Execute the original function
136                // 执行原函数
137                let result = async move {
138                    #fn_block
139                }.await;
140
141                // Commit or rollback based on result
142                // 根据结果提交或回滚
143                match &result {
144                    Ok(_) => {
145                        if let Err(e) = tx_mgr.commit(status).await {
146                            return Err(e.into());
147                        }
148                    }
149                    Err(_err) => {
150                        // Simplified: always rollback on error
151                        // 简化版：错误时总是回滚
152                        let _ = tx_mgr.rollback(status).await;
153                    }
154                }
155
156                result
157            }).await
158        }
159    };
160
161    TokenStream::from(expanded)
162}

164/// Parsed transactional options
165/// 解析的transactional选项
166#[derive(Default)]
167struct TransactionalOptionsParsed
168{
169    propagation: Option<Expr>,
170    isolation: Option<Expr>,
171    timeout_secs: Option<Expr>,
172    read_only: bool,
173}
174
175/// Parse transactional options from attribute token stream
176/// 从属性token流解析transactional选项
177fn parse_transactional_options_attr(attr: &TokenStream) -> TransactionalOptionsParsed
178{
179    let mut options = TransactionalOptionsParsed::default();
180
181    // Parse the attribute as a comma-separated list of name=value pairs
182    // 将属性解析为逗号分隔的 name=value 对列表
183    let attr_str = attr.to_string();
184
185    // Simple parsing - split by comma, then by '='
186    // 简单解析 - 先按逗号分割，再按等号分割
187    for pair in attr_str.split(',')
188    {
189        let pair = pair.trim();
190        if let Some((key, value)) = pair.split_once('=')
191        {
192            let key = key.trim();
193            let value = value.trim().trim_matches('"');
194
195            match key
196            {
197                "propagation" =>
198                {
199                    let prop_expr = match value
200                    {
201                        "REQUIRED" => quote! { .propagation(Propagation::Required) },
202                        "REQUIRES_NEW" => quote! { .propagation(Propagation::RequiresNew) },
203                        "SUPPORTS" => quote! { .propagation(Propagation::Supports) },
204                        "NOT_SUPPORTED" => quote! { .propagation(Propagation::NotSupported) },
205                        "MANDATORY" => quote! { .propagation(Propagation::Mandatory) },
206                        "NEVER" => quote! { .propagation(Propagation::Never) },
207                        "NESTED" => quote! { .propagation(Propagation::Nested) },
208                        _ => quote! {},
209                    };
210                    options.propagation =
211                        Some(syn::parse2(prop_expr).unwrap_or_else(|e| panic!("{}", e)));
212                },
213                "isolation" =>
214                {
215                    let iso_expr = match value
216                    {
217                        "READ_UNCOMMITTED" =>
218                        {
219                            quote! { .isolation(IsolationLevel::ReadUncommitted) }
220                        },
221                        "READ_COMMITTED" => quote! { .isolation(IsolationLevel::ReadCommitted) },
222                        "REPEATABLE_READ" => quote! { .isolation(IsolationLevel::RepeatableRead) },
223                        "SERIALIZABLE" => quote! { .isolation(IsolationLevel::Serializable) },
224                        _ => quote! {},
225                    };
226                    options.isolation =
227                        Some(syn::parse2(iso_expr).unwrap_or_else(|e| panic!("{}", e)));
228                },
229                "timeout" | "timeout_secs" =>
230                {
231                    if let Ok(timeout_val) = value.parse::<u64>()
232                    {
233                        let timeout_expr = quote! { .timeout_secs(#timeout_val) };
234                        options.timeout_secs =
235                            Some(syn::parse2(timeout_expr).unwrap_or_else(|e| panic!("{}", e)));
236                    }
237                },
238                "read_only" =>
239                {
240                    options.read_only = value == "true" || value == "1";
241                },
242                _ =>
243                {},
244            }
245        }
246    }
247
248    options
249}
