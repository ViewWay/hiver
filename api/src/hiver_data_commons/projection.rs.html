1//! Projection support for entity-to-DTO transformation
2//! 实体到 DTO 转换的投影支持
3//!
4//! # Overview / 概述
5//!
6//! This module provides projection types for selecting specific fields
7//! from entities and transforming them into DTOs, similar to Spring Data's
8//! projection interface and SpEL-based projections.
9//! 本模块提供从实体中选择特定字段并转换为 DTO 的投影类型，
10//! 类似于 Spring Data 的 projection 接口和基于 SpEL 的投影。
11//!
12//! # Equivalent to Spring / 等价于 Spring
13//!
14//! | Hiver | Spring Data |
15//! |-------|-------------|
16//! | `Projection` trait | `Projection` interface |
17//! | `ProjectionField` | `@Value` SpEL expression |
18//! | `ProjectionTransformer` | `ProjectionFactory` |
19//!
20//! # Example / 示例
21//!
22//! ```rust,ignore
23//! use hiver_data_commons::projection::{ProjectionField, ProjectionTransformer};
24//!
25//! let fields = vec![
26//!     ProjectionField::new("name", Some("username")),
27//!     ProjectionField::new("email", None),
28//! ];
29//!
30//! // Transform entity to a HashMap-based DTO
31//! let dto = transformer.transform(&entity, &fields);
32//! ```

36/// A single field in a projection definition.
37/// 投影定义中的单个字段。
38///
39/// Describes how a source entity field maps to a projected DTO field.
40/// 描述源实体字段如何映射到投影的 DTO 字段。
41///
42/// # Example / 示例
43///
44/// ```rust
45/// use hiver_data_commons::projection::ProjectionField;
46///
47/// // Simple field mapping
48/// let field = ProjectionField::new("name", None::<&str>);
49/// assert_eq!(field.name(), "name");
50/// assert_eq!(field.alias(), "name");
51///
52/// // Field with alias
53/// let field = ProjectionField::new("firstName", Some("first_name"));
54/// assert_eq!(field.name(), "firstName");
55/// assert_eq!(field.alias(), "first_name");
56/// ```
57#[derive(Debug, Clone, PartialEq, Eq)]
58pub struct ProjectionField
59{
60    /// The source field name on the entity.
61    /// 实体上的源字段名。
62    name: String,
63
64    /// Optional expression or alias for the target DTO field.
65    /// 目标 DTO 字段的可选表达式或别名。
66    expression: Option<String>,
67
68    /// The target alias in the DTO (defaults to name if not set).
69    /// DTO 中的目标别名（未设置时默认为 name）。
70    alias: String,
71}
72
73impl ProjectionField
74{
75    /// Create a new projection field.
76    /// 创建新的投影字段。
77    ///
78    /// # Parameters / 参数
79    ///
80    /// - `name`: Source field name on the entity / 实体上的源字段名
81    /// - `alias`: Optional alias for the target field / 目标字段的可选别名
82    pub fn new(name: impl Into<String>, alias: Option<impl Into<String>>) -> Self
83    {
84        let name = name.into();
85        let alias_str = alias.map(|a| a.into());
86        Self {
87            alias: alias_str.clone().unwrap_or_else(|| name.clone()),
88            expression: alias_str,
89            name,
90        }
91    }
92
93    /// Create a projection field with a SpEL-like expression.
94    /// 创建带有 SpEL 风格表达式的投影字段。
95    ///
96    /// # Parameters / 参数
97    ///
98    /// - `name`: Source field name / 源字段名
99    /// - `expression`: Expression string (e.g., "target.firstName") / 表达式字符串
100    /// - `alias`: Target alias / 目标别名
101    pub fn with_expression(
102        name: impl Into<String>,
103        expression: impl Into<String>,
104        alias: impl Into<String>,
105    ) -> Self
106    {
107        Self {
108            name: name.into(),
109            expression: Some(expression.into()),
110            alias: alias.into(),
111        }
112    }
113
114    /// Get the source field name.
115    /// 获取源字段名。
116    pub fn name(&self) -> &str
117    {
118        &self.name
119    }
120
121    /// Get the expression, if any.
122    /// 获取表达式（如果有）。
123    pub fn expression(&self) -> Option<&str>
124    {
125        self.expression.as_deref()
126    }
127
128    /// Get the target alias.
129    /// 获取目标别名。
130    pub fn alias(&self) -> &str
131    {
132        &self.alias
133    }
134}

136/// Trait for defining a projection (DTO interface).
137/// 定义投影（DTO 接口）的 trait。
138///
139/// Implementors describe which fields should be projected from the source entity.
140/// 实现者描述应从源实体投影哪些字段。
141///
142/// This mirrors Spring Data's projection interface concept where you can
143/// define an interface with getter methods to project only specific fields.
144/// 这镜像了 Spring Data 的投影接口概念，可以定义带有 getter 方法的接口
145/// 来仅投影特定字段。
146///
147/// # Example / 示例
148///
149/// ```rust,ignore
150/// use hiver_data_commons::projection::Projection;
151///
152/// struct UserNameProjection;
153///
154/// impl Projection for UserNameProjection {
155///     fn fields() -> Vec<String> {
156///         vec!["name".to_string()]
157///     }
158/// }
159/// ```
160pub trait Projection
161{
162    /// List of field names included in this projection.
163    /// 此投影中包含的字段名列表。
164    fn fields() -> Vec<String>;
165}

167/// Transformer that converts entities into projected DTOs.
168/// 将实体转换为投影 DTO 的转换器。
169///
170/// This trait abstracts the mechanism of reading entity fields
171/// and mapping them to a target projection structure.
172/// 此 trait 抽象了读取实体字段并将其映射到目标投影结构的机制。
173///
174/// # Example / 示例
175///
176/// ```rust,ignore
177/// use hiver_data_commons::projection::ProjectionTransformer;
178///
179/// let dto: HashMap<String, String> = transformer.transform(&entity, &fields);
180/// ```
181pub trait ProjectionTransformer
182{
183    /// Source entity type.
184    /// 源实体类型。
185    type Source;
186
187    /// Transform a source entity into a map of projected field values.
188    /// 将源实体转换为投影字段值的映射。
189    fn transform(
190        &self,
191        source: &Self::Source,
192        fields: &[ProjectionField],
193    ) -> HashMap<String, String>;
194}

196/// A DTO backed by a `HashMap<String, String>`.
197/// 由 `HashMap<String, String>` 支持的 DTO。
198///
199/// Useful for dynamic projections where the DTO shape is not known at compile time.
200/// 适用于 DTO 形状在编译时未知的动态投影。
201///
202/// # Example / 示例
203///
204/// ```rust
205/// use hiver_data_commons::projection::DtoProjection;
206///
207/// let mut dto = DtoProjection::new();
208/// dto.set("name", "Alice");
209/// dto.set("email", "alice@example.com");
210///
211/// assert_eq!(dto.get("name"), Some("Alice"));
212/// assert_eq!(dto.get("email"), Some("alice@example.com"));
213/// assert_eq!(dto.get("missing"), None);
214/// ```
215#[derive(Debug, Clone, Default)]
216pub struct DtoProjection
217{
218    /// Projected field values.
219    /// 投影的字段值。
220    data: HashMap<String, String>,
221}
222
223impl DtoProjection
224{
225    /// Create a new empty DTO projection.
226    /// 创建新的空 DTO 投影。
227    pub fn new() -> Self
228    {
229        Self::default()
230    }
231
232    /// Set a field value.
233    /// 设置字段值。
234    pub fn set(&mut self, key: impl Into<String>, value: impl Into<String>)
235    {
236        self.data.insert(key.into(), value.into());
237    }
238
239    /// Get a field value.
240    /// 获取字段值。
241    pub fn get(&self, key: &str) -> Option<&str>
242    {
243        self.data.get(key).map(|s| s.as_str())
244    }
245
246    /// Check if a field exists.
247    /// 检查字段是否存在。
248    pub fn contains(&self, key: &str) -> bool
249    {
250        self.data.contains_key(key)
251    }
252
253    /// Get all field names.
254    /// 获取所有字段名。
255    pub fn keys(&self) -> impl Iterator<Item = &str>
256    {
257        self.data.keys().map(|s| s.as_str())
258    }
259
260    /// Get the number of fields.
261    /// 获取字段数量。
262    pub fn len(&self) -> usize
263    {
264        self.data.len()
265    }
266
267    /// Check if the DTO is empty.
268    /// 检查 DTO 是否为空。
269    pub fn is_empty(&self) -> bool
270    {
271        self.data.is_empty()
272    }
273
274    /// Convert to the underlying HashMap.
275    /// 转换为底层 HashMap。
276    pub fn into_map(self) -> HashMap<String, String>
277    {
278        self.data
279    }
280}

282impl Projection for DtoProjection
283{
284    fn fields() -> Vec<String>
285    {
286        // DtoProjection is dynamic; fields are determined at runtime.
287        // DtoProjection 是动态的；字段在运行时确定。
288        Vec::new()
289    }
290}

292/// A closed projection backed by a typed struct.
293/// 由类型化结构体支持的封闭投影。
294///
295/// Unlike [`DtoProjection`], this provides compile-time field safety
296/// by using a known struct type.
297/// 与 [`DtoProjection`] 不同，它通过使用已知结构体类型提供编译时字段安全性。
298///
299/// # Example / 示例
300///
301/// ```rust,ignore
302/// use hiver_data_commons::projection::ClosedProjection;
303///
304/// #[derive(Debug, Clone)]
305/// struct NameDto {
306///     name: String,
307/// }
308///
309/// // Transform entity -> NameDto using projection fields
310/// let dto: NameDto = ClosedProjection::from_entity(&entity, &fields);
311/// ```
312pub trait ClosedProjection: Sized
313{
314    /// Create a closed projection from an entity using the given fields.
315    /// 使用给定字段从实体创建封闭投影。
316    fn from_entity(entity: &impl crate::Entity, fields: &[ProjectionField]) -> Option<Self>;
317}
