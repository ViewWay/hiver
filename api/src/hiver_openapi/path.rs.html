158/// Path operation
159/// 路径操作
160///
161/// Combines a path with a method and operation.
162/// 组合路径、方法和操作。
163#[derive(Debug, Clone)]
164pub struct PathOperation
165{
166    /// Path (e.g., "/users/{id}")
167    /// 路径（例如 "/users/{id}"）
168    pub path: String,
169
170    /// Method
171    /// 方法
172    pub method: PathMethod,
173
174    /// Operation
175    /// 操作
176    pub operation: Operation,
177}
178
179impl PathOperation
180{
181    /// Create a new path operation
182    /// 创建新路径操作
183    pub fn new(path: impl Into<String>, method: PathMethod, operation: Operation) -> Self
184    {
185        Self {
186            path: path.into(),
187            method,
188            operation,
189        }
190    }
191
192    /// Create a GET operation
193    /// 创建GET操作
194    pub fn get(path: impl Into<String>, operation: Operation) -> Self
195    {
196        Self::new(path, PathMethod::Get, operation)
197    }
198
199    /// Create a POST operation
200    /// 创建POST操作
201    pub fn post(path: impl Into<String>, operation: Operation) -> Self
202    {
203        Self::new(path, PathMethod::Post, operation)
204    }
205
206    /// Create a PUT operation
207    /// 创建PUT操作
208    pub fn put(path: impl Into<String>, operation: Operation) -> Self
209    {
210        Self::new(path, PathMethod::Put, operation)
211    }
212
213    /// Create a DELETE operation
214    /// 创建DELETE操作
215    pub fn delete(path: impl Into<String>, operation: Operation) -> Self
216    {
217        Self::new(path, PathMethod::Delete, operation)
218    }
219
220    /// Create a PATCH operation
221    /// 创建PATCH操作
222    pub fn patch(path: impl Into<String>, operation: Operation) -> Self
223    {
224        Self::new(path, PathMethod::Patch, operation)
225    }

228/// Schema reference
229/// 模式引用
230#[derive(Debug, Clone, Serialize, Deserialize)]
231pub struct SchemaRef
232{
233    /// Reference
234    /// 引用
235    #[serde(rename = "$ref")]
236    pub ref_: String,
237}
238
239impl SchemaRef
240{
241    /// Create a new schema reference
242    /// 创建新模式引用
243    pub fn new(ref_: impl Into<String>) -> Self
244    {
245        Self { ref_: ref_.into() }
246    }
247
248    /// Reference to component schema
249    /// 引用组件模式
250    pub fn component(name: impl Into<String>) -> Self
251    {
252        Self::new(format!("#/components/schemas/{}", name.into()))
253    }
254}

256/// Components
257/// 组件
258#[derive(Debug, Clone, Serialize, Deserialize, Default)]
259pub struct Components
260{
261    /// Schemas
262    /// 模式列表
263    #[serde(skip_serializing_if = "Option::is_none")]
264    pub schemas: Option<HashMap<String, Schema>>,
265
266    /// Responses
267    /// 响应列表
268    #[serde(skip_serializing_if = "Option::is_none")]
269    pub responses: Option<HashMap<String, Response>>,
270
271    /// Parameters
272    /// 参数列表
273    #[serde(skip_serializing_if = "Option::is_none")]
274    pub parameters: Option<HashMap<String, Parameter>>,
275
276    /// Examples
277    /// 示例列表
278    #[serde(skip_serializing_if = "Option::is_none")]
279    pub examples: Option<HashMap<String, crate::operation::Example>>,
280
281    /// Request bodies
282    /// 请求体列表
283    #[serde(skip_serializing_if = "Option::is_none")]
284    pub request_bodies: Option<HashMap<String, crate::RequestBody>>,
285
286    /// Security schemes
287    /// 安全方案列表
288    #[serde(skip_serializing_if = "Option::is_none")]
289    pub security_schemes: Option<HashMap<String, crate::SecurityScheme>>,
290
291    /// Links
292    /// 链接列表
293    #[serde(skip_serializing_if = "Option::is_none")]
294    pub links: Option<HashMap<String, serde_json::Value>>,
295
296    /// Callbacks
297    /// 回调列表
298    #[serde(skip_serializing_if = "Option::is_none")]
299    pub callbacks: Option<HashMap<String, serde_json::Value>>,
300}
301
302impl Components
303{
304    /// Create a new components
305    /// 创建新组件
306    pub fn new() -> Self
307    {
308        Self::default()
309    }
310
311    /// Add schema
312    /// 添加模式
313    pub fn add_schema(&mut self, name: impl Into<String>, schema: Schema) -> &mut Self
314    {
315        self.schemas
316            .get_or_insert_with(HashMap::new)
317            .insert(name.into(), schema);
318        self
319    }
320
321    /// Add response
322    /// 添加响应
323    pub fn add_response(&mut self, name: impl Into<String>, response: Response) -> &mut Self
324    {
325        self.responses
326            .get_or_insert_with(HashMap::new)
327            .insert(name.into(), response);
328        self
329    }
330
331    /// Add parameter
332    /// 添加参数
333    pub fn add_parameter(&mut self, name: impl Into<String>, parameter: Parameter) -> &mut Self
334    {
335        self.parameters
336            .get_or_insert_with(HashMap::new)
337            .insert(name.into(), parameter);
338        self
339    }
340}

361                .add_response("200", Response::ok("User found")),
