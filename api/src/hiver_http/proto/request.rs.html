10/// Parse an HTTP/1.1 request from bytes
11/// 从字节解析 HTTP/1.1 请求
12///
13/// # Arguments / 参数
14///
15/// * `data` - The raw bytes to parse / 要解析的原始字节
16/// * `ctx` - The connection context / 连接上下文
17///
18/// # Returns / 返回
19///
20/// * `Ok((request, bytes_used))` - The parsed request and number of bytes consumed
21/// * `Err(Error)` - Parse error
22///
23/// # Example / 示例
24///
25/// ```rust,no_run,ignore
26/// use hiver_http::proto::parse_request;
27/// use hiver_http::proto::context::ConnectionContext;
28///
29/// let data = b"GET / HTTP/1.1\r\nHost: example.com\r\n\r\n";
30/// let ctx = ConnectionContext::new();
31/// let (request, used) = parse_request(data, &ctx)?;
32/// ```
33pub fn parse_request(data: &[u8], ctx: &ConnectionContext) -> Result<(Request, usize)>
34{
35    let mut headers = [httparse::EMPTY_HEADER; 64];
36    let mut req = HttparseRequest::new(&mut headers);
37
38    // Parse the headers
39    let result = req.parse(data).map_err(|e| match e
40    {
41        httparse::Error::HeaderName => Error::InvalidRequest("Invalid header name".to_string()),
42        httparse::Error::HeaderValue => Error::InvalidRequest("Invalid header value".to_string()),
43        httparse::Error::NewLine => Error::InvalidRequest("Invalid newline".to_string()),
44        httparse::Error::Status => Error::InvalidRequest("Invalid status line".to_string()),
45        httparse::Error::Token => Error::InvalidRequest("Invalid token".to_string()),
46        httparse::Error::Version => Error::InvalidRequest("Invalid HTTP version".to_string()),
47        httparse::Error::TooManyHeaders => Error::InvalidRequest("Too many headers".to_string()),
48    })?;
49
50    let bytes_used = match result
51    {
52        httparse::Status::Complete(n) => n,
53        httparse::Status::Partial =>
54        {
55            return Err(Error::IncompleteRequest);
56        },
57    };
58
59    // Extract method and path
60    let method = req
61        .method
62        .ok_or_else(|| Error::InvalidRequest("Missing method".to_string()))?;
63
64    let path = req
65        .path
66        .ok_or_else(|| Error::InvalidRequest("Missing path".to_string()))?;
67
68    // Parse version
69    let version = match req.version
70    {
71        Some(0) => HttpVersion::Http10,
72        Some(1) | None => HttpVersion::Http11,
73        Some(v) => return Err(Error::InvalidRequest(format!("Unsupported HTTP version: {}", v))),
74    };
75
76    // Build the http::Request
77    let mut http_builder = http::Request::builder().method(method).uri(path);
78
79    // Add headers
80    for header in req.headers.iter()
81    {
82        http_builder = http_builder.header(header.name, header.value);
83    }
84
85    // Build the request with empty body for now (body will be read separately)
86    let http_request = http_builder
87        .body(Body::empty())
88        .map_err(|e| Error::InvalidRequest(format!("Failed to build request: {}", e)))?;
89
90    // Convert to our Request type
91    let mut request = Request::new(http_request);
92
93    // Extract the body if present
94    if bytes_used < data.len()
95    {
96        let body_data = data.get(bytes_used..).unwrap_or(&[]);
97        if !body_data.is_empty()
98        {
99            // Update the request with the actual body
100            let content_length = request
101                .header("content-length")
102                .and_then(|v| v.parse::<usize>().ok())
103                .unwrap_or(body_data.len());
104
105            let actual_body_len = body_data.len().min(content_length);
106            let body_slice = body_data.get(..actual_body_len).unwrap_or(&[]);
107            let body = Body::from(Bytes::copy_from_slice(body_slice));
108
109            // Rebuild with body
110            let mut http_builder = http::Request::builder()
111                .method(request.inner().method().clone())
112                .uri(request.inner().uri().clone());
113
114            for (name, value) in request.inner().headers()
115            {
116                http_builder = http_builder.header(name, value);
117            }
118
119            let http_request = http_builder.body(body).map_err(|e| {
120                Error::InvalidRequest(format!("Failed to build request with body: {}", e))
121            })?;
122
123            request = Request::new(http_request);
124        }
125    }
126
127    // Update context based on headers
128    let mut updated_ctx = ctx.clone();
129    updated_ctx.set_version(version);
130    updated_ctx.update_keep_alive_from_header(
131        req.headers
132            .iter()
133            .find(|h| h.name.eq_ignore_ascii_case("connection"))
134            .map(|h| std::str::from_utf8(h.value).unwrap_or("")),
135    );
136    // Store context in request extensions if needed
137
138    Ok((request, bytes_used))
139}

141/// HTTP request parser with state
142/// 带状态的 HTTP 请求解析器
143#[derive(Debug)]
144pub struct RequestParser
145{
146    /// Buffer for incoming data
147    buffer: BytesMut,
148    /// Connection context
149    ctx: ConnectionContext,
150}
151
152impl RequestParser
153{
154    /// Create a new request parser
155    /// 创建新的请求解析器
156    pub fn new() -> Self
157    {
158        Self {
159            buffer: BytesMut::with_capacity(8192),
160            ctx: ConnectionContext::new(),
161        }
162    }
163
164    /// Create a new request parser with custom context
165    /// 使用自定义上下文创建新的请求解析器
166    pub fn with_context(ctx: ConnectionContext) -> Self
167    {
168        Self {
169            buffer: BytesMut::with_capacity(ctx.max_buffer_size()),
170            ctx,
171        }
172    }
173
174    /// Feed data to the parser
175    /// 向解析器提供数据
176    pub fn feed(&mut self, data: &[u8]) -> Result<()>
177    {
178        if self.buffer.len() + data.len() > self.ctx.max_buffer_size()
179        {
180            return Err(Error::InvalidRequest("Buffer overflow".to_string()));
181        }
182        self.buffer.extend_from_slice(data);
183        Ok(())
184    }
185
186    /// Try to parse a request from the buffered data
187    /// 尝试从缓冲数据解析请求
188    ///
189    /// Returns `Ok(None)` if more data is needed.
190    pub fn parse(&mut self) -> Result<Option<(Request, usize)>>
191    {
192        match parse_request(&self.buffer, &self.ctx)
193        {
194            Ok((req, used)) =>
195            {
196                self.buffer.advance(used);
197                Ok(Some((req, used)))
198            },
199            Err(Error::IncompleteRequest) => Ok(None),
200            Err(e) => Err(e),
201        }
202    }
203
204    /// Get the current buffer length
205    /// 获取当前缓冲区长度
206    pub fn buffered_len(&self) -> usize
207    {
208        self.buffer.len()
209    }
210
211    /// Clear the buffer
212    /// 清空缓冲区
213    pub fn clear(&mut self)
214    {
215        self.buffer.clear();
216    }
217
218    /// Get the connection context
219    /// 获取连接上下文
220    pub fn context(&self) -> &ConnectionContext
221    {
222        &self.ctx
223    }
224
225    /// Get mutable reference to the connection context
226    /// 获取连接上下文的可变引用
227    pub fn context_mut(&mut self) -> &mut ConnectionContext
228    {
229        &mut self.ctx
230    }
231}
232
233impl Default for RequestParser
234{
235    fn default() -> Self
236    {
237        Self::new()
238    }
239}

254    #[test]
255    fn test_parse_simple_get()
256    {
257        let data = b"GET / HTTP/1.1\r\nHost: example.com\r\n\r\n";
258        let ctx = ConnectionContext::new();
259        let (req, used) = parse_request(data, &ctx).unwrap();
260        assert_eq!(used, data.len());
261        assert_eq!(req.method(), Method::GET);
262        assert_eq!(req.path(), "/");
263    }
264
265    #[test]
266    fn test_parse_post_with_body()
267    {
268        let data = b"POST /api HTTP/1.1\r\nHost: example.com\r\nContent-Length: 5\r\n\r\nhello";
269        let ctx = ConnectionContext::new();
270        let (req, _used) = parse_request(data, &ctx).unwrap();
271        assert_eq!(req.method(), Method::POST);
272        assert_eq!(req.path(), "/api");
273    }
274
275    #[test]
276    fn test_incomplete_request()
277    {
278        let data = b"GET / HTTP/1.1\r\nHost: example.com";
279        let ctx = ConnectionContext::new();
280        let result = parse_request(data, &ctx);
281        assert!(matches!(result, Err(Error::IncompleteRequest)));
282    }
