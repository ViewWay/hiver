9/// Component scanner (equivalent to @`ComponentScan`)
10/// 组件扫描器（等价于 @`ComponentScan`）
11pub struct ComponentScanner
12{
13    base_packages: Vec<String>,
14}
15
16impl ComponentScanner
17{
18    /// Create a new scanner
19    /// 创建新扫描器
20    pub fn new() -> Self
21    {
22        Self {
23            base_packages: Vec::new(),
24        }
25    }
26
27    /// Add a base package to scan
28    /// 添加要扫描的基础包
29    ///
30    /// # Example / 示例
31    ///
32    /// ```rust,no_run,ignore
33    /// let scanner = ComponentScanner::new()
34    ///     .scan_package("com.example");
35    /// ```
36    pub fn scan_package(mut self, package: impl Into<String>) -> Self
37    {
38        self.base_packages.push(package.into());
39        self
40    }
41
42    /// Scan for components and register them
43    /// 扫描组件并注册它们
44    ///
45    /// Note: In Rust, true runtime component scanning is not possible like in Java.
46    /// Instead, this framework uses proc-macros for compile-time component registration.
47    /// Use the `#[hiver_macros::component]` attribute to register components at compile time.
48    ///
49    /// 注意：在Rust中，像Java那样的真正运行时组件扫描是不可能的。
50    /// 相反，此框架使用proc宏进行编译时组件注册。
51    /// 使用 `#[hiver_macros::component]` 属性在编译时注册组件。
52    ///
53    /// # Example / 示例
54    ///
55    /// ```rust,no_run,ignore
56    /// use hiver_core::container::ComponentScanner;
57    /// use hiver_macros::component;
58    ///
59    /// #[component]
60    /// struct MyService {
61    ///     // Dependencies are automatically injected
62    /// }
63    /// }
64    ///
65    /// // Components are collected at compile time and registered automatically
66    /// // 组件在编译时被收集并自动注册
67    /// ```
68    pub fn scan(&self, _context: &mut ApplicationContext) -> Result<()>
69    {
70        // Component scanning in Rust is done at compile time via proc-macros
71        // The `#[component]` macro generates registration code
72        // 在Rust中，组件扫描通过proc宏在编译时完成
73        // `#[component]` 宏生成注册代码
74        //
75        // This method is a no-op at runtime but exists for API compatibility
76        // with Spring's @ComponentScan pattern
77        // 此方法在运行时是空操作，但存在是为了与Spring的@ComponentScan模式API兼容
78        Ok(())
79    }
80

82    /// 注册组件类型（用于proc宏生成的代码）
83    ///

85    /// This is not intended to be called manually.

92    ) -> Result<()>

95        // proc宏将为每个组件生成对register_bean的调用
96        Ok(())
97    }
