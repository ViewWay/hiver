13/// HTTP Status Codes
14/// HTTP 状态码
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
16pub struct StatusCode(u16);

238/// Returns `StatusCode::OK` (200) as the default.
239/// 返回 `StatusCode::OK` (200) 作为默认值。
240impl Default for StatusCode
241{
242    fn default() -> Self
243    {
244        StatusCode::OK
245    }
246}

248/// Formats the status code as `"CODE Reason"`, e.g. `"404 Not Found"`.
249/// 将状态码格式化为 `"CODE 原因短语"`，例如 `"404 Not Found"`。
250impl fmt::Display for StatusCode
251{
252    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
253    {
254        if let Some(reason) = self.canonical_reason()
255        {
256            write!(f, "{} {}", self.0, reason)
257        }
258        else
259        {
260            write!(f, "{}", self.0)
261        }
262    }
263}

265/// Converts a `u16` into a `StatusCode` without validation.
266/// 将 `u16` 转换为 `StatusCode`，不进行验证。
267impl From<u16> for StatusCode
268{
269    fn from(code: u16) -> Self
270    {
271        StatusCode(code)
272    }
