1//! Spring Boot Style Logging Module
2//! Spring Boot 风格日志模块
3//!
4//! # Overview / 概述
5//!
6//! This module provides Spring Boot-style structured logging functionality.
7//! 本模块提供 Spring Boot 风格的结构化日志功能。
8//!
9//! # Equivalent to Spring Boot / 等价于 Spring Boot
10//!
11//! - SLF4J + Logback → tracing + tracing-subscriber
12//! - Logger → `tracing::info/warn/error/debug/trace`!
13//! - @Slf4j → #[`hiver_observability::logger`] macro
14//!
15//! # Example / 示例
16//!
17//! ```rust,no_run,ignore
18//! use hiver_observability::log::{Logger, LoggerFactory};
19//!
20//! // Get logger (equivalent to LoggerFactory.getLogger(MyClass.class))
21//! let log = LoggerFactory::get("my::module");
22//!
23//! // Log at different levels
24//! log.error("Something went wrong: {}", error);
25//! log.warn("Warning: deprecated API used");
26//! log.info("Application started");
27//! log.debug("Debugging: variable = {}", value);
28//! log.trace("Detailed trace information");
29//! ```
30//!
31//! # Configuration / 配置
32//!
33//! Environment variables:
34//! - `HIVER_LOG_LEVEL`: Global log level (TRACE, DEBUG, INFO, WARN, ERROR)
35//! - `HIVER_LOG_FORMAT`: log format (pretty, json, compact)
36//! - `HIVER_LOG_FILE`: Log file path (e.g., logs/application.log)
37//! - `HIVER_LOG_MAX_FILES`: Maximum number of log files to keep
38//! - `HIVER_LOG_MAX_SIZE`: Maximum size of each log file (in MB)

57/// Spring Boot log levels
58/// Spring Boot 日志级别
59#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
60pub enum LogLevel
61{
62    /// Trace level - most detailed
63    /// TRACE 级别 - 最详细
64    /// TRACE: 追踪信息, 比DEBUG更细粒度的信息事件(除非有特殊用意，否则请使用DEBUG级别替代)
65    Trace = 0,
66    /// Debug level
67    /// DEBUG 级别
68    /// DEBUG: 调试信息, 需要调试时候的关键信息打印
69    Debug = 1,
70    /// Info level (default)
71    /// INFO 级别（默认）
72    /// INFO: 普通信息, 用于记录应用程序正常运行时的一些信息, 例如系统启动完成、请求处理完成等
73    Info = 2,
74    /// Warning level
75    /// WARN 级别
76    /// WARN: 警告信息, 不影响使用, 但需要注意的问题
77    Warn = 3,
78    /// Error level
79    /// ERROR 级别
80    /// ERROR: 错误信息, 级别较高的错误日志信息, 但仍然不影响系统的继续运行
81    Error = 4,
82    /// Off - no logging
83    /// OFF - 不记录日志
84    /// OFF: 不记录日志
85    Off = 5,
86}

138/// Format a log record using a custom pattern
139/// 使用自定义模式格式化日志记录
140///
141/// # Supported Placeholders / 支持的占位符
142///
143/// | Placeholder | Meaning | Example |
144/// |------------|---------|---------|
145/// | `%d` | Date/time | `2024-01-30 10:30:45` |
146/// | `%t` | Thread ID | `12345` |
147/// | `%p` | Priority/level | `INFO` |
148/// | `%c` | Category/logger name | `my.app.Service` |
149/// | `%m` | Message | `User logged in` |
150/// | `%n` | Newline | (actual newline) |
151/// | `%%` | Percent sign | `%` |
152/// | `%X{key}` | MDC value | (from MDC map) |
153///
154/// # Example / 示例
155///
156/// ```text
157/// # "2024-01-30 10:30:45 [12345] INFO my.app.Service - User logged in"
158/// ```
159pub fn format_pattern(
160    pattern: &str,
161    timestamp: &str,
162    thread_id: u64,
163    level: &LogLevel,
164    target: &str,
165    message: &str,
166) -> String
167{
168    let mut chars = pattern.chars().peekable();
169    let mut formatted = String::new();
170
171    while let Some(c) = chars.next()
172    {
173        if c == '%'
174        {
175            match chars.next()
176            {
177                Some('d') =>
178                {
179                    formatted.push_str(timestamp);
180                },
181                Some('t') =>
182                {
183                    formatted.push_str(&thread_id.to_string());
184                },
185                Some('p') =>
186                {
187                    formatted.push_str(&level.to_string());
188                },
189                Some('c') =>
190                {
191                    formatted.push_str(target);
192                },
193                Some('m') =>
194                {
195                    formatted.push_str(message);
196                },
197                Some('n') =>
198                {
199                    formatted.push('\n');
200                },
201                Some('%') | None =>
202                {
203                    formatted.push('%');
204                },
205                Some('X') =>
206                {
207                    // MDC placeholder %X{key}
208                    // MDC 占位符 %X{key}
209                    if chars.next() == Some('{')
210                    {
211                        let mut key = String::new();
212                        while let Some(&ch) = chars.peek()
213                        {
214                            if ch == '}'
215                            {
216                                break;
217                            }
218                            key.push(ch);
219                            chars.next();
220                        }
221                        if chars.next() == Some('}')
222                        {
223                            // For now, just show [key] placeholder for MDC values
224                            // 未来：实际从 MDC map 获取值
225                            let _ = write!(formatted, "[{}]", key);
226                        }
227                    }
228                },
229                Some(other) =>
230                {
231                    formatted.push('%');
232                    formatted.push(other);
233                },
234            }
235        }
236        else
237        {
238            formatted.push(c);
239        }
240    }
241
242    formatted
243}

277/// Log output mode (controls verbosity and performance)
278/// 日志输出模式（控制详细程度和性能）
279///
280/// # Verbose vs Simple / 详细模式 vs 精简模式
281///
282/// | Feature | Verbose | Simple |
283/// |---------|---------|--------|
284/// | Timestamp | ✅ ISO 8601 | ❌ None |
285/// | PID | ✅ Yes | ❌ None |
286/// | Thread | ✅ Yes | ❌ None |
287/// | Target | ✅ Shortened | ✅ Shortened |
288/// | Level | ✅ Colored | ✅ Colored |
289/// | File location | ✅ ERROR/WARN | ❌ None |
290/// | Performance | Baseline | ~30% faster |
291///
292/// # Usage / 使用场景
293///
294/// - **Verbose**: Development, debugging, local testing
295/// - **Simple**: Production, high-throughput scenarios
296#[derive(Debug, Clone, Copy, PartialEq, Eq)]
297pub enum LogMode
298{
299    /// Verbose mode with full context (development)
300    /// 详细模式，包含完整上下文（开发环境）
301    Verbose,
302
303    /// Simple mode with minimal overhead (production)
304    /// 精简模式，最小开销（生产环境）
305    Simple,
306}
307
308impl LogMode
309{
310    /// Parse from string
311    /// 从字符串解析
312    pub fn from_str(s: &str) -> Option<Self>
313    {
314        match s.to_lowercase().as_str()
315        {
316            "verbose" | "full" | "detailed" => Some(LogMode::Verbose),
317            "simple" | "compact" | "minimal" => Some(LogMode::Simple),
318            _ => None,
319        }
320    }
321
322    /// Get default mode for given profile
323    /// 根据配置文件获取默认模式
324    pub fn from_profile(profile: Option<&str>) -> Self
325    {
326        match profile
327        {
328            Some("prod" | "production") => LogMode::Simple,
329            _ => LogMode::Verbose, // Default to verbose for unknown profiles
330        }
331    }
332}

415impl Default for LoggerConfig
416{
417    fn default() -> Self
418    {
419        // Get profile from environment
420        let profile = std::env::var("HIVER_PROFILE").ok();
421
422        // Get mode from environment or derive from profile
423        let mode = if let Ok(mode_str) = std::env::var("HIVER_LOG_MODE")
424        {
425            LogMode::from_str(&mode_str)
426                .unwrap_or_else(|| LogMode::from_profile(profile.as_deref()))
427        }
428        else
429        {
430            LogMode::from_profile(profile.as_deref())
431        };
432
433        let level = std::env::var("HIVER_LOG_LEVEL")
434            .ok()
435            .and_then(|s| LogLevel::from_str(&s))
436            .unwrap_or(LogLevel::Info);
437
438        let format = std::env::var("HIVER_LOG_FORMAT")
439            .ok()
440            .and_then(|s| LogFormat::from_str(&s))
441            .unwrap_or(LogFormat::Pretty);
442
443        let file_path = std::env::var("HIVER_LOG_FILE").ok();
444
445        let rotation = match std::env::var("HIVER_LOG_ROTATION")
446            .unwrap_or_default()
447            .to_lowercase()
448            .as_str()
449        {
450            "hourly" => LogRotation::Hourly,
451            "minutely" => LogRotation::Minutely,
452            _ => LogRotation::Daily,
453        };
454
455        let max_files = std::env::var("HIVER_LOG_MAX_FILES")
456            .ok()
457            .and_then(|s| s.parse().ok())
458            .unwrap_or(7);
459
460        LoggerConfig {
461            level,
462            format,
463            mode,
464            profile,
465            file_path,
466            with_thread: matches!(mode, LogMode::Verbose), // Verbose mode includes thread
467            with_file: matches!(mode, LogMode::Verbose),   // Verbose mode includes file
468            with_target: true,
469            rotation,
470            max_files,
471            custom_pattern: None,
472        }
473    }
474}

480impl Logger
481{
482    /// Initialize the global logger with default configuration
483    /// 使用默认配置初始化全局日志
484    ///
485    /// # Example / 示例
486    ///
487    /// ```rust,no_run,ignore
488    /// use hiver_observability::log::Logger;
489    ///
490    /// Logger::init().unwrap();
491    /// ```
492    pub fn init() -> Result<(), Box<dyn std::error::Error>>
493    {
494        Logger::init_with_config(&LoggerConfig::default())
495    }
496
497    /// Initialize the global logger with custom configuration
498    /// 使用自定义配置初始化全局日志
499    ///
500    /// # Example / 示例
501    ///
502    /// ```rust,no_run,ignore
503    /// use hiver_observability::log::{Logger, LoggerConfig, LogLevel, LogFormat};
504    ///
505    /// let config = LoggerConfig {
506    ///     level: LogLevel::Debug,
507    ///     format: LogFormat::Pretty,
508    ///     file_path: Some("logs/application.log".to_string()),
509    ///     ..Default::default()
510    /// };
511    ///
512    /// Logger::init_with_config(config).unwrap();
513    /// ```
514    pub fn init_with_config(config: &LoggerConfig) -> Result<(), Box<dyn std::error::Error>>
515    {
516        let env_filter = create_env_filter(config.level);
517
518        // Choose formatter based on LogMode first, then LogFormat
519        // 首先根据 LogMode 选择格式化器，然后根据 LogFormat
520        match config.mode
521        {
522            LogMode::Simple =>
523            {
524                // Simple mode - use SimpleFormatter for high performance
525                // 精简模式 - 使用 SimpleFormatter 以获得高性能
526                #[cfg(feature = "hiver-format")]
527                {
528                    let fmt_layer = fmt::layer()
529                        .with_target(config.with_target)
530                        .event_format(SimpleFormatter::new());
531
532                    if let Some(ref path) = config.file_path
533                    {
534                        let file_appender = create_file_appender(path, config.rotation)?;
535                        Registry::default()
536                            .with(env_filter)
537                            .with(fmt_layer)
538                            .with(fmt::layer().with_writer(file_appender).compact())
539                            .try_init()?;
540                    }
541                    else
542                    {
543                        Registry::default()
544                            .with(env_filter)
545                            .with(fmt_layer)
546                            .try_init()?;
547                    }
548                }
549                #[cfg(not(feature = "hiver-format"))]
550                {
551                    // Fallback to standard compact
552                    let fmt_layer = fmt::layer().with_target(config.with_target).compact();
553
554                    if let Some(ref path) = config.file_path
555                    {
556                        let file_appender = create_file_appender(path, config.rotation)?;
557                        Registry::default()
558                            .with(env_filter)
559                            .with(fmt_layer)
560                            .with(fmt::layer().with_writer(file_appender).compact())
561                            .try_init()?;
562                    }
563                    else
564                    {
565                        Registry::default()
566                            .with(env_filter)
567                            .with(fmt_layer)
568                            .try_init()?;
569                    }
570                }
571            },
572            LogMode::Verbose =>
573            {
574                // Verbose mode - full format based on LogFormat
575                // 详细模式 - 根据 LogFormat 使用完整格式
576                match config.format
577                {
578                    LogFormat::Pretty =>
579                    {
580                        #[cfg(feature = "hiver-format")]
581                        {
582                            let mut fmt_layer = fmt::layer()
583                                .with_file(config.with_file)
584                                .with_line_number(config.with_file)
585                                .with_target(config.with_target)
586                                .event_format(HiverFormatter::new());
587                            fmt_layer.set_span_events(FmtSpan::CLOSE);
588
589                            if let Some(ref path) = config.file_path
590                            {
591                                let file_appender = create_file_appender(path, config.rotation)?;
592                                let (non_blocking, _guard) =
593                                    tracing_appender::non_blocking(file_appender);
594
595                                let file_layer = fmt::layer()
596                                    .with_file(config.with_file)
597                                    .with_line_number(config.with_file)
598                                    .with_target(config.with_target)
599                                    .with_writer(non_blocking)
600                                    .compact();
601
602                                Registry::default()
603                                    .with(env_filter)
604                                    .with(fmt_layer)
605                                    .with(file_layer)
606                                    .try_init()?;
607                            }
608                            else
609                            {
610                                Registry::default()
611                                    .with(env_filter)
612                                    .with(fmt_layer)
613                                    .try_init()?;
614                            }
615                        }
616                        #[cfg(not(feature = "hiver-format"))]
617                        {
618                            let mut fmt_layer = fmt::layer()
619                                .with_file(config.with_file)
620                                .with_line_number(config.with_file)
621                                .with_target(config.with_target)
622                                .compact();
623                            fmt_layer.set_span_events(FmtSpan::CLOSE);
624
625                            if let Some(ref path) = config.file_path
626                            {
627                                let file_appender = create_file_appender(path, config.rotation)?;
628                                let (non_blocking, _guard) =
629                                    tracing_appender::non_blocking(file_appender);
630
631                                let file_layer = fmt::layer()
632                                    .with_file(config.with_file)
633                                    .with_line_number(config.with_file)
634                                    .with_target(config.with_target)
635                                    .with_writer(non_blocking)
636                                    .compact();
637
638                                Registry::default()
639                                    .with(env_filter)
640                                    .with(fmt_layer)
641                                    .with(file_layer)
642                                    .try_init()?;
643                            }
644                            else
645                            {
646                                Registry::default()
647                                    .with(env_filter)
648                                    .with(fmt_layer)
649                                    .try_init()?;
650                            }
651                        }
652                    },
653                    LogFormat::Compact =>
654                    {
655                        let mut fmt_layer = fmt::layer()
656                            .with_file(config.with_file)
657                            .with_line_number(config.with_file)
658                            .with_target(config.with_target)
659                            .compact();
660                        fmt_layer.set_span_events(FmtSpan::CLOSE);
661
662                        if let Some(ref path) = config.file_path
663                        {
664                            let file_appender = create_file_appender(path, config.rotation)?;
665
666                            let file_layer = fmt::layer()
667                                .with_file(config.with_file)
668                                .with_line_number(config.with_file)
669                                .with_target(config.with_target)
670                                .with_writer(file_appender)
671                                .compact();
672
673                            Registry::default()
674                                .with(env_filter)
675                                .with(fmt_layer)
676                                .with(file_layer)
677                                .try_init()?;
678                        }
679                        else
680                        {
681                            Registry::default()
682                                .with(env_filter)
683                                .with(fmt_layer)
684                                .try_init()?;
685                        }
686                    },
687                    LogFormat::Json =>
688                    {
689                        let mut fmt_layer = fmt::layer()
690                            .json()
691                            .with_file(config.with_file)
692                            .with_line_number(config.with_file)
693                            .with_target(config.with_target)
694                            .with_current_span(false);
695                        fmt_layer.set_span_events(FmtSpan::CLOSE);
696
697                        if let Some(ref path) = config.file_path
698                        {
699                            let file_appender = create_file_appender(path, config.rotation)?;
700
701                            let file_layer = fmt::layer()
702                                .json()
703                                .with_file(config.with_file)
704                                .with_line_number(config.with_file)
705                                .with_target(config.with_target)
706                                .with_writer(file_appender);
707
708                            Registry::default()
709                                .with(env_filter)
710                                .with(fmt_layer)
711                                .with(file_layer)
712                                .try_init()?;
713                        }
714                        else
715                        {
716                            Registry::default()
717                                .with(env_filter)
718                                .with(fmt_layer)
719                                .try_init()?;
720                        }
721                    },
722                }
723            },
724        }
725
726        Ok(())
727    }

784    /// Initialize logging from hiver-config
785    /// 从 hiver-config 初始化日志
786    ///
787    /// # Supported Properties / 支持的属性
788    ///
789    /// ```toml
790    /// [logging]
791    /// level = "INFO"              # Global log level
792    /// mode = "verbose"            # "verbose" or "simple"
793    /// format = "pretty"           # "pretty", "compact", or "json"
794    /// file = "logs/app.log"       # Optional log file
795    ///
796    /// [logging.rotation]
797    /// policy = "daily"            # "daily", "hourly", or "never"
798    /// max_files = 7               # Number of files to keep
799    /// ```
800    ///
801    /// # Profile-based defaults / 基于 Profile 的默认值
802    ///
803    /// - `dev` profile → `mode = "verbose"`
804    /// - `prod` profile → `mode = "simple"`
805    ///
806    /// # Example / 示例
807    ///
808    /// ```rust,no_run,ignore
809    /// use hiver_observability::log::Logger;
810    /// use hiver_config::Config;
811    ///
812    /// let config = Config::builder()
813    ///     .add_profile("dev")
814    ///     .build()?;
815    ///
816    /// Logger::init_from_config(&config)?;
817    /// ```
818    #[cfg(feature = "config")]
819    pub fn init_from_config(config: &hiver_config::Config)
820    -> Result<(), Box<dyn std::error::Error>>
821    {
822        // Get profile first (affects defaults)
823        let profile = config
824            .get("hiver.profile")
825            .or_else(|| config.get("profile"))
826            .and_then(|v| v.as_str().map(String::from));
827
828        // Build LoggerConfig from Config
829        let mut logger_config = LoggerConfig {
830            profile: profile.clone(),
831            ..Default::default()
832        };
833
834        // Override with explicit config values
835        if let Some(level) = config.get("logging.level")
836        {
837            if let Some(s) = level.as_str()
838            {
839                if let Some(lvl) = LogLevel::from_str(s)
840                {
841                    logger_config.level = lvl;
842                }
843            }
844        }
845
846        if let Some(mode) = config.get("logging.mode")
847        {
848            if let Some(s) = mode.as_str()
849            {
850                if let Some(m) = LogMode::from_str(s)
851                {
852                    logger_config.mode = m;
853                }
854            }
855        }
856        else
857        {
858            // Derive from profile if not explicitly set
859            logger_config.mode = LogMode::from_profile(profile.as_deref());
860        }
861
862        if let Some(format) = config.get("logging.format")
863        {
864            if let Some(s) = format.as_str()
865            {
866                if let Some(f) = LogFormat::from_str(s)
867                {
868                    logger_config.format = f;
869                }
870            }
871        }
872
873        if let Some(file) = config.get("logging.file")
874        {
875            if let Some(s) = file.as_str()
876            {
877                logger_config.file_path = Some(s.to_string());
878            }
879        }
880
881        // Rotation settings
882        if let Some(rotation) = config.get("logging.rotation.policy")
883        {
884            if let Some(s) = rotation.as_str()
885            {
886                match s.to_lowercase().as_str()
887                {
888                    "daily" => logger_config.rotation = LogRotation::Daily,
889                    "hourly" => logger_config.rotation = LogRotation::Hourly,
890                    "never" => logger_config.rotation = LogRotation::Never,
891                    _ =>
892                    {},
893                }
894            }
895        }
896
897        if let Some(max) = config.get("logging.rotation.max_files")
898        {
899            if let Some(n) = max.as_i64()
900            {
901                logger_config.max_files = n.max(0) as usize;
902            }
903        }
904
905        Logger::init_with_config(&logger_config)
906    }
907}

909/// Create environment filter for log level
910/// 创建日志级别的环境过滤器
911#[allow(clippy::indexing_slicing)]
912fn create_env_filter(default_level: LogLevel) -> EnvFilter
913{
914    let base_filter = if let Some(level) = default_level.to_tracing_level()
915    {
916        EnvFilter::builder()
917            .with_default_directive(level.into())
918            .from_env_lossy()
919    }
920    else
921    {
922        EnvFilter::builder()
923            .with_default_directive(Level::INFO.into())
924            .from_env_lossy()
925    };
926
927    // Support Spring Boot style: logging.level.<package>=<LEVEL>
928    // 支持Spring Boot风格：logging.level.<package>=<LEVEL>
929
930    if let Ok(level_str) = std::env::var("LOGGING_LEVEL")
931    {
932        let parts: Vec<&str> = level_str.split('=').collect();
933        if parts.len() == 2
934        {
935            let target = parts[0];
936            let level = parts[1];
937            if let Some(lvl) = LogLevel::from_str(level).and_then(LogLevel::to_tracing_level)
938            {
939                base_filter.add_directive(
940                    format!("{}={}", target, lvl)
941                        .parse()
942                        .unwrap_or_else(|_| lvl.into()),
943                )
944            }
945            else
946            {
947                base_filter
948            }
949        }
950        else
951        {
952            base_filter
953        }
954    }
955    else
956    {
957        base_filter
958    }
959}

961/// Create rolling file appender
962/// 创建滚动文件附加器
963#[allow(clippy::indexing_slicing)]
964fn create_file_appender(
965    path: &str,
966    rotation: LogRotation,
967) -> Result<RollingFileAppender, std::io::Error>
968{
969    let (directory, prefix) = if path.contains('/')
970    {
971        let parts: Vec<&str> = path.rsplitn(2, '/').collect();
972        (parts[1], parts[0])
973    }
974    else
975    {
976        (".", path)
977    };
978
979    let rotation = match rotation
980    {
981        LogRotation::Never => Rotation::NEVER,
982        LogRotation::Daily => Rotation::DAILY,
983        LogRotation::Hourly => Rotation::HOURLY,
984        LogRotation::Minutely => Rotation::MINUTELY,
985    };
986
987    Ok(RollingFileAppender::new(rotation, directory, prefix))
988}
