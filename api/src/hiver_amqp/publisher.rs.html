9/// Message publisher
10/// 消息发布者
11///
12/// # Spring Equivalent / Spring等价物
13///
14/// ```java
15/// @Autowired
16/// private RabbitTemplate rabbitTemplate;
17///
18/// rabbitTemplate.convertAndSend("my_exchange", "routing.key", message);
19/// ```
20#[derive(Clone)]
21pub struct Publisher
22{
23    /// Connection
24    /// 连接
25    connection: Arc<AmqpConnection>,
26
27    /// Default publishing options
28    /// 默认发布选项
29    default_options: PublishingOptions,
30}

32/// Publishing options
33/// 发布选项
34#[derive(Clone, Debug, Default)]
35pub struct PublishingOptions
36{
37    /// Exchange
38    /// 交换机
39    pub exchange: String,
40
41    /// Routing key
42    /// 路由键
43    pub routing_key: String,
44
45    /// Mandatory flag
46    /// 强制标志
47    pub mandatory: bool,
48
49    /// Immediate flag
50    /// 立即标志
51    pub immediate: bool,
52
53    /// Delivery mode
54    /// 传递模式
55    pub delivery_mode: Option<crate::DeliveryMode>,
56
57    /// Priority (0-9)
58    /// 优先级（0-9）
59    pub priority: Option<u8>,
60
61    /// Expiration (milliseconds)
62    /// 过期时间（毫秒）
63    pub expiration: Option<String>,
64
65    /// Message ID
66    /// 消息ID
67    pub message_id: Option<String>,
68
69    /// Correlation ID
70    /// 关联ID
71    pub correlation_id: Option<String>,
72
73    /// Reply to
74    /// 回复到
75    pub reply_to: Option<String>,
76
77    /// Content type
78    /// 内容类型
79    pub content_type: Option<String>,
80
81    /// Content encoding
82    /// 内容编码
83    pub content_encoding: Option<String>,
84}

160impl Publisher

162    /// Create new publisher
163    /// 创建新的发布者
164    pub fn new(connection: Arc<AmqpConnection>) -> Self
165    {
166        Self {
167            connection,
168            default_options: PublishingOptions::default(),
169        }
170    }
171
172    /// Create with default options
173    /// 使用默认选项创建
174    pub fn with_options(mut self, options: PublishingOptions) -> Self
175    {
176        self.default_options = options;
177        self
178    }

180    /// Publish message
181    /// 发布消息
182    pub fn publish(&self, exchange: &str, routing_key: &str, payload: &[u8]) -> Result<(), String>
183    {
184        // Mock implementation
185        // In a real implementation, this would publish to AMQP
186        // 模拟实现
187        // 在实际实现中，这将发布到AMQP
188        tracing::debug!(
189            "Publishing to exchange '{}' with routing key '{}': {} bytes",
190            exchange,
191            routing_key,
192            payload.len()
193        );
194        Ok(())
195    }
196
197    /// Publish message with options
198    /// 使用选项发布消息
199    pub fn publish_with_options(
200        &self,
201        exchange: &str,
202        routing_key: &str,
203        payload: &[u8],
204        _options: &PublishingOptions,
205    ) -> Result<(), String>
206    {
207        tracing::debug!(
208            "Publishing to exchange '{}' with routing key '{}': {} bytes",
209            exchange,
210            routing_key,
211            payload.len()
212        );
213        Ok(())
214    }
215
216    /// Publish message as JSON
217    /// 发布JSON消息
218    pub fn publish_json<T: serde::Serialize>(
219        &self,
220        exchange: &str,
221        routing_key: &str,
222        payload: &T,
223    ) -> Result<(), String>
224    {
225        let json =
226            serde_json::to_vec(payload).map_err(|e| format!("Failed to serialize JSON: {}", e))?;
227        self.publish(exchange, routing_key, &json)
228    }
229
230    /// Convert and send (like Spring's convertAndSend)
231    /// 转换并发送（类似Spring的convertAndSend）
232    pub fn convert_and_send<T: serde::Serialize>(
233        &self,
234        exchange: &str,
235        routing_key: &str,
236        payload: &T,
237    ) -> Result<(), String>
238    {
239        self.publish_json(exchange, routing_key, payload)
240    }
241
242    /// Send to default exchange
243    /// 发送到默认交换机
244    pub fn send(&self, routing_key: &str, payload: &[u8]) -> Result<(), String>
245    {
246        self.publish("", routing_key, payload)
247    }
248
249    /// Send to default exchange with JSON
250    /// 发送JSON到默认交换机
251    pub fn send_json<T: serde::Serialize>(
252        &self,
253        routing_key: &str,
254        payload: &T,
255    ) -> Result<(), String>
256    {
257        self.publish_json("", routing_key, payload)
258    }
259}
