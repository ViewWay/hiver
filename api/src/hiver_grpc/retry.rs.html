18use std::{future::Future, time::Duration};
19
20/// Retry policy for gRPC calls.
21/// gRPC 调用的重试策略。
22///
23/// Equivalent to Spring Retry's `RetryTemplate` / gRPC service config retry policy.
24/// 等价于 Spring Retry 的 `RetryTemplate` / gRPC 服务配置重试策略。
25#[derive(Debug, Clone)]
26pub struct RetryPolicy
27{
28    /// Maximum number of retry attempts.
29    /// 最大重试次数。
30    max_attempts: usize,
31
32    /// Initial delay between retries.
33    /// 重试之间的初始延迟。
34    initial_delay: Duration,
35
36    /// Maximum delay cap.
37    /// 最大延迟上限。
38    max_delay: Duration,
39
40    /// Multiplier for exponential backoff.
41    /// 指数退避的乘数。
42    multiplier: f64,
43
44    /// Retryable gRPC status codes.
45    /// 可重试的 gRPC 状态码。
46    retryable_codes: Vec<tonic::Code>,
47}
48
49impl RetryPolicy
50{
51    /// Create a fixed-delay retry policy with the given max attempts.
52    /// 创建固定延迟的重试策略。
53    pub fn fixed(max_attempts: usize) -> Self
54    {
55        Self {
56            max_attempts,
57            initial_delay: Duration::from_millis(100),
58            max_delay: Duration::from_secs(10),
59            multiplier: 1.0,
60            retryable_codes: vec![
61                tonic::Code::Unavailable,
62                tonic::Code::DeadlineExceeded,
63                tonic::Code::Aborted,
64                tonic::Code::ResourceExhausted,
65            ],
66        }
67    }
68
69    /// Create an exponential backoff retry policy.
70    /// 创建指数退避重试策略。
71    pub fn exponential(max_attempts: usize) -> Self
72    {
73        Self {
74            max_attempts,
75            initial_delay: Duration::from_millis(100),
76            max_delay: Duration::from_secs(10),
77            multiplier: 2.0,
78            retryable_codes: vec![
79                tonic::Code::Unavailable,
80                tonic::Code::DeadlineExceeded,
81                tonic::Code::Aborted,
82                tonic::Code::ResourceExhausted,
83            ],
84        }
85    }
86
87    /// No retries.
88    /// 无重试。
89    pub fn none() -> Self
90    {
91        Self {
92            max_attempts: 0,
93            initial_delay: Duration::ZERO,
94            max_delay: Duration::ZERO,
95            multiplier: 1.0,
96            retryable_codes: vec![],
97        }
98    }
99
100    /// Set the initial delay.
101    /// 设置初始延迟。
102    pub fn with_initial_delay(mut self, delay: Duration) -> Self
103    {
104        self.initial_delay = delay;
105        self
106    }
107
108    /// Set the maximum delay cap.
109    /// 设置最大延迟上限。
110    pub fn with_max_delay(mut self, delay: Duration) -> Self
111    {
112        self.max_delay = delay;
113        self
114    }
115
116    /// Set the backoff multiplier.
117    /// 设置退避乘数。
118    pub fn with_multiplier(mut self, m: f64) -> Self
119    {
120        self.multiplier = m;
121        self
122    }
123
124    /// Set which gRPC status codes are retryable.
125    /// 设置哪些 gRPC 状态码可重试。
126    pub fn with_retryable_codes(mut self, codes: Vec<tonic::Code>) -> Self
127    {
128        self.retryable_codes = codes;
129        self
130    }
131
132    /// Maximum number of attempts.
133    /// 最大尝试次数。
134    pub fn max_attempts(&self) -> usize
135    {
136        self.max_attempts
137    }
138
139    /// Whether the given status code is retryable.
140    /// 给定状态码是否可重试。
141    pub fn is_retryable(&self, code: tonic::Code) -> bool
142    {
143        self.retryable_codes.contains(&code)
144    }
145
146    /// Calculate the delay before the given attempt (0-indexed).
147    /// 计算给定尝试（0 索引）之前的延迟。
148    pub fn delay_for_attempt(&self, attempt: usize) -> Duration
149    {
150        if attempt == 0
151        {
152            return Duration::ZERO;
153        }
154        let base = self.initial_delay.as_secs_f64();
155        let delay_secs = base * self.multiplier.powi((attempt - 1) as i32);
156        let delay = Duration::from_secs_f64(delay_secs);
157        delay.min(self.max_delay)
158    }
159
160    /// Execute an async operation with retry logic.
161    /// 使用重试逻辑执行异步操作。
162    pub async fn execute<F, Fut, T>(&self, mut f: F) -> Result<T, tonic::Status>
163    where
164        F: FnMut() -> Fut,
165        Fut: Future<Output = Result<T, tonic::Status>>,
166    {
167        let mut last_err = None;
168        for attempt in 0..=self.max_attempts
169        {
170            match f().await
171            {
172                Ok(v) => return Ok(v),
173                Err(status) =>
174                {
175                    if attempt >= self.max_attempts || !self.is_retryable(status.code())
176                    {
177                        return Err(status);
178                    }
179                    last_err = Some(status);
180                    let delay = self.delay_for_attempt(attempt);
181                    if !delay.is_zero()
182                    {
183                        tokio::time::sleep(delay).await;
184                    }
185                },
186            }
187        }
188        Err(last_err.expect("retry loop should always produce an error"))
189    }
190}
191
192impl Default for RetryPolicy
193{
194    fn default() -> Self
195    {
196        Self::exponential(3)
197    }
198}

211
212    #[test]
213    fn test_fixed_policy()
214    {
215        let p = RetryPolicy::fixed(3);
216        assert_eq!(p.max_attempts(), 3);
217        assert!(p.is_retryable(tonic::Code::Unavailable));
218        assert!(!p.is_retryable(tonic::Code::NotFound));
219    }
220
221    #[test]
222    fn test_exponential_delay()
223    {
224        let p = RetryPolicy::exponential(5)
225            .with_initial_delay(Duration::from_millis(100))
226            .with_multiplier(2.0);
227
228        assert_eq!(p.delay_for_attempt(0), Duration::ZERO);
229        assert_eq!(p.delay_for_attempt(1), Duration::from_millis(100));
230        assert_eq!(p.delay_for_attempt(2), Duration::from_millis(200));
231        assert_eq!(p.delay_for_attempt(3), Duration::from_millis(400));
232    }
233
234    #[test]
235    fn test_delay_cap()
236    {
237        let p = RetryPolicy::exponential(10)
238            .with_initial_delay(Duration::from_secs(1))
239            .with_max_delay(Duration::from_secs(5))
240            .with_multiplier(3.0);
241
242        assert_eq!(p.delay_for_attempt(1), Duration::from_secs(1));
243        assert_eq!(p.delay_for_attempt(2), Duration::from_secs(3));
244        assert_eq!(p.delay_for_attempt(3), Duration::from_secs(5)); // capped
245        assert_eq!(p.delay_for_attempt(4), Duration::from_secs(5)); // capped
246    }

249    fn test_none_policy()
250    {

253    }

255    #[tokio::test]
256    async fn test_retry_succeeds_on_second_attempt()
257    {
258        let p = RetryPolicy::fixed(2).with_initial_delay(Duration::from_millis(1));
259        let mut attempts = 0;
260        let result = p
261            .execute(|| {
262                attempts += 1;
263                async move {
264                    if attempts <= 1
265                    {
266                        Err(tonic::Status::unavailable("not ready"))
267                    }
268                    else
269                    {
270                        Ok(42)
271                    }
272                }
273            })
274            .await;
275        assert_eq!(result.unwrap(), 42);
276    }

278    #[tokio::test]
279    async fn test_retry_exhausted()
280    {
281        let p = RetryPolicy::fixed(1).with_initial_delay(Duration::from_millis(1));
282        let result: Result<i32, tonic::Status> = p
283            .execute(|| async { Err(tonic::Status::unavailable("down")) })
284            .await;
285        assert!(result.is_err());
286        assert_eq!(result.unwrap_err().code(), tonic::Code::Unavailable);
287    }

289    #[tokio::test]
290    async fn test_non_retryable_error()
291    {
292        let p = RetryPolicy::fixed(5).with_initial_delay(Duration::from_millis(1));
293        let mut attempts = 0;
294        let result: Result<i32, tonic::Status> = p
295            .execute(|| {
296                attempts += 1;
297                async { Err(tonic::Status::not_found("nope")) }
298            })
299            .await;
300        assert!(result.is_err());
301        assert_eq!(attempts, 1); // not retried
302    }
