1//! Remember-Me authentication — persistent login via cookie tokens.
2//! 记住我认证 —— 通过 Cookie 令牌实现持久登录。
3//!
4//! Equivalent to Spring Security's `RememberMeServices` + `TokenBasedRememberMeServices`.
5
6#![allow(dead_code)]
7use std::{collections::HashMap, sync::RwLock};
8
9use chrono::{DateTime, Utc};
10use hmac::{Hmac, Mac};
11use rand::RngCore;
12use sha2::Sha256;
13
14type HmacSha256 = Hmac<Sha256>;

16/// A remember-me token stored in the repository.
17/// 存储在仓库中的记住我令牌。
18#[derive(Debug, Clone)]
19pub struct RememberMeToken
20{
21    /// Random series identifier — ties all tokens for a login session.
22    pub series: String,
23    /// Random token value — rotated on each successful authentication.
24    pub value: String,
25    /// The username this token belongs to.
26    pub username: String,
27    /// Last time this token was used.
28    pub last_used: DateTime<Utc>,
29}
30
31impl RememberMeToken
32{
33    /// Encode token as `series:token` for a cookie value.
34    /// 将令牌编码为 series:token 格式的 Cookie 值。
35    pub fn to_cookie_value(&self) -> String
36    {
37        format!("{}:{}", self.series, self.value)
38    }
39
40    /// Parse a cookie value back into (series, value).
41    /// 从 Cookie 值解析出 (series, value)。
42    pub fn from_cookie_value(cookie: &str) -> Option<(String, String)>
43    {
44        let parts: Vec<&str> = cookie.splitn(2, ':').collect();
45        if parts.len() == 2 && !parts[0].is_empty() && !parts[1].is_empty()
46        {
47            Some((parts[0].to_string(), parts[1].to_string()))
48        }
49        else
50        {
51            None
52        }
53    }
54
55    /// Refresh with a new random value, updating last_used.
56    /// 用新的随机值刷新令牌，更新 last_used。
57    pub fn refresh(&mut self)
58    {
59        self.value = random_hex(16);
60        self.last_used = Utc::now();
61    }
62}
63
64/// In-memory token repository.
65/// 内存令牌仓库。
66pub struct InMemoryTokenRepository
67{
68    tokens: RwLock<HashMap<String, RememberMeToken>>,
69}
70
71impl InMemoryTokenRepository
72{
73    /// Create an empty repository.
74    /// 创建空仓库。
75    pub fn new() -> Self
76    {
77        Self {
78            tokens: RwLock::new(HashMap::new()),
79        }
80    }
81
82    /// Save a new token.
83    /// 保存新令牌。
84    pub fn save(&self, token: RememberMeToken)
85    {
86        self.tokens
87            .write()
88            .unwrap()
89            .insert(token.series.clone(), token);
90    }
91
92    /// Find a token by its series identifier.
93    /// 通过 series 标识查找令牌。
94    pub fn find_by_series(&self, series: &str) -> Option<RememberMeToken>
95    {
96        self.tokens.read().unwrap().get(series).cloned()
97    }
98
99    /// Update an existing token.
100    /// 更新现有令牌。
101    pub fn update(&self, token: &RememberMeToken)
102    {
103        self.tokens
104            .write()
105            .unwrap()
106            .insert(token.series.clone(), token.clone());
107    }
108
109    /// Remove a token by series.
110    /// 通过 series 移除令牌。
111    pub fn remove(&self, series: &str)
112    {
113        self.tokens.write().unwrap().remove(series);
114    }
115
116    /// Remove all tokens for a given user.
117    /// 移除指定用户的所有令牌。
118    pub fn remove_user_tokens(&self, username: &str)
119    {
120        self.tokens
121            .write()
122            .unwrap()
123            .retain(|_, t| t.username != username);
124    }
125}
126
127impl Default for InMemoryTokenRepository
128{
129    fn default() -> Self
130    {
131        Self::new()
132    }
133}

135/// Configuration for Remember-Me services.
136/// 记住我服务的配置。
137pub struct RememberMeConfig
138{
139    /// Cookie name (default: "remember-me").
140    pub cookie_name: String,
141    /// How long the token is valid, in seconds.
142    pub token_validity_secs: u64,
143    /// Whether to set the Secure flag on the cookie.
144    pub secure_cookie: bool,
145    /// HMAC key for token-based signatures.
146    pub key: String,
147}
148
149impl Default for RememberMeConfig
150{
151    fn default() -> Self
152    {
153        Self {
154            cookie_name: "remember-me".to_string(),
155            token_validity_secs: 14 * 24 * 3600, // 2 weeks
156            secure_cookie: true,
157            key: "hiver-remember-me-key".to_string(),
158        }
159    }
160}
161
162impl RememberMeConfig
163{
164    /// Create config with a custom key.
165    /// 用自定义密钥创建配置。
166    pub fn with_key(mut self, key: impl Into<String>) -> Self
167    {
168        self.key = key.into();
169        self
170    }
171}

173/// Remember-Me authentication services.
174/// 记住我认证服务。
175///
176/// Equivalent to Spring Security's `TokenBasedRememberMeServices` when using
177/// the token-based strategy, or `PersistentTokenBasedRememberMeServices` when
178/// using the persistent repository.
179pub struct RememberMeServices
180{
181    config: RememberMeConfig,
182    repository: InMemoryTokenRepository,
183}
184
185impl RememberMeServices
186{
187    /// Create new services with default config.
188    /// 用默认配置创建服务。
189    pub fn new(config: RememberMeConfig) -> Self
190    {
191        Self {
192            config,
193            repository: InMemoryTokenRepository::new(),
194        }
195    }
196
197    /// Called after a successful interactive login — creates a new persistent token.
198    /// 交互式登录成功后调用 — 创建新的持久令牌。
199    pub fn login_success(&self, username: &str) -> RememberMeToken
200    {
201        let token = RememberMeToken {
202            series: random_hex(16),
203            value: random_hex(16),
204            username: username.to_string(),
205            last_used: Utc::now(),
206        };
207        self.repository.save(token.clone());
208        token
209    }
210
211    /// Attempt auto-login from a cookie value.
212    /// 尝试从 Cookie 值自动登录。
213    ///
214    /// Returns the username if valid, or `None` if the token is missing,
215    /// expired, or stolen (series found but value mismatch → all tokens removed).
216    pub fn auto_login(&self, cookie_value: &str) -> Option<String>
217    {
218        let (series, value) = RememberMeToken::from_cookie_value(cookie_value)?;
219
220        let mut token = self.repository.find_by_series(&series)?;
221
222        // Token value mismatch → potential theft; invalidate all user tokens
223        // 令牌值不匹配 → 可能被盗；清除该用户所有令牌
224        if token.value != value
225        {
226            let username = token.username.clone();
227            self.repository.remove_user_tokens(&username);
228            tracing::warn!(
229                "Remember-me token mismatch for user={}, series={}. Possible token theft.",
230                username,
231                series
232            );
233            return None;
234        }
235
236        // Check expiry
237        // 检查过期
238        let elapsed = Utc::now()
239            .signed_duration_since(token.last_used)
240            .num_seconds();
241        if elapsed < 0 || elapsed as u64 > self.config.token_validity_secs
242        {
243            self.repository.remove(&series);
244            return None;
245        }
246
247        // Rotate token value
248        // 轮换令牌值
249        let username = token.username.clone();
250        token.refresh();
251        self.repository.update(&token);
252
253        Some(username)
254    }
255
256    /// Logout — remove the token identified by the cookie.
257    /// 登出 — 移除 Cookie 对应的令牌。
258    pub fn logout(&self, cookie_value: &str)
259    {
260        if let Some((series, _)) = RememberMeToken::from_cookie_value(cookie_value)
261        {
262            self.repository.remove(&series);
263        }
264    }
265
266    /// Generate a token-based signature (HMAC-SHA256).
267    /// 生成基于令牌的签名 (HMAC-SHA256)。
268    pub fn hash_token(&self, token: &RememberMeToken) -> String
269    {
270        let data =
271            format!("{}:{}:{}:{}", token.username, token.series, token.value, self.config.key);
272        let mut mac = HmacSha256::new_from_slice(self.config.key.as_bytes())
273            .expect("HMAC key length is valid");
274        mac.update(data.as_bytes());
275        let result = mac.finalize();
276        hex::encode(result.into_bytes())
277    }
278}

280/// Generate `n` random bytes as a hex string.
281/// 生成 n 个随机字节的十六进制字符串。
282fn random_hex(n: usize) -> String
283{
284    let mut buf = vec![0u8; n];
285    rand::rng().fill_bytes(&mut buf);
286    hex::encode(&buf)
287}

422        // Old cookie value (pre-rotation) should fail and trigger theft detection
