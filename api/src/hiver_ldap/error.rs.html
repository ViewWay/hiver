1//! Error types for hiver-ldap / hiver-ldap的错误类型
2use thiserror::Error;
3
4/// LDAP error types / LDAP错误类型
5#[derive(Error, Debug)]
6pub enum LdapError
7{
8    /// Connection error / 连接错误
9    #[error("LDAP connection error: {0}")]
10    Connection(String),
11
12    /// Authentication error / 认证错误
13    #[error("LDAP authentication error: {0}")]
14    Authentication(String),
15
16    /// Operation error / 操作错误
17    #[error("LDAP operation error: {0}")]
18    Operation(String),
19
20    /// Entry not found error / 条目未找到错误
21    #[error("LDAP entry not found: {0}")]
22    NotFound(String),
23
24    /// Schema violation error / 模式违规错误
25    #[error("LDAP schema violation: {0}")]
26    SchemaViolation(String),
27}
28
29/// Result type for LDAP operations / LDAP操作的结果类型
30pub type LdapResult<T> = Result<T, LdapError>;
