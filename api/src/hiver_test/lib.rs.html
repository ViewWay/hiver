18//! - `@TestMethodOrder` - `TestOrder`

20//! # Example / 示例
21//!
22//! ```rust,no_run,ignore
23//! use hiver_test::{HiverTest, TestClient};

26//! #[tokio::test]
27//! async fn test_user_endpoint() {
28//!     let client = TestClient::new().await;

30//!     let response = client.get("/api/users/1")
