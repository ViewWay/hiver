4/// Generate Cargo.toml content.
5/// 生成 Cargo.toml 内容。
6pub fn generate_cargo_toml(name: &str, modules: &[String]) -> String
7{
8    let mut deps = vec![("hiver-runtime", r#""0.1""#)];
9
10    for module in modules
11    {
12        match module.as_str()
13        {
14            "web" =>
15            {
16                deps.push(("hiver-http", r#""0.1""#));
17                deps.push(("hiver-router", r#""0.1""#));
18                deps.push(("hiver-middleware", r#""0.1""#));
19                deps.push(("hiver-extractors", r#""0.1""#));
20                deps.push(("hiver-response", r#""0.1""#));
21                deps.push(("serde", r#"{ version = "1", features = ["derive"] }"#));
22                deps.push(("serde_json", r#""0.1""#));
23            },
24            "security" => deps.push(("hiver-security", r#""0.1""#)),
25            "data" =>
26            {
27                deps.push(("hiver-data-rdbc", r#""0.1""#));
28                deps.push(("hiver-data-orm", r#""0.1""#));
29                deps.push(("hiver-tx", r#""0.1""#));
30            },
31            "cache" => deps.push(("hiver-data-redis", r#""0.1""#)),
32            "schedule" => deps.push(("hiver-schedule", r#""0.1""#)),
33            "actuator" => deps.push(("hiver-actuator", r#""0.1""#)),
34            "web3" => deps.push(("hiver-web3", r#""0.1""#)),
35            "graphql" => deps.push(("hiver-graphql", r#""0.1""#)),
36            "grpc" => deps.push(("hiver-grpc", r#""0.1""#)),
37            "ai" => deps.push(("hiver-ai", r#""0.1""#)),
38            _ =>
39            {},
40        }
41    }
42
43    if !modules.contains(&"web".to_string())
44    {
45        deps.push(("serde", r#"{ version = "1", features = ["derive"] }"#));
46    }
47
48    let deps_str = deps
49        .iter()
50        .map(|(name, version)| format!("{} = {}", name, version))
51        .collect::<Vec<_>>()
52        .join("\n");
53
54    format!(
55        "[package]\nname = \"{}\"\nversion = \"0.1.0\"\nedition = \"2024\"\nrust-version = \
56         \"1.85\"\n\n[dependencies]\n{}\n\ntracing = \"0.1\"\ntracing-subscriber = {{ version = \
57         \"0.3\", features = [\"env-filter\"] }}\n",
58        name, deps_str
59    )
60}

62/// Generate application.toml content.
63/// 生成 application.toml 内容。
64pub fn generate_application_toml(modules: &[String]) -> String
65{
66    let mut config = String::from(
67        r##"# Hiver Application Configuration
68# Hiver 应用配置
69
70[server]
71host = "127.0.0.1"
72port = 8080
73workers = 4
74
75[logging]
76level = "info"
77mode = "verbose"  # verbose for dev, simple for prod
78
79"##,
80    );
81
82    if modules.contains(&"data".to_string())
83    {
84        config.push_str(
85            r##"[data.source]
86url = "postgresql://localhost:5432/mydb"
87username = "user"
88password = "pass"
89max_connections = 20
90
91"##,
92        );
93    }
94
95    if modules.contains(&"cache".to_string())
96    {
97        config.push_str(
98            r##"[cache]
99type = "redis"
100url = "redis://localhost:6379"
101
102"##,
103        );
104    }
105
106    if modules.contains(&"security".to_string())
107    {
108        config.push_str(
109            r##"[security]
110jwt_secret = "change-me-in-production"
111jwt_expiration = 3600
112
113"##,
114        );
115    }
116
117    config
118}
