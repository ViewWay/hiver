56    Toml,

62    Json,

68impl ConfigFormat

70    /// 从文件扩展名解析格式
71    pub fn from_extension(ext: &str) -> Option<Self>
72    {
73        match ext.to_lowercase().as_str()
74        {
75            "toml" => Some(ConfigFormat::Toml),
76            "yaml" | "yml" => Some(ConfigFormat::Yaml),
77            "json" => Some(ConfigFormat::Json),
78            "properties" | "props" => Some(ConfigFormat::Properties),
79            _ => None,
80        }
81    }

85    {
86        match self
87        {
88            ConfigFormat::Toml => "application.toml",
89            ConfigFormat::Yaml => "application.yml",
90            ConfigFormat::Json => "application.json",
91            ConfigFormat::Properties => "application.properties",
92        }
93    }

100/// 配置加载器
101/// Configuration loader
102///
103/// 从多个来源加载配置，并按优先级合并。
104/// Loads configuration from multiple sources and merges them by priority.
105///
106/// # 示例 / Example
107///
108/// ```rust,ignore
109/// let loader = ConfigurationLoader::new();
110/// loader.load().await?;
111///
112/// let port = loader.get("server.port")
113///     .and_then(|p| p.parse::<u16>().ok())
114///     .unwrap_or(8080);
115/// ```
116#[derive(Debug, Clone)]
117pub struct ConfigurationLoader
118{
119    /// 配置属性
120    properties: HashMap<String, String>,
121
122    /// 当前环境
123    environment: Arc<Environment>,
124
125    /// 配置文件搜索路径
126    search_paths: Vec<String>,
127
128    /// 是否加载环境变量
129    load_env_vars: bool,
130
131    /// 配置文件格式
132    format: ConfigFormat,
133}

170    /// 加载配置
171    /// Load configuration
172    ///
173    /// 按优先级顺序从多个来源加载配置。
174    /// Loads configuration from multiple sources in priority order.
175    pub async fn load(&mut self) -> Result<()>
176    {
177        tracing::debug!("Loading configuration...");
178
179        // 1. 加载默认配置文件
180        self.load_default_config().await?;
181
182        // 2. 加载环境特定配置文件
183        self.load_profile_config().await?;
184
185        // 3. 加载环境变量
186        if self.load_env_vars
187        {
188            self.load_environment_variables();
189        }
190
191        // 4. 加载系统属性
192        self.load_system_properties();
193
194        tracing::debug!("Configuration loaded: {} properties", self.properties.len());
195        Ok(())
196    }
197
198    /// 加载默认配置文件
199    async fn load_default_config(&mut self) -> Result<()>
200    {
201        let filename = self.format.default_filename().to_string();
202        self.load_config_file(&filename).await
203    }
204
205    /// 加载环境特定配置文件
206    async fn load_profile_config(&mut self) -> Result<()>
207    {
208        let profile = self.environment.active_profile();
209        let filename = format!("application-{}.{}", profile, self.format.extension());
210        self.load_config_file(&filename).await
211    }
212
213    /// 加载配置文件
214    async fn load_config_file(&mut self, filename: &str) -> Result<()>
215    {
216        for search_path in &self.search_paths
217        {
218            let file_path = Path::new(search_path).join(filename);
219            if file_path.exists()
220            {
221                tracing::info!("Loading configuration from: {}", file_path.display());
222                self.load_config_from_path(&file_path).await?;
223                return Ok(());
224            }
225        }
226        // 文件不存在不是错误，跳过
227        Ok(())
228    }
229
230    /// 从路径加载配置
231    async fn load_config_from_path(&mut self, path: &Path) -> Result<()>
232    {
233        let content = fs::read_to_string(path).await?;
234
235        match self.format
236        {
237            ConfigFormat::Toml =>
238            {
239                self.parse_toml(&content)?;
240            },
241            ConfigFormat::Yaml =>
242            {
243                self.parse_yaml(&content)?;
244            },
245            ConfigFormat::Json =>
246            {
247                self.parse_json(&content)?;
248            },
249            ConfigFormat::Properties =>
250            {
251                self.parse_properties(&content)?;
252            },
253        }
254
255        Ok(())
256    }
257
258    /// 解析 TOML 配置
259    /// Parse TOML configuration
260    fn parse_toml(&mut self, content: &str) -> Result<()>
261    {
262        // 使用 toml crate 解析 / Parse using toml crate
263        let value: toml::Value =
264            toml::from_str(content).map_err(|e| anyhow::anyhow!("Failed to parse TOML: {}", e))?;
265
266        // 递归解析 TOML 值并展平为点分隔的键
267        // Recursively parse TOML value and flatten to dot-separated keys
268        self.insert_toml_value(String::new(), &value);
269        Ok(())
270    }
271
272    /// 递归插入 TOML 值
273    /// Recursively insert TOML value
274    fn insert_toml_value(&mut self, prefix: String, value: &toml::Value)
275    {
276        match value
277        {
278            toml::Value::String(s) =>
279            {
280                self.properties.insert(prefix, s.clone());
281            },
282            toml::Value::Integer(i) =>
283            {
284                self.properties.insert(prefix.clone(), i.to_string());
285            },
286            toml::Value::Float(f) =>
287            {
288                self.properties.insert(prefix.clone(), f.to_string());
289            },
290            toml::Value::Boolean(b) =>
291            {
292                self.properties.insert(prefix.clone(), b.to_string());
293            },
294            toml::Value::Array(arr) =>
295            {
296                for (i, item) in arr.iter().enumerate()
297                {
298                    let key = format!("{}[{}]", prefix, i);
299                    self.insert_toml_value(key, item);
300                }
301            },
302            toml::Value::Table(table) =>
303            {
304                for (k, v) in table
305                {
306                    let key = if prefix.is_empty()
307                    {
308                        k.clone()
309                    }
310                    else
311                    {
312                        format!("{}.{}", prefix, k)
313                    };
314                    self.insert_toml_value(key, v);
315                }
316            },
317            toml::Value::Datetime(dt) =>
318            {
319                self.properties.insert(prefix, dt.to_string());
320            },
321        }
322    }
323
324    /// 解析 YAML 配置
325    /// Parse YAML configuration
326    fn parse_yaml(&mut self, content: &str) -> Result<()>
327    {
328        // 使用 serde_yaml 解析 / Parse using serde_yaml
329        let value: serde_yaml::Value = serde_yaml::from_str(content)
330            .map_err(|e| anyhow::anyhow!("Failed to parse YAML: {}", e))?;
331
332        // 递归解析 YAML 值并展平为点分隔的键
333        // Recursively parse YAML value and flatten to dot-separated keys
334        self.insert_yaml_value(String::new(), &value);
335        Ok(())
336    }
337
338    /// 递归插入 YAML 值
339    /// Recursively insert YAML value
340    fn insert_yaml_value(&mut self, prefix: String, value: &serde_yaml::Value)
341    {
342        match value
343        {
344            serde_yaml::Value::String(s) =>
345            {
346                self.properties.insert(prefix, s.clone());
347            },
348            serde_yaml::Value::Number(n) =>
349            {
350                self.properties.insert(prefix.clone(), n.to_string());
351            },
352            serde_yaml::Value::Bool(b) =>
353            {
354                self.properties.insert(prefix.clone(), b.to_string());
355            },
356            serde_yaml::Value::Null =>
357            {
358                // 跳过 null 值 / Skip null values
359            },
360            serde_yaml::Value::Sequence(arr) =>
361            {
362                for (i, item) in arr.iter().enumerate()
363                {
364                    let key = format!("{}[{}]", prefix, i);
365                    self.insert_yaml_value(key, item);
366                }
367            },
368            serde_yaml::Value::Mapping(map) =>
369            {
370                for (k, v) in map
371                {
372                    if let Some(key_str) = k.as_str()
373                    {
374                        let key = if prefix.is_empty()
375                        {
376                            key_str.to_string()
377                        }
378                        else
379                        {
380                            format!("{}.{}", prefix, key_str)
381                        };
382                        self.insert_yaml_value(key, v);
383                    }
384                }
385            },
386            // Tagged values are handled as their underlying value
387            serde_yaml::Value::Tagged(tagged) =>
388            {
389                self.insert_yaml_value(prefix, &tagged.value);
390            },
391        }
392    }
393
394    /// 解析 JSON 配置
395    fn parse_json(&mut self, content: &str) -> Result<()>
396    {
397        let map: HashMap<String, serde_json::Value> = serde_json::from_str(content)?;
398        for (key, value) in map
399        {
400            if let Some(str_value) = value.as_str()
401            {
402                self.properties.insert(key, str_value.to_string());
403            }
404            else
405            {
406                self.properties.insert(key, value.to_string());
407            }
408        }
409        Ok(())
410    }
411
412    /// 解析 Properties 配置
413    fn parse_properties(&mut self, content: &str) -> Result<()>
414    {
415        for line in content.lines()
416        {
417            let line = line.trim();
418            if line.is_empty() || line.starts_with('#') || line.starts_with('!')
419            {
420                continue;
421            }
422            if let Some(eq_pos) = line.find('=')
423            {
424                let key = line[..eq_pos].trim();
425                let value = line[eq_pos + 1..].trim();
426                self.properties.insert(key.to_string(), value.to_string());
427            }
428        }
429        Ok(())
430    }
431
432    /// 加载环境变量
433    fn load_environment_variables(&mut self)
434    {
435        // 加载 HIVER_* 前缀的环境变量
436        for (key, value) in std::env::vars()
437        {
438            if let Some(rest) = key.strip_prefix("HIVER_")
439            {
440                let config_key = rest.to_lowercase().replace('_', ".");
441                self.properties.insert(config_key, value);
442            }
443            else if let Some(rest) = key.strip_prefix("APP_")
444            {
445                let config_key = rest.to_lowercase().replace('_', ".");
446                self.properties.insert(config_key, value);
447            }
448        }
449    }
450
451    /// 加载系统属性
452    fn load_system_properties(&mut self)
453    {
454        // 加载常见的系统属性
455        if let Ok(user_dir) = std::env::var("USER")
456        {
457            self.properties.insert("user.dir".to_string(), user_dir);
458        }
459        if let Ok(user_home) = std::env::var("HOME")
460        {
461            self.properties.insert("user.home".to_string(), user_home);
462        }
463    }

465    /// 获取配置值
466    /// Get configuration value
467    pub fn get(&self, key: &str) -> Option<String>
468    {
469        self.properties.get(key).cloned()
470    }

472    /// 获取配置值或默认值
473    /// Get configuration value or default
474    pub fn get_or(&self, key: &str, default: &str) -> String
475    {
476        self.get(key).unwrap_or_else(|| default.to_string())
477    }

479    /// 获取配置值并解析
480    /// Get configuration value and parse
481    pub fn get_parsed<T>(&self, key: &str) -> Option<T>
482    where
483        T: std::str::FromStr,
484    {
485        self.get(key)?.parse().ok()
