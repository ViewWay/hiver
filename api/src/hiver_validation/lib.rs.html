1//! Hiver Validation - 验证模块 / Validation Module
2//!
3//! 提供请求参数校验功能，等价于 Spring Validation / Provides request parameter validation,
4//! equivalent to Spring Validation
5
6#![allow(clippy::result_large_err)]
7//! # Spring Equivalent / Spring等价物
8//!
9//! - `@Valid` → `#[Valid]` - Cascading validation / 级联验证
10//! - `@Validated` → `#[Validated]` - Group validation / 分组验证
11//! - `@NotNull` → `#[NotNull]` - Non-null check / 非空检查
12//! - `@Size` → `#[Size]` - Size validation / 大小验证
13//! - `@Min/@Max` → `#[Min]/#[Max]` - Numeric bounds / 数值边界
14//! - `@Email` → `#[Email]` - Email validation / 邮箱验证
15//! - `@Pattern` → `#[Pattern]` - Regex validation / 正则验证
16//!
17//! # Basic Usage / 基本使用
18//!
19//! ```rust,ignore
20//! use hiver_validation::{Valid, Validated, NotNull, Size, Email};
21//! use serde::Deserialize;
22//!
23//! #[derive(Debug, Deserialize)]
24//! struct CreateUserRequest {
25//!     #[NotNull]
26//!     #[Size(min = 3, max = 20)]
27//!     username: String,
28//!
29//!     #[Email]
30//!     email: String,
31//!
32//!     #[Min(18)]
33//!     #[Max(120)]
34//!     age: u32,
35//! }
36//!
37//! // Auto-validation with #[Valid] annotation
38//! // 使用#[Valid]注解自动验证
39//! #[hiver_macros::post("/users")]
40//! async fn create_user(
41//!     #[Valid] request: CreateUserRequest,
42//! ) -> Result<Json<User>, Error> {
43//!     // request is already validated
44//!     // request 已经被验证
45//!     Ok(Json(user))
46//! }
47//!
48//! // Group validation with #[Validated]
49//! // 使用#[Validated]进行分组验证
50//! #[hiver_macros::post("/users")]
51//! async fn create_user_validated(
52//!     #[Validated(CreateGroup::default())] request: CreateUserRequest,
53//! ) -> Result<Json<User>, Error> {
54//!     Ok(Json(user))
55//! }
56//! ```
57//!
58//! # Validation Groups / 验证分组
59//!
60//! ```rust,ignore
61//! use hiver_validation::Validated;
62//!
63//! #[derive(Debug, Default)]
64//! struct CreateGroup;
65//!
66//! #[derive(Debug, Default)]
67//! struct UpdateGroup;
68//!
69//! #[derive(Debug, Deserialize)]
70//! struct UserRequest {
71//!     #[NotNull(group = "CreateGroup")]
72//!     username: String,
73//!
74//!     #[NotNull(group = "UpdateGroup")]
75//!     id: u64,
76//! }
77//!
78//! // Only validates fields marked with CreateGroup
79//! // 仅验证标记为CreateGroup的字段
80//! async fn create_user(
81//!     #[Validated(CreateGroup)] request: UserRequest,
82//! ) -> Result<Json<User>, Error> {
83//!     Ok(Json(user))
84//! }
85//! ```

132/// 验证上下文 / Validation context
133#[derive(Debug, Clone)]
134pub struct ValidationContext
135{
136    /// 字段名 / Field name
137    pub field: String,
138    /// 字段值 / Field value
139    pub value: String,
140    /// 自定义消息 / Custom message
141    pub message: Option<String>,
142    /// 代码 / Code
143    pub code: String,
144}
145
146impl ValidationContext
147{
148    /// Create a new validation context
149    /// 创建新的验证上下文
150    pub fn new(field: impl Into<String>, value: impl Into<String>) -> Self
151    {
152        Self {
153            field: field.into(),
154            value: value.into(),
155            message: None,
156            code: "validation_failed".to_string(),
157        }
158    }
159
160    /// Set custom message
161    /// 设置自定义消息
162    pub fn with_message(mut self, message: impl Into<String>) -> Self
163    {
164        self.message = Some(message.into());
165        self
166    }
167
168    /// Set error code
169    /// 设置错误代码
170    pub fn with_code(mut self, code: impl Into<String>) -> Self
171    {
172        self.code = code.into();
173        self
174    }
175}

177/// 验证规则 / Validation rules
178#[derive(Debug, Clone, Copy, PartialEq, Eq)]
179pub enum ValidationRule
180{
181    /// 非空 / Not empty
182    NotEmpty,
183    /// 长度范围 / Length range
184    Length
185    {
186        /// Minimum length
187        /// 最小长度
188        min: Option<usize>,
189        /// Maximum length
190        /// 最大长度
191        max: Option<usize>,
192    },
193    /// 数值范围 / Range
194    Range
195    {
196        /// Minimum value
197        /// 最小值
198        min: Option<i64>,
199        /// Maximum value
200        /// 最大值
201        max: Option<i64>,
202    },
203    /// 邮箱 / Email
204    Email,
205    /// URL
206    Url,
207    /// 正则表达式 / Regex
208    Regex(&'static str),
209    /// 自定义 / Custom
210    Custom(&'static str),
211}
212
213impl fmt::Display for ValidationRule
214{
215    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
216    {
217        match self
218        {
219            ValidationRule::NotEmpty => write!(f, "not_empty"),
220            ValidationRule::Length { min, max } =>
221            {
222                write!(f, "length(min={:?}, max={:?})", min, max)
223            },
224            ValidationRule::Range { min, max } =>
225            {
226                write!(f, "range(min={:?}, max={:?})", min, max)
227            },
228            ValidationRule::Email => write!(f, "email"),
229            ValidationRule::Url => write!(f, "url"),
230            ValidationRule::Regex(pattern) => write!(f, "regex({})", pattern),
231            ValidationRule::Custom(name) => write!(f, "custom({})", name),
232        }
233    }
