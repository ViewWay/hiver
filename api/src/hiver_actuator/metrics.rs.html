17/// Metric type
18/// 指标类型
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
20#[serde(rename_all = "lowercase")]
21pub enum MetricType
22{
23    /// A gauge can go up or down
24    /// 仪表盘，可以上下变化
25    Gauge,
26
27    /// A counter can only increase
28    /// 计数器，只能增加
29    Counter,
30
31    /// A timing measurement
32    /// 时间测量
33    Timer,
34}
35
36/// Metric value
37/// 指标值
38#[derive(Debug, Clone, Serialize, Deserialize)]
39#[serde(untagged)]
40pub enum MetricValue
41{
42    /// Integer value
43    /// 整数值
44    Integer(i64),
45
46    /// Floating point value
47    /// 浮点数值
48    Float(f64),
49
50    /// String value
51    /// 字符串值
52    String(String),
53}
54
55/// A metric
56/// 指标
57#[derive(Debug, Clone, Serialize, Deserialize)]
58pub struct Metric
59{
60    /// Metric name
61    /// 指标名称
62    pub name: String,
63
64    /// Metric description
65    /// 指标描述
66    #[serde(skip_serializing_if = "Option::is_none")]
67    pub description: Option<String>,
68
69    /// Metric type
70    /// 指标类型
71    #[serde(rename = "type")]
72    pub metric_type: MetricType,
73
74    /// Metric value
75    /// 指标值
76    pub value: MetricValue,
77
78    /// Available measurements (for timers/gauges)
79    /// 可用的测量值（用于计时器/仪表盘）
80    #[serde(skip_serializing_if = "HashMap::is_empty")]
81    pub measurements: HashMap<String, f64>,
82
83    /// Available tags
84    /// 可用的标签
85    #[serde(skip_serializing_if = "HashMap::is_empty")]
86    pub tags: HashMap<String, String>,
87}
88
89impl Metric
90{
91    /// Create a new gauge metric
92    /// 创建新的仪表盘指标
93    pub fn gauge(name: impl Into<String>, value: i64) -> Self
94    {
95        Self {
96            name: name.into(),
97            description: None,
98            metric_type: MetricType::Gauge,
99            value: MetricValue::Integer(value),
100            measurements: HashMap::new(),
101            tags: HashMap::new(),
102        }
103    }
104
105    /// Create a new counter metric
106    /// 创建新的计数器指标
107    pub fn counter(name: impl Into<String>, value: u64) -> Self
108    {
109        Self {
110            name: name.into(),
111            description: None,
112            metric_type: MetricType::Counter,
113            value: MetricValue::Integer(value as i64),
114            measurements: HashMap::new(),
115            tags: HashMap::new(),
116        }
117    }
118
119    /// Set the description
120    /// 设置描述
121    pub fn with_description(mut self, description: impl Into<String>) -> Self
122    {
123        self.description = Some(description.into());
124        self
125    }
126
127    /// Add a tag
128    /// 添加标签
129    pub fn with_tag(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
130    {
131        self.tags.insert(key.into(), value.into());
132        self
133    }
134}

136/// Metrics registry
137/// 指标注册表
138#[derive(Debug, Default)]
139pub struct MetricsRegistry
140{
141    counters: HashMap<String, Arc<AtomicU64>>,
142    gauges: HashMap<String, Arc<AtomicI64>>,
143}
144
145impl MetricsRegistry
146{
147    /// Create a new metrics registry
148    /// 创建新的指标注册表
149    pub fn new() -> Self
150    {
151        Self {
152            counters: HashMap::new(),
153            gauges: HashMap::new(),
154        }
155    }
156
157    /// Increment a counter
158    /// 增加计数器
159    pub fn increment(&mut self, name: &str) -> u64
160    {
161        let counter = self
162            .counters
163            .entry(name.to_string())
164            .or_insert_with(|| Arc::new(AtomicU64::new(0)));
165        counter.fetch_add(1, Ordering::Relaxed)
166    }
167
168    /// Increment a counter by a specific amount
169    /// 按指定数量增加计数器
170    pub fn increment_by(&mut self, name: &str, amount: u64) -> u64
171    {
172        let counter = self
173            .counters
174            .entry(name.to_string())
175            .or_insert_with(|| Arc::new(AtomicU64::new(0)));
176        counter.fetch_add(amount, Ordering::Relaxed)
177    }
178
179    /// Get a counter value
180    /// 获取计数器值
181    pub fn get_counter(&self, name: &str) -> Option<u64>
182    {
183        self.counters.get(name).map(|c| c.load(Ordering::Relaxed))
184    }
185
186    /// Set a gauge value
187    /// 设置仪表盘值
188    pub fn set_gauge(&mut self, name: &str, value: i64)
189    {
190        let gauge = self
191            .gauges
192            .entry(name.to_string())
193            .or_insert_with(|| Arc::new(AtomicI64::new(0)));
194        gauge.store(value, Ordering::Relaxed);
195    }
196
197    /// Get a gauge value
198    /// 获取仪表盘值
199    pub fn get_gauge(&self, name: &str) -> Option<i64>
200    {
201        self.gauges.get(name).map(|g| g.load(Ordering::Relaxed))
202    }
203
204    /// Collect all metrics
205    /// 收集所有指标
206    pub fn collect(&self) -> Vec<Metric>
207    {
208        let mut metrics = Vec::new();
209
210        // Collect counters
211        for (name, counter) in &self.counters
212        {
213            let value = counter.load(Ordering::Relaxed);
214            metrics.push(Metric::counter(name, value).with_description("Counter metric"));
215        }
216
217        // Collect gauges
218        for (name, gauge) in &self.gauges
219        {
220            let value = gauge.load(Ordering::Relaxed);
221            metrics.push(Metric::gauge(name, value).with_description("Gauge metric"));
222        }
223
224        metrics
225    }
226
227    /// Get metric names list
228    /// 获取指标名称列表
229    pub fn names(&self) -> Vec<String>
230    {
231        let mut names = Vec::new();
232        names.extend(self.counters.keys().cloned());
233        names.extend(self.gauges.keys().cloned());
234        names.sort();
235        names.dedup();
236        names
237    }
238}

240/// System metrics collector
241/// 系统指标收集器
242pub struct SystemMetrics
243{
244    _registry: Arc<MetricsRegistry>,
245}
246
247impl SystemMetrics
248{
249    /// Create new system metrics
250    /// 创建新的系统指标
251    pub fn new() -> Self
252    {
253        Self {
254            _registry: Arc::new(MetricsRegistry::new()),
255        }
256    }
257
258    /// Collect system metrics
259    /// 收集系统指标
260    pub fn collect(&self) -> Vec<Metric>
261    {
262        vec![
263            Metric::gauge("jvm.memory.max", 1024 * 1024 * 512).with_description("Maximum memory"),
264            Metric::gauge("jvm.memory.used", 1024 * 1024 * 128).with_description("Used memory"),
265            Metric::gauge("process.cpu.usage", 5).with_description("Process CPU usage"),
266            Metric::gauge("system.cpu.count", 4).with_description("Number of processors"),
267        ]
268    }
269}
270
271impl Default for SystemMetrics
272{
273    fn default() -> Self
274    {
275        Self::new()
276    }
