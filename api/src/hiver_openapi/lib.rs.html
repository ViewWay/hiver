3//! Hiver `OpenAPI` - OpenAPI/Swagger documentation support
4//! Hiver `OpenAPI` - OpenAPI/Swagger 文档支持
5//!
6//! # Equivalent to Spring / 等价于 Spring
7//!
8//! | Hiver | Spring |
9//! |-------|--------|
10//! | `OpenApi` | `@OpenAPIDefinition` |
11//! | `SwaggerUi` | `springdoc-openapi-ui` |
12//! | `ToSchema` | `@Schema` |
13//! | `IntoParams` | `@Parameter` |
14//! | `#[openapi]` | `@Operation` |
15//!
16//! # Features / 功能
17//!
18//! - `OpenAPI` 3.0 specification generation / `OpenAPI` 3.0 规范生成
19//! - Swagger UI integration / Swagger UI 集成
20//! - Type-safe schema definitions / 类型安全的模式定义
21//! - HTTP framework integration / HTTP 框架集成
22//! - Spring Boot compatible API / Spring Boot 兼容 API
23//!
24//! # Quick Start / 快速开始
25//!
26//! ```rust,ignore
27//! use hiver_openapi::{OpenApi, OpenApiConfig, SwaggerUi};
28//! use hiver_openapi::{InfoConfig, OpenApiBuilder};
29//! use hiver_openapi::{Schema, PathItem, Operation, Response};
30//!
31//! // Create OpenAPI specification
32//! // 创建 OpenAPI 规范
33//! let openapi = OpenApiBuilder::new()
34//!     .title("My API")
35//!     .version("1.0.0")
36//!     .description("My API description")
37//!     .add_path("/users", PathItem::new()
38//!         .get(Operation::new()
39//!             .summary("List users")
40//!             .add_response("200", Response::ok("Success"))
41//!         )
42//!     )
43//!     .build();
44//!
45//! // Create Swagger UI handler
46//! // 创建 Swagger UI 处理器
47//! let swagger = SwaggerUi::new(openapi);
48//!
49//! // Handle HTTP request
50//! // 处理 HTTP 请求
51//! let (body, status, headers) = swagger.handle("/swagger");
52//! ```
53//!
54//! # Modules / 模块
55//!
56//! - [`config`] - Configuration types / 配置类型
57//! - [`schema`] - Schema definitions / 模式定义
58//! - [`operation`] - Operation definitions / 操作定义
59//! - [`response`] - Response definitions / 响应定义
60//! - [`path`] - Path definitions / 路径定义
61//! - [`openapi`] - `OpenAPI` builder / `OpenAPI` 构建器
62//! - [`swagger`] - Swagger UI integration / Swagger UI 集成
63//! - [`http`] - HTTP framework integration / HTTP 框架集成
64//! - [`macros`] - Re-exported utoipa macros / 重新导出的 utoipa 宏
65//!
66//! # Examples / 示例
67//!
68//! More examples are available in the [OpenAPI documentation](https://hiver.viewway.io/openapi).
69//! 更多示例请参考 [OpenAPI 文档](https://hiver.viewway.io/openapi)。

141/// `OpenApi` trait for generating documentation
142/// `OpenApi` trait 用于生成文档
143///
144/// Equivalent to `SpringDoc`'s `OpenAPI` annotation.
145/// `等价于SpringDoc的OpenAPI注解`。
146///
147/// # Spring Equivalent / Spring等价物
148///
149/// ```java
150/// @OpenAPIDefinition(
151///     info = @Info(
152///         title = "My API",
153///         version = "1.0.0"
154///     )
155/// )
156/// ```
157pub trait GenerateOpenApi
158{
159    /// Generate `OpenAPI` specification
160    /// `生成OpenAPI规范`
161    fn generate(&self) -> Result<utoipa::openapi::OpenApi, String>;
162}
163
164/// Default implementation for `utoipa::OpenApi`
165impl<T: utoipa::OpenApi> GenerateOpenApi for T
166{
167    fn generate(&self) -> Result<utoipa::openapi::OpenApi, String>
168    {
169        Ok(T::openapi())
170    }
171}
