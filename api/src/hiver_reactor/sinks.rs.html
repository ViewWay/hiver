1//! `Sinks` — multicast/broadcast hot sources (Project Reactor `Sinks` equivalent).
2//! `Sinks`——多播/广播热源（Project Reactor `Sinks` 等价）。

13/// Error returned by [`Sinks::try_emit`] when the value cannot be accepted.
14/// 当 [`Sinks::try_emit`] 无法接受值时返回的错误。
15#[derive(Debug, thiserror::Error)]
16pub enum TrySendError
17{
18    /// There are no active subscribers, so the value was dropped.
19    /// 没有活跃订阅者，值被丢弃。
20    #[error("no active subscribers / 无活跃订阅者")]
21    NoSubscribers,
22
23    /// The internal buffer is full (overflow under the configured strategy).
24    /// 内部缓冲区已满（在配置策略下溢出）。
25    #[error("buffer full (overflow) / 缓冲区满（溢出）")]
26    Full,
27
28    /// The sink is closed.
29    /// sink 已关闭。
30    #[error("sink closed / sink 已关闭")]
31    Closed,
32}
33
34/// A multicast, hot `Sinks.many()`-style publisher. Each subscriber receives
35/// values emitted after it subscribes (broadcast semantics).
36///
37/// 多播、热 `Sinks.many()` 风格发布者。每个订阅者收到其订阅后发射的值（广播语义）。
38///
39/// Backed by `tokio::sync::broadcast`. The `capacity` bounds the per-subscriber
40/// lag buffer; the [`BackpressureStrategy`] governs what `try_emit` does when a
41/// subscriber lags (it can only **drop** lagged receivers — broadcast channels
42/// cannot block a producer).
43///
44/// 由 `tokio::sync::broadcast` 支撑。`capacity` 限定每订阅者的滞后缓冲；
45/// [`BackpressureStrategy`] 决定订阅者滞后时 `try_emit` 的行为（只能**丢弃**
46/// 滞后的接收者——广播 channel 无法阻塞生产者）。
47pub struct Sinks<T: Clone>
48{
49    tx: broadcast::Sender<T>,
50    strategy: BackpressureStrategy,
51}
52
53impl<T: Clone + Send + 'static> Sinks<T>
54{
55    /// Create a new broadcast sink with the given buffer `capacity` and
56    /// overflow `strategy`.
57    /// 创建具有给定缓冲 `capacity` 与溢出 `strategy` 的新广播 sink。
58    #[must_use]
59    pub fn many(capacity: usize, strategy: BackpressureStrategy) -> Self
60    {
61        let (tx, _rx) = broadcast::channel(capacity.max(1));
62        Self { tx, strategy }
63    }
64
65    /// Create a sink with the default strategy (Buffer).
66    /// 以默认策略（Buffer）创建 sink。
67    #[must_use]
68    pub fn with_capacity(capacity: usize) -> Self
69    {
70        Self::many(capacity, BackpressureStrategy::default())
71    }
72
73    /// Try to emit a value to all current subscribers.
74    /// 向所有当前订阅者尝试发射一个值。
75    ///
76    /// - `Buffer` / `Drop` / `DropLatest`: emit and silently let lagged subscribers miss values
77    ///   (returns `Ok(())` if there are subscribers).
78    /// - `Error`: returns [`TrySendError::Full`] if no subscriber could receive.
79    /// - Returns [`TrySendError::NoSubscribers`] if there are no receivers.
80    ///
81    /// - `Buffer` / `Drop` / `DropLatest`：发射并静默让滞后订阅者错过值 （有订阅者时返回
82    ///   `Ok(())`）。
83    /// - `Error`：若无订阅者能接收，返回 [`TrySendError::Full`]。
84    /// - 无接收者时返回 [`TrySendError::NoSubscribers`]。
85    pub fn try_emit(&self, value: T) -> Result<(), TrySendError>
86    {
87        let n = self.tx.receiver_count();
88        if n == 0
89        {
90            // Honor Error strategy's "no subscribers" semantics; otherwise it's a
91            // no-op for a hot source.
92            return match self.strategy
93            {
94                BackpressureStrategy::Error => Err(TrySendError::NoSubscribers),
95                _ => Ok(()),
96            };
97        }
98        match self.tx.send(value)
99        {
100            Ok(_) => Ok(()),
101            Err(broadcast::error::SendError(_)) => Err(TrySendError::Closed),
102        }
103    }
104
105    /// Emit a value, blocking until at least one subscriber receives it.
106    /// Requires the `tokio` feature (broadcast is tokio-only).
107    /// 发射一个值，阻塞直到至少一个订阅者接收。需要 `tokio` 特性
108    /// （broadcast 仅 tokio 提供）。
109    pub async fn emit(&self, value: T) -> ReactorResult<()>
110    {
111        // broadcast::send is non-blocking; "blocking until received" is approximated
112        // by reporting NoSubscribers immediately when there are none.
113        // broadcast::send 非阻塞；"阻塞直到被接收"以无订阅者时立即上报
114        // NoSubscribers 来近似。
115        self.try_emit(value).map_err(|e| match e
116        {
117            TrySendError::NoSubscribers => ReactorError::Overflow("no subscribers".into()),
118            TrySendError::Full => ReactorError::Overflow("buffer full".into()),
119            TrySendError::Closed => ReactorError::SinkClosed,
120        })
121    }
122
123    /// Subscribe to the sink, returning a [`Flux`] of emitted values. A lagged
124    /// subscriber yields an error (it must resubscribe).
125    /// 订阅 sink，返回发射值的 [`Flux`]。滞后订阅者会产生错误（须重新订阅）。
126    #[must_use]
127    pub fn as_flux(&self) -> Flux<T>
128    {
129        let rx = self.tx.subscribe();
130        // Drive the broadcast receiver as a futures Stream via `unfold`, where the
131        // receiver itself is the unfold state (so it is never re-moved per call).
132        // 通过 `unfold` 将广播接收者驱动为 futures Stream，接收者本身作为
133        // unfold 状态（因此每次调用不会重新 move 它）。
134        let stream = unfold(rx, |mut rx| async move {
135            match rx.recv().await
136            {
137                Ok(v) => Some((Ok(v), rx)),
138                Err(broadcast::error::RecvError::Lagged(n)) =>
139                {
140                    Some((Err(ReactorError::Overflow(format!("lagged by {n}"))), rx))
141                },
142                Err(broadcast::error::RecvError::Closed) => None,
143            }
144        });
145        Flux::from_stream(stream)
146    }
147
148    /// Current number of active subscribers.
149    /// 当前活跃订阅者数量。
150    #[must_use]
151    pub fn subscriber_count(&self) -> usize
152    {
153        self.tx.receiver_count()
154    }
155}

157impl<T: Clone> std::fmt::Debug for Sinks<T>
158{
159    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result
160    {
161        f.debug_struct("Sinks")
162            .field("subscribers", &self.tx.receiver_count())
163            .field("strategy", &self.strategy)
164            .finish()
165    }
