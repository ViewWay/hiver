1//! API documentation generator (Markdown / HTML)
2//! API 文档生成器（Markdown / HTML）
3//!
4//! Generates structured documentation from an `OpenAPI` 3.0 specification
5//! in Markdown or printable HTML format (which can be exported to PDF).
6//! 从 `OpenAPI` 3.0 规范生成结构化文档，支持 Markdown 或可打印 HTML 格式（可导出为 PDF）。
7//!
8//! # Example / 示例
9//!
10//! ```rust,ignore
11//! use hiver_openapi::doc_pdf::ApiDocPdf;
12//!
13//! let doc = ApiDocPdf::from_openapi(&openapi);
14//! let md = doc.to_markdown();
15//! let html = doc.to_html();
16//! ```

196    /// Generate documentation from an `OpenApi` specification
197    /// 从 `OpenApi` 规范生成文档
198    pub fn from_openapi(openapi: &OpenApi) -> Self
199    {
200        let mut tag_sections: HashMap<String, DocSection> = HashMap::new();
201
202        // Pre-populate sections from tags
203        // 从标签预填充章节
204        for tag in &openapi.tags
205        {
206            tag_sections.insert(tag.name.clone(), DocSection {
207                title: tag.name.clone(),
208                description: tag.description.clone(),
209                endpoints: Vec::new(),
210            });
211        }
212
213        // Walk paths and operations
214        // 遍历路径和操作
215        for (path_str, path_item) in &openapi.paths
216        {
217            let ops: Vec<(&str, &crate::Operation)> = vec![
218                ("GET", path_item.get.as_ref()),
219                ("POST", path_item.post.as_ref()),
220                ("PUT", path_item.put.as_ref()),
221                ("DELETE", path_item.delete.as_ref()),
222                ("PATCH", path_item.patch.as_ref()),
223                ("HEAD", path_item.head.as_ref()),
224                ("OPTIONS", path_item.options.as_ref()),
225                ("TRACE", path_item.trace.as_ref()),
226            ]
227            .into_iter()
228            .filter_map(|(m, o)| o.map(|op| (m, op)))
229            .collect();
230
231            for (method, op) in ops
232            {
233                let params: Vec<ParamDoc> = path_item
234                    .parameters
235                    .iter()
236                    .chain(op.parameters.iter())
237                    .map(|p| ParamDoc {
238                        name: p.name.clone(),
239                        location: format!("{:?}", p.location).to_lowercase(),
240                        description: p.description.clone(),
241                        required: p.required,
242                        type_: p.schema.as_ref().map(|s| {
243                            s.schema_type.as_ref().map_or_else(
244                                || "any".to_string(),
245                                |t| format!("{:?}", t).to_lowercase(),
246                            )
247                        }),
248                    })
249                    .collect();
250
251                let mut responses: Vec<ResponseDoc> = op
252                    .responses
253                    .iter()
254                    .map(|(code, resp)| ResponseDoc {
255                        code: code.clone(),
256                        description: resp.description.clone(),
257                    })
258                    .collect();
259                responses.sort_by(|a, b| {
260                    let na: u16 = a.code.parse().unwrap_or(999);
261                    let nb: u16 = b.code.parse().unwrap_or(999);
262                    na.cmp(&nb)
263                });
264
265                let request_body = op.request_body.as_ref().map(|body| {
266                    let mut desc_parts = Vec::new();
267                    if let Some(d) = &body.description
268                    {
269                        desc_parts.push(d.clone());
270                    }
271                    for (ct, media) in &body.content
272                    {
273                        desc_parts.push(format!("Content-Type: {}", ct));
274                        if let Some(schema) = &media.schema
275                        {
276                            desc_parts.push(format!("Schema: {}", describe_schema(schema)));
277                        }
278                    }
279                    desc_parts.join("\n")
280                });
281
282                let endpoint = EndpointDoc {
283                    method: method.to_string(),
284                    path: path_str.clone(),
285                    summary: op.summary.clone(),
286                    description: op.description.clone(),
287                    operation_id: op.operation_id.clone(),
288                    parameters: params,
289                    request_body,
290                    responses,
291                    deprecated: op.deprecated,
292                };
293
294                let tag = op
295                    .tags
296                    .first()
297                    .cloned()
298                    .unwrap_or_else(|| "default".to_string());
299
300                tag_sections
301                    .entry(tag.clone())
302                    .or_insert_with(|| DocSection {
303                        title: tag.clone(),
304                        description: None,
305                        endpoints: Vec::new(),
306                    })
307                    .endpoints
308                    .push(endpoint);
309            }
310        }
311
312        // Build ordered sections
313        // 构建有序章节
314        let mut sections = Vec::new();
315        for tag in &openapi.tags
316        {
317            if let Some(section) = tag_sections.remove(&tag.name)
318            {
319                sections.push(section);
320            }
321        }
322        // Remaining sections not in tag order
323        // 不在标签顺序中的剩余章节
324        for (_, section) in tag_sections
325        {
326            sections.push(section);
327        }
328
329        Self {
330            title: openapi.info.title.clone(),
331            version: openapi.info.version.clone(),
332            description: openapi.info.description.clone(),
333            sections,
334            contact: openapi.info.contact.as_ref().map(|c| ContactDoc {
335                name: c.name.clone(),
336                email: c.email.clone(),
337                url: c.url.clone(),
338            }),
339            license: openapi.info.license.as_ref().map(|l| LicenseDoc {
340                name: l.name.clone(),
341                url: l.url.clone(),
342            }),
343        }
344    }

346    /// Generate Markdown documentation
347    /// 生成 Markdown 文档
348    pub fn to_markdown(&self) -> String
349    {
350        let mut md = String::new();
351
352        // Title
353        // 标题
354        let _ = writeln!(md, "# {}\n", self.title);
355        let _ = writeln!(md, "**Version:** {}\n", self.version);
356
357        if let Some(desc) = &self.description
358        {
359            let _ = writeln!(md, "{}\n", desc);
360        }
361
362        // Contact & License
363        // 联系 & 许可证
364        if let Some(contact) = &self.contact
365        {
366            md.push_str("## Contact\n\n");
367            if let Some(name) = &contact.name
368            {
369                let _ = writeln!(md, "- **Name:** {}", name);
370            }
371            if let Some(email) = &contact.email
372            {
373                let _ = writeln!(md, "- **Email:** {}", email);
374            }
375            if let Some(url) = &contact.url
376            {
377                let _ = writeln!(md, "- **URL:** {}", url);
378            }
379            md.push('\n');
380        }
381
382        if let Some(license) = &self.license
383        {
384            let _ = writeln!(md, "## License\n\n{}\n", license.name);
385            if let Some(url) = &license.url
386            {
387                let _ = writeln!(md, "[License URL]({})\n", url);
388            }
389        }
390
391        // Table of Contents
392        // 目录
393        md.push_str("## Table of Contents\n\n");
394        for (i, section) in self.sections.iter().enumerate()
395        {
396            let _ = writeln!(md, "{}. {}", i + 1, section.title);
397        }
398        md.push('\n');
399
400        // Sections
401        // 章节
402        for section in &self.sections
403        {
404            let _ = writeln!(md, "## {}\n", section.title);
405            if let Some(desc) = &section.description
406            {
407                let _ = writeln!(md, "{}\n", desc);
408            }
409
410            for endpoint in &section.endpoints
411            {
412                let _ = writeln!(md, "### {} `{}`\n", endpoint.method, endpoint.path);
413
414                if endpoint.deprecated
415                {
416                    md.push_str("> **DEPRECATED**\n>\n");
417                }
418
419                if let Some(summary) = &endpoint.summary
420                {
421                    let _ = writeln!(md, "**{}**\n", summary);
422                }
423                if let Some(desc) = &endpoint.description
424                {
425                    let _ = writeln!(md, "{}\n", desc);
426                }
427                if let Some(op_id) = &endpoint.operation_id
428                {
429                    let _ = writeln!(md, "**Operation ID:** `{}`\n", op_id);
430                }
431
432                // Parameters
433                // 参数
434                if !endpoint.parameters.is_empty()
435                {
436                    md.push_str("#### Parameters\n\n");
437                    md.push_str("| Name | In | Required | Type | Description |\n");
438                    md.push_str("|------|----|----------|------|-------------|\n");
439                    for p in &endpoint.parameters
440                    {
441                        let desc = p.description.as_deref().unwrap_or("-");
442                        let type_ = p.type_.as_deref().unwrap_or("-");
443                        let _ = writeln!(
444                            md,
445                            "| `{}` | {} | {} | {} | {} |\n",
446                            p.name, p.location, p.required, type_, desc
447                        );
448                    }
449                    md.push('\n');
450                }
451
452                // Request body
453                // 请求体
454                if let Some(body) = &endpoint.request_body
455                {
456                    md.push_str("#### Request Body\n\n");
457                    md.push_str("```text\n");
458                    md.push_str(body);
459                    md.push_str("\n```\n\n");
460                }
461
462                // Responses
463                // 响应
464                if !endpoint.responses.is_empty()
465                {
466                    md.push_str("#### Responses\n\n");
467                    md.push_str("| Code | Description |\n");
468                    md.push_str("|------|-------------|\n");
469                    for r in &endpoint.responses
470                    {
471                        let _ = writeln!(md, "| `{}` | {} |", r.code, r.description);
472                    }
473                    md.push('\n');
474                }
475            }
476        }
477
478        md
479    }

481    /// Generate printable HTML documentation (can be exported to PDF)
482    /// 生成可打印 HTML 文档（可导出为 PDF）
483    pub fn to_html(&self) -> String
484    {
485        let mut html = String::new();
486
487        html.push_str("<!DOCTYPE html>\n");
488        html.push_str("<html lang=\"en\">\n<head>\n");
489        let _ = writeln!(html, "  <title>{}</title>", escape_html(&self.title));
490        html.push_str("  <meta charset=\"UTF-8\">\n");
491        html.push_str(
492            "  <meta name=\"viewport\" content=\"width=device-width, initial-scale=1.0\">\n",
493        );
494        html.push_str("  <style>\n");
495        html.push_str(STYLESHEET);
496        html.push_str("  </style>\n");
497        html.push_str("</head>\n<body>\n");
498
499        // Header
500        // 头部
501        let _ =
502            writeln!(html, "  <div class=\"header\">\n    <h1>{}</h1>\n", escape_html(&self.title));
503        let _ =
504            writeln!(html, "    <p class=\"version\">Version: {}</p>", escape_html(&self.version));
505        html.push_str("  </div>\n");
506
507        // Description
508        // 描述
509        if let Some(desc) = &self.description
510        {
511            let _ =
512                writeln!(html, "  <div class=\"description\"><p>{}</p></div>", escape_html(desc));
513        }
514
515        // Table of Contents
516        // 目录
517        html.push_str("  <div class=\"toc\">\n    <h2>Table of Contents</h2>\n    <ol>\n");
518        for section in &self.sections
519        {
520            let _ = writeln!(
521                html,
522                "      <li><a href=\"#section-{}\">{}</a></li>\n",
523                slug(&section.title),
524                escape_html(&section.title)
525            );
526        }
527        html.push_str("    </ol>\n  </div>\n");
528
529        // Sections
530        // 章节
531        for section in &self.sections
532        {
533            let _ = writeln!(
534                html,
535                "  <div class=\"section\" id=\"section-{}\">\n    <h2>{}</h2>\n",
536                slug(&section.title),
537                escape_html(&section.title)
538            );
539
540            if let Some(desc) = &section.description
541            {
542                let _ = writeln!(html, "    <p class=\"section-desc\">{}</p>", escape_html(desc));
543            }
544
545            for endpoint in &section.endpoints
546            {
547                let method_class = endpoint.method.to_lowercase();
548                html.push_str("    <div class=\"endpoint\">\n");
549                let _ = writeln!(
550                    html,
551                    "      <h3><span class=\"method method-{}\">{}</span> <code>{}</code></h3>\n",
552                    method_class,
553                    endpoint.method,
554                    escape_html(&endpoint.path)
555                );
556
557                if endpoint.deprecated
558                {
559                    html.push_str("      <p class=\"deprecated\">DEPRECATED</p>\n");
560                }
561
562                if let Some(summary) = &endpoint.summary
563                {
564                    let _ = writeln!(
565                        html,
566                        "      <p class=\"summary\"><strong>{}</strong></p>",
567                        escape_html(summary)
568                    );
569                }
570                if let Some(desc) = &endpoint.description
571                {
572                    let _ = writeln!(
573                        html,
574                        "      <p class=\"endpoint-desc\">{}</p>",
575                        escape_html(desc)
576                    );
577                }
578
579                // Parameters table
580                // 参数表格
581                if !endpoint.parameters.is_empty()
582                {
583                    html.push_str("      <table class=\"params\">\n");
584                    html.push_str(
585                        "        <thead><tr><th>Name</th><th>In</th><th>Required</th><th>Type</\
586                         th><th>Description</th></tr></thead>\n",
587                    );
588                    html.push_str("        <tbody>\n");
589                    for p in &endpoint.parameters
590                    {
591                        let desc = escape_html(p.description.as_deref().unwrap_or("-"));
592                        let type_ = escape_html(p.type_.as_deref().unwrap_or("-"));
593                        let req = if p.required { "Yes" } else { "No" };
594                        let _ = writeln!(
595                            html,
596                            "          <tr><td><code>{}</code></td><td>{}</td><td>{}</td><td>{}</\
597                             td><td>{}</td></tr>\n",
598                            escape_html(&p.name),
599                            p.location,
600                            req,
601                            type_,
602                            desc
603                        );
604                    }
605                    html.push_str("        </tbody>\n      </table>\n");
606                }
607
608                // Request body
609                // 请求体
610                if let Some(body) = &endpoint.request_body
611                {
612                    html.push_str("      <div class=\"request-body\">\n");
613                    html.push_str("        <h4>Request Body</h4>\n");
614                    let _ = writeln!(html, "        <pre>{}</pre>", escape_html(body));
615                    html.push_str("      </div>\n");
616                }
617
618                // Responses table
619                // 响应表格
620                if !endpoint.responses.is_empty()
621                {
622                    html.push_str("      <table class=\"responses\">\n");
623                    html.push_str(
624                        "        <thead><tr><th>Code</th><th>Description</th></tr></thead>\n",
625                    );
626                    html.push_str("        <tbody>\n");
627                    for r in &endpoint.responses
628                    {
629                        let _ = writeln!(
630                            html,
631                            "          <tr><td><code>{}</code></td><td>{}</td></tr>\n",
632                            escape_html(&r.code),
633                            escape_html(&r.description)
634                        );
635                    }
636                    html.push_str("        </tbody>\n      </table>\n");
637                }
638
639                html.push_str("    </div>\n");
640            }
641
642            html.push_str("  </div>\n");
643        }
644
645        // Footer
646        // 页脚
647        html.push_str("  <div class=\"footer\">\n");
648        let _ = writeln!(
649            html,
650            "    <p>Generated by Hiver OpenAPI | API Documentation v{}</p>",
651            escape_html(&self.version)
652        );
653        html.push_str("  </div>\n");
654
655        html.push_str("</body>\n</html>\n");
656        html
657    }
658}

664/// Describe a schema as a human-readable string
665/// 将模式描述为可读字符串
666fn describe_schema(schema: &crate::Schema) -> String
667{
668    if let Some(ref_) = &schema.ref_
669    {
670        return ref_.clone();
671    }
672    match schema.schema_type.as_ref()
673    {
674        Some(t) => format!("{:?}", t).to_lowercase(),
675        None => "object".to_string(),
676    }
677}
678
679/// Escape HTML special characters
680/// 转义 HTML 特殊字符
681fn escape_html(s: &str) -> String
682{
683    s.replace('&', "&amp;")
684        .replace('<', "&lt;")
685        .replace('>', "&gt;")
686        .replace('"', "&quot;")
687        .replace('\'', "&#39;")
688}
689
690/// Create a URL-safe slug from text
691/// 从文本创建 URL 安全的 slug
692fn slug(text: &str) -> String
693{
694    text.chars()
695        .map(|c| {
696            if c.is_ascii_alphanumeric()
697            {
698                c.to_ascii_lowercase()
699            }
700            else
701            {
702                '-'
703            }
704        })
705        .collect()
706}

772    {

774            .title("Pet Store API")
775            .version("1.0.0")
776            .description("A sample pet store API")
777            .add_path(

782                        .add_tag("pets")
783                        .add_parameter(Parameter::query("limit").description("Max results"))

785                        .add_response("401", Response::unauthorized("Unauthorized")),
786                ),
787            )
788            .add_path(
789                "/pets/{id}",
790                PathItem::new().get(
791                    Operation::new()
792                        .summary("Get pet by ID")
793                        .add_tag("pets")
794                        .add_parameter(Parameter::path("id").description("Pet ID"))
795                        .add_response("200", Response::ok("Pet found"))
796                        .add_response("404", Response::not_found("Pet not found")),
797                ),
798            )
799            .build();

802

805        assert_eq!(doc.sections.len(), 1);
806        assert_eq!(doc.sections[0].title, "pets");
