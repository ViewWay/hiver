1//! Security context module
2//! 安全上下文模块
3//!
4//! Provides per-task async-scoped security context propagation via
5//! `tokio::task_local!`, as well as a legacy global context for
6//! backward compatibility.
7//! 通过 `tokio::task_local!` 提供每个任务级别的异步范围安全上下文传播，
8//! 以及一个用于向后兼容的全局上下文。
9//!
10//! # Spring Equivalent / Spring等价物
11//!
12//! ```java
13//! SecurityContext context = SecurityContextHolder.getContext();
14//! Authentication auth = context.getAuthentication();
15//! ```

26/// Security context
27/// 安全上下文
28///
29/// Holds the current authentication and security information.
30/// 保存当前认证和安全信息。
31///
32/// Equivalent to Spring's `SecurityContext`.
33/// `等价于Spring的SecurityContext`。
34///
35/// # Spring Equivalent / Spring等价物
36///
37/// ```java
38/// SecurityContext context = SecurityContextHolder.getContext();
39/// Authentication auth = context.getAuthentication();
40/// ```
41pub struct SecurityContext
42{
43    /// Current authentication
44    /// 当前认证
45    authentication: Arc<tokio::sync::RwLock<Option<Authentication>>>,
46}
47
48impl SecurityContext
49{
50    /// Create a new security context
51    /// 创建新的安全上下文
52    pub fn new() -> Self
53    {
54        Self {
55            authentication: Arc::new(tokio::sync::RwLock::new(None)),
56        }
57    }
58
59    /// Get current authentication
60    /// 获取当前认证
61    pub async fn get_authentication(&self) -> Option<Authentication>
62    {
63        self.authentication.read().await.clone()
64    }
65
66    /// Set authentication
67    /// 设置认证
68    pub async fn set_authentication(&self, auth: Authentication)
69    {
70        let mut auth_guard = self.authentication.write().await;
71        *auth_guard = Some(auth);
72    }
73
74    /// Clear authentication
75    /// 清除认证
76    pub async fn clear(&self)
77    {
78        let mut auth_guard = self.authentication.write().await;
79        *auth_guard = None;
80    }
81
82    /// Check if authenticated
83    /// 检查是否已认证
84    pub async fn is_authenticated(&self) -> bool
85    {
86        self.authentication
87            .read()
88            .await
89            .as_ref()
90            .is_some_and(|a| a.authenticated)
91    }
92
93    /// Get current username
94    /// 获取当前用户名
95    pub async fn get_username(&self) -> Option<String>
96    {
97        self.authentication
98            .read()
99            .await
100            .as_ref()
101            .map(|a| a.principal.clone())
102    }
103
104    /// Check if user has authority
105    /// 检查用户是否有权限
106    pub async fn has_authority(&self, authority: &crate::Authority) -> bool
107    {
108        self.authentication
109            .read()
110            .await
111            .as_ref()
112            .is_some_and(|a| a.has_authority(authority))
113    }
114
115    /// Check if user has role
116    /// 检查用户是否有角色
117    pub async fn has_role(&self, role: &crate::Role) -> bool
118    {
119        self.authentication
120            .read()
121            .await
122            .as_ref()
123            .is_some_and(|a| a.has_role(role))
124    }
125}
126
127impl Default for SecurityContext
128{
129    fn default() -> Self
130    {
131        Self::new()
132    }
133}

203// ---------------------------------------------------------------------------
204// Async task-local context propagation
205// ---------------------------------------------------------------------------
206
207// Task-local storage for the per-task security context.
208// 每个任务的安全上下文任务本地存储。
209//
210// Unlike the global context, `task_local` values are scoped to the
211// current async task and automatically propagate through `.await` points.
212// This prevents cross-request leakage in server environments where many
213// tasks share the same thread.
214// 与全局上下文不同，`task_local` 值的作用域限定在当前异步任务内，
215// 并自动通过 `.await` 点传播。这可以防止在多任务共享同一线程的
216// 服务器环境中出现跨请求泄漏。
217tokio::task_local! {
218    static CURRENT_SECURITY_CONTEXT: Arc<SecurityContext>;
219}
220
221/// RAII guard that installs a [`SecurityContext`] for the duration of a scope.
222/// RAII守卫，在作用域期间安装 [`SecurityContext`]。
223///
224/// The guard keeps the [`SecurityContext`] alive via `Arc`. Use
225/// [`SecurityContextGuard::scope`] or [`SecurityContextGuard::scope_async`]
226/// to run code with the context installed as a task-local value.
227/// 守卫通过 `Arc` 保持 [`SecurityContext`] 存活。
228/// 使用 [`SecurityContextGuard::scope`] 或 [`SecurityContextGuard::scope_async`]
229/// 在安装上下文作为任务本地值的情况下运行代码。
230///
231/// # Example / 示例
232///
233/// ```rust,no_run,ignore
234/// use hiver_security::context::SecurityContextGuard;
235/// use hiver_security::SecurityContext;
236/// use hiver_security::Authentication;
237///
238/// let ctx = SecurityContext::new();
239/// // ... set authentication on ctx ...
240///
241/// let guard = SecurityContextGuard::new(ctx);
242/// let result = guard.scope(|| {
243///     // Inside this scope, `get_security_context()` returns Some(ctx).
244///     42
245/// });
246/// ```
247pub struct SecurityContextGuard
248{
249    /// The wrapped security context.
250    /// 包装的安全上下文。
251    ctx: Arc<SecurityContext>,
252}
253
254impl SecurityContextGuard
255{
256    /// Create a new guard wrapping the given [`SecurityContext`].
257    /// 创建新的守卫，包装给定的 [`SecurityContext`]。
258    pub fn new(ctx: SecurityContext) -> Self
259    {
260        Self { ctx: Arc::new(ctx) }
261    }
262
263    /// Run a synchronous closure with this context installed as the
264    /// task-local value.
265    /// 在安装此上下文为任务本地值的情况下运行同步闭包。
266    pub fn scope<F, R>(&self, f: F) -> R
267    where
268        F: FnOnce() -> R,
269    {
270        CURRENT_SECURITY_CONTEXT.sync_scope(self.ctx.clone(), f)
271    }
272
273    /// Run an async future with this context installed as the task-local
274    /// value. The context propagates through all `.await` points inside
275    /// the future.
276    /// 在安装此上下文为任务本地值的情况下运行异步 future。
277    /// 上下文在 future 内所有 `.await` 点传播。
278    pub async fn scope_async<F, Fut, R>(&self, f: F) -> R
279    where
280        F: FnOnce() -> Fut,
281        Fut: Future<Output = R>,
282    {
283        CURRENT_SECURITY_CONTEXT.scope(self.ctx.clone(), f()).await
284    }
285
286    /// Get a clone of the inner `Arc<SecurityContext>`.
287    /// 获取内部 `Arc<SecurityContext>` 的克隆。
288    pub fn context(&self) -> Arc<SecurityContext>
289    {
290        self.ctx.clone()
291    }
292}
293
294/// Set the security context and run a synchronous closure with it.
295/// 设置安全上下文并使用它运行同步闭包。
296///
297/// Shorthand for `SecurityContextGuard::new(ctx).scope(f)`.
298/// `SecurityContextGuard::new(ctx).scope(f)` 的简写。
299///
300/// # Example / 示例
301///
302/// ```rust,no_run,ignore
303/// use hiver_security::context::set_security_context;
304/// use hiver_security::SecurityContext;
305///
306/// let ctx = SecurityContext::new();
307/// let result = set_security_context(ctx, || {
308///     // ctx is now active in this scope
309///     42
310/// });
311/// ```
312pub fn set_security_context<F, R>(ctx: SecurityContext, f: F) -> R
313where
314    F: FnOnce() -> R,
315{
316    let guard = SecurityContextGuard::new(ctx);
317    guard.scope(f)
318}
319
320/// Get the current task-local [`SecurityContext`], if one has been set.
321/// 获取当前任务本地的 [`SecurityContext`]（如果已设置）。
322///
323/// Returns `None` if no context has been installed for the current task.
324/// 如果当前任务未安装上下文，则返回 `None`。
325///
326/// This function may be called both in async and sync contexts.
327/// 此函数可以在异步和同步上下文中调用。
328pub fn get_security_context() -> Option<Arc<SecurityContext>>
329{
330    CURRENT_SECURITY_CONTEXT.try_with(Clone::clone).ok()
331}
332
333/// Run a closure with the given [`SecurityContext`] installed as the
334/// task-local value for the current async task.
335/// 使用给定的 [`SecurityContext`] 作为当前异步任务的任务本地值运行闭包。
336///
337/// This is the most ergonomic way to scope a context for a single
338/// operation.  The context is automatically removed when the closure
339/// returns.
340/// 这是为单个操作限定上下文范围的最符合人体工程学的方式。
341/// 闭包返回时上下文会自动移除。
342///
343/// # Example / 示例
344///
345/// ```rust,no_run,ignore
346/// use hiver_security::context::with_security_context;
347/// use hiver_security::SecurityContext;
348///
349/// let ctx = SecurityContext::new();
350/// let result = with_security_context(ctx, || {
351///     // Inside this closure, `get_security_context()` returns Some.
352///     42
353/// });
354/// assert_eq!(result, 42);
355/// ```
356pub fn with_security_context<F, R>(ctx: SecurityContext, f: F) -> R
357where
358    F: FnOnce() -> R,
359{
360    CURRENT_SECURITY_CONTEXT.sync_scope(Arc::new(ctx), f)
361}
362
363/// Run an async closure with the given [`SecurityContext`] installed as
364/// the task-local value.  The context propagates through all `.await`
365/// points inside the future.
366/// 使用给定的 [`SecurityContext`] 作为任务本地值运行异步闭包。
367/// 上下文在 future 内所有 `.await` 点传播。
368///
369/// # Example / 示例
370///
371/// ```rust,no_run,ignore
372/// use hiver_security::context::with_security_context_async;
373/// use hiver_security::SecurityContext;
374///
375/// let ctx = SecurityContext::new();
376/// let result = with_security_context_async(ctx, || async {
377///     // Inside this block, `get_security_context()` returns Some.
378///     some_async_fn().await
379/// }).await;
380/// ```
381pub async fn with_security_context_async<F, Fut, R>(ctx: SecurityContext, f: F) -> R
382where
383    F: FnOnce() -> Fut,
384    Fut: Future<Output = R>,
385{
386    let arc = Arc::new(ctx);
387    CURRENT_SECURITY_CONTEXT.scope(arc, f()).await
388}

543    #[tokio::test]
544    async fn test_context_propagates_across_await()
545    {
546        let ctx = SecurityContext::new();
547        ctx.set_authentication(Authentication {
548            principal: "dave".to_string(),
549            credentials: None,
550            authorities: vec![],
551            authenticated: true,
552            details: None,
553            login_time: chrono::Utc::now(),
554        })
555        .await;
556
557        let guard = SecurityContextGuard::new(ctx);
558
559        let username = guard
560            .scope_async(|| async {
561                // First check before yield.
562                let before = get_security_context().unwrap().get_username().await;
563                assert_eq!(before, Some("dave".to_string()));
564
565                // Yield to the scheduler — context must survive.
566                tokio::task::yield_now().await;
567
568                // Check after yield.
569                let after = get_security_context().unwrap().get_username().await;
570                assert_eq!(after, Some("dave".to_string()));
571
572                after
573            })
574            .await;
575
576        assert_eq!(username, Some("dave".to_string()));
577        // After the scope, context is gone.
578        assert!(get_security_context().is_none());
579    }

598        // We need to spawn from within the scope.
