4use std::path::PathBuf;
5
6use crate::error::{GrpcError, GrpcResult};

8/// TLS configuration for gRPC server and client.
9/// gRPC 服务端和客户端的 TLS 配置。
10///
11/// # Example / 示例
12/// ```rust,ignore
13/// use hiver_grpc::tls::TlsConfig;
14///
15/// let tls = TlsConfig::new("certs/server.pem", "certs/server.key")
16///     .with_ca_cert("certs/ca.pem");
17/// let server_tls = tls.server_tls_config()?;
18/// ```
19#[allow(clippy::struct_field_names)]
20pub struct TlsConfig
21{
22    cert_path: PathBuf,
23    key_path: PathBuf,
24    ca_cert_path: Option<PathBuf>,
25}
26
27impl TlsConfig
28{
29    /// Create a new TLS config with certificate and private key paths.
30    /// 使用证书和私钥路径创建新的 TLS 配置。
31    pub fn new(cert: impl Into<PathBuf>, key: impl Into<PathBuf>) -> Self
32    {
33        Self {
34            cert_path: cert.into(),
35            key_path: key.into(),
36            ca_cert_path: None,
37        }
38    }
39
40    /// Set the CA certificate path (required for mTLS).
41    /// 设置 CA 证书路径（mTLS 必需）。
42    pub fn with_ca_cert(mut self, ca: impl Into<PathBuf>) -> Self
43    {
44        self.ca_cert_path = Some(ca.into());
45        self
46    }
47
48    /// Build a tonic `ServerTlsConfig`.
49    /// 构建 tonic ServerTlsConfig。
50    pub fn server_tls_config(&self) -> GrpcResult<tonic::transport::ServerTlsConfig>
51    {
52        let identity = tonic::transport::Identity::from_pem(
53            read_file(&self.cert_path)?,
54            read_file(&self.key_path)?,
55        );
56        let mut config = tonic::transport::ServerTlsConfig::new().identity(identity);
57        if let Some(ca) = &self.ca_cert_path
58        {
59            let ca_cert = read_file(ca)?;
60            config = config.client_ca_root(tonic::transport::Certificate::from_pem(ca_cert));
61        }
62        Ok(config)
63    }
64
65    /// Build a tonic `ClientTlsConfig`.
66    /// 构建 tonic ClientTlsConfig。
67    pub fn client_tls_config(&self) -> GrpcResult<tonic::transport::ClientTlsConfig>
68    {
69        let mut config = tonic::transport::ClientTlsConfig::new();
70        if let Some(ca) = &self.ca_cert_path
71        {
72            let ca_cert = read_file(ca)?;
73            config = config.ca_certificate(tonic::transport::Certificate::from_pem(ca_cert));
74        }
75        Ok(config)
76    }
77}
78
79fn read_file(path: &PathBuf) -> GrpcResult<Vec<u8>>
80{
81    std::fs::read(path)
82        .map_err(|e| GrpcError::config(format!("failed to read file {}: {e}", path.display())))
83}
