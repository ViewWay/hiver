1//! Extended mock bean support with Mockito-style `when()` chaining.
2//! 具有Mockito风格 `when()` 链式调用的扩展模拟bean支持。
3//!
4//! This module provides a type-safe wrapper around the generic [`MockRegistry`]
5//! (see [`crate::mock_bean`]) and introduces a Mockito-inspired fluent API
6//! for configuring mock behaviour in tests.
7//!
8//! 本模块提供了通用 [`MockRegistry`]（参见 [`crate::mock_bean`]）的类型安全包装，
9//! 并引入了受Mockito启发的流式API，用于在测试中配置模拟行为。
10//!
11//! # Equivalent to Spring Boot / 等价于 Spring Boot
12//!
13//! ```java
14//! @MockBean
15//! private UserService userService;
16//!
17//! @Test
18//! void testWithMock() {
19//!     when(userService.findById(1L))
20//!         .thenReturn(Optional.of(new User("Alice")));
21//!
22//!     verify(userService, times(1)).findById(1L);
23//! }
24//! ```
25//!
26//! # Example / 示例
27//!
28//! ```rust,no_run,ignore
29//! use hiver_test::mockito_ext::{MockitoHelper, MockBeanWrapper};
30//! use hiver_test::MockRegistry;
31//!
32//! #[tokio::test]
33//! async fn test_with_mockito_helper() {
34//!     let registry = MockRegistry::new();
35//!     let mock = MockitoHelper::new(&registry);
36//!
37//!     // Configure mock: when "userService::findById" is called, return a user.
38//!     // 配置模拟：当调用 "userService::findById" 时，返回一个用户。
39//!     mock.when("userService", "findById")
40//!         .then_return(Box::new(Some("Alice".to_string())));
41//!
42//!     // Call the mock.
43//!     // 调用模拟。
44//!     let result = mock.call::<String>("userService", "findById", vec![]).await;
45//!     assert_eq!(result, Some("Alice".to_string()));
46//! }
47//! ```

55// MockBeanWrapper<T>
56// ---------------------------------------------------------------------------
57
58/// A type-safe wrapper that replaces a real bean with a mock in the
59/// application context.
60///
61/// 一个类型安全的包装器，在应用上下文中用模拟替换真实的bean。
62///
63/// When dropped, the wrapper automatically restores the original bean
64/// (if one was present).
65///
66/// 当丢弃时，包装器自动恢复原始bean（如果存在的话）。
67///
68/// # Spring Equivalent / Spring等价物
69///
70/// ```java
71/// @MockBean
72/// private UserService userService;
73/// ```
74///
75/// # Type Parameters / 类型参数
76///
77/// * `T` - The trait or concrete type being mocked. 被模拟的trait或具体类型。
78pub struct MockBeanWrapper<T: ?Sized>
79{
80    /// The mock implementation.
81    /// 模拟实现。
82    mock: Arc<T>,
83
84    /// The bean name this mock replaces.
85    /// 此模拟替换的bean名称。
86    bean_name: String,
87
88    /// Reference to the mock registry for cleanup.
89    /// 模拟注册表的引用，用于清理。
90    registry: MockRegistry,
91}
92
93impl<T: ?Sized> MockBeanWrapper<T>
94{
95    /// Create a new mock bean wrapper.
96    /// 创建新的模拟bean包装器。
97    ///
98    /// # Arguments / 参数
99    ///
100    /// * `bean_name` - The name under which the mock is registered. 模拟注册的名称。
101    /// * `mock` - The mock implementation. 模拟实现。
102    /// * `registry` - The registry to register the mock in. 注册模拟的注册表。
103    pub fn new(bean_name: impl Into<String>, mock: Arc<T>, registry: &MockRegistry) -> Self
104    {
105        Self {
106            mock,
107            bean_name: bean_name.into(),
108            registry: registry.clone(),
109        }
110    }
111
112    /// Get a reference to the underlying mock.
113    /// 获取底层模拟的引用。
114    pub fn mock(&self) -> &Arc<T>
115    {
116        &self.mock
117    }
118
119    /// Get the bean name this mock is registered under.
120    /// 获取此模拟注册的bean名称。
121    pub fn bean_name(&self) -> &str
122    {
123        &self.bean_name
124    }
125}
126
127impl<T: ?Sized + fmt::Debug> fmt::Debug for MockBeanWrapper<T>
128{
129    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
130    {
131        f.debug_struct("MockBeanWrapper")
132            .field("bean_name", &self.bean_name)
133            .field("mock", &self.mock)
134            .finish()
135    }
136}

138// ---------------------------------------------------------------------------
139// MockInteraction
140// ---------------------------------------------------------------------------
141
142/// A fluent interaction builder for configuring a mock's behaviour.
143///
144/// 用于配置模拟行为的流式交互构建器。
145///
146/// Created via [`MockitoHelper::when`], this struct allows you to chain
147/// return values or error conditions onto a specific mock method.
148///
149/// 通过 [`MockitoHelper::when`] 创建，此结构体允许你将返回值或错误条件
150/// 链接到特定的模拟方法上。
151///
152/// # Example / 示例
153///
154/// ```rust,no_run,ignore
155/// helper.when("userService", "findById")
156///     .then_return(Box::new(some_user));
157/// ```
158pub struct MockInteraction<'a>
159{
160    /// Reference to the parent helper for registration.
161    /// 父帮助器的引用，用于注册。
162    helper: &'a MockitoHelper,
163
164    /// Bean name the mock is associated with.
165    /// 模拟关联的bean名称。
166    bean_name: String,
167
168    /// Method name being mocked.
169    /// 被模拟的方法名称。
170    method_name: String,
171}
172
173impl MockInteraction<'_>
174{
175    /// Register a return value for this mock interaction.
176    /// 为此模拟交互注册返回值。
177    ///
178    /// The value is wrapped in `Box<dyn Any>` and will be returned
179    /// whenever the mocked method is called.
180    ///
181    /// 值被包装在 `Box<dyn Any>` 中，每当模拟方法被调用时都会返回。
182    pub async fn then_return(self, value: Box<dyn Any + Send + Sync>)
183    {
184        let shared: Arc<dyn Any + Send + Sync> = Arc::from(value);
185        self.helper
186            .registry
187            .register_mock(&self.bean_name, &self.method_name, move |_args| {
188                // Wrap in Arc for safe cloning across multiple invocations.
189                // 将值包装在 Arc 中以安全地在多次调用间克隆。
190                let cloned: Arc<dyn Any + Send + Sync> = Arc::clone(&shared);
191                Box::new(cloned) as Box<dyn Any + Send + Sync>
192            })
193            .await;
194    }
195
196    /// Register a return value that can be called multiple times (requires `Clone`).
197    /// 注册可以多次调用的返回值（需要 `Clone`）。
198    ///
199    /// The value is wrapped in `Arc` so that it can be returned on every call.
200    /// 值被包装在 `Arc` 中，以便在每次调用时都能返回。
201    pub async fn then_return_clone<T: Clone + Any + Send + Sync + 'static>(self, value: T)
202    {
203        let shared = Arc::new(value);
204        self.helper
205            .registry
206            .register_mock(&self.bean_name, &self.method_name, move |_args| {
207                let cloned: T = (*shared).clone();
208                Box::new(cloned) as Box<dyn Any + Send + Sync>
209            })
210            .await;
211    }
212
213    /// Register a function that computes the return value from arguments.
214    /// 注册一个从参数计算返回值的函数。
215    pub async fn then_answer<
216        T: Any + Send + Sync + 'static,
217        F: Fn(Vec<Arc<dyn Any + Send + Sync>>) -> T + Send + Sync + 'static,
218    >(
219        self,
220        f: F,
221    )
222    {
223        self.helper
224            .registry
225            .register_mock(&self.bean_name, &self.method_name, move |args| {
226                let result = f(args);
227                Box::new(result) as Box<dyn Any + Send + Sync>
228            })
229            .await;
230    }
231}

233// ---------------------------------------------------------------------------
234// MockitoHelper
235// ---------------------------------------------------------------------------
236
237/// A Mockito-style helper for creating and configuring mocks.
238///
239/// 一个Mockito风格的帮助器，用于创建和配置模拟。
240///
241/// Provides the `when()` entry point and utility methods for verifying
242/// mock interactions.
243///
244/// 提供 `when()` 入口点和用于验证模拟交互的实用方法。
245///
246/// # Spring Equivalent / Spring等价物
247///
248/// ```java
249/// // Mockito.when()
250/// when(userService.findById(1L)).thenReturn(Optional.of(user));
251///
252/// // Mockito.verify()
253/// verify(userService, times(1)).findById(1L);
254///
255/// // Mockito.reset()
256/// reset(userService);
257///
258/// // Mockito.verifyNoInteractions()
259/// verifyNoInteractions(userService);
260/// ```
261///
262/// # Example / 示例
263///
264/// ```rust,no_run,ignore
265/// let registry = MockRegistry::new();
266/// let mock = MockitoHelper::new(&registry);
267///
268/// // Configure mock behaviour.
269/// // 配置模拟行为。
270/// mock.when("userService", "findById")
271///     .then_return_clone(Some(String::from("Alice")));
272///
273/// // Call the mock.
274/// // 调用模拟。
275/// let result = mock.call::<Option<String>>("userService", "findById", vec![]).await;
276/// assert_eq!(result, Some(String::from("Alice")));
277///
278/// // Verify interactions.
279/// // 验证交互。
280/// assert!(mock.verify("userService", "findById").await);
281/// assert_eq!(mock.call_count("userService", "findById").await, 1);
282///
283/// // Reset the mock.
284/// // 重置模拟。
285/// mock.reset("userService", "findById").await;
286/// ```
287pub struct MockitoHelper
288{
289    /// The underlying mock registry.
290    /// 底层模拟注册表。
291    registry: MockRegistry,
292}
293
294impl MockitoHelper
295{
296    /// Create a new Mockito helper backed by the given registry.
297    /// 创建由给定注册表支持的新Mockito帮助器。
298    pub fn new(registry: &MockRegistry) -> Self
299    {
300        Self {
301            registry: registry.clone(),
302        }
303    }
304
305    /// Create a new Mockito helper backed by the global mock registry.
306    /// 创建由全局模拟注册表支持的新Mockito帮助器。
307    pub fn global() -> Self
308    {
309        Self {
310            registry: crate::mock_bean::global_mock_registry().clone(),
311        }
312    }
313
314    /// Begin configuring a mock interaction: `when(bean::method)`.
315    /// 开始配置模拟交互：`when(bean::method)`。
316    ///
317    /// Returns a [`MockInteraction`] that can be chained with
318    /// `then_return()` or `then_answer()`.
319    ///
320    /// 返回可以与 `then_return()` 或 `then_answer()` 链接的 [`MockInteraction`]。
321    pub fn when(&self, bean_name: &str, method_name: &str) -> MockInteraction<'_>
322    {
323        MockInteraction {
324            helper: self,
325            bean_name: bean_name.to_string(),
326            method_name: method_name.to_string(),
327        }
328    }
329
330    /// Invoke a mocked method by bean name and method name.
331    /// 通过bean名称和方法名称调用模拟方法。
332    ///
333    /// Returns the mock result downcast to `T`, or `None` if no mock is
334    /// registered or the downcast fails.
335    ///
336    /// 返回向下转换为 `T` 的模拟结果，如果没有注册模拟或向下转换失败则返回 `None`。
337    pub async fn call<T: Any + Send + Sync + Clone + 'static>(
338        &self,
339        bean_name: &str,
340        method_name: &str,
341        args: Vec<Arc<dyn Any + Send + Sync>>,
342    ) -> Option<T>
343    {
344        let result = self
345            .registry
346            .call_mock(bean_name, method_name, args)
347            .await?;
348
349        result.downcast::<T>().ok().map(|v| *v)
350    }
351
352    /// Verify that a mocked method was called at least once.
353    /// 验证模拟方法至少被调用过一次。
354    ///
355    /// Equivalent to Mockito's `verify(mock).method()`.
356    /// 等价于Mockito的 `verify(mock).method()`。
357    pub async fn verify(&self, bean_name: &str, method_name: &str) -> bool
358    {
359        self.registry.verify_called(bean_name, method_name).await
360    }
361
362    /// Get the number of times a mocked method was called.
363    /// 获取模拟方法被调用的次数。
364    ///
365    /// Equivalent to Mockito's `verify(mock, times(n)).method()`.
366    /// 等价于Mockito的 `verify(mock, times(n)).method()`。
367    pub async fn call_count(&self, bean_name: &str, method_name: &str) -> usize
368    {
369        self.registry.call_count(bean_name, method_name).await
370    }
371
372    /// Verify that a mocked method was called exactly `n` times.
373    /// 验证模拟方法恰好被调用了 `n` 次。
374    pub async fn verify_times(&self, bean_name: &str, method_name: &str, expected: usize) -> bool
375    {
376        self.registry
377            .verify_call_count(bean_name, method_name, expected)
378            .await
379    }
380
381    /// Reset a specific mock (clear call counts and behaviour).
382    /// 重置特定模拟（清除调用计数和行为）。
383    ///
384    /// Equivalent to Mockito's `reset(mock)`.
385    /// 等价于Mockito的 `reset(mock)`。
386    pub async fn reset(&self, bean_name: &str, method_name: &str)
387    {
388        self.registry.reset_mock(bean_name, method_name).await;
389    }
390
391    /// Reset all mocks.
392    /// 重置所有模拟。
393    pub async fn reset_all(&self)
394    {
395        self.registry.reset_all().await;
396    }
397
398    /// Get a reference to the underlying mock registry.
399    /// 获取底层模拟注册表的引用。
400    pub fn registry(&self) -> &MockRegistry
401    {
402        &self.registry
403    }
404}

410        f.debug_struct("MockitoHelper").finish()
411    }
412}
