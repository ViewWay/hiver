149    /// 打印服务器启动完成（Spring Boot 风格）
150    /// Print server started (Spring Boot style)
151    pub fn print_started(&self, class_name: &str, port: u16)
152    {
153        let elapsed = self.start_time.elapsed().as_millis();
154        let timestamp = format_timestamp();
155
156        println!();
157        println!(
158            "{} {} {} --- [           main] {} : Tomcat started on port(s): {} (http)",
159            timestamp,
160            "INFO".green(),
161            pid(),
162            "o.s.b.w.e.tomcat.TomcatWebServer".gray(),
163            port.to_string().cyan()
164        );
165        println!(
166            "{} {} {} --- [           main] {} : Started Application in {} seconds (JVM running \
167             for {})",
168            timestamp,
169            "INFO".green(),
170            pid(),
171            class_name,
172            format!("{}.{:03}", elapsed / 1000, elapsed % 1000).cyan(),
173            format!("{}.{:03}", elapsed / 1000, elapsed % 1000).cyan()
174        );
175        println!();
176    }

179/// 格式化时间戳（ISO 8601 格式）
180/// Format timestamp (ISO 8601 format)
181fn format_timestamp() -> String
182{
183    use std::time::{SystemTime, UNIX_EPOCH};
184    let duration = SystemTime::now()
185        .duration_since(UNIX_EPOCH)
186        .unwrap_or_default();
187    let secs = duration.as_secs();
188    let millis = duration.subsec_millis();
189
190    // Simple format: 2024-01-29T10:30:45 123
191    let days_since_epoch = secs / 86400;
192    let year = 1970 + (days_since_epoch / 365);
193    let day_of_year = (days_since_epoch % 365) as u32;
194    let month = (day_of_year / 30) + 1;
195    let day = (day_of_year % 30) + 1;
196
197    format!(
198        "{:04}-{:02}-{:02}T{:02}:{:02}:{:02} {:03}",
199        year,
200        month,
201        day,
202        (secs % 86400 / 3600) as u32,
203        (secs % 3600 / 60) as u32,
204        (secs % 60) as u32,
205        millis
206    )
207}

258/// 初始化 Hiver 运行时日志
259/// Initialize Hiver runtime logging
260///
261/// 使用 hiver-observability 统一日志系统。
262/// Uses hiver-observability unified logging system.
263///
264/// # 配置 / Configuration
265///
266/// 通过环境变量或配置文件控制：
267/// - `HIVER_LOG_LEVEL`: 日志级别 (TRACE, DEBUG, INFO, WARN, ERROR)
268/// - `HIVER_LOG_MODE`: 日志模式 (verbose, simple)
269/// - `HIVER_PROFILE`: Profile (dev→verbose, prod→simple)
270///
271/// # Example / 示例
272///
273/// ```rust,no_run,ignore
274/// use hiver_starter::core::logging::init_runtime_logging;
275///
276/// // 使用默认配置
277/// init_runtime_logging(None)?;
278///
279/// // 指定 profile
280/// init_runtime_logging(Some("dev"))?;
281/// ```
282pub fn init_runtime_logging(profile: Option<&str>) -> anyhow::Result<()>
283{
284    #[cfg(feature = "observability")]
285    {
286        // 使用 hiver-observability 统一日志系统
287        // Use hiver-observability unified logging system
288        use hiver_observability::log::{LogLevel, LogMode, Logger, LoggerConfig};
289
290        // 从环境变量或 profile 获取配置
291        let level = std::env::var("HIVER_LOG_LEVEL")
292            .ok()
293            .and_then(|s| LogLevel::from_str(&s))
294            .unwrap_or(LogLevel::Info);
295
296        let mode = if let Ok(mode_str) = std::env::var("HIVER_LOG_MODE")
297        {
298            LogMode::from_str(&mode_str).unwrap_or(LogMode::from_profile(profile))
299        }
300        else
301        {
302            LogMode::from_profile(profile)
303        };
304
305        let config = LoggerConfig {
306            level,
307            mode,
308            profile: profile.map(String::from),
309            ..Default::default()
310        };
311
312        Logger::init_with_config(&config)
313            .map_err(|e| anyhow::anyhow!("Logger init failed: {e}"))?;
314        Ok(())
315    }
316
317    #[cfg(not(feature = "observability"))]
318    {
319        // 回退到简单日志
320        let level = std::env::var("HIVER_LOG_LEVEL")
321            .or_else(|_| std::env::var("RUST_LOG"))
322            .ok()
323            .and_then(|s| s.parse().ok())
324            .unwrap_or(tracing::Level::INFO);
325
326        let subscriber = tracing_subscriber::fmt()
327            .with_max_level(level)
328            .with_target(false)
329            .with_thread_ids(false)
330            .with_file(false)
331            .with_line_number(false)
332            .with_level(true)
333            .with_ansi(true)
334            .finish();
335
336        let _ = tracing::subscriber::set_global_default(subscriber);
337        Ok(())
338    }
339}

341/// 日志级别图标（保留用于其他地方）
342/// Log level icons
343pub fn level_icon(level: &tracing::Level) -> &'static str
344{
345    match *level
346    {
347        tracing::Level::ERROR => "❌",
348        tracing::Level::WARN => "⚠️ ",
349        tracing::Level::INFO => "✨",
350        tracing::Level::DEBUG => "🔍",
351        tracing::Level::TRACE => "📝",
352    }
353}

358pub trait Colorize
359{
360    /// Convert to cyan color
361    /// 转换为青色
362    fn cyan(self) -> String;
363
364    /// Convert to green color
365    /// 转换为绿色

368    /// Convert to yellow color
369    /// 转换为黄色
370    fn yellow(self) -> String;
371
372    /// Convert to red color
373    /// 转换为红色
374    fn red(self) -> String;
375
376    /// Convert to gray color
377    /// 转换为灰色
378    fn gray(self) -> String;
379
380    /// Convert to bold text
381    /// 转换为粗体

383}

385impl Colorize for &str
386{
387    /// Convert to cyan color
388    /// 转换为青色

390    {
391        format!("\x1b[36m{}\x1b[0m", self)
392    }
393
394    /// Convert to green color
395    /// 转换为绿色
396    fn green(self) -> String
397    {
398        format!("\x1b[32m{}\x1b[0m", self)
399    }
400
401    /// Convert to yellow color
402    /// 转换为黄色
403    fn yellow(self) -> String
404    {
405        format!("\x1b[33m{}\x1b[0m", self)
406    }
407
408    /// Convert to red color
409    /// 转换为红色
410    fn red(self) -> String
411    {
412        format!("\x1b[31m{}\x1b[0m", self)
413    }
414
415    /// Convert to gray color
416    /// 转换为灰色
417    fn gray(self) -> String
418    {
419        format!("\x1b[90m{}\x1b[0m", self)
420    }
421
422    /// Convert to bold text
423    /// 转换为粗体
424    fn bold(self) -> String
