1//! Metrics module
2//! 指标模块
3//!
4//! # Overview / 概述
5//!
6//! This module provides metrics collection functionality compatible with Prometheus.
7//! It includes counters, gauges, histograms, and a metrics registry.
8//!
9//! 本模块提供与 Prometheus 兼容的指标收集功能。
10//! 包括计数器、仪表、直方图和指标注册表。
11//!
12//! # Equivalent to Spring Boot / 等价于 Spring Boot
13//!
14//! - Micrometer Metrics
15//! - Spring Boot Actuator Metrics
16//! - Prometheus `MeterRegistry`
17//!
18//! # Example / 示例
19//!
20//! ```rust,no_run,ignore
21//! use hiver_observability::metrics::{MetricsRegistry, Counter, Gauge, Histogram};
22//!
23//! let registry = MetricsRegistry::new();
24//!
25//! // Counter - monotonically increasing
26//! let counter = registry.counter("http_requests_total");
27//! counter.increment();
28//! counter.increment_by(5);
29//!
30//! // Gauge - can go up or down
31//! let gauge = registry.gauge("active_connections");
32//! gauge.set(10);
33//! gauge.increment();
34//! gauge.decrement();
35//!
36//! // Histogram - observe distributions
37//! let histogram = registry.histogram("request_duration_seconds");
38//! histogram.observe(0.123);
39//! histogram.observe(0.456);
40//! ```

53/// Metric ID
54/// 指标ID
55#[derive(Debug, Clone, PartialEq, Eq, Hash)]
56pub struct MetricId
57{
58    /// Metric name
59    /// 指标名称
60    pub name: String,
61
62    /// Metric labels
63    /// 指标标签
64    pub labels: Vec<(String, String)>,
65}
66
67impl MetricId
68{
69    /// Create a new metric ID
70    /// 创建新的指标ID
71    pub fn new(name: impl Into<String>) -> Self
72    {
73        Self {
74            name: name.into(),
75            labels: Vec::new(),
76        }
77    }
78
79    /// Create a new metric ID from a reference
80    /// 从引用创建新的指标ID
81    fn new_from_ref(name: &str) -> Self
82    {
83        Self {
84            name: name.to_string(),
85            labels: Vec::new(),
86        }
87    }
88
89    /// Add a label
90    /// 添加标签
91    pub fn with_label(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
92    {
93        self.labels.push((key.into(), value.into()));
94        self
95    }
96
97    /// Create a labeled metric ID
98    /// 创建带标签的指标ID
99    pub fn with_labels(mut self, labels: Vec<(String, String)>) -> Self
100    {
101        self.labels = labels;
102        self
103    }
104
105    /// Create a new `MetricId` with a suffix added to the name
106    /// 创建名称带后缀的新 `MetricId`
107    fn with_suffix(&self, suffix: &str) -> MetricId
108    {
109        MetricId {
110            name: format!("{}_{}", self.name, suffix),
111            labels: self.labels.clone(),
112        }
113    }
114}

157/// Counter metric - monotonically increasing value
158/// 计数器指标 - 单调递增值
159///
160/// Counters are used for things that only increase (e.g., number of requests).
161/// 计数器用于只增加的内容（例如请求数）。
162#[derive(Clone)]
163pub struct Counter
164{
165    /// Metric ID
166    /// 指标ID
167    id: MetricId,
168
169    /// Counter value
170    /// 计数器值
171    value: Arc<AtomicU64>,
172
173    /// Help text (interior mutable for shared updates)
174    /// 帮助文本（内部可变以支持共享更新）
175    help: Arc<RwLock<Option<String>>>,
176}
177
178impl Counter
179{
180    /// Create a new counter
181    /// 创建新的计数器
182    pub fn new(name: impl Into<String>) -> Self
183    {
184        Self {
185            id: MetricId::new(name),
186            value: Arc::new(AtomicU64::new(0)),
187            help: Arc::new(RwLock::new(None)),
188        }
189    }
190
191    /// Create a new labeled counter
192    /// 创建带标签的计数器
193    pub fn with_labels(name: impl Into<String>, labels: Vec<(String, String)>) -> Self
194    {
195        Self {
196            id: MetricId::new(name).with_labels(labels),
197            value: Arc::new(AtomicU64::new(0)),
198            help: Arc::new(RwLock::new(None)),
199        }
200    }
201
202    /// Set help text
203    /// 设置帮助文本
204    pub fn with_help(&self, help: impl Into<String>) -> Self
205    {
206        if let Ok(mut h) = self.help.write()
207        {
208            *h = Some(help.into());
209        }
210        self.clone()
211    }
212
213    /// Get the metric ID
214    /// 获取指标ID
215    pub fn id(&self) -> &MetricId
216    {
217        &self.id
218    }
219
220    /// Get the current value
221    /// 获取当前值
222    pub fn get(&self) -> u64
223    {
224        self.value.load(Ordering::Relaxed)
225    }
226
227    /// Increment by 1
228    /// 增加1
229    pub fn increment(&self)
230    {
231        self.value.fetch_add(1, Ordering::Relaxed);
232    }
233
234    /// Increment by a specific amount
235    /// 增加特定数量
236    pub fn increment_by(&self, amount: u64)
237    {
238        self.value.fetch_add(amount, Ordering::Relaxed);
239    }
240
241    /// Reset to zero
242    /// 重置为零
243    pub fn reset(&self)
244    {
245        self.value.store(0, Ordering::Relaxed);
246    }
247
248    /// Get help text
249    /// 获取帮助文本
250    pub fn help(&self) -> Option<String>
251    {
252        self.help.read().ok().and_then(|h| h.clone())
253    }
254}

373/// Histogram bucket
374/// 直方图桶
375#[derive(Debug, Clone)]
376pub struct Bucket
377{
378    /// Upper bound (None means +Inf)
379    /// 上界（None表示+Inf）
380    pub upper_bound: Option<f64>,
381
382    /// Cumulative count
383    /// 累积计数
384    pub count: u64,
385}
386
387impl Bucket
388{
389    /// Create a new bucket
390    /// 创建新的桶
391    pub fn new(upper_bound: Option<f64>) -> Self
392    {
393        Self {
394            upper_bound,
395            count: 0,
396        }
397    }
398
399    /// Get the label for this bucket
400    /// 获取此桶的标签
401    pub fn label(&self) -> String
402    {
403        match self.upper_bound
404        {
405            Some(bound) => format!("{:.6}", bound),
406            None => "+Inf".to_string(),
407        }
408    }
409
410    /// Check if this is the infinity bucket
411    /// 检查这是否是无穷大桶
412    pub fn is_infinity(&self) -> bool
413    {
414        self.upper_bound.is_none()
415    }
416}

427    /// 指标ID
428    id: MetricId,

430    /// Histogram data
431    /// 直方图数据
432    data: Arc<RwLock<HistogramData>>,
433
434    /// Help text (interior mutable for shared updates)
435    /// 帮助文本（内部可变以支持共享更新）
436    help: Arc<RwLock<Option<String>>>,
437}

439/// Histogram data (interior mutable)
440/// 直方图数据（内部可变）
441#[derive(Debug)]
442struct HistogramData
443{
444    /// Total sum of observed values
445    /// 观察值的总和
446    sum: f64,
447
448    /// Total count of observations
449    /// 观察总数
450    count: u64,
451
452    /// Bucket definitions
453    /// 桶定义
454    buckets: Vec<Option<f64>>,
455
456    /// Bucket counts
457    /// 桶计数
458    bucket_counts: Vec<u64>,
459}
460
461impl HistogramData
462{
463    /// Create new histogram data with default buckets
464    /// 使用默认桶创建新的直方图数据
465    fn new() -> Self
466    {
467        // Default Prometheus buckets
468        let buckets = vec![
469            Some(0.005),
470            Some(0.01),
471            Some(0.025),
472            Some(0.05),
473            Some(0.1),
474            Some(0.25),
475            Some(0.5),
476            Some(1.0),
477            Some(2.5),
478            Some(5.0),
479            Some(10.0),
480            None, // +Inf
481        ];
482        Self {
483            sum: 0.0,
484            count: 0,
485            buckets: buckets.clone(),
486            bucket_counts: vec![0; buckets.len()],
487        }
488    }
489
490    /// Create new histogram data with custom buckets
491    /// 使用自定义桶创建新的直方图数据
492    fn with_buckets(mut buckets: Vec<Option<f64>>) -> Self
493    {
494        // Ensure buckets are sorted and end with +Inf
495        let has_infinity = buckets.iter().any(Option::is_none);
496        buckets.sort_by(|a, b| match (a, b)
497        {
498            (Some(va), Some(vb)) => va.partial_cmp(vb).unwrap_or(std::cmp::Ordering::Equal),
499            (Some(_), None) => std::cmp::Ordering::Less,
500            (None, Some(_)) => std::cmp::Ordering::Greater,
501            (None, None) => std::cmp::Ordering::Equal,
502        });
503        if !has_infinity
504        {
505            buckets.push(None);
506        }
507
508        Self {
509            sum: 0.0,
510            count: 0,
511            bucket_counts: vec![0; buckets.len()],
512            buckets,
513        }
514    }
515
516    /// Observe a value
517    /// 观察值
518    fn observe(&mut self, value: f64)
519    {
520        self.sum += value;
521        self.count += 1;
522
523        // Find the appropriate bucket
524        for (i, upper_bound) in self.buckets.iter().enumerate()
525        {
526            if upper_bound.is_none_or(|ub| value <= ub)
527                && let Some(count) = self.bucket_counts.get_mut(i)
528            {
529                *count += 1;
530            }
531        }
532    }
533
534    /// Get all buckets with cumulative counts
535    /// 获取所有桶及其累积计数
536    fn get_buckets(&self) -> Vec<Bucket>
537    {
538        let mut cumulative = 0u64;
539        self.buckets
540            .iter()
541            .zip(self.bucket_counts.iter())
542            .map(|(upper_bound, &count)| {
543                cumulative += count;
544                Bucket {
545                    upper_bound: *upper_bound,
546                    count: cumulative,
547                }
548            })
549            .collect()
550    }
551}

724/// Metrics registry - central storage for metrics
725/// 指标注册表 - 指标的中央存储
726///
727/// The registry manages all metrics and provides export functionality.
728/// 注册表管理所有指标并提供导出功能。
729#[derive(Clone)]
730pub struct MetricsRegistry
731{
732    /// Registered metrics
733    /// 已注册的指标
734    metrics: Arc<RwLock<HashMap<MetricId, Metric>>>,
735
736    /// Common labels to add to all metrics
737    /// 添加到所有指标的公共标签
738    common_labels: Arc<Vec<(String, String)>>,
739}
740
741impl Default for MetricsRegistry
742{
743    fn default() -> Self
744    {
745        Self::new()
746    }
747}
748
749impl MetricsRegistry
750{
751    /// Create a new metrics registry
752    /// 创建新的指标注册表
753    pub fn new() -> Self
754    {
755        Self {
756            metrics: Arc::new(RwLock::new(HashMap::new())),
757            common_labels: Arc::new(Vec::new()),
758        }
759    }
760
761    /// Create a registry with common labels
762    /// 创建带有公共标签的注册表
763    pub fn with_common_labels(labels: Vec<(String, String)>) -> Self
764    {
765        Self {
766            metrics: Arc::new(RwLock::new(HashMap::new())),
767            common_labels: Arc::new(labels),
768        }
769    }
770
771    /// Register or retrieve a counter
772    /// 注册或检索计数器
773    pub fn counter(&self, name: impl Into<String>) -> Counter
774    {
775        self.counter_with_labels(name, Vec::new())
776    }
777
778    /// Register or retrieve a labeled counter
779    /// 注册或检索带标签的计数器
780    pub fn counter_with_labels(
781        &self,
782        name: impl Into<String>,
783        labels: Vec<(String, String)>,
784    ) -> Counter
785    {
786        let name = name.into();
787        let mut all_labels = (*self.common_labels).clone();
788        all_labels.extend(labels);
789
790        let id = MetricId::new_from_ref(&name).with_labels(all_labels.clone());
791
792        let mut metrics = self
793            .metrics
794            .write()
795            .unwrap_or_else(std::sync::PoisonError::into_inner);
796        if let Some(Metric::Counter(counter)) = metrics.get(&id)
797        {
798            return counter.clone();
799        }
800
801        let counter = Counter::with_labels(name, all_labels);
802        metrics.insert(id, Metric::Counter(counter.clone()));
803        counter
804    }
805
806    /// Register or retrieve a gauge
807    /// 注册或检索仪表
808    pub fn gauge(&self, name: impl Into<String>) -> Gauge
809    {
810        self.gauge_with_labels(name, Vec::new())
811    }

816    -> Gauge
817    {
818        let name = name.into();
819        let mut all_labels = (*self.common_labels).clone();
820        all_labels.extend(labels);
821
822        let id = MetricId::new_from_ref(&name).with_labels(all_labels.clone());
823
824        let mut metrics = self
825            .metrics
826            .write()
827            .unwrap_or_else(std::sync::PoisonError::into_inner);
828        if let Some(Metric::Gauge(gauge)) = metrics.get(&id)
829        {
830            return gauge.clone();
831        }
832
833        let gauge = Gauge::with_labels(name, all_labels);
834        metrics.insert(id, Metric::Gauge(gauge.clone()));
835        gauge
836    }

883    /// Export metrics in Prometheus text format
884    /// 以Prometheus文本格式导出指标
885    pub fn export_prometheus(&self) -> String
886    {
887        let mut output = String::new();
888
889        let metrics = self.get_all();
890
891        // Group by name (ignoring labels)
892        let mut by_name: HashMap<String, Vec<&Metric>> = HashMap::new();
893        for metric in &metrics
894        {
895            by_name
896                .entry(metric.id().name.clone())
897                .or_default()
898                .push(metric);
899        }
900
901        for (name, group) in by_name
902        {
903            // Get help text from first metric in group
904            if let Some(help) = group.first().and_then(|m| m.help())
905            {
906                output.push_str("# HELP ");
907                output.push_str(&name);
908                output.push(' ');
909                output.push_str(&help);
910                output.push('\n');
911            }
912
913            // Get type
914            if let Some(first) = group.first()
915            {
916                output.push_str("# TYPE ");
917                output.push_str(&name);
918                output.push(' ');
919                output.push_str(first.metric_type().as_str());
920                output.push('\n');
921            }
922
923            // Export each metric
924            for metric in group
925            {
926                match metric
927                {
928                    Metric::Counter(counter) =>
929                    {
930                        output.push_str(&export_metric_line(counter.id(), counter.get()));
931                    },
932                    Metric::Gauge(gauge) =>
933                    {
934                        output.push_str(&export_metric_line(gauge.id(), gauge.get() as u64));
935                    },
936                    Metric::Histogram(histogram) =>
937                    {
938                        // Export bucket counts with _bucket suffix
939                        let bucket_name = histogram.id().with_suffix("bucket");
940                        for bucket in histogram.buckets()
941                        {
942                            let mut bucket_id = bucket_name.clone();
943                            bucket_id.labels.push(("le".to_string(), bucket.label()));
944                            output.push_str(&export_metric_line(&bucket_id, bucket.count));
945                        }
946                        // Export sum and count with proper suffixes
947                        let sum_id = histogram.id().with_suffix("sum");
948                        output.push_str(&export_metric_line(&sum_id, histogram.sum() as u64));
949
950                        let count_id = histogram.id().with_suffix("count");
951                        output.push_str(&export_metric_line(&count_id, histogram.count()));
952                    },
953                }
954            }
955        }
956
957        output
958    }

971/// Export a single metric line in Prometheus format
972/// 以Prometheus格式导出单个指标行
973fn export_metric_line(id: &MetricId, value: u64) -> String
974{
975    let mut line = id.name.clone();
976
977    if !id.labels.is_empty()
978    {
979        line.push('{');
980        let labels: Vec<String> = id
981            .labels
982            .iter()
983            .map(|(k, v)| format!("{}=\"{}\"", k, v))
984            .collect();
985        line.push_str(&labels.join(","));
986        line.push('}');
987    }
988
989    // Handle suffixes like _sum, _count which have empty value labels
990    let labels: Vec<_> = id
991        .labels
992        .iter()
993        .filter(|(k, _)| !k.starts_with('_'))
994        .collect();
995    if !labels.is_empty() && id.labels.len() != labels.len()
996    {
997        // Rebuild line without empty labels
998        line.clone_from(&id.name);
999        line.push('{');
1000        let label_strings: Vec<String> = labels
1001            .iter()
1002            .map(|(k, v)| format!("{}=\"{}\"", k, v))
1003            .collect();
1004        line.push_str(&label_strings.join(","));
1005        line.push('}');
1006    }
1007    else if labels.is_empty() && !id.labels.is_empty()
1008    {
1009        // Only suffix labels, remove braces
1010        line.clone_from(&id.name);
1011    }
1012
1013    // Handle histogram suffixes
1014    for (k, _) in &id.labels
1015    {
1016        if k.starts_with('_') && k != "_sum" && k != "_count"
1017        {
1018            // This is a bucket label, don't add to name
1019            break;
1020        }
1021        if k == "_sum" || k == "_count"
1022        {
1023            line = format!("{}{}", id.name, k);
1024        }
1025    }
1026
1027    // Format value
1028    let formatted_value = value.to_string();
1029
1030    line.push(' ');
1031    line.push_str(&formatted_value);
1032    line.push('\n');
1033
1034    line
1035}
