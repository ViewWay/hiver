1//! Unified API Response Structure / 统一 API 响应结构
2//!
3//! # Equivalent to Spring Boot / 等价于 Spring Boot
4//!
5//! - `ResponseEntity<T>` - Wrapper for API responses
6//! - `@ResponseBody` + `ResponseAdvice` - Automatic response wrapping
7//! - `Result<T>` / `ApiResponse<T>` - Standardized response format
8//!
9//! # Example / 示例
10//!
11//! ```rust,no_run,ignore
12//! use hiver_http::ApiResponse;
13//! use hiver_http::response::Response;
14//!
15//! #[hiver_macros::get("/users/{id}")]
16//! fn get_user(id: u64) -> Response {
17//!     let user = user_service.find_by_id(id)?;
18//!     ApiResponse::success(user).into_response()
19//! }
20//!
21//! // Error handling
22//! #[hiver_macros::get("/users/{id}")]
23//! fn get_user(id: u64) -> Response {
24//!     ApiResponse::not_found()
25//!         .code("USER_NOT_FOUND")
26//!         .message(&format!("User {} not found", id))
27//!         .into_response()
28//! }
29//! ```

41/// Standard result codes
42/// 标准结果码
43#[derive(Debug, Clone, Copy, PartialEq, Eq)]
44pub enum ResultCode
45{
46    // Success / 成功
47    /// 200 OK - Request succeeded / 请求成功
48    Success = 200,
49    /// 201 Created - Resource created / 资源已创建
50    Created = 201,
51    /// 202 Accepted - Request accepted / 请求已接受
52    Accepted = 202,
53    /// 204 No Content - Success with no response body / 成功但无响应体
54    NoContent = 204,
55
56    // Client errors / 客户端错误
57    /// 400 Bad Request - Invalid request / 无效请求
58    BadRequest = 400,
59    /// 401 Unauthorized - Authentication required / 需要认证
60    Unauthorized = 401,
61    /// 403 Forbidden - Access denied / 访问被拒绝
62    Forbidden = 403,
63    /// 404 Not Found - Resource not found / 资源未找到
64    NotFound = 404,
65    /// 405 Method Not Allowed - Method not supported / 方法不支持
66    MethodNotAllowed = 405,
67    /// 409 Conflict - Resource conflict / 资源冲突
68    Conflict = 409,
69    /// 422 Unprocessable Entity - Invalid data / 无效数据
70    UnprocessableEntity = 422,
71    /// 429 Too Many Requests - Rate limit exceeded / 超过速率限制
72    TooManyRequests = 429,
73
74    // Server errors / 服务端错误
75    /// 500 Internal Server Error - Server error / 服务器错误
76    InternalError = 500,
77    /// 501 Not Implemented - Feature not implemented / 功能未实现
78    NotImplemented = 501,
79    /// 502 Bad Gateway - Invalid gateway response / 网关响应无效
80    BadGateway = 502,
81    /// 503 Service Unavailable - Service unavailable / 服务不可用
82    ServiceUnavailable = 503,
83}

160/// Unified API response structure
161/// 统一 API 响应结构
162///
163/// This is equivalent to Spring's `ResponseEntity<T>` or a custom `Result<T>` wrapper.
164/// 这等价于 Spring 的 `ResponseEntity<T>` 或自定义 `Result<T>` 包装器。
165///
166/// # JSON Format / JSON 格式
167///
168/// Success response / 成功响应:
169/// ```json
170/// {
171///   "code": 200,
172///   "message": "Success",
173///   "data": { "id": 1, "name": "Alice" },
174///   "timestamp": "2024-01-29T10:30:45Z"
175/// }
176/// ```
177///
178/// Error response / 错误响应:
179/// ```json
180/// {
181///   "code": 404,
182///   "message": "Not Found",
183///   "error": "USER_NOT_FOUND",
184///   "timestamp": "2024-01-29T10:30:45Z"
185/// }
186/// ```
187#[derive(Debug, Clone, Serialize, Deserialize)]
188pub struct ApiResponse<T>
189{
190    /// Response code / 响应码
191    pub code: u16,
192
193    /// Response message / 响应消息
194    pub message: String,
195
196    /// Response data / 响应数据
197    #[serde(skip_serializing_if = "Option::is_none")]
198    pub data: Option<T>,
199
200    /// Error code (for error responses) / 错误码（用于错误响应）
201    #[serde(skip_serializing_if = "Option::is_none")]
202    pub error: Option<String>,
203
204    /// Request path / 请求路径
205    #[serde(skip_serializing_if = "Option::is_none")]
206    pub path: Option<String>,
207
208    /// Response timestamp / 响应时间戳
209    pub timestamp: String,
210
211    /// Additional errors (for validation errors) / 额外错误（用于校验错误）
212    #[serde(skip_serializing_if = "HashMap::is_empty")]
213    pub errors: HashMap<String, String>,
214}

216impl<T> ApiResponse<T>
217{
218    /// Create a new API response
219    /// 创建新的 API 响应
220    pub fn new(code: u16, message: impl Into<String>) -> Self
221    {
222        Self {
223            code,
224            message: message.into(),
225            data: None,
226            error: None,
227            path: None,
228            timestamp: format_timestamp(),
229            errors: HashMap::new(),
230        }
231    }
232
233    /// Create a success response with data
234    /// 创建带数据的成功响应
235    pub fn success_data(data: T) -> Self
236    {
237        Self {
238            code: ResultCode::Success.code(),
239            message: ResultCode::Success.message().to_string(),
240            data: Some(data),
241            error: None,
242            path: None,
243            timestamp: format_timestamp(),
244            errors: HashMap::new(),
245        }
246    }
247
248    /// Create a 201 Created response
249    /// 创建 201 Created 响应
250    pub fn created(data: T) -> Self
251    {
252        Self {
253            code: ResultCode::Created.code(),
254            message: ResultCode::Created.message().to_string(),
255            data: Some(data),
256            error: None,
257            path: None,
258            timestamp: format_timestamp(),
259            errors: HashMap::new(),
260        }
261    }
262
263    /// Set the response code
264    /// 设置响应码
265    pub fn code(mut self, code: u16) -> Self
266    {
267        self.code = code;
268        self
269    }
270
271    /// Set the response message
272    /// 设置响应消息
273    pub fn message(mut self, message: impl Into<String>) -> Self
274    {
275        self.message = message.into();
276        self
277    }
278
279    /// Set the error code
280    /// 设置错误码
281    pub fn error(mut self, error: impl Into<String>) -> Self
282    {
283        self.error = Some(error.into());
284        self
285    }
286
287    /// Set the request path
288    /// 设置请求路径
289    pub fn path(mut self, path: impl Into<String>) -> Self
290    {
291        self.path = Some(path.into());
292        self
293    }
294
295    /// Add an error field
296    /// 添加错误字段
297    pub fn add_error(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
298    {
299        self.errors.insert(key.into(), value.into());
300        self
301    }
302
303    /// Convert to HTTP Response
304    /// 转换为 HTTP Response
305    pub fn into_response(self) -> Response
306    where
307        T: Serialize,
308    {
309        let json_body = serde_json::to_string(&self).unwrap_or_default();
310        let status = StatusCode::from_u16(self.code);
311
312        Response::builder()
313            .status(status)
314            .header("content-type", "application/json")
315            .body(Body::from(json_body))
316            .unwrap_or_else(|_| Response::internal_server_error())
317    }
318
319    /// Check if this is a success response (2xx)
320    /// 检查是否为成功响应 (2xx)
321    pub fn is_success(&self) -> bool
322    {
323        (200..300).contains(&self.code)
324    }
325
326    /// Check if this is a client error (4xx)
327    /// 检查是否为客户端错误 (4xx)
328    pub fn is_client_error(&self) -> bool
329    {
330        (400..500).contains(&self.code)
331    }
332
333    /// Check if this is a server error (5xx)
334    /// 检查是否为服务端错误 (5xx)
335    pub fn is_server_error(&self) -> bool
336    {
337        (500..600).contains(&self.code)
338    }
339}

444/// Paginated response structure
445/// 分页响应结构
446///
447/// Equivalent to Spring's `Page<T>` or `PageResult<T>`.
448/// 等价于 Spring 的 `Page<T>` 或 `PageResult<T>`。
449#[derive(Debug, Clone, Serialize, Deserialize)]
450pub struct PageResponse<T>
451{
452    /// Page content / 页面内容
453    pub content: Vec<T>,
454
455    /// Current page number (0-based) / 当前页码（从0开始）
456    pub page: u32,
457
458    /// Page size / 页面大小
459    pub size: u32,
460
461    /// Total number of elements / 总元素数
462    pub total_elements: u64,
463
464    /// Total number of pages / 总页数
465    pub total_pages: u32,
466
467    /// Whether this is the first page / 是否为第一页
468    pub first: bool,
469
470    /// Whether this is the last page / 是否为最后一页
471    pub last: bool,
472}
473
474impl<T> PageResponse<T>
475{
476    /// Create a new page response
477    /// 创建新的分页响应
478    pub fn new(content: Vec<T>, page: u32, size: u32, total_elements: u64) -> Self
479    {
480        #[allow(clippy::cast_precision_loss)]
481        let total_pages = if size == 0
482        {
483            0
484        }
485        else
486        {
487            ((total_elements as f64) / (size as f64)).ceil() as u32
488        };
489
490        Self {
491            content,
492            page,
493            size,
494            total_elements,
495            total_pages,
496            first: page == 0,
497            last: page + 1 >= total_pages,
498        }
499    }
500
501    /// Create an empty page response
502    /// 创建空的分页响应
503    pub fn empty() -> Self
504    {
505        Self::new(vec![], 0, 10, 0)
506    }
507
508    /// Wrap the page content in an `ApiResponse`
509    /// 将分页内容包装为 `ApiResponse`
510    pub fn into_api_response(self) -> ApiResponse<Vec<T>>
511    {
512        ApiResponse::success_data(self.content)
513            .add_error("page", self.page.to_string())
514            .add_error("size", self.size.to_string())
515            .add_error("total_elements", self.total_elements.to_string())
516            .add_error("total_pages", self.total_pages.to_string())
517    }
518}

524/// Format current timestamp in ISO 8601 format
525/// 格式化当前时间戳为 ISO 8601 格式
526fn format_timestamp() -> String
527{
528    use std::time::{SystemTime, UNIX_EPOCH};
529    let duration = SystemTime::now()
530        .duration_since(UNIX_EPOCH)
531        .unwrap_or_default();
532    let secs = duration.as_secs();
533
534    format!(
535        "{:04}-{:02}-{:02}T{:02}:{:02}:{:02}Z",
536        (1970 + secs / 31_536_000) % 10000,
537        ((secs % 31_536_000) / 2_592_000 + 1) % 13,
538        ((secs % 2_592_000) / 86_400 + 1) % 32,
539        (secs % 86_400) / 3600,
540        (secs % 3600) / 60,
541        secs % 60
542    )
543}
