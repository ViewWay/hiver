1//! 统一响应结构 / Unified response structure
2//!
3//! 等价于 Spring Boot 的 Result<T> / Equivalent to Spring Boot's Result<T>
4//!
5//! # Example / 示例
6//!
7//! ```rust,ignore
8//! use hiver_response::Result;
9//! use serde::Serialize;
10//!
11//! #[derive(Serialize)]
12//! struct User {
13//!     id: u64,
14//!     name: String,
15//! }
16//!
17//! // Success response
18//! let response: Result<User> = Result::success(user);
19//!

31/// 响应码枚举 / Response code enum
32#[derive(Debug, Clone, Copy, PartialEq, Eq)]
33pub enum ResultCode
34{
35    /// 成功 / Success
36    Success = 200,
37    /// 已创建 / Created
38    Created = 201,
39    /// 无内容 / No Content
40    NoContent = 204,
41    /// 错误请求 / Bad Request
42    BadRequest = 400,
43    /// 未认证 / Unauthorized
44    Unauthorized = 401,
45    /// 禁止访问 / Forbidden
46    Forbidden = 403,
47    /// 未找到 / Not Found
48    NotFound = 404,
49    /// 冲突 / Conflict
50    Conflict = 409,
51    /// 服务器错误 / Internal Server Error
52    InternalError = 500,
53}
54
55impl ResultCode
56{
57    /// 获取HTTP状态码 / Get HTTP status code
58    pub fn status_code(&self) -> StatusCode
59    {
60        match self
61        {
62            ResultCode::Success => StatusCode::OK,
63            ResultCode::Created => StatusCode::CREATED,
64            ResultCode::NoContent => StatusCode::NO_CONTENT,
65            ResultCode::BadRequest => StatusCode::BAD_REQUEST,
66            ResultCode::Unauthorized => StatusCode::UNAUTHORIZED,
67            ResultCode::Forbidden => StatusCode::FORBIDDEN,
68            ResultCode::NotFound => StatusCode::NOT_FOUND,
69            ResultCode::Conflict => StatusCode::CONFLICT,
70            ResultCode::InternalError => StatusCode::INTERNAL_SERVER_ERROR,
71        }
72    }
73
74    /// 获取描述 / Get description
75    pub fn description(&self) -> &'static str
76    {
77        match self
78        {
79            ResultCode::Success => "操作成功 / Operation successful",
80            ResultCode::Created => "创建成功 / Created successfully",
81            ResultCode::NoContent => "无内容 / No content",
82            ResultCode::BadRequest => "请求参数错误 / Bad request",
83            ResultCode::Unauthorized => "未认证 / Unauthorized",
84            ResultCode::Forbidden => "无权限 / Forbidden",
85            ResultCode::NotFound => "资源不存在 / Resource not found",
86            ResultCode::Conflict => "资源冲突 / Resource conflict",
87            ResultCode::InternalError => "服务器错误 / Internal server error",
88        }
89    }
90}
91
92impl From<ResultCode> for StatusCode
93{
94    fn from(code: ResultCode) -> Self
95    {
96        code.status_code()
97    }
98}

100/// 统一响应结构 / Unified response structure
101///
102/// 等价于 Spring Boot 的 Result<T>
103#[derive(Debug, Clone, Serialize)]
104pub struct Result<T>
105{
106    /// 响应码 / Response code
107    pub code: u16,
108    /// 响应消息 / Response message
109    pub message: String,
110    /// 响应数据 / Response data
111    #[serde(skip_serializing_if = "Option::is_none")]
112    pub data: Option<T>,
113    /// 时间戳 / Timestamp
114    pub timestamp: i64,
115    /// 请求路径 / Request path
116    #[serde(skip_serializing_if = "Option::is_none")]
117    pub path: Option<String>,
118    /// 错误详情 / Error details
119    #[serde(skip_serializing_if = "HashMap::is_empty")]
120    pub errors: HashMap<String, String>,
121}
122
123impl<T> Result<T>
124{
125    /// 创建成功响应（无数据） / Create success response without data
126    pub fn ok() -> Self
127    {
128        Self::success_with_message("成功 / Success".to_string())
129    }
130
131    /// 创建成功响应（带数据） / Create success response with data
132    pub fn success(data: T) -> Self
133    {
134        Self {
135            code: ResultCode::Success as u16,
136            message: ResultCode::Success.description().to_string(),
137            data: Some(data),
138            timestamp: chrono::Utc::now().timestamp(),
139            path: None,
140            errors: HashMap::new(),
141        }
142    }
143
144    /// 创建成功响应（自定义消息） / Create success response with custom message
145    pub fn success_with_message(message: String) -> Self
146    {
147        Self {
148            code: ResultCode::Success as u16,
149            message,
150            data: None,
151            timestamp: chrono::Utc::now().timestamp(),
152            path: None,
153            errors: HashMap::new(),
154        }
155    }
156
157    /// 创建错误响应 / Create error response
158    pub fn error(code: u16, message: impl Into<String>) -> Self
159    {
160        Self {
161            code,
162            message: message.into(),
163            data: None,
164            timestamp: chrono::Utc::now().timestamp(),
165            path: None,
166            errors: HashMap::new(),
167        }
168    }
169
170    /// 创建错误响应（ResultCode） / Create error response with `ResultCode`
171    pub fn from_code(code: ResultCode) -> Self
172    {
173        Self::error(code as u16, code.description())
174    }
175
176    /// 创建400错误 / Create 400 error
177    pub fn bad_request(message: impl Into<String>) -> Self
178    {
179        Self::error(ResultCode::BadRequest as u16, message)
180    }
181
182    /// 创建401错误 / Create 401 error
183    pub fn unauthorized(message: impl Into<String>) -> Self
184    {
185        Self::error(ResultCode::Unauthorized as u16, message)
186    }
187
188    /// 创建403错误 / Create 403 error
189    pub fn forbidden(message: impl Into<String>) -> Self
190    {
191        Self::error(ResultCode::Forbidden as u16, message)
192    }
193
194    /// 创建404错误 / Create 404 error
195    pub fn not_found(resource: impl Into<String>, id: impl Into<String>) -> Self
196    {
197        Self::error(
198            ResultCode::NotFound as u16,
199            format!("{} {}: 不存在 / not found", resource.into(), id.into()),
200        )
201    }
202
203    /// 创建409错误 / Create 409 error
204    pub fn conflict(message: impl Into<String>) -> Self
205    {
206        Self::error(ResultCode::Conflict as u16, message)
207    }
208
209    /// 创建500错误 / Create 500 error
210    pub fn internal_error(message: impl Into<String>) -> Self
211    {
212        Self::error(ResultCode::InternalError as u16, message)
213    }
214
215    /// 设置请求路径 / Set request path
216    pub fn with_path(mut self, path: impl Into<String>) -> Self
217    {
218        self.path = Some(path.into());
219        self
220    }
221
222    /// 设置错误详情 / Set error details
223    pub fn with_errors(mut self, errors: HashMap<String, String>) -> Self
224    {
225        self.errors = errors;
226        self
227    }
228
229    /// 添加单个错误详情 / Add single error detail
230    pub fn add_error(mut self, field: impl Into<String>, message: impl Into<String>) -> Self
231    {
232        self.errors.insert(field.into(), message.into());
233        self
234    }
235
236    /// 检查是否成功 / Check if successful
237    pub fn is_success(&self) -> bool
238    {
239        self.code >= 200 && self.code < 300
240    }
241
242    /// 检查是否失败 / Check if failed
243    pub fn is_error(&self) -> bool
244    {
245        !self.is_success()
246    }
247}
248
249impl<T> IntoResponse for Result<T>
250where
251    T: Serialize,
252{
253    fn into_response(self) -> Response
254    {
255        let _ = StatusCode::from_u16(self.code).unwrap_or(StatusCode::INTERNAL_SERVER_ERROR);
256
257        Json(self).into_response()
258    }
259}

261/// 分页结果 / Paginated result
262#[derive(Debug, Clone, Serialize)]
263pub struct PageResult<T>
264{
265    /// 数据列表 / Data list
266    pub content: Vec<T>,
267    /// 当前页 / Current page (0-indexed)
268    pub page: u32,
269    /// 每页数量 / Page size
270    pub size: u32,
271    /// 总元素数 / Total elements
272    pub total_elements: u64,
273    /// 总页数 / Total pages
274    pub total_pages: u32,
275}
276
277impl<T> PageResult<T>
278{
279    /// 创建分页结果 / Create paginated result
280    #[allow(clippy::cast_precision_loss)]
281    pub fn new(content: Vec<T>, page: u32, size: u32, total_elements: u64) -> Self
282    {
283        let total_pages = if size == 0
284        {
285            0
286        }
287        else
288        {
289            ((total_elements as f64) / (size as f64)).ceil() as u32
290        };
291
292        Self {
293            content,
294            page,
295            size,
296            total_elements,
297            total_pages,
298        }
299    }
300
301    /// 创建空分页结果 / Create empty paginated result
302    pub fn empty() -> Self
303    {
304        Self::new(vec![], 0, 10, 0)
305    }
306
307    /// 创建成功的分页响应 / Create success paginated response
308    pub fn success(self) -> Result<Vec<T>>
309    {
310        Result::success(self.content)
311    }
312}
