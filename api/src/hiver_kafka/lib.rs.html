1//! Hiver Kafka - Apache Kafka module
2//! Hiver Kafka - Apache Kafka模块
3//!
4//! # Equivalent to Spring Boot / 等价于 Spring Boot
5//!
6//! - `@KafkaListener` - `ConsumerListener`
7//! - `KafkaTemplate` - Producer
8//! - `@TopicPartition` - `TopicPartition`
9//! - `@Header` - `MessageHeader`
10//! - `ConsumerConfig` - `ConsumerProperties`
11//! - `ProducerConfig` - `ProducerProperties`
12//!
13//! # Example / 示例
14//!
15//! ```rust,no_run,ignore
16//! use hiver_kafka::{Producer, Consumer, ConsumerConfig, TopicPartition};
17//!
18//! #[tokio::main]
19//! async fn main() {
20//!     let producer = Producer::new("localhost:9092").await.unwrap();
21//!     producer.send("my_topic", "key", b"Hello, Kafka!").await.unwrap();
22//!
23//!     let config = ConsumerConfig::new("my_group");
24//!     let consumer = Consumer::new("localhost:9092", &config).await.unwrap();
25//!     consumer.subscribe(&["my_topic"]).await.unwrap();
26//!
27//!     while let Some(message) = consumer.recv().await {
28//!         println!("Received: {:?}", message);
29//!     }
30//! }
31//! ```
