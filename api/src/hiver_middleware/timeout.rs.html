18/// Timeout middleware
19/// 超时中间件
20///
21/// Equivalent to Spring's:
22/// - `@RequestTimeout`
23/// - `TimeoutWebHandlerExecutor`
24/// - Resilience4j's `TimeLimiter`
25///
26/// 这等价于Spring的：
27/// - `@RequestTimeout`
28/// - `TimeoutWebHandlerExecutor`
29/// - Resilience4j的 `TimeLimiter`
30///
31/// # Example / 示例
32///
33/// ```rust,no_run,ignore
34/// use hiver_router::Router;
35/// use hiver_middleware::TimeoutMiddleware;
36/// use std::sync::Arc;
37/// use std::time::Duration;
38///
39/// let timeout = Arc::new(TimeoutMiddleware::new(Duration::from_secs(30)));
40/// let router = Router::new()
41///     .middleware(timeout)
42///     .get("/", handler);
43/// ```
44#[derive(Clone)]
45pub struct TimeoutMiddleware
46{
47    /// Request timeout duration
48    /// 请求超时时长
49    pub timeout: Duration,
50}
51
52impl TimeoutMiddleware
53{
54    /// Create a new timeout middleware
55    /// 创建新的超时中间件
56    pub fn new(timeout: Duration) -> Self
57    {
58        Self { timeout }
59    }
60
61    /// Set the timeout duration
62    /// 设置超时时长
63    pub fn with_timeout(mut self, timeout: Duration) -> Self
64    {
65        self.timeout = timeout;
66        self
67    }
68}

70impl<S> Middleware<S> for TimeoutMiddleware
71where
72    S: Send + Sync + 'static,
73{
74    fn call(
75        &self,
76        req: Request,
77        state: Arc<S>,
78        next: Next<S>,
79    ) -> Pin<Box<dyn Future<Output = Result<Response>> + Send>>
80    {
81        let timeout = self.timeout;
82
83        Box::pin(async move {
84            // Race the handler against a timer. IMPORTANT: use Hiver's own
85            // runtime timer (hiver_runtime::time::sleep), NOT tokio::time::timeout.
86            // The HTTP server runs on hiver_runtime; tokio's time driver is not
87            // active there, so tokio::time::timeout would never resolve and the
88            // response would silently never be returned.
89            // 将处理程序与定时器竞争。重要:使用 Hiver 自有 runtime 的定时器
90            // (hiver_runtime::time::sleep),而非 tokio::time::timeout。
91            // HTTP 服务端运行于 hiver_runtime;tokio 的 time driver 在那里未激活,
92            // 故 tokio::time::timeout 永不完成,响应会被静默丢弃。
93            use futures::future::FutureExt;
94            let handler_fut = next.call(req, state).boxed();
95            let timer_fut = hiver_runtime::time::sleep(timeout);
96            match futures::future::select(handler_fut, timer_fut).await
97            {
98                futures::future::Either::Left((response, _)) => response,
99                futures::future::Either::Right((_, _)) =>
100                {
101                    tracing::warn!("Request timed out after {:?}", timeout);
102                    Err(hiver_http::Error::Timeout(format!(
103                        "Request timed out after {:?}",
104                        timeout
105                    )))
106                },
107            }
108        })
109    }
110}
