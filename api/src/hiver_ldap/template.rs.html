1//! LDAP template pattern — simplified LDAP operations
2//! LDAP模板模式 — 简化的LDAP操作
3//!
4//! Equivalent to Spring LDAP's `LdapTemplate`.
5//! When the `ldap` feature is enabled, operations connect to a real LDAP server.
6//! Otherwise they return safe defaults for API compatibility testing.
7//!
8//! 等价于 Spring LDAP 的 `LdapTemplate`。
9//! 启用 `ldap` feature 时，操作会连接到真实的LDAP服务器。
10//! 否则返回安全的默认值以支持API兼容性测试。
11

20/// Central class for LDAP operations, wrapping a `ContextSource`.
21/// LDAP操作的中心类，包装 `ContextSource`。
22///
23/// # Example / 示例
24///
25/// ```rust,no_run,ignore
26/// use hiver_ldap::{LdapTemplate, LdapContextSource};
27///
28/// let context_source = LdapContextSource::new("ldap://localhost:389", "dc=example,dc=com");
29/// let template = LdapTemplate::new(context_source);
30/// ```
31#[derive(Debug, Clone)]
32pub struct LdapTemplate
33{
34    context_source: LdapContextSource,
35}
36
37impl LdapTemplate

51    /// Authenticate a user against the LDAP server / 对LDAP服务器认证用户
52    ///
53    /// Performs a simple bind to verify credentials.
54    /// 执行简单绑定以验证凭据。
55    #[allow(clippy::unused_async)]
56    pub async fn authenticate(&self, user_dn: &str, password: &str) -> LdapResult<bool>
57    {
58        #[cfg(feature = "ldap")]
59        {
60            let mut conn = self.context_source.get_context().await?;
61            match conn.simple_bind(user_dn, password).await
62            {
63                Ok(()) => Ok(true),
64                Err(LdapError::Authentication(_)) => Ok(false),
65                Err(e) => Err(e),
66            }
67        }
68        #[cfg(not(feature = "ldap"))]
69        {
70            let _ = (user_dn, password);
71            Ok(true)
72        }
73    }

75    /// Search for entries matching the filter / 搜索匹配过滤器的条目
76    ///
77    /// Uses `AttributesMapper` to convert raw LDAP attributes into Rust types.
78    /// 使用 `AttributesMapper` 将原始LDAP属性转换为Rust类型。
79    #[allow(clippy::unused_async)]
80    pub async fn search<T, M: AttributesMapper<T>>(
81        &self,
82        base: &str,
83        filter: &str,
84        mapper: &M,
85    ) -> LdapResult<Vec<T>>
86    {
87        #[cfg(feature = "ldap")]
88        {
89            let scope = ldap3::Scope::Subtree;
90            let mut conn = self.context_source.get_context().await?;
91            let results = conn.search(base, scope, filter, &["*"]).await?;
92            let mapped = results
93                .into_iter()
94                .map(|(_dn, attrs)| {
95                    let _attr_slices: Vec<(&str, &[String])> = Vec::new();
96                    // Convert owned attrs to borrowed for mapper
97                    let attr_pairs: Vec<(&str, Vec<&str>)> = attrs
98                        .iter()
99                        .map(|(k, v)| (k.as_str(), v.iter().map(String::as_str).collect()))
100                        .collect();
101                    let attr_ref_pairs: Vec<(&str, &[&str])> =
102                        attr_pairs.iter().map(|(k, v)| (*k, v.as_slice())).collect();
103                    mapper.map_attributes(&attr_ref_pairs)
104                })
105                .collect();
106            Ok(mapped)
107        }
108        #[cfg(not(feature = "ldap"))]
109        {
110            let _ = (base, filter, mapper);
111            Ok(Vec::new())
112        }
113    }

115    /// Search and return results as `AttrMap` / 搜索并返回 `AttrMap` 结果
116    ///
117    /// Convenience method that returns attribute maps instead of requiring a mapper.
118    /// 便捷方法，返回属性映射而不需要提供mapper。
119    #[allow(clippy::unused_async)]
120    pub async fn search_attrs(&self, base: &str, filter: &str) -> LdapResult<Vec<AttrMap>>
121    {
122        #[cfg(feature = "ldap")]
123        {
124            let scope = ldap3::Scope::Subtree;
125            let mut conn = self.context_source.get_context().await?;
126            let results = conn.search(base, scope, filter, &["*"]).await?;
127            let maps = results
128                .into_iter()
129                .map(|(_dn, attrs)| {
130                    let mut map = AttrMap::new();
131                    for (key, values) in attrs
132                    {
133                        let refs: Vec<&str> = values.iter().map(String::as_str).collect();
134                        map.add(&key, &refs);
135                    }
136                    map
137                })
138                .collect();
139            Ok(maps)
140        }
141        #[cfg(not(feature = "ldap"))]
142        {
143            let _ = (base, filter);
144            Ok(Vec::new())
145        }
146    }

148    /// Look up a single entry by DN / 通过DN查找单个条目
149    #[allow(clippy::unused_async)]
150    pub async fn lookup<T, M: ContextMapper<T>>(
151        &self,
152        dn: &str,
153        mapper: &M,
154    ) -> LdapResult<Option<T>>
155    {
156        #[cfg(feature = "ldap")]
157        {
158            let scope = ldap3::Scope::Base;
159            let mut conn = self.context_source.get_context().await?;
160            let results = conn.search(dn, scope, "(objectClass=*)", &["*"]).await?;
161            if let Some((_dn, _attrs)) = results.into_iter().next()
162            {
163                Ok(Some(mapper.map_from_context(dn)))
164            }
165            else
166            {
167                Ok(None)
168            }
169        }
170        #[cfg(not(feature = "ldap"))]
171        {
172            let _ = (dn, mapper);
173            Ok(None)
174        }
175    }

177    /// Bind (create) a new LDAP entry / 绑定（创建）新LDAP条目
178    ///
179    /// `attrs` is a list of `(attribute_name, values)` pairs.
180    /// `attrs` 是 `(属性名, 值列表)` 对的列表。
181    #[allow(clippy::unused_async)]
182    pub async fn bind(&self, dn: &str, attrs: &[(&str, &[&str])]) -> LdapResult<()>
183    {
184        #[cfg(feature = "ldap")]
185        {
186            use std::collections::HashSet;
187            let mut conn = self.context_source.get_context().await?;
188            let ldap_attrs: Vec<(String, HashSet<String>)> = attrs
189                .iter()
190                .map(|(k, v)| {
191                    (
192                        k.to_string(),
193                        v.iter()
194                            .map(ToString::to_string)
195                            .collect::<HashSet<String>>(),
196                    )
197                })
198                .collect();
199            conn.add(dn, ldap_attrs).await
200        }
201        #[cfg(not(feature = "ldap"))]
202        {
203            let _ = (dn, attrs);
204            Ok(())
205        }
206    }

208    /// Unbind (delete) an LDAP entry / 解绑（删除）LDAP条目
209    #[allow(clippy::unused_async)]
210    pub async fn unbind(&self, dn: &str) -> LdapResult<()>
211    {
212        #[cfg(feature = "ldap")]

216        }
217        #[cfg(not(feature = "ldap"))]

224    /// Modify an existing LDAP entry / 修改现有LDAP条目
225    ///
226    /// `modifications` is a list of `(attribute_name, new_values)` pairs.
227    /// Each pair replaces the attribute with the given values.
228    ///
229    /// `modifications` 是 `(属性名, 新值列表)` 对的列表。
230    /// 每对将属性替换为给定的值。
231    #[allow(clippy::unused_async)]
232    pub async fn modify(&self, dn: &str, modifications: &[(&str, &[&str])]) -> LdapResult<()>
233    {
234        #[cfg(feature = "ldap")]
235        {
236            let mut conn = self.context_source.get_context().await?;
237            let mods: Vec<ldap3::Mod<String>> = modifications
238                .iter()
239                .map(|(attr, values)| {
240                    ldap3::Mod::<String>::Replace(
241                        attr.to_string(),
242                        values.iter().map(ToString::to_string).collect(),
243                    )
244                })
245                .collect();
246            conn.modify(dn, mods).await
247        }
248        #[cfg(not(feature = "ldap"))]
249        {
250            let _ = (dn, modifications);
251            Ok(())
252        }
253    }

255    /// Check if an entry exists by performing a base search / 通过基础搜索检查条目是否存在
256    #[allow(clippy::unused_async)]
257    pub async fn exists(&self, dn: &str) -> LdapResult<bool>
258    {
259        #[cfg(feature = "ldap")]
260        {
261            let scope = ldap3::Scope::Base;
262            let mut conn = self.context_source.get_context().await?;
263            let results = conn.search(dn, scope, "(objectClass=*)", &["1.1"]).await?;
264            Ok(!results.is_empty())
265        }
266        #[cfg(not(feature = "ldap"))]
267        {
268            let _ = dn;
269            Ok(false)
270        }
271    }

273    /// Count entries matching a filter / 统计匹配过滤器的条目数
274    #[allow(clippy::unused_async)]
275    pub async fn count(&self, base: &str, filter: &str) -> LdapResult<usize>
276    {
277        #[cfg(feature = "ldap")]
278        {
279            let scope = ldap3::Scope::Subtree;
280            let mut conn = self.context_source.get_context().await?;
281            let results = conn.search(base, scope, filter, &["1.1"]).await?;
282            Ok(results.len())
283        }
284        #[cfg(not(feature = "ldap"))]
285        {
286            let _ = (base, filter);
287            Ok(0)
288        }
289    }
