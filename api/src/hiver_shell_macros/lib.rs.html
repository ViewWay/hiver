6//! - `#[shell_component]` — Marks a struct as a shell command component

9use proc_macro::TokenStream;
10use quote::quote;
11use syn::{ItemStruct, parse_macro_input};

13/// Marks a struct as a shell component that contains shell commands.
14/// 将结构体标记为包含shell命令的shell组件。
15///
16/// # Example / 示例
17///
18/// ```rust,ignore
19/// use hiver_shell_macros::shell_component;
20///
21/// #[shell_component]
22/// struct MyCommands;
23/// ```
24#[proc_macro_attribute]
25pub fn shell_component(_attr: TokenStream, item: TokenStream) -> TokenStream
26{
27    let input = parse_macro_input!(item as ItemStruct);
28    let name = &input.ident;
29
30    let expanded = quote! {
31        #input
32
33        impl #name {
34            /// Auto-generated method to register commands
35            pub fn register_commands(registry: &mut hiver_shell::command::CommandRegistry) {
36                // Commands are registered at compile time via attribute macros
37                let _ = registry;
38            }
39        }
40    };
41
42    TokenStream::from(expanded)
43}

45/// Marks a method as a shell command.
46/// 将方法标记为shell命令。
47///
48/// # Attributes / 属性
49///
50/// - `name` — Command name / 命令名称
51/// - `description` — Command description / 命令描述
52///
53/// # Example / 示例
54///
55/// ```rust,ignore
56/// use hiver_shell_macros::shell_method;
57///
58/// #[shell_component]
59/// struct GreetCommand;
60///
61/// impl GreetCommand {
62///     #[shell_method("greet", "Greet someone")]
63///     fn greet(&self, args: &[&str]) -> Result<String, Box<dyn std::error::Error>> {
64///         Ok(format!("Hello, {}!", args.first().unwrap_or(&"World")))
65///     }
66/// }
67/// ```
68#[proc_macro_attribute]
69pub fn shell_method(attr: TokenStream, item: TokenStream) -> TokenStream
70{
71    // Parse the attributes: first token is command name, rest is description
72    let args: Vec<String> = attr
73        .to_string()
74        .split(',')
75        .map(|s| s.trim().trim_matches('"').to_string())
76        .collect();
77
78    let cmd_name = args.first().map_or("unknown", String::as_str);
79    let cmd_desc = args.get(1).map_or("", String::as_str);
80
81    let input_fn = parse_macro_input!(item as syn::ImplItemFn);
82    let fn_name = &input_fn.sig.ident;
83
84    // Generate a const that registers this method
85    let register_name = quote::format_ident!("__REGISTER_{}", fn_name.to_string().to_uppercase());
86
87    let expanded = quote! {
88        #input_fn
89
90        #[doc(hidden)]
91        #[allow(non_upper_case_globals)]
92        pub const #register_name: () = {
93            // This forces the method to be registered at compile time
94        };
95    };
96
97    // Attach documentation attribute
98    let doc_comment = format!(" Command: {} — {}", cmd_name, cmd_desc);
99    let expanded = quote! {
100        #[doc = #doc_comment]
101        #expanded
102    };
103
104    TokenStream::from(expanded)
105}
