1//! Test application support for booting a test instance of the Hiver application.
2//! 测试应用支持，用于启动Hiver应用的测试实例。
3//!
4//! # Equivalent to Spring Boot / 等价于 Spring Boot
5//!
6//! ```java
7//! @SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
8//! @ActiveProfiles("test")
9//! @TestPropertySource(properties = "server.port=0")
10//! public class MyTests {
11//!     @Autowired
12//!     private ApplicationContext context;
13//!
14//!     @LocalServerPort
15//!     private int port;
16//!
17//!     @Test
18//!     void contextLoads() {
19//!         assertThat(context).isNotNull();
20//!     }
21//! }
22//! ```
23//!
24//! # Example / 示例
25//!
26//! ```rust,no_run,ignore
27//! use hiver_test::TestApplication;
28//!
29//! #[tokio::test]
30//! async fn test_application_startup() {
31//!     let app = TestApplication::builder()
32//!         .with_profile("test")
33//!         .with_property("server.port", "0")
34//!         .with_bean("myService", || MyService::new())
35//!         .start()
36//!         .await
37//!         .expect("application should start");
38//!
39//!     let port = app.port();
40//!     assert!(port > 0);
41//!
42//!     // app automatically shuts down on drop
43//! }
44//! ```

53/// Errors that can occur when starting or managing a test application.
54/// 启动或管理测试应用时可能发生的错误。
55#[derive(Debug, Error)]
56pub enum TestApplicationError
57{
58    /// The application failed to start.
59    /// 应用启动失败。
60    #[error("failed to start test application: {0}")]
61    StartupFailed(String),
62
63    /// A required bean was not found.
64    /// 未找到所需的bean。
65    #[error("bean not found: {0}")]
66    BeanNotFound(String),
67
68    /// A bean with the given name already exists.
69    /// 具有给定名称的bean已存在。
70    #[error("bean already registered: {0}")]
71    BeanAlreadyExists(String),
72
73    /// The application has been shut down.
74    /// 应用已关闭。
75    #[error("application is shut down")]
76    ShutDown,
77}
78
79/// Result type for test application operations.
80/// 测试应用操作的结果类型。
81pub type TestAppResult<T> = Result<T, TestApplicationError>;

83/// A test application instance that boots a lightweight Hiver context.
84///
85/// 测试应用实例，启动轻量级的Hiver上下文。
86///
87/// Automatically shuts down when dropped (equivalent to Spring's
88/// `@DirtiesContext` cleanup).
89///
90/// 在丢弃时自动关闭（等价于Spring的 `@DirtiesContext` 清理）。
91///
92/// # Spring Equivalent / Spring等价物
93///
94/// ```java
95/// @SpringBootTest
96/// @AutoConfigureMockMvc
97/// public class ApplicationTests {
98///     @Autowired private ApplicationContext context;
99///     @LocalServerPort private int port;
100/// }
101/// ```
102pub struct TestApplication
103{
104    /// Test application context holding beans and config.
105    /// 持有bean和配置的测试应用上下文。
106    context: TestApplicationContext,
107
108    /// The resolved configuration for this test run.
109    /// 本次测试运行的解析配置。
110    config: TestConfig,
111
112    /// The actual port the server is listening on (0 if not started).
113    /// 服务器实际监听的端口（未启动时为0）。
114    port: u16,
115
116    /// Whether the application has been started.
117    /// 应用是否已启动。
118    started: bool,
119
120    /// Whether the application has been shut down.
121    /// 应用是否已关闭。
122    shut_down: bool,
123
124    /// Registered bean factory closures (stored for deferred initialisation).
125    /// 注册的bean工厂闭包（存储用于延迟初始化）。
126    bean_factories:
127        Arc<RwLock<HashMap<String, Arc<dyn Fn() -> Box<dyn Any + Send + Sync> + Send + Sync>>>>,
128
129    /// Pre-registered bean instances (stored for deferred registration).
130    /// 预注册的bean实例（存储用于延迟注册）。
131    pre_registered: Vec<(String, Arc<dyn Any + Send + Sync>)>,
132}

137    /// 创建新的默认测试应用构建器。

139    /// # Example / 示例

152    /// Create a test application with default configuration.
153    /// 使用默认配置创建测试应用。
154    ///
155    /// Equivalent to `@SpringBootTest` with no customisation.
156    /// 等价于没有自定义的 `@SpringBootTest`。
157    pub fn new() -> Self
158    {
159        Self {
160            context: TestApplicationContext::new(),
161            config: TestConfig::new(),
162            port: 0,
163            started: false,
164            shut_down: false,
165            bean_factories: Arc::new(RwLock::new(HashMap::new())),
166            pre_registered: Vec::new(),
167        }
168    }

170    /// Build the application with the given configuration.
171    /// 使用给定配置构建应用。
172    pub fn with_config(mut self, config: TestConfig) -> Self
173    {
174        self.config = config;
175        self
176    }

178    /// Activate the given Spring-style profile (e.g. `"test"`, `"integration"`).
179    /// 激活给定的Spring风格配置文件（例如 `"test"`、`"integration"`）。
180    ///
181    /// Equivalent to `@ActiveProfiles("test")`.
182    /// 等价于 `@ActiveProfiles("test")`。
183    pub fn with_profile(mut self, profile: impl Into<String>) -> Self
184    {
185        self.config.profiles.push(profile.into());
186        self
187    }

189    /// Override or register a named bean in the test context.
190    /// 在测试上下文中覆盖或注册命名bean。
191    ///
192    /// Equivalent to `@TestConfiguration @Bean`.
193    /// 等价于 `@TestConfiguration @Bean`。
194    pub fn with_bean<T: Any + Send + Sync + 'static>(self, name: impl Into<String>, bean: T)
195    -> Self
196    {
197        let rt = tokio::runtime::Handle::current();
198        let name = name.into();
199        rt.block_on(self.context.register_bean(name.as_str(), bean));
200        self
201    }

203    /// Start the test application.
204    /// 启动测试应用。
205    ///
206    /// Initialises the application context, registers all bean factories,
207    /// and marks the application as running.
208    ///
209    /// 初始化应用上下文，注册所有bean工厂，并将应用标记为运行中。
210    pub async fn start(mut self) -> TestAppResult<Self>
211    {
212        if self.shut_down
213        {
214            return Err(TestApplicationError::ShutDown);
215        }
216
217        // Load configuration from environment overrides.
218        // 从环境覆盖加载配置。
219        {
220            let mut cfg = self.config.clone();
221            cfg.load_from_env();
222            self.config = cfg;
223        }
224
225        // Apply bean factories that were registered via the builder.
226        // 应用通过构建器注册的bean工厂。
227        {
228            let factories = self.bean_factories.read().await;
229            for (name, factory) in factories.iter()
230            {
231                let bean = factory();
232                self.context.register_boxed_bean(name.as_str(), bean).await;
233            }
234        }
235
236        // Apply pre-registered bean instances.
237        // 应用预注册的bean实例。
238        {
239            for (name, bean) in self.pre_registered.drain(..)
240            {
241                let mut beans = self.context.beans_mut().await;
242                beans.insert(name, bean);
243            }
244        }
245
246        // Resolve port from config.
247        // 从配置解析端口。
248        self.port = self.config.server.port;
249
250        // In a full implementation this is where the HTTP server,
251        // database connections, and other infrastructure would be
252        // bootstrapped.
253        // 在完整实现中，这里会引导HTTP服务器、数据库连接和其他基础设施。
254
255        self.started = true;
256        tracing::info!(
257            port = self.port,
258            profiles = ?self.config.profiles,
259            "TestApplication started"
260        );
261        Ok(self)
262    }
263
264    /// Shut down the test application, releasing all resources.
265    /// 关闭测试应用，释放所有资源。
266    ///
267    /// Equivalent to Spring context close.
268    /// 等价于Spring上下文关闭。
269    pub async fn shutdown(&mut self)
270    {
271        if !self.shut_down
272        {
273            self.context.clear_beans().await;
274            self.context.clear_config().await;
275            self.started = false;
276            self.shut_down = true;
277            tracing::info!("TestApplication shut down");
278        }
279    }
280

282    /// 返回测试服务器监听的端口。
283    ///
284    /// Equivalent to `@LocalServerPort`.
285    /// 等价于 `@LocalServerPort`。
286    pub fn port(&self) -> u16

288        self.port
289    }

299    /// 返回应用是否已启动。

305    /// Retrieve a bean by type from the application context.
306    /// 从应用上下文按类型检索bean。
307    ///
308    /// # Errors
309    ///
310    /// Returns [`TestApplicationError::BeanNotFound`] if no bean of the
311    /// requested type is registered.
312    pub async fn get_bean<T: 'static + Send + Sync + Clone>(&self) -> TestAppResult<T>
313    {
314        self.context.get_bean::<T>().await.ok_or_else(|| {
315            TestApplicationError::BeanNotFound(std::any::type_name::<T>().to_string())
316        })
317    }
318
319    /// Retrieve a bean by name and type from the application context.
320    /// 从应用上下文按名称和类型检索bean。
321    ///
322    /// # Errors
323    ///
324    /// Returns [`TestApplicationError::BeanNotFound`] if no bean with the
325    /// given name exists.
326    pub async fn get_bean_by_name<T: 'static + Send + Sync + Clone>(
327        &self,
328        name: &str,
329    ) -> TestAppResult<T>
330    {
331        self.context
332            .get_bean_by_name::<T>(name)
333            .await
334            .ok_or_else(|| TestApplicationError::BeanNotFound(name.to_string()))
335    }

374impl Drop for TestApplication
375{
376    fn drop(&mut self)
377    {
378        if self.started && !self.shut_down
379        {
380            // Spawn a task to perform async cleanup.
381            // 生成一个任务来执行异步清理。
382            let context = self.context.clone();
383            tokio::spawn(async move {
384                context.clear_beans().await;
385                context.clear_config().await;
386            });
387            tracing::info!("TestApplication dropped — cleanup scheduled");
388        }
389    }
390}

396/// Builder for constructing a [`TestApplication`] with custom configuration.
397///
398/// 用于构建具有自定义配置的 [`TestApplication`] 的构建器。
399///
400/// # Spring Equivalent / Spring等价物
401///
402/// ```java
403/// @SpringBootTest(
404///     webEnvironment = WebEnvironment.RANDOM_PORT,
405///     properties = { "server.port=0", "spring.datasource.url=jdbc:h2:mem:test" }
406/// )
407/// @ActiveProfiles({ "test", "security" })
408/// ```
409///
410/// # Example / 示例
411///
412/// ```rust,no_run,ignore
413/// let app = TestApplication::builder()
414///     .with_profile("test")
415///     .with_property("server.port", "0")
416///     .with_property("spring.datasource.url", "jdbc:h2:mem:test")
417///     .start()
418///     .await?;
419/// ```
420pub struct TestApplicationBuilder
421{
422    /// Resolved configuration for this build.
423    /// 本次构建的解析配置。
424    config: TestConfig,
425
426    /// Bean factory closures keyed by bean name.
427    /// 按bean名称键控的bean工厂闭包。
428    bean_factories: HashMap<String, Arc<dyn Fn() -> Box<dyn Any + Send + Sync> + Send + Sync>>,
429
430    /// Pre-registered bean instances (name, Arc<dyn Any + Send + Sync>).
431    /// 预注册的bean实例（名称, Arc<dyn Any + Send + Sync>）。
432    pre_registered: Vec<(String, Arc<dyn Any + Send + Sync>)>,
433}
434
435impl TestApplicationBuilder
436{
437    /// Create a new builder with default test configuration.
438    /// 使用默认测试配置创建新的构建器。
439    pub fn new() -> Self
440    {
441        Self {
442            config: TestConfig::new(),
443            bean_factories: HashMap::new(),
444            pre_registered: Vec::new(),
445        }
446    }
447
448    /// Set the test configuration directly.
449    /// 直接设置测试配置。
450    pub fn with_config(mut self, config: TestConfig) -> Self
451    {
452        self.config = config;
453        self
454    }
455
456    /// Activate an additional profile (can be called multiple times).
457    /// 激活额外的配置文件（可以多次调用）。
458    ///
459    /// Equivalent to `@ActiveProfiles({ "test", "security" })`.
460    /// 等价于 `@ActiveProfiles({ "test", "security" })`。
461    pub fn with_profile(mut self, profile: impl Into<String>) -> Self
462    {
463        self.config.profiles.push(profile.into());
464        self
465    }
466
467    /// Set the test mode (unit, integration, e2e).
468    /// 设置测试模式（unit、integration、e2e）。
469    pub fn with_test_mode(mut self, mode: crate::test_config::TestMode) -> Self
470    {
471        self.config.test_mode = mode;
472        self
473    }
474
475    /// Add a configuration property.
476    /// 添加配置属性。
477    ///
478    /// Equivalent to `@TestPropertySource(properties = "...")`.
479    /// 等价于 `@TestPropertySource(properties = "...")`。
480    pub fn with_property(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
481    {
482        self.config.properties.insert(key.into(), value.into());
483        self
484    }
485
486    /// Override or register a named bean instance.
487    /// 覆盖或注册命名bean实例。
488    ///
489    /// The bean is stored as-is and registered during [`start()`](TestApplication::start).
490    /// bean按原样存储，并在 [`start()`](TestApplication::start) 期间注册。
491    pub fn with_bean<T: Any + Send + Sync + 'static>(
492        mut self,
493        name: impl Into<String>,
494        bean: T,
495    ) -> Self
496    {
497        self.pre_registered.push((name.into(), Arc::new(bean)));
498        self
499    }
500
501    /// Register a bean factory function that will be invoked during
502    /// [`start()`](TestApplication::start). 注册一个bean工厂函数，将在
503    /// [`start()`](TestApplication::start) 期间调用。
504    pub fn with_bean_factory<T: Any + Send + Sync + 'static, F>(
505        mut self,
506        name: impl Into<String>,
507        factory: F,
508    ) -> Self
509    where
510        F: Fn() -> T + Send + Sync + 'static,
511    {
512        let name = name.into();
513        let factory = Arc::new(move || -> Box<dyn Any + Send + Sync> { Box::new(factory()) });
514        self.bean_factories.insert(name, factory);
515        self
516    }
517
518    /// Set the server port (0 for random).
519    /// 设置服务器端口（0表示随机）。
520    pub fn with_port(mut self, port: u16) -> Self
521    {
522        self.config.server.port = port;
523        self
524    }
525
526    /// Set the database URL.
527    /// 设置数据库URL。
528    pub fn with_database_url(mut self, url: impl Into<String>) -> Self
529    {
530        self.config.database.url = Some(url.into());
531        self.config.database.in_memory = false;
532        self
533    }
534
535    /// Use an in-memory database.
536    /// 使用内存数据库。
537    pub fn with_in_memory_db(mut self) -> Self
538    {
539        self.config.database.in_memory = true;
540        self.config.database.url = None;
541        self
542    }
543
544    /// Build the [`TestApplication`] without starting it.
545    /// 构建但不启动 [`TestApplication`]。
546    ///
547    /// Call [`TestApplication::start()`] to boot the application.
548    /// 调用 [`TestApplication::start()`] 启动应用。
549    pub fn build(self) -> TestAppResult<TestApplication>
550    {
551        let mut app = TestApplication::new().with_config(self.config);
552
553        // Store pre-registered beans directly on the app for later registration during start().
554        // 将预注册的bean直接存储在app上，以便在start()期间注册。
555        app.pre_registered = self.pre_registered;
556
557        for (name, factory) in self.bean_factories
558        {
559            app.bean_factories
560                .try_write()
561                .map_err(|_| {
562                    TestApplicationError::StartupFailed("failed to acquire write lock".to_string())
563                })?
564                .insert(name, factory);
565        }
566
567        Ok(app)
568    }
569
570    /// Build and immediately start the test application.
571    /// 构建并立即启动测试应用。
572    pub async fn start(self) -> TestAppResult<TestApplication>
573    {
574        let app = self.build()?;
575        app.start().await
576    }
577}
