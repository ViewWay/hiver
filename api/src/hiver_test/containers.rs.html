1//! Testcontainers integration — Docker-managed containers for integration tests.
2//! Testcontainers 集成 — 用于集成测试的 Docker 管理容器。
3//!
4//! # Description / 描述
5//!
6//! Provides type-safe wrappers around `testcontainers` and `testcontainers-modules`
7//! so Hiver integration tests can start real Postgres, Redis, and Kafka containers
8//! with a Spring Boot-like fluent API.
9//!
10//! 围绕 testcontainers 和 testcontainers-modules 提供类型安全的包装器，
11//! 使 Hiver 集成测试可以使用类似 Spring Boot 的流畅 API 启动真实的
12//! Postgres、Redis 和 Kafka 容器。
13//!
14//! # Example / 示例
15//! ```rust,ignore
16//! use hiver_test::containers::PostgresContainer;
17//!
18//! #[tokio::test]
19//! async fn test_with_postgres() {
20//!     let pg = PostgresContainer::start().await;
21//!     let url = pg.connection_url();
22//!     // use url to configure your DatabaseClient
23//! }
24//! ```
25
26#![allow(dead_code)]

35/// A running PostgreSQL test container.
36/// 正在运行的 PostgreSQL 测试容器。
37///
38/// Equivalent to Spring's `@Testcontainers` + `@Container` with `PostgreSQLContainer`.
39/// 等价于 Spring 的 @Testcontainers + @Container 与 PostgreSQLContainer。
40///
41/// The container is automatically stopped when this struct is dropped.
42/// 此结构体被丢弃时，容器会自动停止。
43pub struct PostgresContainer
44{
45    container: ContainerAsync<Postgres>,
46    host: String,
47    port: u16,
48}
49
50impl PostgresContainer
51{
52    /// Start a PostgreSQL container with the default image and credentials.
53    /// 使用默认镜像和凭据启动 PostgreSQL 容器。
54    ///
55    /// Default: `postgres:16-alpine`, user=`postgres`, pass=`postgres`, db=`postgres`
56    /// 默认值：postgres:16-alpine，user=postgres，pass=postgres，db=postgres
57    pub async fn start() -> Self
58    {
59        Self::with_options("postgres", "postgres", "postgres").await
60    }
61
62    /// Start with explicit credentials.
63    /// 使用显式凭据启动。
64    pub async fn with_options(user: &str, password: &str, db: &str) -> Self
65    {
66        let image = Postgres::default()
67            .with_user(user)
68            .with_password(password)
69            .with_db_name(db);
70        let container = testcontainers::runners::AsyncRunner::start(image)
71            .await
72            .expect("PostgreSQL container failed to start");
73        let host = container.get_host().await.expect("get host").to_string();
74        let port = container.get_host_port_ipv4(5432).await.expect("get port");
75        info!("PostgreSQL container started at {}:{}", host, port);
76        Self {
77            container,
78            host,
79            port,
80        }
81    }
82
83    /// JDBC-style connection URL (compatible with most Rust crates).
84    /// JDBC 风格的连接 URL（兼容大多数 Rust crate）。
85    ///
86    /// Format: `postgresql://postgres:postgres@host:port/postgres`
87    pub fn connection_url(&self) -> String
88    {
89        format!("postgresql://postgres:postgres@{}:{}/postgres", self.host, self.port)
90    }
91
92    /// Host the container is reachable on.
93    /// 可访问容器的主机。
94    pub fn host(&self) -> &str
95    {
96        &self.host
97    }
98
99    /// Mapped host port for PostgreSQL (default container port: 5432).
100    /// PostgreSQL 的映射主机端口（默认容器端口：5432）。
101    pub fn port(&self) -> u16
102    {
103        self.port
104    }
105}

224// ContainerSet — start multiple containers together

227/// Helper that starts multiple containers and exposes their connection info.
228/// 启动多个容器并暴露其连接信息的助手。
229///
230/// # Example / 示例
231/// ```rust,ignore
232/// let set = ContainerSet::builder()
233///     .postgres()
234///     .redis()
235///     .build()
236///     .await;
237/// let pg_url = set.postgres_url().unwrap();
238/// let redis_url = set.redis_url().unwrap();
239/// ```
240#[derive(Default)]
241pub struct ContainerSetBuilder
242{
243    need_postgres: bool,
244    need_redis: bool,
245    need_kafka: bool,
246}
247
248impl ContainerSetBuilder
249{
250    /// Create a new builder.
251    /// 创建新构建器。
252    pub fn new() -> Self
253    {
254        Self::default()
255    }
256
257    /// Include a PostgreSQL container.
258    /// 包含 PostgreSQL 容器。
259    pub fn postgres(mut self) -> Self
260    {
261        self.need_postgres = true;
262        self
263    }
264
265    /// Include a Redis container.
266    /// 包含 Redis 容器。
267    pub fn redis(mut self) -> Self
268    {
269        self.need_redis = true;
270        self
271    }
272
273    /// Include a Kafka container.
274    /// 包含 Kafka 容器。
275    pub fn kafka(mut self) -> Self
276    {
277        self.need_kafka = true;
278        self
279    }
280
281    /// Start all requested containers concurrently.
282    /// 并发启动所有请求的容器。
283    pub async fn build(self) -> ContainerSet
284    {
285        let (pg, redis, kafka) = tokio::join!(
286            async {
287                if self.need_postgres
288                {
289                    Some(PostgresContainer::start().await)
290                }
291                else
292                {
293                    None
294                }
295            },
296            async {
297                if self.need_redis
298                {
299                    Some(RedisContainer::start().await)
300                }
301                else
302                {
303                    None
304                }
305            },
306            async {
307                if self.need_kafka
308                {
309                    Some(KafkaContainer::start().await)
310                }
311                else
312                {
313                    None
314                }
315            },
316        );
317        ContainerSet {
318            postgres: pg,
319            redis,
320            kafka,
321        }
322    }
323}

325/// A running set of test containers.
326/// 一组正在运行的测试容器。
327pub struct ContainerSet
328{
329    /// Optional PostgreSQL container.
330    pub postgres: Option<PostgresContainer>,
331    /// Optional Redis container.
332    pub redis: Option<RedisContainer>,
333    /// Optional Kafka container.
334    pub kafka: Option<KafkaContainer>,
335}
336
337impl ContainerSet
338{
339    /// Create a builder.
340    /// 创建构建器。
341    pub fn builder() -> ContainerSetBuilder
342    {
343        ContainerSetBuilder::new()
344    }
345
346    /// PostgreSQL connection URL if a container was started.
347    /// 若容器已启动，返回 PostgreSQL 连接 URL。
348    pub fn postgres_url(&self) -> Option<String>
349    {
350        self.postgres
351            .as_ref()
352            .map(PostgresContainer::connection_url)
353    }
354
355    /// Redis connection URL if a container was started.
356    /// 若容器已启动，返回 Redis 连接 URL。
357    pub fn redis_url(&self) -> Option<String>
358    {
359        self.redis.as_ref().map(RedisContainer::connection_url)
360    }
361
362    /// Kafka bootstrap servers if a container was started.
363    /// 若容器已启动，返回 Kafka bootstrap 服务器地址。
364    pub fn kafka_bootstrap(&self) -> Option<String>
365    {
366        self.kafka.as_ref().map(KafkaContainer::bootstrap_servers)
367    }
