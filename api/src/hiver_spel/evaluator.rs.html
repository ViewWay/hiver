11/// Compiles and evaluates a SpEL expression against a context.
12/// 编译并针对上下文求值 SpEL 表达式。
13pub struct SpelEvaluator
14{
15    expr: Result<SpelExpr, SpelError>,
16}
17
18impl SpelEvaluator
19{
20    /// Creates a new evaluator from an expression string.
21    /// 从表达式字符串创建新的求值器。
22    pub fn new(expression: &str) -> Self
23    {
24        Self {
25            expr: parser::parse(expression),
26        }
27    }
28
29    /// Evaluates the expression against the given context.
30    /// 针对给定上下文求值表达式。
31    pub fn evaluate(&self, ctx: &SpelContext) -> Result<bool, SpelError>
32    {
33        let expr = self
34            .expr
35            .as_ref()
36            .map_err(|e| SpelError::Parse(e.to_string()))?;
37        eval_bool(expr, ctx)
38    }
39}
40
41fn eval_bool(expr: &SpelExpr, ctx: &SpelContext) -> Result<bool, SpelError>
42{
43    match expr
44    {
45        SpelExpr::HasRole(role) => Ok(ctx.has_role(role)),
46        SpelExpr::HasAuthority(auth) => Ok(ctx.has_authority(auth)),
47        SpelExpr::HasAnyRole(roles) => Ok(ctx.has_any_role(roles)),
48        SpelExpr::PermitAll => Ok(true),
49        SpelExpr::DenyAll => Ok(false),
50        SpelExpr::IsAuthenticated => Ok(ctx.is_authenticated()),
51        SpelExpr::IsAnonymous => Ok(ctx.is_anonymous()),
52        SpelExpr::And(a, b) => Ok(eval_bool(a, ctx)? && eval_bool(b, ctx)?),
53        SpelExpr::Or(a, b) => Ok(eval_bool(a, ctx)? || eval_bool(b, ctx)?),
54        SpelExpr::Not(e) => Ok(!eval_bool(e, ctx)?),
55        SpelExpr::Compare(left, op, right) =>
56        {
57            let lv = eval_value(left, ctx)?;
58            let rv = eval_value(right, ctx)?;
59            compare(&lv, &rv, *op)
60        },
61        SpelExpr::LiteralBool(b) => Ok(*b),
62        _ => Err(SpelError::Evaluation("expression does not evaluate to bool".into())),
63    }
64}
65
66fn eval_value(expr: &SpelExpr, ctx: &SpelContext) -> Result<Value, SpelError>
67{
68    match expr
69    {
70        SpelExpr::Variable(name) => ctx
71            .get_variable(name)
72            .cloned()
73            .ok_or_else(|| SpelError::Evaluation(format!("undefined variable: '{name}'"))),
74        SpelExpr::PropertyAccess(obj, prop) =>
75        {
76            let val = eval_value(obj, ctx)?;
77            match &val
78            {
79                Value::Object(map) => map
80                    .get(prop)
81                    .cloned()
82                    .ok_or_else(|| SpelError::Evaluation(format!("property '{prop}' not found"))),
83                Value::String(s) => eval_string_property(s, prop),
84                Value::Array(arr) => eval_array_property(arr, prop),
85                Value::Number(n) => eval_number_property(n, prop),
86                _ => Err(SpelError::Evaluation(format!(
87                    "cannot access property '{prop}' on {:?}",
88                    val
89                ))),
90            }
91        },
92        SpelExpr::LiteralNumber(n) => Ok(Value::from(*n)),
93        SpelExpr::LiteralString(s) => Ok(Value::from(s.as_str())),
94        SpelExpr::LiteralBool(b) => Ok(Value::from(*b)),
95        _ => Err(SpelError::Evaluation("expected a value expression".into())),
96    }
97}

99/// Evaluate property access on string values.
100/// 对字符串值进行属性访问求值。
101fn eval_string_property(s: &str, prop: &str) -> Result<Value, SpelError>
102{
103    match prop
104    {
105        "length" | "len" => Ok(Value::from(s.len() as f64)),
106        "isEmpty" | "is_empty" => Ok(Value::from(s.is_empty())),
107        "trim" => Ok(Value::from(s.trim())),
108        "toUpper" | "toUpperCase" | "to_uppercase" => Ok(Value::from(s.to_uppercase())),
109        "toLower" | "toLowerCase" | "to_lowercase" => Ok(Value::from(s.to_lowercase())),
110        _ => Err(SpelError::Evaluation(format!("unknown string property: '{prop}'"))),
111    }
112}
113
114/// Evaluate property access on array values.
115/// 对数组值进行属性访问求值。
116fn eval_array_property(arr: &[Value], prop: &str) -> Result<Value, SpelError>
117{
118    match prop
119    {
120        "length" | "len" | "size" => Ok(Value::from(arr.len() as f64)),
121        "isEmpty" | "is_empty" => Ok(Value::from(arr.is_empty())),
122        _ => Err(SpelError::Evaluation(format!("unknown array property: '{prop}'"))),
123    }
124}

126/// Evaluate property access on number values.
127/// 对数值进行属性访问求值。
128fn eval_number_property(n: &serde_json::Number, prop: &str) -> Result<Value, SpelError>
129{
130    match prop
131    {
132        "intValue" | "as_i64" if n.is_i64() => Ok(Value::from(n.as_i64().unwrap() as f64)),
133        "doubleValue" | "as_f64" if n.is_f64() => Ok(Value::from(n.as_f64().unwrap())),
134        _ => Err(SpelError::Evaluation(format!("unknown number property: '{prop}'"))),
135    }
136}

138#[allow(clippy::float_cmp)]
139fn compare(l: &Value, r: &Value, op: CmpOp) -> Result<bool, SpelError>
140{
141    let ln = l.as_f64();
142    let rn = r.as_f64();
143    if let (Some(lv), Some(rv)) = (ln, rn)
144    {
145        return Ok(match op
146        {
147            CmpOp::Eq => lv == rv,
148            CmpOp::NotEq => lv != rv,
149            CmpOp::Gt => lv > rv,
150            CmpOp::Lt => lv < rv,
151            CmpOp::GtEq => lv >= rv,
152            CmpOp::LtEq => lv <= rv,
153        });
154    }
155    let ls = l.as_str();
156    let rs = r.as_str();
157    if let (Some(lv), Some(rv)) = (ls, rs)
158    {
159        return Ok(match op
160        {
161            CmpOp::Eq => lv == rv,
162            CmpOp::NotEq => lv != rv,
163            CmpOp::Gt => lv > rv,
164            CmpOp::Lt => lv < rv,
165            CmpOp::GtEq => lv >= rv,
166            CmpOp::LtEq => lv <= rv,
167        });
168    }
169    let lb = l.as_bool();
170    let rb = r.as_bool();
171    if let (Some(lv), Some(rv)) = (lb, rb)
172    {
173        return Ok(match op
174        {
175            CmpOp::Eq => lv == rv,
176            CmpOp::NotEq => lv != rv,
177            _ =>
178            {
179                return Err(SpelError::Evaluation(
180                    "ordered comparison not supported for booleans".into(),
181                ));
182            },
183        });
184    }
185    Ok(l == r)
186}
