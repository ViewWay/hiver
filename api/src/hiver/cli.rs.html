4use clap::{Parser, Subcommand};

6use crate::commands;

8/// Hiver CLI - Project scaffolding tool.
9/// Hiver CLI - 项目脚手架工具。
10#[derive(Parser)]
11#[command(name = "hiver")]
12#[command(
13    version,
14    about = "Hiver project scaffolding tool / Hiver 项目脚手架工具"
15)]
16pub struct Cli
17{
18    #[command(subcommand)]
19    command: Commands,
20}
21
22impl Cli
23{
24    /// Parse arguments and execute the command.
25    /// 解析参数并执行命令。
26    pub fn parse() -> Self
27    {
28        Parser::parse()
29    }
30
31    /// Run the selected command.
32    /// 执行选中的命令。
33    pub fn run(self) -> Result<(), Box<dyn std::error::Error>>
34    {
35        match self.command
36        {
37            Commands::New(args) => commands::new::run(&args),
38            Commands::Run(args) => commands::run::run(&args),
39            Commands::Build(args) => commands::run::build(&args),
40            Commands::Add(args) => commands::add::run(&args),
41            Commands::Generate(args) => commands::generate::run(&args),
42            Commands::Check => commands::run::check(),
43            Commands::Test(args) => commands::run::test(&args),
44        }
45    }
46}
47
48/// Available commands.
49/// 可用命令。
50#[derive(Subcommand)]
51enum Commands
52{
53    /// Create a new Hiver project / 创建新的 Hiver 项目
54    #[command(alias = "n")]
55    New(NewArgs),
56
57    /// Run the project (cargo run) / 运行项目
58    #[command(alias = "r")]
59    Run(RunArgs),
60
61    /// Build the project (cargo build) / 构建项目
62    #[command(alias = "b")]
63    Build(BuildArgs),
64
65    /// Add a module dependency / 添加模块依赖
66    #[command(alias = "a")]
67    Add(AddArgs),
68
69    /// Generate code from templates / 从模板生成代码
70    #[command(alias = "g")]
71    Generate(GenerateArgs),
72
73    /// Check compilation (cargo check) / 检查编译
74    #[command(alias = "c")]
75    Check,
76
77    /// Run tests (cargo test) / 运行测试
78    #[command(alias = "t")]
79    Test(TestArgs),
80}
