2//! Hiver Transaction - Transaction management module
3//! Hiver事务 - 事务管理模块
4//!
5//! # Equivalent to Spring Boot / 等价于 Spring Boot
6//!
7//! - `@Transactional` - Transactional
8//! - `TransactionTemplate` - `TransactionTemplate`
9//! - `TransactionManager` - `TransactionManager`
10//! - `PlatformTransactionManager` - `PlatformTransactionManager`
11//! - `@EnableTransactionManagement` - `EnableTransactionManagement`
12//!
13//! # Example / 示例
14//!
15//! ```rust,no_run,ignore
16//! use hiver_tx::{Transactional, TransactionTemplate};
17//!
18//! struct UserService {
19//!     // ... fields
20//! }
21//!
22//! impl UserService {
23//!     // Equivalent to @Transactional
24//!     #[transactional]
25//!     async fn create_user(&self, user: User) -> Result<User, Error> {
26//!         // Database operations
27//!         Ok(user)
28//!     }
29//!
30//!     // With specific isolation level
31//!     #[transactional(isolation = "SERIALIZABLE")]
32//!     async fn transfer_money(&self, from: u64, to: u64, amount: f64) -> Result<(), Error> {
33//!         // Transfer logic
34//!         Ok(())
35//!     }
36//! }
37//! ```
