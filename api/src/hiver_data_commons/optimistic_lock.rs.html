1//! Optimistic locking support for concurrent entity updates
2//! 并发实体更新的乐观锁支持
3//!
4//! # Overview / 概述
5//!
6//! This module provides optimistic locking abstractions to prevent
7//! lost updates when multiple transactions modify the same entity
8//! concurrently. This mirrors Spring Data's `@Version` annotation
9//! and optimistic locking mechanism.
10//! 本模块提供乐观锁抽象，防止多个事务并发修改同一实体时
11//! 发生更新丢失。这镜像了 Spring Data 的 `@Version` 注解
12//! 和乐观锁机制。
13//!
14//! # Equivalent to Spring / 等价于 Spring
15//!
16//! | Hiver | Spring Data |
17//! |-------|-------------|
18//! | `Version` trait | `@Version` annotation |
19//! | `OptimisticLockError` | `OptimisticLockingFailureException` |
20//! | `Versioned<T>` | Version-managed entity |
21//!
22//! # How it works / 工作原理
23//!
24//! 1. Entity carries a version number (typically starting at 0). 实体携带一个版本号（通常从 0
25//!    开始）。
26//! 2. On read, the version is captured. 读取时捕获版本号。
27//! 3. On update, the version is checked: if it has changed since read, the update is rejected with
28//!    `OptimisticLockError`. 更新时检查版本号：如果自读取以来版本号已改变， 更新将被拒绝并返回
29//!    `OptimisticLockError`。
30//! 4. On successful update, the version is incremented. 更新成功后，版本号递增。

36/// Error indicating an optimistic locking failure.
37/// 表示乐观锁失败的错误。
38///
39/// Thrown when an entity's version has changed between read and write,
40/// indicating that another transaction has modified the entity concurrently.
41/// 当实体的版本在读取和写入之间发生变化时抛出，
42/// 表明另一个事务已并发修改了该实体。
43///
44/// # Example / 示例
45///
46/// ```rust
47/// use hiver_data_commons::optimistic_lock::OptimisticLockError;
48///
49/// let err = OptimisticLockError::new("User", "42", 1, 3);
50/// assert_eq!(err.type_name(), "User");
51/// assert_eq!(err.entity_id(), "42");
52/// assert_eq!(err.expected_version(), 1);
53/// assert_eq!(err.actual_version(), 3);
54/// ```
55#[derive(Debug, Clone)]
56pub struct OptimisticLockError
57{
58    /// Entity type name / 实体类型名
59    type_name: String,
60    /// Entity ID / 实体 ID
61    entity_id: String,
62    /// Expected version (the version when the entity was read) / 预期版本
63    expected_version: i64,
64    /// Actual version (the current version in the database) / 实际版本
65    actual_version: i64,
66}
67
68impl OptimisticLockError
69{
70    /// Create a new optimistic lock error.
71    /// 创建新的乐观锁错误。
72    ///
73    /// # Parameters / 参数
74    ///
75    /// - `type_name`: Entity type name / 实体类型名
76    /// - `entity_id`: Entity identifier / 实体标识符
77    /// - `expected_version`: Version expected by the caller / 调用者期望的版本
78    /// - `actual_version`: Current version in the store / 存储中的当前版本
79    pub fn new(
80        type_name: impl Into<String>,
81        entity_id: impl Into<String>,
82        expected_version: i64,
83        actual_version: i64,
84    ) -> Self
85    {
86        Self {
87            type_name: type_name.into(),
88            entity_id: entity_id.into(),
89            expected_version,
90            actual_version,
91        }
92    }
93
94    /// Get the entity type name.
95    /// 获取实体类型名。
96    pub fn type_name(&self) -> &str
97    {
98        &self.type_name
99    }
100
101    /// Get the entity ID.
102    /// 获取实体 ID。
103    pub fn entity_id(&self) -> &str
104    {
105        &self.entity_id
106    }
107
108    /// Get the expected version.
109    /// 获取预期版本。
110    pub fn expected_version(&self) -> i64
111    {
112        self.expected_version
113    }
114
115    /// Get the actual version.
116    /// 获取实际版本。
117    pub fn actual_version(&self) -> i64
118    {
119        self.actual_version
120    }
121}

123impl fmt::Display for OptimisticLockError
124{
125    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
126    {
127        write!(
128            f,
129            "Optimistic lock failure for {}[{}]: expected version {}, but found {}",
130            self.type_name, self.entity_id, self.expected_version, self.actual_version
131        )
132    }
133}

135impl std::error::Error for OptimisticLockError {}

137impl From<OptimisticLockError> for Error
138{
139    fn from(err: OptimisticLockError) -> Self
140    {
141        Error::optimistic_locking_failure(err.type_name.clone(), err.entity_id.clone())
142    }

145/// Trait for versioned entities supporting optimistic locking.
146/// 支持乐观锁的版本化实体 trait。
147///
148/// Entities implementing this trait carry a monotonically increasing
149/// version number that is checked on update to prevent lost updates.
150/// 实现此 trait 的实体携带一个单调递增的版本号，
151/// 在更新时进行检查以防止更新丢失。
152///
153/// This is the Rust equivalent of Spring Data's `@Version` annotation.
154/// 这是 Spring Data `@Version` 注解的 Rust 等价物。
155///
156/// # Example / 示例
157///
158/// ```rust
159/// use hiver_data_commons::optimistic_lock::Version;
160///
161/// #[derive(Debug, Clone)]
162/// struct VersionedUser
163/// {
164///     id: i32,
165///     name: String,
166///     version: i64,
167/// }
168///
169/// impl Version for VersionedUser
170/// {
171///     fn version(&self) -> i64
172///     {
173///         self.version
174///     }
175///
176///     fn set_version(&mut self, version: i64)
177///     {
178///         self.version = version;
179///     }
180/// }
181/// ```
182pub trait Version
183{
184    /// Get the current version number.
185    /// 获取当前版本号。
186    fn version(&self) -> i64;
187
188    /// Set the version number.
189    /// 设置版本号。
190    fn set_version(&mut self, version: i64);
191
192    /// Increment the version number by one.
193    /// 将版本号递增一。
194    fn increment_version(&mut self)
195    {
196        let current = self.version();
197        self.set_version(current + 1);
198    }
199}

201/// A wrapper that pairs an entity with its version for safe updates.
202/// 将实体与其版本配对的安全更新包装器。
203///
204/// This type is used when reading an entity for later update. The version
205/// is captured at read time and can be verified at write time.
206/// 此类型用于读取实体以供后续更新。版本在读取时捕获，
207/// 并可在写入时进行验证。
208///
209/// # Example / 示例
210///
211/// ```rust
212/// use hiver_data_commons::optimistic_lock::{OptimisticLockError, Version, Versioned};
213///
214/// #[derive(Debug, Clone)]
215/// struct User
216/// {
217///     id: i32,
218///     name: String,
219///     ver: i64,
220/// }
221///
222/// impl Version for User
223/// {
224///     fn version(&self) -> i64
225///     {
226///         self.ver
227///     }
228///
229///     fn set_version(&mut self, v: i64)
230///     {
231///         self.ver = v;
232///     }
233/// }
234///
235/// let user = User {
236///     id: 1,
237///     name: "Alice".into(),
238///     ver: 3,
239/// };
240/// let versioned = Versioned::new(user);
241///
242/// assert_eq!(versioned.version(), 3);
243/// assert_eq!(versioned.entity().name, "Alice");
244///
245/// // Verify version before update
246/// let current_db_version: i64 = 3;
247/// versioned
248///     .verify_version("User", "1", current_db_version)
249///     .unwrap();
250/// ```
251#[derive(Debug, Clone)]
252pub struct Versioned<T: Version>
253{
254    /// The versioned entity.
255    /// 版本化实体。
256    entity: T,
257}
258
259impl<T: Version> Versioned<T>
260{
261    /// Create a new versioned wrapper around an entity.
262    /// 创建围绕实体的新版本化包装器。
263    pub fn new(entity: T) -> Self
264    {
265        Self { entity }
266    }
267
268    /// Get a reference to the inner entity.
269    /// 获取内部实体的引用。
270    pub fn entity(&self) -> &T
271    {
272        &self.entity
273    }
274
275    /// Get a mutable reference to the inner entity.
276    /// 获取内部实体的可变引用。
277    pub fn entity_mut(&mut self) -> &mut T
278    {
279        &mut self.entity
280    }
281
282    /// Get the captured version.
283    /// 获取捕获的版本。
284    pub fn version(&self) -> i64
285    {
286        self.entity.version()
287    }
288
289    /// Verify that the captured version matches the current database version.
290    /// 验证捕获的版本是否匹配当前数据库版本。
291    ///
292    /// Returns `Ok(())` if versions match, or `Err(OptimisticLockError)` if they differ.
293    /// 如果版本匹配返回 `Ok(())`，如果不同返回 `Err(OptimisticLockError)`。
294    pub fn verify_version(
295        &self,
296        type_name: &str,
297        entity_id: &str,
298        current_db_version: i64,
299    ) -> Result<(), OptimisticLockError>
300    {
301        let expected = self.version();
302        if expected == current_db_version
303        {
304            Ok(())
305        }
306        else
307        {
308            Err(OptimisticLockError::new(type_name, entity_id, expected, current_db_version))
309        }
310    }
311
312    /// Consume the wrapper, increment the version, and return the entity.
313    /// 消费包装器，递增版本，并返回实体。
314    ///
315    /// Call this after a successful update to prepare for future updates.
316    /// 在成功更新后调用此方法，为未来的更新做准备。
317    pub fn into_updated(mut self) -> T
318    {
319        self.entity.increment_version();
320        self.entity
321    }
322
323    /// Consume the wrapper and return the inner entity without modification.
324    /// 消费包装器并返回内部实体，不做修改。
325    pub fn into_inner(self) -> T
326    {
327        self.entity
328    }
329}

331/// Trait for performing a version-checked update.
332/// 执行版本检查更新的 trait。
333///
334/// Implementors provide a way to atomically compare-and-set
335/// the entity version in the data store.
336/// 实现者提供在数据存储中原子化比较并设置实体版本的方法。
337///
338/// # Example / 示例
339///
340/// ```rust,ignore
341/// use hiver_data_commons::optimistic_lock::VersionCheckedUpdate;
342///
343/// async fn save_versioned(&self, versioned: Versioned<User>) -> Result<User, Error> {
344///     let entity = versioned.entity();
345///     let expected = versioned.version();
346///     // UPDATE users SET ... WHERE id = ? AND version = ?
347///     let rows = db.execute("UPDATE users SET name = ?, version = version + 1 WHERE id = ? AND version = ?",
348///         &[&entity.name, &entity.id, &expected]).await?;
349///     if rows == 0 {
350///         return Err(OptimisticLockError::new("User", &entity.id.to_string(), expected, ...).into());
351///     }
352///     Ok(versioned.into_updated())
353/// }
354/// ```
355pub trait VersionCheckedUpdate<T: Version>
356{
357    /// Error type for the update operation.
358    /// 更新操作的错误类型。
359    type Error: std::fmt::Debug + Send + Sync;
360
361    /// Attempt to update the entity only if the version matches.
362    /// 仅当版本匹配时尝试更新实体。
363    ///
364    /// Returns `Ok(updated_entity)` on success, or `Err` if the version
365    /// has changed (optimistic lock failure).
366    /// 成功返回 `Ok(updated_entity)`，如果版本已改变返回 `Err`（乐观锁失败）。
367    fn update_if_version_matches(&self, versioned: Versioned<T>) -> Result<T, Self::Error>;
368}
