1//! Validation groups support / 验证分组支持
2//!
3//! # Spring Equivalent / Spring等价物
4//!
5//! ```java
6//! // Define validation groups
7//! public interface CreateGroup {}
8//! public interface UpdateGroup {}
9//!
10//! @NotNull(groups = CreateGroup.class)
11//! private String username;
12//!
13//! @PostMapping("/users")
14//! public ResponseEntity<User> createUser(
15//!     @Validated(CreateGroup.class) @RequestBody UserRequest request
16//! ) {
17//!     // Only validates fields marked with CreateGroup
18//! }
19//! ```
20//!
21//! # Example / 示例
22//!
23//! ```rust,ignore
24//! use hiver_validation::Validated;
25//!
26//! #[derive(Debug, Default)]
27//! struct CreateGroup;
28//!
29//! #[derive(Debug, Default)]
30//! struct UpdateGroup;
31//!
32//! #[derive(Debug, Deserialize)]
33//! struct UserRequest {
34//!     #[NotNull(group = "CreateGroup")]
35//!     username: String,
36//!
37//!     #[NotNull(group = "UpdateGroup")]
38//!     id: u64,
39//! }
40//!
41//! // Only validates fields with CreateGroup
42//! #[hiver_macros::post("/users")]
43//! async fn create_user(
44//!     #[Validated(CreateGroup)] request: UserRequest,
45//! ) -> Result<Json<User>, Error> {
46//!     Ok(Json(user))
47//! }

58/// Validation group trait / 验证分组trait
59///
60/// Marker trait for validation groups.
61/// 验证分组的标记trait。
62pub trait ValidationGroup: Any + Send + Sync + 'static
63{
64    /// Get the group name / 获取分组名称
65    fn name(&self) -> &'static str
66    {
67        let type_name = std::any::type_name::<Self>();
68        // Extract just the type name from the full path
69        // 从完整路径中提取类型名称
70        type_name.rsplit("::").next().unwrap_or(type_name)
71    }
72
73    /// Get the group `TypeId` / `获取分组TypeId`
74    fn type_id(&self) -> TypeId
75    {
76        TypeId::of::<Self>()
77    }
78}

86/// Validation group set / 验证分组集合
87///
88/// Represents a set of active validation groups.
89/// 表示一组活跃的验证分组。
90#[derive(Clone)]
91pub struct GroupSet
92{
93    /// Active group type IDs / `活跃分组TypeId`
94    groups: HashSet<TypeId>,
95    /// Group names for display / 用于显示的分组名称
96    names: HashSet<String>,
97}
98
99impl GroupSet
100{
101    /// Create a new empty group set / 创建新的空分组集合
102    pub fn new() -> Self
103    {
104        Self {
105            groups: HashSet::new(),
106            names: HashSet::new(),
107        }
108    }
109
110    /// Create with default group / 使用默认分组创建
111    pub fn default_group() -> Self
112    {
113        Self::with_group(DefaultGroup)
114    }
115
116    /// Create with a single group / 使用单个分组创建
117    #[allow(clippy::needless_pass_by_value)]
118    pub fn with_group<G: ValidationGroup>(group: G) -> Self
119    {
120        let mut set = Self::new();
121        set.add(&group);
122        set
123    }
124
125    /// Create with multiple groups / 使用多个分组创建
126    pub fn with_groups<G: ValidationGroup>(groups: impl IntoIterator<Item = G>) -> Self
127    {
128        let mut set = Self::new();
129        for group in groups
130        {
131            set.add(&group);
132        }
133        set
134    }
135
136    /// Add a group / 添加分组
137    pub fn add<G: ValidationGroup>(&mut self, group: &G)
138    {
139        self.groups.insert(TypeId::of::<G>());
140        self.names.insert(group.name().to_string());
141    }
142
143    /// Check if a group is active / 检查分组是否活跃
144    pub fn contains<G: ValidationGroup>(&self) -> bool
145    {
146        self.groups.contains(&TypeId::of::<G>())
147    }
148
149    /// Check if default group is active / 检查默认分组是否活跃
150    pub fn contains_default(&self) -> bool
151    {
152        self.contains::<DefaultGroup>()
153    }
154
155    /// Check if the set is empty / 检查集合是否为空
156    pub fn is_empty(&self) -> bool
157    {
158        self.groups.is_empty()
159    }
160
161    /// Get the number of groups / 获取分组数量
162    pub fn len(&self) -> usize
163    {
164        self.groups.len()
165    }
166
167    /// Get all group names / 获取所有分组名称
168    pub fn names(&self) -> &HashSet<String>
169    {
170        &self.names
171    }
172}
173
174impl Default for GroupSet
175{
176    fn default() -> Self
177    {
178        Self::default_group()
179    }
180}
181
182impl fmt::Debug for GroupSet
183{
184    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
185    {
186        f.debug_set().entries(self.names.iter()).finish()
187    }
188}

190/// Validated wrapper for group-based validation
191/// 用于基于分组验证的Validated包装器
192///
193/// Extracts and validates request data using specific validation groups.
194/// 使用特定验证分组提取和验证请求数据。
195///
196/// # Spring Equivalent / Spring等价物
197///
198/// ```java
199/// @PostMapping("/users")
200/// public ResponseEntity<User> createUser(
201///     @Validated(CreateGroup.class) @RequestBody UserRequest request
202/// ) { }
203/// ```
204///
205/// # Example / 示例
206///
207/// ```rust,ignore
208/// use hiver_validation::Validated;
209///
210/// #[derive(Debug, Default)]
211/// struct CreateGroup;
212///
213/// #[hiver_macros::post("/users")]
214/// async fn create_user(
215///     #[Validated(CreateGroup)] request: CreateUserRequest,
216/// ) -> Result<Json<User>, Error> {
217///     // Only validates fields marked with CreateGroup
218/// }
219/// ```
220pub struct Validated<T, G = DefaultGroup>
221{
222    /// The validated value / 已验证的值
223    pub value: T,
224    /// The validation group / 验证分组
225    pub group: G,
226}
227
228impl<T, G> Validated<T, G>
229{
230    /// Create a new Validated wrapper / 创建新的Validated包装器
231    pub fn new(value: T, group: G) -> Self
232    {
233        Self { value, group }
234    }
235
236    /// Consume and return the inner value / 消耗并返回内部值
237    pub fn into_inner(self) -> T
238    {
239        self.value
240    }
241
242    /// Get reference to the inner value / 获取内部值的引用
243    pub fn get(&self) -> &T
244    {
245        &self.value
246    }
247}
248
249impl<T> Validated<T, DefaultGroup>
250{
251    /// Create with default group / 使用默认分组创建
252    pub fn with_default(value: T) -> Self
253    {
254        Self {
255            value,
256            group: DefaultGroup,
257        }
258    }
259}

269    /// Get the group `TypeId` / `获取分组TypeId`

276impl<T: fmt::Debug, G> fmt::Debug for Validated<T, G>
277{

298/// Validated extractor implementation
299/// Validated提取器实现
300impl<T, G> Validated<T, G>
301where
302    T: serde::de::DeserializeOwned + crate::Validate,
303    G: ValidationGroup,
304{
305    /// Validate with group / 使用分组验证
306    pub fn validate_with_group(data: T, _group: G) -> Result<Self, ValidationError>
307    {
308        // For now, just validate using the Validate trait
309        // The group-based validation will be enhanced with macro support
310        // 目前，仅使用Validate trait进行验证
311        // 基于分组的验证将通过宏支持增强
312        data.validate().map_err(ValidationError::from)?;
313        Ok(Self {
314            value: data,
315            group: _group,
316        })
317    }
318}

320/// Macro to create validation groups / 创建验证分组的宏
321///
322/// # Example / 示例
323///
324/// ```rust,ignore
325/// use hiver_validation::validation_groups;
326///
327/// validation_groups! {
328///     CreateGroup,
329///     UpdateGroup,
330///     DeleteGroup,
331/// }
332/// ```
333#[macro_export]
334macro_rules! validation_groups {
335    (
336        $(
337            $(#[$meta:meta])*
338            $group:ident
339        ),* $(,)?
340    ) => {
341        $(
342            $(#[$meta])*
343            #[allow(unreachable_pub)]
344            #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
345            pub struct $group;
346            impl $crate::groups::ValidationGroup for $group {}
347        )*
348    };
349}
