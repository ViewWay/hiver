1//! Type-level retry error classification — a Rust advantage over Spring.
2//! 类型级重试错误分类 — Rust 相比 Spring 的独特优势。
3//!
4//! Spring checks exception classes at runtime; Hiver uses marker traits
5//! to classify errors at compile time, ensuring no retryable error is missed.
6//!
7//! Spring 在运行时检查异常类；Hiver 使用标记 trait 在编译期分类错误，
8//! 确保不会遗漏可重试错误。
9
10use std::fmt;
11
12/// Marker trait for errors that are safe to retry.
13/// 可安全重试的错误的标记 trait。
14///
15/// Types implementing this trait will be automatically retried by
16/// `RetryTemplate::execute_retryable()`. The compiler enforces that only
17/// explicitly-marked errors trigger retries.
18///
19/// 实现此 trait 的类型会被 `RetryTemplate::execute_retryable()` 自动重试。
20/// 编译器确保只有显式标记的错误才会触发重试。
21///
22/// # Example / 示例
23///
24/// ```rust,ignore
25/// use hiver_retry::Retryable;
26///
27/// #[derive(Debug, thiserror::Error)]
28/// #[error("Connection timeout")]
29/// struct ConnectionTimeout;
30///
31/// impl Retryable for ConnectionTimeout {}
32///
33/// // Now ConnectionTimeout is retryable at compile time
34/// ```
35pub trait Retryable: std::error::Error + Send + Sync {}
36
37/// Marker trait for errors that should NOT be retried.
38/// 不应重试的错误的标记 trait。
39///
40/// Non-retryable errors short-circuit the retry loop immediately,
41/// regardless of remaining attempts.
42///
43/// 不可重试错误会立即短路重试循环，不管剩余尝试次数。
44pub trait NonRetryable: std::error::Error + Send + Sync {}

50/// A generic retryable error wrapper.
51/// 通用可重试错误包装器。
52///
53/// Wraps any error to make it retryable.
54/// 包装任何错误使其可重试。
55#[derive(Debug)]
56pub struct RetryableError<E>
57{
58    /// The wrapped error.
59    pub inner: E,
60}
61
62impl<E> RetryableError<E>
63{
64    /// Creates a new retryable error wrapper.
65    /// 创建新的可重试错误包装器。
66    #[must_use]
67    pub fn new(error: E) -> Self
68    {
69        Self { inner: error }
70    }
71}
72
73impl<E: std::error::Error + Send + Sync> Retryable for RetryableError<E> {}
74
75impl<E: fmt::Display> fmt::Display for RetryableError<E>
76{
77    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result
78    {
79        write!(f, "[retryable] {}", self.inner)
80    }
81}
82
83impl<E: std::error::Error> std::error::Error for RetryableError<E>
84{
85    fn source(&self) -> Option<&(dyn std::error::Error + 'static)>
86    {
87        self.inner.source()
88    }
89}

133/// Result of classifying an error for retry.
134/// 错误重试分类结果。
135#[derive(Debug, Clone, Copy, PartialEq, Eq)]
136pub enum RetryDecision
137{
138    /// This error is retryable — continue retrying.
139    /// 此错误可重试 — 继续重试。
140    Retry,
141    /// This error is fatal — stop retrying immediately.
142    /// 此错误是致命的 — 立即停止重试。
143    Stop,
144}
