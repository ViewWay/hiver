7/// Transaction error
8/// 事务错误
9///
10/// Equivalent to Spring's `TransactionException`.
11/// `等价于Spring的TransactionException`。
12#[derive(Error, Debug)]
13pub enum TransactionError
14{
15    /// Transaction creation failed
16    /// 事务创建失败
17    #[error("Failed to create transaction: {0}")]
18    CreationFailed(String),
19
20    /// Transaction commit failed
21    /// 事务提交失败
22    #[error("Failed to commit transaction: {0}")]
23    CommitFailed(String),
24
25    /// Transaction rollback failed
26    /// 事务回滚失败
27    #[error("Failed to rollback transaction: {0}")]
28    RollbackFailed(String),
29
30    /// Transaction timeout
31    /// 事务超时
32    #[error("Transaction timeout after {0} seconds")]
33    Timeout(u64),
34
35    /// Invalid transaction state
36    /// 无效事务状态
37    #[error("Invalid transaction state: {0}")]
38    InvalidState(String),
39
40    /// Transaction not found
41    /// 事务未找到
42    #[error("Transaction not found: {0}")]
43    NotFound(String),
44
45    /// Concurrent modification
46    /// 并发修改
47    #[error("Concurrent modification: {0}")]
48    ConcurrentModification(String),
49
50    /// Deadlock detected
51    /// 检测到死锁
52    #[error("Deadlock detected: {0}")]
53    Deadlock(String),
54
55    /// Database error
56    /// 数据库错误
57    #[error("Database error: {0}")]
58    Database(String),
59
60    /// IO error
61    /// IO错误
62    #[error("IO error: {0}")]
63    Io(#[from] std::io::Error),
64
65    /// Other error
66    /// 其他错误
67    #[error("Transaction error: {0}")]
68    Other(String),
69}
70
71/// Transaction result type
72/// 事务结果类型
73pub type TransactionResult<T> = Result<T, TransactionError>;

75/// Transaction exception runtime type
76/// 事务异常运行时类型
77///
78/// Equivalent to Spring's `TransactionSystemException`.
79/// `等价于Spring的TransactionSystemException`。
80#[derive(Error, Debug)]
81#[error("Transaction system exception: {message}")]
82pub(crate) struct TransactionSystemException
83{
84    /// Error message
85    /// 错误消息
86    pub message: String,
87
88    /// Underlying error
89    /// 底层错误
90    #[source]
91    pub source: Option<Box<dyn std::error::Error + Send + Sync>>,
92}
93
94impl TransactionSystemException
95{
96    /// Create a new transaction system exception
97    /// 创建新的事务系统异常
98    pub(crate) fn new(message: impl Into<String>) -> Self
99    {
100        Self {
101            message: message.into(),
102            source: None,
103        }
104    }
105
106    /// Create with underlying error
107    /// 使用底层错误创建
108    pub(crate) fn with_source(
109        message: impl Into<String>,
110        source: impl Into<Box<dyn std::error::Error + Send + Sync>>,
111    ) -> Self
112    {
113        Self {
114            message: message.into(),
115            source: Some(source.into()),
116        }
117    }
118}
