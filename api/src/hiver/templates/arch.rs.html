6/// Supported architecture patterns.
7/// 支持的架构模式。
8#[derive(Debug, Clone, Copy, PartialEq, Eq)]
9pub enum Architecture
10{
11    /// Layered Architecture (handler / service / repository / model)
12    /// 分层架构
13    Layered,
14    /// Hexagonal / Ports & Adapters Architecture
15    /// 六角架构 / 端口适配器架构
16    Hexagonal,
17    /// Clean Architecture (domain / usecase / interface / infrastructure)
18    /// 整洁架构
19    Clean,
20    /// Domain-Driven Design
21    /// 领域驱动设计
22    Ddd,
23    /// Microservice Architecture (service-per-boundary, API gateway)
24    /// 微服务架构
25    Microservice,
26}
27
28impl Architecture
29{
30    /// Parse from string, returns None for unknown values.
31    /// 从字符串解析，未知值返回 None。
32    pub fn from_str_opt(s: &str) -> Option<Self>
33    {
34        match s.to_lowercase().as_str()
35        {
36            "layered" => Some(Self::Layered),
37            "hexagonal" | "hex" | "ports" => Some(Self::Hexagonal),
38            "clean" => Some(Self::Clean),
39            "ddd" | "domain" => Some(Self::Ddd),
40            "microservice" | "micro" | "ms" => Some(Self::Microservice),
41            _ => None,
42        }
43    }
44
45    /// Get all valid architecture names for help text.
46    /// 获取所有有效的架构名称，用于帮助文本。
47    pub fn valid_names() -> &'static str
48    {
49        "layered, hexagonal, clean, ddd, microservice"
50    }
51}

53/// Directory entry to create.
54/// 要创建的目录条目。
55struct DirTemplate
56{
57    /// Relative path from src/.
58    /// 相对于 src/ 的路径。
59    path: &'static str,
60    /// Module file content.
61    /// 模块文件内容。
62    content: &'static str,
63}
64
65/// Create directory structure for the given architecture.
66/// 为指定架构创建目录结构。
67pub fn create_arch_dirs(
68    base: &Path,
69    arch: Architecture,
70    modules: &[String],
71) -> Result<(), std::io::Error>
72{
73    let src = base.join("src");
74    let resources = base.join("resources");
75
76    std::fs::create_dir_all(&resources)?;
77
78    let entries = arch_entries(arch, modules);
79    for entry in &entries
80    {
81        let dir = src.join(entry.path);
82        std::fs::create_dir_all(&dir)?;
83        let mod_path = dir.join("mod.rs");
84        if !mod_path.exists()
85        {
86            std::fs::write(&mod_path, entry.content)?;
87        }
88    }
89
90    Ok(())
91}

93/// Generate main.rs content for the given architecture.
94/// 为指定架构生成 main.rs 内容。
95pub fn generate_arch_main_rs(arch: Architecture, modules: &[String]) -> String
96{
97    match arch
98    {
99        Architecture::Layered => layered_main_rs(modules),
100        Architecture::Hexagonal => hexagonal_main_rs(modules),
101        Architecture::Clean => clean_main_rs(modules),
102        Architecture::Ddd => ddd_main_rs(modules),
103        Architecture::Microservice => microservice_main_rs(modules),
104    }
105}
