80/// Response building example / 响应构建示例
81///
82/// Demonstrates various ways to build HTTP responses.
83/// 演示构建HTTP响应的各种方法。
84fn response_building_example()
85{
86    // Simple text response / 简单文本响应
87    let response = Response::builder()
88        .status(StatusCode::OK)
89        .header(header::CONTENT_TYPE, content_type::TEXT)
90        .body(Body::from("Hello, World!"))
91        .unwrap();
92    println!("  Text response: {:?}", response.status());
93
94    // JSON response / JSON响应
95    let response = Response::builder()
96        .status(StatusCode::OK)
97        .header(header::CONTENT_TYPE, content_type::JSON)
98        .body(Body::from(r#"{"message": "Hello"}"#))
99        .unwrap();
100    println!("  JSON response: {:?}", response.status());
101
102    // Empty response (204 No Content) / 空响应
103    let response = Response::builder()
104        .status(StatusCode::NO_CONTENT)
105        .body(Body::empty())
106        .unwrap();
107    println!("  Empty response: {:?}", response.status());
108}

110/// JSON response example / JSON响应示例
111///
112/// Demonstrates the Json wrapper for automatic serialization.
113/// 演示Json包装器的自动序列化。
114///
115/// Equivalent to Spring's `@ResponseBody` annotation.
116/// 等价于 Spring 的 `@ResponseBody` 注解。
117fn json_response_example()
118{
119    // Create a user response / 创建用户响应
120    let user = User {
121        id: 1,
122        name: "Alice".to_string(),
123        email: "alice@example.com".to_string(),
124    };
125
126    // Wrap in Json for automatic serialization / 使用Json包装自动序列化
127    let json_user = Json::new(user);
128
129    println!("  Json<User> created");
130    println!("  Inner value: {:?}", json_user.0);
131    println!("  Content-Type would be: {}", content_type::JSON);
132
133    // Get inner value / 获取内部值
134    let inner = json_user.get();
135    println!("  Retrieved user: {}", inner.name);
136}

156/// `IntoResponse` trait example / `IntoResponse` trait示例
157///
158/// Demonstrates automatic conversion to Response.
159/// 演示自动转换为Response。
160fn into_response_example()
161{
162    // String → Response / 字符串自动转响应
163    let response: Response = "Hello, World!".into_response();
164    println!("  String → Response: {:?}", response.status());
165
166    // &str → Response / 字符串切片转响应
167    let response: Response = "Static text".into_response();
168    println!("  &str → Response: {:?}", response.status());
169
170    // () → Response (204 No Content) / 空元组转204响应
171    let response: Response = ().into_response();
172    println!("  () → Response: {:?}", response.status());
173
174    // StatusCode → Response / 状态码转响应
175    let response: Response = StatusCode::NOT_FOUND.into_response();
176    println!("  StatusCode → Response: {:?}", response.status());
177
178    // Vec<u8> → Response / 字节向量转响应
179    let data = vec![1, 2, 3, 4];
180    let response: Response = data.into_response();
181    println!("  Vec<u8> → Response: {:?}", response.status());
182}

234/// JSON parsing example / JSON解析示例
235///
236/// Demonstrates parsing JSON from request body.
237/// 演示从请求体解析JSON。
238fn json_parsing_example()
239{
240    let json_body = r#"{"id":1,"name":"Alice","email":"alice@example.com"}"#;
241
242    // Parse JSON string / 解析JSON字符串
243    if let Ok(user) = serde_json::from_str::<User>(json_body)
244    {
245        println!("  Parsed user: {} (email: {})", user.name, user.email);
246    }
247
248    // Create Json wrapper / 创建Json包装器
249    let user = User {
250        id: 1,
251        name: "Alice".to_string(),
252        email: "alice@example.com".to_string(),
253    };
254    let json_user = Json::new(user);
255    println!("  Json<User>: {} (id: {})", json_user.get().name, json_user.get().id);
256}

346/// Example: Error handling endpoint / 示例：错误处理端点
347///
348/// Demonstrates returning different error responses.
349/// 演示返回不同的错误响应。
350fn example_error_handler(error_type: &str) -> Response
351{
352    let (status, error_msg) = match error_type
353    {
354        "not_found" => (StatusCode::NOT_FOUND, "User not found"),
355        "bad_request" => (StatusCode::BAD_REQUEST, "Invalid input"),
356        "unauthorized" => (StatusCode::UNAUTHORIZED, "Authentication required"),
357        "forbidden" => (StatusCode::FORBIDDEN, "Access denied"),
358        "internal" => (StatusCode::INTERNAL_SERVER_ERROR, "Internal server error"),
359        _ => (StatusCode::OK, "Success"),
360    };
361
362    Response::builder()
363        .status(status)
364        .header(header::CONTENT_TYPE, content_type::JSON)
365        .body(Body::from(format!(r#"{{"error":"{}"}}"#, error_msg)))
366        .unwrap()
367}

369/// Example: Pagination response / 示例：分页响应
370///
371/// Demonstrates a paginated response pattern.
372/// 演示分页响应模式。
373fn example_paginated_response(page: u32, size: u32) -> Response
374{
375    let users = vec![
376        User {
377            id: 1,
378            name: "Alice".to_string(),
379            email: "alice@example.com".to_string(),
380        },
381        User {
382            id: 2,
383            name: "Bob".to_string(),
384            email: "bob@example.com".to_string(),
385        },
386    ];
387
388    let pagination = PaginationResponse {
389        data: users,
390        page,
391        size,
392        total: 100,
393        has_more: (page * size) < 100,
394    };
395
396    let json = serde_json::to_string(&pagination).unwrap();
397
398    Response::builder()
399        .status(StatusCode::OK)
400        .header(header::CONTENT_TYPE, content_type::JSON)
401        .body(Body::from(json))
402        .unwrap()
403}
