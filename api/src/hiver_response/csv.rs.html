1//! CSV export module
2//! CSV 导出模块
3//!
4//! # Overview / 概述
5//!
6//! This module provides CSV (Comma-Separated Values) export capabilities.
7//! It generates RFC 4180 compliant CSV output with configurable delimiters,
8//! quoting rules, and streaming support.
9//!
10//! 本模块提供 CSV（逗号分隔值）导出功能。
11//! 它生成符合 RFC 4180 的 CSV 输出，支持可配置分隔符、引号规则和流式写入。
12//!
13//! # Equivalent to Spring / 等价于 Spring
14//!
15//! - OpenCSV / `CsvMapper`
16//! - EasyExcel CSV export
17//! - Spring Batch CSV `FlatFileItemWriter`
18//!
19//! # Example / 示例
20//!
21//! ```rust,ignore
22//! use hiver_response::csv::{CsvExportConfig, CsvExporter};
23//!
24//! let config = CsvExportConfig::new()
25//!     .header("Name")
26//!     .header("Age")
27//!     .header("Email");
28//!
29//! let mut exporter = CsvExporter::new(config);
30//! exporter.add_row(vec!["Alice", "30", "alice@example.com"]);
31//! exporter.add_row(vec!["Bob", "25", "bob@example.com"]);
32//!
33//! let csv_string = exporter.to_string().unwrap();
34//! ```

45/// CSV export error.
46/// CSV 导出错误。
47#[derive(Debug, thiserror::Error)]
48pub enum CsvError
49{
50    /// I/O error during write.
51    /// 写入过程中的 I/O 错误。
52    #[error("I/O error: {0}")]
53    Io(#[from] io::Error),
54
55    /// Formatting error.
56    /// 格式化错误。
57    #[error("Format error: {0}")]
58    Format(String),
59}
60
61/// Result type for CSV operations.
62/// CSV 操作的结果类型。
63pub type Result<T> = std::result::Result<T, CsvError>;

69/// CSV export configuration.
70/// CSV 导出配置。
71///
72/// Equivalent to Spring's OpenCSV `CSVWriter` settings.
73/// 等价于 Spring 的 OpenCSV `CSVWriter` 设置。
74#[derive(Debug, Clone)]
75pub struct CsvExportConfig
76{
77    /// Column headers / 列标题
78    pub headers: Vec<String>,
79    /// Field delimiter (default: comma) / 字段分隔符（默认：逗号）
80    pub delimiter: char,
81    /// Quote character (default: double-quote) / 引号字符（默认：双引号）
82    pub quote_char: char,
83    /// Line terminator / 行终止符
84    pub line_terminator: String,
85    /// Always quote fields / 总是引用字段
86    pub always_quote: bool,
87    /// Include BOM for Excel compatibility / 包含 BOM 以兼容 Excel
88    pub include_bom: bool,
89}
90
91impl Default for CsvExportConfig
92{
93    fn default() -> Self
94    {
95        Self {
96            headers: Vec::new(),
97            delimiter: ',',
98            quote_char: '"',
99            line_terminator: "\r\n".to_string(),
100            always_quote: false,
101            include_bom: false,
102        }
103    }
104}

175// ---------------------------------------------------------------------------
176// CSV exporter
177// ---------------------------------------------------------------------------

179/// CSV exporter for generating CSV output.
180/// 用于生成 CSV 输出的导出器。
181///
182/// Equivalent to Spring's OpenCSV `CSVWriter` or EasyExcel CSV export.
183/// 等价于 Spring 的 OpenCSV `CSVWriter` 或 EasyExcel CSV 导出。
184///
185/// # Example / 示例
186///
187/// ```rust,ignore
188/// use hiver_response::csv::{CsvExportConfig, CsvExporter};
189///
190/// let config = CsvExportConfig::new()
191///     .header("ID")
192///     .header("Name");
193///
194/// let mut exporter = CsvExporter::new(config);
195/// exporter.add_row(vec!["1", "Alice"]);
196/// exporter.add_row(vec!["2", "Bob"]);
197///
198/// let output = exporter.to_string().unwrap();
199/// assert!(output.contains("ID,Name"));
200/// assert!(output.contains("1,Alice"));
201/// ```
202pub struct CsvExporter
203{
204    config: CsvExportConfig,
205    rows: Vec<Vec<String>>,
206}
207
208impl CsvExporter
209{
210    /// Create a new CSV exporter with the given configuration.
211    /// 使用给定配置创建新的 CSV 导出器。
212    pub fn new(config: CsvExportConfig) -> Self
213    {
214        Self {
215            config,
216            rows: Vec::new(),
217        }
218    }
219
220    /// Add a data row.
221    /// 添加数据行。
222    pub fn add_row(&mut self, fields: Vec<impl Into<String>>)
223    {
224        self.rows.push(fields.into_iter().map(Into::into).collect());
225    }
226
227    /// Add a pre-formatted row (already `String`).
228    /// 添加预格式化行（已是 `String`）。
229    pub fn add_row_str(&mut self, fields: Vec<String>)
230    {
231        self.rows.push(fields);
232    }
233
234    /// Returns the number of data rows (excluding header).
235    /// 返回数据行数（不包括标题）。
236    pub fn row_count(&self) -> usize
237    {
238        self.rows.len()
239    }
240
241    /// Generate the CSV output as a string.
242    /// 将 CSV 输出生成为字符串。
243    pub fn to_string(&self) -> Result<String>
244    {
245        let mut buf = String::new();
246
247        // UTF-8 BOM for Excel compatibility
248        // UTF-8 BOM 用于 Excel 兼容性
249        if self.config.include_bom
250        {
251            buf.push('\u{FEFF}');
252        }
253
254        // Header row
255        // 标题行
256        if !self.config.headers.is_empty()
257        {
258            let line = format_csv_line(&self.config.headers, &self.config);
259            buf.push_str(&line);
260            buf.push_str(&self.config.line_terminator);
261        }
262
263        // Data rows
264        // 数据行
265        for row in &self.rows
266        {
267            let line = format_csv_line(row, &self.config);
268            buf.push_str(&line);
269            buf.push_str(&self.config.line_terminator);
270        }
271
272        Ok(buf)
273    }
274
275    /// Generate the CSV output as bytes.
276    /// 将 CSV 输出生成为字节。
277    pub fn to_bytes(&self) -> Result<Vec<u8>>
278    {
279        Ok(self.to_string()?.into_bytes())
280    }
281
282    /// Write the CSV output to a writer.
283    /// 将 CSV 输出写入 writer。
284    pub fn write_to_writer(&self, mut writer: impl Write) -> Result<()>
285    {
286        let data = self.to_bytes()?;
287        writer.write_all(&data)?;
288        Ok(())
289    }
290
291    /// Write the CSV output to a file.
292    /// 将 CSV 输出写入文件。
293    pub fn write_to(&self, path: impl AsRef<Path>) -> Result<()>
294    {
295        let data = self.to_bytes()?;
296        std::fs::write(path, data)?;
297        Ok(())
298    }
299}

301// ---------------------------------------------------------------------------
302// CsvTable trait
303// ---------------------------------------------------------------------------
304
305/// Trait for types that can be auto-exported to CSV.
306/// 可自动导出到 CSV 的类型的 trait。
307///
308/// Equivalent to Spring Batch's `FieldSetMapper` or OpenCSV's bean mapping.
309/// 等价于 Spring Batch 的 `FieldSetMapper` 或 OpenCSV 的 Bean 映射。
310///
311/// # Example / 示例
312///
313/// ```rust,ignore
314/// use hiver_response::csv::{CsvTable, export_to_csv};
315///
316/// struct User {
317///     name: String,
318///     email: String,
319/// }
320///
321/// impl CsvTable for User {
322///     fn csv_headers() -> Vec<String> {
323///         vec!["Name".into(), "Email".into()]
324///     }
325///
326///     fn csv_row(&self) -> Vec<String> {
327///         vec![self.name.clone(), self.email.clone()]
328///     }
329/// }
330/// ```
331pub trait CsvTable
332{
333    /// Return the column headers for this type.
334    /// 返回此类型的列标题。
335    fn csv_headers() -> Vec<String>;
336
337    /// Convert this instance into a row of string fields.
338    /// 将此实例转换为一行字符串字段。
339    fn csv_row(&self) -> Vec<String>;
340}
341
342/// Export a slice of `CsvTable` implementors to CSV string.
343/// 将 `CsvTable` 实现者切片导出为 CSV 字符串。
344///
345/// # Example / 示例
346///
347/// ```rust,ignore
348/// use hiver_response::csv::{CsvExportConfig, CsvTable, export_to_csv};
349///
350/// let data = vec![User { name: "Alice".into(), email: "a@b.com".into() }];
351/// let csv = export_to_csv(&data, CsvExportConfig::new()).unwrap();
352/// ```
353pub fn export_to_csv<T: CsvTable>(data: &[T], config: CsvExportConfig) -> Result<String>
354{
355    let mut new_config = config;
356    if new_config.headers.is_empty() && !data.is_empty()
357    {
358        new_config.headers = T::csv_headers();
359    }
360
361    let mut exporter = CsvExporter::new(new_config);
362    for item in data
363    {
364        exporter.add_row_str(item.csv_row());
365    }
366    exporter.to_string()
367}

369// ---------------------------------------------------------------------------
370// CSV line formatting
371// ---------------------------------------------------------------------------
372
373/// Format a single CSV line from field values.
374/// 将字段值格式化为单行 CSV。
375fn format_csv_line(fields: &[String], config: &CsvExportConfig) -> String
376{
377    let mut line = String::new();
378    for (i, field) in fields.iter().enumerate()
379    {
380        if i > 0
381        {
382            line.push(config.delimiter);
383        }
384        if config.always_quote || needs_quoting(field, config)
385        {
386            let quoted = quote_field(field, config);
387            line.push_str(&quoted);
388        }
389        else
390        {
391            line.push_str(field);
392        }
393    }
394    line
395}
396
397/// Check if a field value requires quoting per RFC 4180.
398/// 检查字段值是否需要按 RFC 4180 加引号。
399fn needs_quoting(field: &str, config: &CsvExportConfig) -> bool
400{
401    field.contains(config.delimiter)
402        || field.contains(config.quote_char)
403        || field.contains('\n')
404        || field.contains('\r')
405}
406
407/// Quote a field value, escaping internal quote characters.
408/// 引用字段值，转义内部引号字符。
409fn quote_field(field: &str, config: &CsvExportConfig) -> String
410{
411    let escaped =
412        field.replace(config.quote_char, &format!("{}{}", config.quote_char, config.quote_char));
413    format!("{}{}{}", config.quote_char, escaped, config.quote_char)
414}

418// ---------------------------------------------------------------------------
419
420/// Wrapper type that enables CSV output to be used as an HTTP response.
421/// 包装类型，使 CSV 输出可以用作 HTTP 响应。
422///
423/// Equivalent to Spring's `ResponseEntity` with `Content-Type: text/csv`.
424/// 等价于 Spring 的 `ResponseEntity` 配合 `Content-Type: text/csv`。
425pub struct Csv(pub CsvExporter);
426
427impl crate::IntoResponse for Csv
428{
429    fn into_response(self) -> crate::Response
430    {
431        match self.0.to_bytes()
432        {
433            Ok(bytes) => crate::Response::builder()
434                .header("content-type", "text/csv; charset=utf-8")
435                .header("content-disposition", "attachment; filename=\"export.csv\"")
436                .body(bytes)
437                .unwrap_or_else(|_| crate::Response::new()),
438            Err(_) => crate::Response::builder()
439                .status(http::StatusCode::INTERNAL_SERVER_ERROR)
440                .body("Failed to generate CSV")
441                .unwrap_or_else(|_| crate::Response::new()),
442        }
443    }
444}
