6/// Delivery mode
7/// 传递模式
8#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize, Default)]
9pub enum DeliveryMode
10{
11    /// Transient (not persisted)
12    /// 瞬态（不持久化）
13    #[default]
14    Transient = 1,
15
16    /// Persistent (survives broker restart)
17    /// 持久化（代理重启后存活）
18    Persistent = 2,
19}

21/// Message properties
22/// 消息属性
23///
24/// # Spring Equivalent / Spring等价物
25///
26/// ```java
27/// MessageProperties props = MessageProperties.builder()
28///     .setContentType("application/json")
29///     .setDeliveryMode(MessageDeliveryMode.PERSISTENT)
30///     .setExpiration("60000")
31///     .build();
32/// ```
33#[derive(Clone, Debug, Serialize, Deserialize, Default)]
34pub struct MessageProperties
35{
36    /// Content type
37    /// 内容类型
38    #[serde(skip_serializing_if = "Option::is_none")]
39    pub content_type: Option<String>,
40
41    /// Content encoding
42    /// 内容编码
43    #[serde(skip_serializing_if = "Option::is_none")]
44    pub content_encoding: Option<String>,
45
46    /// Delivery mode
47    /// 传递模式
48    #[serde(default)]
49    pub delivery_mode: DeliveryMode,
50
51    /// Priority (0-9)
52    /// 优先级（0-9）
53    #[serde(skip_serializing_if = "Option::is_none")]
54    pub priority: Option<u8>,
55
56    /// Correlation ID
57    /// 关联ID
58    #[serde(skip_serializing_if = "Option::is_none")]
59    pub correlation_id: Option<String>,
60
61    /// Reply to
62    /// 回复到
63    #[serde(skip_serializing_if = "Option::is_none")]
64    pub reply_to: Option<String>,
65
66    /// Expiration (milliseconds)
67    /// 过期时间（毫秒）
68    #[serde(skip_serializing_if = "Option::is_none")]
69    pub expiration: Option<String>,
70
71    /// Message ID
72    /// 消息ID
73    #[serde(skip_serializing_if = "Option::is_none")]
74    pub message_id: Option<String>,
75
76    /// Timestamp
77    /// 时间戳
78    #[serde(skip_serializing_if = "Option::is_none")]
79    pub timestamp: Option<u64>,
80
81    /// Type
82    /// 类型
83    #[serde(skip_serializing_if = "Option::is_none")]
84    pub message_type: Option<String>,
85
86    /// User ID
87    /// 用户ID
88    #[serde(skip_serializing_if = "Option::is_none")]
89    pub user_id: Option<String>,
90
91    /// Application ID
92    /// 应用ID
93    #[serde(skip_serializing_if = "Option::is_none")]
94    pub app_id: Option<String>,
95
96    /// Headers
97    /// 头
98    #[serde(default)]
99    pub headers: std::collections::HashMap<String, serde_json::Value>,
100}

102impl MessageProperties
103{
104    /// Create new message properties
105    /// 创建新的消息属性
106    pub fn new() -> Self

111    /// Set content type
112    /// 设置内容类型

115        self.content_type = Some(content_type.into());
116        self
117    }
118
119    /// Set content encoding
120    /// 设置内容编码
121    pub fn with_content_encoding(mut self, encoding: impl Into<String>) -> Self
122    {
123        self.content_encoding = Some(encoding.into());
124        self
125    }
126
127    /// Set delivery mode
128    /// 设置传递模式
129    pub fn with_delivery_mode(mut self, mode: DeliveryMode) -> Self
130    {
131        self.delivery_mode = mode;
132        self
133    }
134
135    /// Set priority
136    /// 设置优先级
137    pub fn with_priority(mut self, priority: u8) -> Self
138    {
139        self.priority = Some(priority.min(9));
140        self
141    }
142
143    /// Set correlation ID
144    /// 设置关联ID
145    pub fn with_correlation_id(mut self, id: impl Into<String>) -> Self
146    {
147        self.correlation_id = Some(id.into());
148        self
149    }
150
151    /// Set reply to
152    /// 设置回复到
153    pub fn with_reply_to(mut self, reply_to: impl Into<String>) -> Self
154    {
155        self.reply_to = Some(reply_to.into());
156        self
157    }
158
159    /// Set expiration
160    /// 设置过期时间
161    pub fn with_expiration(mut self, expiration: impl Into<String>) -> Self
162    {
163        self.expiration = Some(expiration.into());
164        self
165    }
166
167    /// Add header
168    /// 添加头
169    pub fn with_header(
170        mut self,
171        key: impl Into<String>,
172        value: impl Into<serde_json::Value>,
173    ) -> Self
174    {
175        self.headers.insert(key.into(), value.into());
176        self
177    }
178}

180/// AMQP message
181/// AMQP消息
182///
183/// # Spring Equivalent / Spring等价物
184///
185/// ```java
186/// Message<String> message = MessageBuilder
187///     .withPayload("Hello, RabbitMQ!")
188///     .setHeader("key", "value")
189///     .build();
190/// ```
191#[derive(Clone, Debug, Serialize, Deserialize)]
192pub struct Message
193{
194    /// Payload
195    /// 有效载荷
196    pub payload: Vec<u8>,
197
198    /// Properties
199    /// 属性
200    #[serde(default)]
201    pub properties: MessageProperties,
202}
203
204impl Message
205{
206    /// Create new message
207    /// 创建新消息
208    pub fn new(payload: Vec<u8>) -> Self
209    {
210        Self {
211            payload,
212            properties: MessageProperties::default(),
213        }
214    }
215
216    /// Create from string
217    /// 从字符串创建
218    pub fn from_string(s: impl Into<String>) -> Self
219    {
220        Self::new(s.into().into_bytes())
221    }
222
223    /// Create with properties
224    /// 使用属性创建
225    pub fn with_properties(mut self, properties: MessageProperties) -> Self
226    {
227        self.properties = properties;
228        self
229    }
230
231    /// Get payload as string
232    /// 获取payload的字符串表示
233    pub fn payload_as_string(&self) -> String
234    {
235        String::from_utf8_lossy(&self.payload).to_string()
236    }
237}
238
239impl From<Vec<u8>> for Message
240{
241    fn from(payload: Vec<u8>) -> Self
242    {
243        Self::new(payload)
244    }
245}
246
247impl From<String> for Message
248{
249    fn from(s: String) -> Self
250    {
251        Self::from_string(s)
252    }
253}
254
255impl From<&str> for Message
256{
257    fn from(s: &str) -> Self
258    {
259        Self::from_string(s)
260    }
261}

263/// AMQP message with metadata
264/// 带元数据的AMQP消息
265///
266/// # Spring Equivalent / Spring等价物
267///
268/// ```java
269/// @RabbitListener(queues = "my_queue")
270/// public void handleMessage(Message message) {
271///     MessageHeaders headers = message.getHeaders();
272///     Object payload = message.getPayload();
273/// }
274/// ```
275#[derive(Clone)]
276pub struct AmqpMessage
277{
278    /// Message
279    /// 消息
280    pub message: Message,
281
282    /// Exchange
283    /// 交换机
284    pub exchange: String,
285
286    /// Routing key
287    /// 路由键
288    pub routing_key: String,
289
290    /// Delivery tag
291    /// 传递标签
292    pub delivery_tag: u64,
293
294    /// Redelivered flag
295    /// 重新传递标志
296    pub redelivered: bool,
297}
298
299impl AmqpMessage
300{
301    /// Create new AMQP message
302    /// 创建新的AMQP消息
303    pub fn new(message: Message) -> Self
304    {
305        Self {
306            message,
307            exchange: String::new(),
308            routing_key: String::new(),
309            delivery_tag: 0,
310            redelivered: false,
311        }
312    }
313
314    /// Get payload
315    /// 获取payload
316    pub fn payload(&self) -> &[u8]
317    {
318        &self.message.payload
319    }
320
321    /// Get payload as string
322    /// 获取payload的字符串表示
323    pub fn payload_as_string(&self) -> String
324    {
325        self.message.payload_as_string()
326    }
327
328    /// Acknowledge the message
329    /// 确认消息
330    pub fn ack(&self) -> Result<(), String>
331    {
332        // Mock implementation
333        // 模拟实现
334        tracing::debug!("Acknowledging message with delivery tag: {}", self.delivery_tag);
335        Ok(())
336    }
337
338    /// Reject the message
339    /// 拒绝消息
340    pub fn reject(&self, requeue: bool) -> Result<(), String>
341    {
342        // Mock implementation
343        // 模拟实现
344        tracing::debug!(
345            "Rejecting message with delivery tag: {}, requeue: {}",
346            self.delivery_tag,
347            requeue
348        );
349        Ok(())
350    }
351
352    /// Negative acknowledgement
353    /// 负向确认
354    pub fn nack(&self, requeue: bool) -> Result<(), String>
355    {
356        self.reject(requeue)
357    }
358}
