13/// Holds a live database transaction behind a mutex.
14/// 在互斥锁后持有活动的数据库事务。
15pub trait LiveTransaction: Send
16{
17    /// Commit the underlying transaction.
18    fn commit_boxed(
19        self: Box<Self>,
20    ) -> std::pin::Pin<Box<dyn Future<Output = TransactionResult<()>> + Send>>;
21    /// Roll back the underlying transaction.
22    fn rollback_boxed(
23        self: Box<Self>,
24    ) -> std::pin::Pin<Box<dyn Future<Output = TransactionResult<()>> + Send>>;
25}
26
27tokio::task_local! {
28    static ACTIVE_TX: Arc<Mutex<HashMap<String, (TransactionStatus, Box<dyn LiveTransaction>)>>>;
29}
30
31fn active_map() -> Arc<Mutex<HashMap<String, (TransactionStatus, Box<dyn LiveTransaction>)>>>
32{
33    ACTIVE_TX.try_with(Clone::clone).expect(
34        "No active transaction scope: wrap your transactional code with `with_transaction_scope`",
35    )
36}
37
38/// Initialize the task-local transaction map for the duration of `f`.
39/// 为 `f` 的执行期间初始化任务本地事务映射。
40///
41/// The `#[transactional]` macro wraps the annotated method body with this call so
42/// that `bind_transaction` / `take_transaction` share the same `Arc<Mutex<HashMap>>`
43/// throughout the whole begin → commit/rollback lifecycle.
44///
45/// `#[transactional]` 宏将带注解的方法体包裹在此调用中，使
46/// `bind_transaction` / `take_transaction` 在整个 begin → commit/rollback
47/// 生命周期中共享同一个 `Arc<Mutex<HashMap>>`。
48pub async fn with_transaction_scope<F, R>(f: F) -> R
49where
50    F: Future<Output = R>,
51{
52    ACTIVE_TX
53        .scope(Arc::new(Mutex::new(HashMap::new())), f)
54        .await
55}
56
57/// Bind a live transaction to a status name in the current task.
58/// 将活动事务绑定到当前任务中的状态名称。
59pub async fn bind_transaction(status: TransactionStatus, tx: Box<dyn LiveTransaction>)
60{
61    let map = active_map();
62    let mut guard = map.lock().await;
63    guard.insert(status.name().to_string(), (status, tx));
64}
65
66/// Take and remove a live transaction for commit/rollback.
67/// 取出并移除用于提交/回滚的活动事务。
68pub async fn take_transaction(status: &TransactionStatus) -> Option<Box<dyn LiveTransaction>>
69{
70    let map = active_map();
71    let mut guard = map.lock().await;
72    guard.remove(status.name()).map(|(_, tx)| tx)
73}
74
75/// Returns an existing active status for REQUIRED propagation, if any.
76/// 对于 REQUIRED 传播，返回现有活动状态（如有）。
77pub async fn current_status() -> Option<TransactionStatus>
78{
79    let map = active_map();
80    let guard = map.lock().await;
81    guard.values().next().map(|(s, _)| s.clone())
82}
83
84/// Resolve propagation: returns existing status or indicates a new transaction is needed.
85/// 解析传播行为：返回现有状态或指示需要新事务。
86pub async fn resolve_propagation(
87    definition: &TransactionDefinition,
88) -> Result<Option<TransactionStatus>, TransactionError>
89{
90    match definition.propagation
91    {
92        Propagation::Required | Propagation::Supports => Ok(current_status().await),
93        Propagation::Mandatory => match current_status().await
94        {
95            Some(s) => Ok(Some(s)),
96            None =>
97            {
98                Err(TransactionError::InvalidState("No existing transaction for MANDATORY".into()))
99            },
100        },
101        Propagation::Never =>
102        {
103            if current_status().await.is_some()
104            {
105                Err(TransactionError::InvalidState("Existing transaction present for NEVER".into()))
106            }
107            else
108            {
109                Ok(None)
110            }
111        },
112        Propagation::NotSupported | Propagation::RequiresNew | Propagation::Nested => Ok(None),
113    }
114}
