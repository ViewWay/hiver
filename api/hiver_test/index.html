§Example / 示例

ⓘ
use hiver_test::{HiverTest, TestClient};
use hiver_http::{Request, Response};

#[tokio::test]
async fn test_user_endpoint() {
    let client = TestClient::new().await;

    let response = client.get("/api/users/1")
        .send()
        .await;

    assert_eq!(response.status(), 200);
}
