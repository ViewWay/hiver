1//! WSDL generation from contract-first XSD
2//! 契约优先XSD的WSDL生成
3//!
4//! Equivalent to Spring WS WSDL generation
5//! 等价于 Spring WS WSDL生成
6
7use std::fmt::Write;
8
9use serde::{Deserialize, Serialize};
10
11/// WSDL definition / WSDL定义
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct WsdlDefinition
14{
15    /// Target namespace / 目标命名空间
16    pub target_namespace: String,
17    /// Service name / 服务名称
18    pub service_name: String,
19    /// Port type name / 端口类型名称
20    pub port_type_name: String,
21    /// List of operations / 操作列表
22    pub operations: Vec<WsdlOperation>,
23    /// Optional embedded XSD types schema / 可选的内嵌XSD类型模式
24    pub types_schema: Option<String>,
25}
26
27/// WSDL operation / WSDL操作
28#[derive(Debug, Clone, Serialize, Deserialize)]
29pub struct WsdlOperation
30{
31    /// Operation name / 操作名称
32    pub name: String,
33    /// Input message name / 输入消息名称
34    pub input_message: String,
35    /// Output message name / 输出消息名称
36    pub output_message: String,
37    /// SOAP action URI / SOAP操作URI
38    pub soap_action: String,
39}
40
41/// WSDL generator / WSDL生成器
42pub struct WsdlGenerator
43{
44    namespace: String,
45    service: String,
46    location: String,
47}
48
49impl WsdlGenerator
50{
51    /// Create a new WSDL generator / 创建新的WSDL生成器
52    pub fn new(namespace: &str, service: &str, location: &str) -> Self
53    {
54        Self {
55            namespace: namespace.to_string(),
56            service: service.to_string(),
57            location: location.to_string(),
58        }
59    }
60
61    /// Generate WSDL with operations / 生成包含操作的WSDL
62    pub fn generate(&self, operations: Vec<WsdlOperation>) -> WsdlDefinition
63    {
64        WsdlDefinition {
65            target_namespace: self.namespace.clone(),
66            service_name: format!("{}Service", self.service),
67            port_type_name: format!("{}Port", self.service),
68            operations,
69            types_schema: None,
70        }
71    }
72
73    /// Generate WSDL as XML string / 生成WSDL的XML字符串
74    pub fn to_xml(&self, def: &WsdlDefinition) -> String
75    {
76        let mut ops_xml = String::new();
77        for op in &def.operations
78        {
79            let _ = write!(
80                ops_xml,
81                r#"  <wsdl:operation name="{}">
82    <soap:operation soapAction="{}"/>
83    <wsdl:input message="tns:{}"/>
84    <wsdl:output message="tns:{}"/>
85  </wsdl:operation>
86"#,
87                escape_xml(&op.name),
88                escape_xml(&op.soap_action),
89                escape_xml(&op.input_message),
90                escape_xml(&op.output_message)
91            );
92        }
93
94        format!(
95            r#"<?xml version="1.0" encoding="UTF-8"?>
96<wsdl:definitions xmlns:wsdl="http://schemas.xmlsoap.org/wsdl/"
97  xmlns:soap="http://schemas.xmlsoap.org/wsdl/soap/"
98  xmlns:tns="{}"
99  targetNamespace="{}">
100  <wsdl:portType name="{}">
101{}
102  </wsdl:portType>
103  <wsdl:binding name="{}Binding" type="tns:{}">
104    <soap:binding transport="http://schemas.xmlsoap.org/soap/http"/>
105  </wsdl:binding>
106  <wsdl:service name="{}">
107    <wsdl:port name="{}Port" binding="tns:{}Binding">
108      <soap:address location="{}"/>
109    </wsdl:port>
110  </wsdl:service>
111</wsdl:definitions>"#,
112            escape_xml(&self.namespace),
113            escape_xml(&self.namespace),
114            escape_xml(&def.port_type_name),
115            ops_xml,
116            escape_xml(&def.port_type_name),
117            escape_xml(&def.port_type_name),
118            escape_xml(&def.service_name),
119            escape_xml(&def.port_type_name),
120            escape_xml(&def.port_type_name),
121            escape_xml(&self.location)
122        )
123    }
124}
125
126fn escape_xml(s: &str) -> String
127{
128    let mut out = String::with_capacity(s.len());
129    for c in s.chars()
130    {
131        match c
132        {
133            '&' => out.push_str("&amp;"),
134            '<' => out.push_str("&lt;"),
135            '>' => out.push_str("&gt;"),
136            '"' => out.push_str("&quot;"),
137            '\'' => out.push_str("&apos;"),
138            _ => out.push(c),
139        }
140    }
141    out
142}

152mod tests

154    use super::*;
155
156    #[test]
157    fn test_wsdl_generation()
158    {
159        let generator =
160            WsdlGenerator::new("http://example.com/ws", "User", "http://localhost:8080/ws");
161        let ops = vec![WsdlOperation {
162            name: "GetUser".into(),
163            input_message: "GetUserRequest".into(),
164            output_message: "GetUserResponse".into(),
165            soap_action: "urn:GetUser".into(),
166        }];
167        let def = generator.generate(ops);
168        let xml = generator.to_xml(&def);
169        assert!(xml.contains("GetUser"));
170        assert!(xml.contains("wsdl:definitions"));
171    }
