18/// Resource bundle message source
19/// 资源包消息源
20///
21/// Loads messages from properties files (like Java's ResourceBundle).
22/// 从属性文件加载消息（类似Java的ResourceBundle）。
23///
24/// # Spring Equivalent / Spring等价物
25///
26/// ```java
27/// @Bean
28/// public MessageSource messageSource() {
29///     ResourceBundleMessageSource source = new ResourceBundleMessageSource();
30///     source.setBasenames("messages", "errors");
31///     source.setDefaultEncoding("UTF-8");
32///     source.setCacheSeconds(3600);
33///     return source;
34/// }
35/// ```
36///
37/// # File Structure / 文件结构
38///
39/// ```text
40/// resources/
41/// ├── messages.properties              (default)
42/// ├── messages_en.properties
43/// ├── messages_en_US.properties
44/// ├── messages_zh.properties
45/// ├── messages_zh_CN.properties
46/// └── errors.properties
47/// ```
48pub struct ResourceBundleMessageSource
49{
50    /// Base names for resource bundles (e.g., "messages", "errors")
51    /// 资源包的基础名称（如"messages", "errors"）
52    basenames: Vec<String>,
53
54    /// Default locale (fallback when locale-specific file not found)
55    /// 默认语言环境（找不到特定语言环境文件时的回退）
56    default_locale: String,
57
58    /// Cache for loaded messages
59    /// 已加载消息的缓存
60    cache: Arc<RwLock<HashMap<String, HashMap<String, String>>>>,
61
62    /// Base resource path
63    /// 基础资源路径
64    base_path: PathBuf,
65
66    /// Cache duration in seconds
67    /// 缓存时长（秒）
68    cache_seconds: u64,
69
70    /// Last cache reload time
71    /// 上次缓存重载时间
72    last_reload: Arc<RwLock<tokio::time::Instant>>,
73}

148    /// Load properties file
149    /// 加载属性文件
150    async fn load_properties(
151        &self,
152        basename: &str,
153        locale: &str,
154    ) -> I18nResult<HashMap<String, String>>
155    {
156        let locale = Locale::parse(locale)?;
157
158        // Try locale-specific files first, then fall back to default
159        let file_names = self.get_file_names(basename, &locale);
160
161        for file_name in file_names
162        {
163            let file_path = self.base_path.join(&file_name);
164
165            if let Ok(content) = fs::read_to_string(&file_path).await
166            {
167                return self.parse_properties(&content);
168            }
169        }
170
171        // Try default (no locale)
172        let default_path = self.base_path.join(format!("{}.properties", basename));
173        if let Ok(content) = fs::read_to_string(&default_path).await
174        {
175            return self.parse_properties(&content);
176        }
177
178        // Return empty map if no file found (will use default message)
179        Ok(HashMap::new())
180    }
181
182    /// Get file names to try (in order)
183    /// 获取要尝试的文件名（按顺序）
184    fn get_file_names(&self, basename: &str, locale: &Locale) -> Vec<String>
185    {
186        let mut names = Vec::new();
187
188        // Try language_country_variant (e.g., messages_en_US.properties)
189        if let (Some(country), Some(_variant)) = (&locale.country, &locale.variant)
190        {
191            names.push(format!("{}_{}_{}.properties", basename, locale.language, country));
192        }
193
194        // Try language_country (e.g., messages_en_US.properties)
195        if let Some(country) = &locale.country
196        {
197            names.push(format!("{}_{}_{}.properties", basename, locale.language, country));
198        }
199
200        // Try language only (e.g., messages_en.properties)
201        names.push(format!("{}_{}.properties", basename, locale.language));
202
203        names
204    }
205
206    /// Parse properties file content
207    /// 解析属性文件内容
208    fn parse_properties(&self, content: &str) -> I18nResult<HashMap<String, String>>
209    {
210        let mut messages = HashMap::new();
211
212        for (line_num, line) in content.lines().enumerate()
213        {
214            let line = line.trim();
215
216            // Skip empty lines and comments
217            if line.is_empty() || line.starts_with('#') || line.starts_with('!')
218            {
219                continue;
220            }
221
222            // Parse key=value
223            if let Some(eq_pos) = line.find('=')
224            {
225                let key = line[..eq_pos].trim();
226                let value = line[eq_pos + 1..].trim();
227
228                // Handle unicode escapes (\uXXXX)
229                let value = self.decode_unicode(value);
230
231                messages.insert(key.to_string(), value);
232            }
233            else if !line.is_empty()
234            {
235                // Invalid line
236                tracing::warn!("Invalid line {} in properties file", line_num + 1);
237            }
238        }
239
240        Ok(messages)
241    }
242
243    /// Decode unicode escapes
244    /// 解码Unicode转义
245    fn decode_unicode(&self, s: &str) -> String
246    {
247        let mut result = String::new();
248        let mut chars = s.chars().peekable();
249
250        while let Some(c) = chars.next()
251        {
252            if c == '\\' && chars.peek() == Some(&'u')
253            {
254                chars.next(); // consume 'u'
255
256                // Read 4 hex digits
257                let mut hex = String::new();
258                for _ in 0..4
259                {
260                    if let Some(&c) = chars.peek()
261                        && c.is_ascii_hexdigit()
262                    {
263                        hex.push(c);
264                        chars.next();
265                    }
266                }
267
268                if let Ok(code) = u32::from_str_radix(&hex, 16)
269                    && let Some(c) = char::from_u32(code)
270                {
271                    result.push(c);
272                }
273            }
274            else
275            {
276                result.push(c);
277            }
278        }
279
280        result
281    }

283    /// Get messages for locale (with caching)
284    /// 获取语言环境的消息（带缓存）
285    async fn get_messages(&self, basename: &str, locale: &str) -> HashMap<String, String>
286    {
287        let cache_key = format!("{}_{}", basename, locale);
288
289        // Check cache
290        {
291            let cache = self.cache.read().await;
292            if let Some(messages) = cache.get(&cache_key)
293            {
294                return messages.clone();
295            }
296        }
297
298        // Load from file
299        match self.load_properties(basename, locale).await
300        {
301            Ok(messages) =>
302            {
303                let mut cache = self.cache.write().await;
304                cache.insert(cache_key, messages.clone());
305                messages
306            },
307            Err(_) => HashMap::new(),
308        }
309    }
310
311    /// Format message with arguments
312    /// 格式化带参数的消息
313    fn format_message(&self, template: &str, args: &[String]) -> String
314    {
315        let mut result = template.to_string();
316        for (i, arg) in args.iter().enumerate()
317        {
318            result = result.replace(&format!("{{{}}}", i), arg);
319        }
320        result
321    }

347#[async_trait::async_trait]
348impl MessageSource for ResourceBundleMessageSource
349{
350    async fn get_message(&self, code: &str, args: &[String], locale: &str) -> I18nResult<String>
351    {
352        // Try each basename in order
353        for basename in &self.basenames
354        {
355            let messages = self.get_messages(basename, locale).await;
356
357            if let Some(template) = messages.get(code)
358            {
359                return Ok(self.format_message(template, args));
360            }
361        }
362
363        // Try default locale
364        if locale != self.default_locale
365        {
366            for basename in &self.basenames
367            {
368                let messages = self.get_messages(basename, &self.default_locale).await;
369
370                if let Some(template) = messages.get(code)
371                {
372                    return Ok(self.format_message(template, args));
373                }
374            }
375        }
376
377        Err(I18nError::MessageNotFound {
378            code: code.to_string(),
379            locale: locale.to_string(),
380        })
381    }
382
383    async fn get_message_with_default(
384        &self,
385        code: &str,
386        args: &[String],
387        default_message: &str,
388        locale: &str,
389    ) -> String
390    {
391        match self.get_message(code, args, locale).await
392        {
393            Ok(msg) => msg,
394            Err(_) => self.format_message(default_message, args),
395        }
396    }
397}

399/// Resource bundle source
400/// 资源包源
401///
402/// Simple representation of a loaded resource bundle.
403/// 已加载资源包的简单表示。
404#[derive(Debug, Clone)]
405pub struct ResourceBundleSource
406{
407    /// Bundle basename
408    /// 包基础名称
409    pub basename: String,
410
411    /// Locale
412    /// 语言环境
413    pub locale: String,
414
415    /// Messages
416    /// 消息
417    pub messages: HashMap<String, String>,
418}
419
420impl ResourceBundleSource
421{
422    /// Create new resource bundle source
423    /// 创建新资源包源
424    pub fn new(basename: impl Into<String>, locale: impl Into<String>) -> Self
425    {
426        Self {
427            basename: basename.into(),
428            locale: locale.into(),
429            messages: HashMap::new(),
430        }
431    }
432
433    /// Add message
434    /// 添加消息
435    pub fn add_message(mut self, key: impl Into<String>, value: impl Into<String>) -> Self
436    {
437        self.messages.insert(key.into(), value.into());
438        self
439    }
440
441    /// Get message
442    /// 获取消息
443    pub fn get_message(&self, key: &str) -> Option<&str>
444    {
445        self.messages.get(key).map(|s| s.as_str())
446    }
447}
