41/// Transactional options
42/// Transactional选项
43///
44/// Equivalent to Spring's @Transactional annotation parameters.
45/// 等价于Spring的@Transactional注解参数。
46///
47/// # Spring Equivalent / Spring等价物
48///
49/// ```java
50/// @Transactional(
51///     propagation = Propagation.REQUIRES_NEW,
52///     isolation = Isolation.SERIALIZABLE,
53///     timeout = 30,
54///     readOnly = true,
55///     rollbackFor = {Exception.class},
56///     noRollbackFor = {RuntimeException.class}
57/// )
58/// ```
59#[derive(Debug, Clone, Serialize, Deserialize)]
60pub struct TransactionalOptions
61{
62    /// Transaction name
63    /// 事务名称
64    #[serde(default)]
65    pub name: Option<String>,
66
67    /// Propagation behavior
68    /// 传播行为
69    #[serde(default)]

73    /// 隔离级别
74    #[serde(default)]
75    pub isolation: Option<IsolationLevel>,
76
77    /// Timeout in seconds
78    /// 超时时间（秒）
79    #[serde(default)]
80    pub timeout_secs: Option<u64>,
81
82    /// Read-only flag
83    /// 只读标志
84    #[serde(default = "default_read_only")]
85    pub read_only: bool,
86
87    /// Rollback for exceptions
88    /// 回滚异常
89    #[serde(default)]
90    pub rollback_for: Vec<String>,
91
92    /// No rollback for exceptions
93    /// 不回滚异常
94    #[serde(default)]

169    /// 添加不回滚异常类型

173        self

175
176    /// Check if should rollback for error
177    /// 检查错误是否应回滚
178    pub fn should_rollback(&self, error: &TransactionError) -> bool
179    {
180        // Default behavior: rollback for all errors
181        if self.rollback_for.is_empty() && self.no_rollback_for.is_empty()
182        {
183            return true;
184        }
185
186        // Check no_rollback_for first
187        for exception in &self.no_rollback_for
188        {
189            // Simplified check - in real implementation would use type matching
190            if error.to_string().contains(exception)
191            {
192                return false;
193            }
194        }
195
196        // Check rollback_for
197        if self.rollback_for.is_empty()
198        {
199            return true;
200        }
201
202        for exception in &self.rollback_for
203        {
204            if error.to_string().contains(exception)
205            {
206                return true;
207            }
208        }
209
210        false
211    }
212}

222/// Transactional trait
223/// Transactional trait
224///
225/// Implement this trait to make a struct transactional.
226/// 实现此trait以使结构体支持事务。
227///
228/// Equivalent to Spring's @Transactional on a class.
229/// 等价于Spring类上的@Transactional。
230pub trait Transactional
231{
232    /// Execute a function within a transaction
233    /// 在事务中执行函数
234    async fn in_transaction<F, T, E>(&self, f: F) -> TransactionResult<T>
235    where
236        F: FnOnce() -> futures::future::BoxFuture<'static, Result<T, E>> + Send + Sync,
237        T: Send + 'static,
238        E: Into<TransactionError> + Send + 'static,
239    {
240        // Default implementation uses global transaction manager
241        // In real implementation, this would use injected transaction manager
242        let result = f().await;
243
244        match result
245        {
246            Ok(value) => Ok(value),
247            Err(e) => Err(e.into()),
248        }
249    }
250
251    /// Execute with options
252    /// 使用选项执行
253    async fn in_transaction_with_options<F, T, E>(
254        &self,
255        options: &TransactionalOptions,
256        f: F,
257    ) -> TransactionResult<T>
258    where
259        F: FnOnce() -> futures::future::BoxFuture<'static, Result<T, E>> + Send + Sync,
260        T: Send + 'static,
261        E: Into<TransactionError> + Send + 'static,
262    {
263        let result = f().await;
264
265        match result
266        {
267            Ok(value) => Ok(value),
268            Err(e) =>
269            {
270                let tx_error = e.into();
271                if options.should_rollback(&tx_error)
272                {
273                    return Err(tx_error);
274                }
275                // Even if we don't rollback, the operation failed
276                // 即使不回滚，操作仍然失败
277                Err(tx_error)
278            },
279        }
280    }
281}

287/// Transaction guard for RAII-style transaction management
288/// RAII风格事务管理的事务守卫
289///
290/// Automatically commits or rolls back when dropped.
291/// 删除时自动提交或回滚。
292///
293/// Equivalent to Spring's `TransactionTemplate` with try-with-resources.
294/// 等价于Spring与try-with-resources的TransactionTemplate。
295pub(crate) struct TransactionGuard<'a>
296{
297    /// Transaction status
298    /// 事务状态
299    status: crate::TransactionStatus,
300
301    /// Transaction manager
302    /// 事务管理器
303    manager: &'a dyn crate::TransactionManager,
304
305    /// Committed flag
306    /// 已提交标志
307    committed: Arc<std::sync::atomic::AtomicBool>,
308}
309
310impl<'a> TransactionGuard<'a>
311{
312    /// Create a new transaction guard
313    /// 创建新的事务守卫
314    pub(crate) fn new(
315        status: crate::TransactionStatus,
316        manager: &'a dyn crate::TransactionManager,
317    ) -> Self
318    {
319        Self {
320            status,
321            manager,
322            committed: Arc::new(std::sync::atomic::AtomicBool::new(false)),
323        }
324    }
325
326    /// Commit the transaction
327    /// 提交事务
328    pub(crate) async fn commit(self) -> TransactionResult<()>
329    {
330        self.committed
331            .store(true, std::sync::atomic::Ordering::SeqCst);
332        self.manager.commit(self.status.clone()).await
333    }
334
335    /// Rollback the transaction
336    /// 回滚事务
337    pub(crate) async fn rollback(self) -> TransactionResult<()>
338    {
339        self.manager.rollback(self.status.clone()).await
340    }
341
342    /// Mark for rollback only
343    /// 标记为仅回滚
344    pub(crate) fn set_rollback_only(&self)
345    {
346        self.status.set_rollback_only();
347    }
348}

350impl Drop for TransactionGuard<'_>
351{
352    fn drop(&mut self)
353    {
354        if !self.committed.load(std::sync::atomic::Ordering::SeqCst)
355        {
356            // In a real async Drop scenario, we'd need to handle this differently
357            // For now, this is a simplified implementation
358            if !self.status.is_completed()
359            {
360                // Would trigger rollback
361            }
362        }
363    }

393    #[test]
394    fn test_should_rollback()
395    {
396        // Test 1: Default behavior - rollback for all errors
397        // 测试 1: 默认行为 - 所有错误都回滚
398        let default_options = TransactionalOptions::new();
399        let any_error = TransactionError::CommitFailed("Some error".to_string());
400        assert!(default_options.should_rollback(&any_error));
401
402        // Test 2: Explicit no-rollback for specific error
403        // 测试 2: 明确指定特定错误不回滚
404        let options = TransactionalOptions::new().no_rollback_for("Invalid");
405
406        let validation_error = TransactionError::InvalidState("Invalid input".to_string());
407        // Should not rollback for "Invalid" errors (explicitly excluded)
408        // "Invalid transaction state: Invalid input" contains "Invalid"
409        assert!(!options.should_rollback(&validation_error));
410
411        // Other errors should still rollback
412        let commit_error = TransactionError::CommitFailed("Connection failed".to_string());
413        assert!(options.should_rollback(&commit_error));
414    }
