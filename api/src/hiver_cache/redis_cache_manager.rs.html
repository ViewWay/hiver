1//! Redis-backed CacheManager implementation
2//! Redis 缓存管理器实现
3//!
4//! Auto-selects between Redis and in-memory backends based on configuration
5//! and connectivity. When Redis is unavailable and `fallback_to_memory` is
6//! enabled, operations silently fall back to `SimpleCacheManager`.
7//! 根据配置和连接性自动选择 Redis 或内存后端。当 Redis 不可用且
8//! `fallback_to_memory` 启用时，操作会静默回退到 `SimpleCacheManager`。

20// ---------------------------------------------------------------------------
21// RedisCacheWorker — type-erased wrapper around RedisCache
22// RedisCacheWorker — RedisCache 的类型擦除包装器
23// ---------------------------------------------------------------------------
24
25/// Type-erased wrapper that exposes `RedisCache<K, V>` through the
26/// `CacheWorker` trait, allowing heterogeneous storage in the manager.
27/// 类型擦除包装器，通过 `CacheWorker` trait 暴露 `RedisCache<K, V>`，
28/// 允许在管理器中异构存储。
29pub struct RedisCacheWorker
30{
31    name: String,
32    config: CacheConfig,
33    stats: Arc<RwLock<CacheStats>>,
34    size_fn: Box<dyn Fn() -> PinnedFuture<usize> + Send + Sync>,
35    clear_fn: Box<dyn Fn() -> PinnedFuture<()> + Send + Sync>,
36}
37
38/// Type alias for a pinned, boxed, sendable future.
39/// 固定、装箱、可发送的 future 的类型别名。
40type PinnedFuture<T> = std::pin::Pin<Box<dyn Future<Output = T> + Send>>;
41
42impl RedisCacheWorker
43{
44    /// Create a new `RedisCacheWorker` from a typed `RedisCache`.
45    /// 从类型化的 `RedisCache` 创建新的 `RedisCacheWorker`。
46    pub fn new<K, V>(cache: RedisCache<K, V>, name: String) -> Self
47    where
48        K: Hash + Eq + Send + Sync + serde::Serialize + serde::de::DeserializeOwned + 'static,
49        V: Clone + Send + Sync + serde::Serialize + serde::de::DeserializeOwned + 'static,
50    {
51        let config = cache.config().clone();
52        let stats = Arc::clone(cache.stats_handle());
53
54        let cache_ptr = Arc::new(cache);
55        let size_cache = cache_ptr.clone();
56        let clear_cache = cache_ptr.clone();
57
58        Self {
59            name,
60            config,
61            stats,
62            size_fn: Box::new(move || {
63                let c = size_cache.clone();
64                Box::pin(async move { c.size().await })
65            }),
66            clear_fn: Box::new(move || {
67                let c = clear_cache.clone();
68                Box::pin(async move { c.clear().await })
69            }),
70        }
71    }
72}

104// RedisCacheManager
105// ---------------------------------------------------------------------------
106
107/// Cache manager backed by Redis, implementing the `CacheManager` trait.
108/// 基于 Redis 的缓存管理器，实现 `CacheManager` trait。
109///
110/// On construction, attempts to connect to Redis. If the connection fails
111/// and `fallback_to_memory` is enabled in `RedisConfig`, all operations
112/// delegate to an in-memory `SimpleCacheManager`.
113/// 构造时尝试连接 Redis。如果连接失败且 `RedisConfig` 中
114/// `fallback_to_memory` 启用，所有操作委托给内存 `SimpleCacheManager`。
115pub struct RedisCacheManager
116{
117    /// Per-name Redis configuration.
118    /// 每个缓存名称的 Redis 配置。
119    redis_config: RedisConfig,
120
121    /// Registered caches.
122    /// 已注册的缓存。
123    caches: RwLock<HashMap<String, Arc<dyn CacheWorker>>>,
124
125    /// Memory fallback manager (used when Redis is unavailable).
126    /// 内存回退管理器（Redis 不可用时使用）。
127    fallback: SimpleCacheManager,
128
129    /// Whether we are currently operating in fallback mode.
130    /// 当前是否处于回退模式。
131    is_fallback: std::sync::atomic::AtomicBool,
132}
133
134impl RedisCacheManager
135{
136    /// Create a new `RedisCacheManager` with the given Redis configuration.
137    /// 使用给定的 Redis 配置创建新的 `RedisCacheManager`。
138    ///
139    /// The `default_cache_config` is used as the template for newly created
140    /// caches when `create_cache` is called without a specific config.
141    /// `default_cache_config` 用作调用 `create_cache` 时新创建缓存的模板。
142    pub fn new(redis_config: RedisConfig, default_cache_config: CacheConfig) -> Self
143    {
144        Self {
145            redis_config,
146            caches: RwLock::new(HashMap::new()),
147            fallback: SimpleCacheManager::with_config(default_cache_config),
148            is_fallback: std::sync::atomic::AtomicBool::new(false),
149        }
150    }
151
152    /// Check whether this manager is operating in fallback (in-memory) mode.
153    /// 检查此管理器是否在回退（内存）模式下运行。
154    pub fn is_fallback(&self) -> bool
155    {
156        self.is_fallback.load(std::sync::atomic::Ordering::Relaxed)
157    }
158
159    /// Set the fallback mode flag.
160    /// 设置回退模式标志。
161    pub fn set_fallback(&self, value: bool)
162    {
163        self.is_fallback
164            .store(value, std::sync::atomic::Ordering::Relaxed);
165    }
166
167    /// Create a typed `RedisCache` and register it with the manager.
168    /// 创建类型化的 `RedisCache` 并注册到管理器中。
169    ///
170    /// If the manager is in fallback mode, this creates an in-memory cache
171    /// via the `SimpleCacheManager` instead.
172    /// 如果管理器处于回退模式，则通过 `SimpleCacheManager` 创建内存缓存。
173    pub async fn create_redis_cache<K, V>(
174        &self,
175        name: &str,
176        config: CacheConfig,
177    ) -> Arc<RedisCache<K, V>>
178    where
179        K: Hash + Eq + Send + Sync + serde::Serialize + serde::de::DeserializeOwned + 'static,
180        V: Clone + Send + Sync + serde::Serialize + serde::de::DeserializeOwned + 'static,
181    {
182        let named_redis_config = RedisConfig {
183            key_prefix: format!("cache:{}:", name),
184            ..self.redis_config.clone()
185        };
186
187        let cache = RedisCache::new(config.clone(), named_redis_config).await;
188
189        // If the newly created cache is not connected, switch to fallback.
190        // 如果新创建的缓存未连接，切换到回退模式。
191        if !cache.is_connected().await
192        {
193            self.set_fallback(true);
194        }
195
196        let worker = RedisCacheWorker::new(cache.clone(), name.to_string());
197        let mut caches = self.caches.write().await;
198        caches.insert(name.to_string(), Arc::new(worker));
199
200        Arc::new(cache)
201    }
202}
203
204impl CacheManager for RedisCacheManager
205{
206    fn get_cache(&self, name: &str) -> Option<Arc<dyn CacheWorker>>
207    {
208        // Try synchronous read — since `caches` is a tokio RwLock we use
209        // try_read. If that fails (contended), fall through to the memory
210        // fallback manager.
211        // 尝试同步读取 — 因为 `caches` 是 tokio RwLock，我们使用 try_read。
212        // 如果失败（竞争），回退到内存管理器。
213        if let Ok(caches) = self.caches.try_read()
214        {
215            if let Some(worker) = caches.get(name)
216            {
217                return Some(worker.clone());
218            }
219        }
220
221        // Delegate to fallback manager
222        // 委托给回退管理器
223        self.fallback.get_cache(name)
224    }
225
226    fn get_cache_names(&self) -> Vec<String>
227    {
228        if let Ok(caches) = self.caches.try_read()
229        {
230            caches.keys().cloned().collect()
231        }
232        else
233        {
234            Vec::new()
235        }
236    }
237
238    fn create_cache(&self, name: &str, config: CacheConfig) -> Option<Arc<dyn CacheWorker>>
239    {
240        // Note: Synchronous context — cannot call async `create_redis_cache`.
241        // Instead, create a MemoryCache as a placeholder that will be replaced
242        // when the async version is called.
243        // 注意：同步上下文 — 无法调用异步 `create_redis_cache`。
244        // 而是创建一个 MemoryCache 作为占位符，在调用异步版本时替换。
245        self.fallback.create_cache(name, config)
246    }
247}

272    async fn test_redis_cache_manager_creates_cache_in_fallback()
273    {
274        let redis_cfg = RedisConfig::new("redis://127.0.0.1:1").fallback_to_memory();
275        let manager = RedisCacheManager::new(redis_cfg, test_config("default"));
276
277        let cache: Arc<RedisCache<String, String>> = manager
278            .create_redis_cache("users", test_config("users"))
279            .await;
280
281        // Should be in fallback mode since Redis is unreachable.
282        // 由于 Redis 不可达，应处于回退模式。

285        // The created cache should still work via memory fallback.
286        // 创建的缓存应通过内存回退仍然工作。
