2//! Hiver Exceptions - Spring @`ControllerAdvice` 等价功能
3//!
4//! # Equivalent to Spring / 等价于 Spring
5//!
6//! - `@ControllerAdvice` - `ControllerAdvice` trait
7//! - `@ExceptionHandler` - `ExceptionHandler` trait
8//! - `ResponseEntityExceptionHandler` - `DefaultExceptionHandler`
9//! - `@ResponseStatus` - `status_code()` method
10//!
11//! # Example / 示例
12//!
13//! ```rust,no_run,ignore
14//! use hiver_exceptions::{ControllerAdvice, ExceptionHandler};
15//! use hiver_http::{Request, Response, StatusCode};
16//!
17//! struct MyExceptionHandler;
18//!
19//! impl ControllerAdvice for MyExceptionHandler {
20//!     // Auto-generated from @ExceptionHandler methods
21//! }
22//!
23//! impl ExceptionHandler<ValidationError> for MyExceptionHandler {
24//!     fn handle(&self, err: ValidationError, _req: &Request) -> Response {
25//!         Response::bad_request().json(serde_json::json!({
26//!             "error": err.message
27//!         }))
28//!     }
29//! }
30//! ```
